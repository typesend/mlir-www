40 /// Module import implementation class that provides methods to import globals
41 /// and functions from an LLVM module into an MLIR module. It holds mappings
42 /// between the original and translated globals, basic blocks, and values used
43 /// during the translation. Additionally, it keeps track of the current constant
44 /// insertion point since LLVM immediate values translate to MLIR operations
45 /// that are introduced at the beginning of the region.
46 class ModuleImport {
47 public:
48  ModuleImport(ModuleOp mlirModule, std::unique_ptr<llvm::Module> llvmModule);

69  /// Provides write-once access to store the MLIR value corresponding to the
70  /// given LLVM value.
71  Value &mapValue(llvm::Value *value) {
72  Value &mlir = valueMapping[value];
73  assert(mlir == nullptr &&
74  "attempting to map a value that is already mapped");
75  return mlir;

81  /// Stores a mapping between an LLVM instruction and the imported MLIR
82  /// operation if the operation returns no result. Asserts if the operation
83  /// returns a result and should be added to valueMapping instead.
84  void mapNoResultOp(llvm::Instruction *llvm, Operation *mlir) {
85  mapNoResultOp(llvm) = mlir;

88  /// Provides write-once access to store the MLIR operation corresponding to
89  /// the given LLVM instruction if the operation returns no result. Asserts if
90  /// the operation returns a result and should be added to valueMapping
91  /// instead.
92  Operation *&mapNoResultOp(llvm::Instruction *inst) {
93  Operation *&mlir = noResultOpMapping[inst];
94  assert(inst->getType()->isVoidTy() &&
95  "attempting to map an operation that returns a result");
96  assert(mlir == nullptr &&
97  "attempting to map an operation that is already mapped");
98  return mlir;
99  }

101  /// Returns the MLIR operation mapped to the given LLVM instruction. Queries
102  /// valueMapping and noResultOpMapping to support operations with and without
103  /// result.
104  Operation *lookupOperation(llvm::Instruction *inst) {
105  if (Value value = lookupValue(inst))
106  return value.getDefiningOp();
107  return noResultOpMapping.lookup(inst);
108  }

110  /// Stores the mapping between an LLVM block and its MLIR counterpart.
111  void mapBlock(llvm::BasicBlock *llvm, Block *mlir) {
112  auto result = blockMapping.try_emplace(llvm, mlir);
113  (void)result;
114  assert(result.second && "attempting to map a block that is already mapped");
115  }

117  /// Returns the MLIR block mapped to the given LLVM block.
118  Block *lookupBlock(llvm::BasicBlock *block) const {
119  return blockMapping.lookup(block);
