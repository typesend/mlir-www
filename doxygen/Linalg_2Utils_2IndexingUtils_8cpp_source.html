45 Value createOrFoldDimOp(OpBuilder &b, Location loc, Value val, int64_t dim) {
46  if (val.getType().isa<UnrankedMemRefType, MemRefType>())
47  return b.createOrFold<memref::DimOp>(loc, val, dim);
48  if (val.getType().isa<UnrankedTensorType, RankedTensorType>())
49  return b.createOrFold<tensor::DimOp>(loc, val, dim);
50  llvm_unreachable("Expected MemRefType or TensorType");
51 }

53 OpFoldResult createFoldedDimOp(OpBuilder &b, Location loc, Value val,
54  int64_t dim) {
55  auto shapedType = val.getType().cast<ShapedType>();
56  if (!shapedType.hasRank() || shapedType.isDynamicDim(dim))
57  return createOrFoldDimOp(b, loc, val, dim);
58  return b.getIndexAttr(shapedType.getDimSize(dim));
59 }

61 SmallVector<Value> createDynamicDimensions(OpBuilder &b, Location loc,
62  Value val) {
63  auto shapedType = val.getType().cast<ShapedType>();
64  assert(shapedType.hasRank() && "`val` must have a static rank");
65  SmallVector<Value> res;
66  res.reserve(shapedType.getRank());
67  for (const auto &dim : llvm::enumerate(shapedType.getShape())) {
68  if (dim.value() == ShapedType::kDynamic)
69  res.push_back(createOrFoldDimOp(b, loc, val, dim.index()));
70  }
71  return res;
72 }

74 SmallVector<OpFoldResult> getMixedDimensions(OpBuilder &b, Location loc,
75  Value val) {
76  auto shapedType = val.getType().cast<ShapedType>();
77  assert(shapedType.hasRank() && "`val` must have a static rank");
78  SmallVector<Value> dynamicDims = createDynamicDimensions(b, loc, val);
79  return getMixedValues(shapedType.getShape(), dynamicDims, b);
80 }
