222 Attribute SparseTensorEncodingAttr::parse(AsmParser &parser, Type type) {
223 #define RETURN_ON_FAIL(stmt) \
224  if (failed(stmt)) { \
225  return {}; \
226  }
227 #define ERROR_IF(COND, MSG) \
228  if (COND) { \
229  parser.emitError(parser.getNameLoc(), MSG); \
230  return {}; \
231  }
232 
233  RETURN_ON_FAIL(parser.parseLess())
234  RETURN_ON_FAIL(parser.parseLBrace())
235 
236  // Process the data from the parsed dictionary value into struct-like data.
237  SmallVector<DimLevelType> lvlTypes;
238  SmallVector<SparseTensorDimSliceAttr> slices;
239  AffineMap dimOrd = {};
240  AffineMap higherOrd = {};
241  unsigned posWidth = 0;
242  unsigned crdWidth = 0;
243 
244  StringRef attrName;
245  // Exactly 6 keys.
246  SmallVector<StringRef, 6> keys = {"dimLevelType", "dimOrdering",
247  "higherOrdering", "posWidth",
248  "crdWidth", "slice"};
249  while (succeeded(parser.parseOptionalKeyword(&attrName))) {
250  if (!llvm::is_contained(keys, attrName)) {
251  parser.emitError(parser.getNameLoc(), "unexpected key: ") << attrName;
252  return {};
253  }
254 
255  // Consume the `=` after keys
256  RETURN_ON_FAIL(parser.parseEqual())
257  // FIXME: using `operator==` below duplicates the string comparison
258  // cost of the `is_contained` check above. Should instead use some
259  // "find" function that returns the index into `keys` so that we can
260  // dispatch on that instead.
261  if (attrName == "dimLevelType") {
262  Attribute attr;
263  RETURN_ON_FAIL(parser.parseAttribute(attr));
264  auto arrayAttr = attr.dyn_cast<ArrayAttr>();
265  ERROR_IF(!arrayAttr, "expected an array for dimension level types")
266  for (auto i : arrayAttr) {
267  auto strAttr = i.dyn_cast<StringAttr>();
268  ERROR_IF(!strAttr, "expected a string value in dimension level types")
269  auto strVal = strAttr.getValue();
270  if (auto optDLT = parseDLT(strVal)) {
271  lvlTypes.push_back(optDLT.value());
272  } else {
273  parser.emitError(parser.getNameLoc(),
274  "unexpected dimension level type: ")
275  << strVal;
276  return {};
277  }
278  }
279  } else if (attrName == "dimOrdering") {
280  Attribute attr;
281  RETURN_ON_FAIL(parser.parseAttribute(attr))
282  auto affineAttr = attr.dyn_cast<AffineMapAttr>();
283  ERROR_IF(!affineAttr, "expected an affine map for dimension ordering")
284  dimOrd = affineAttr.getValue();
285  } else if (attrName == "higherOrdering") {
286  Attribute attr;
287  RETURN_ON_FAIL(parser.parseAttribute(attr))
288  auto affineAttr = attr.dyn_cast<AffineMapAttr>();
289  ERROR_IF(!affineAttr, "expected an affine map for higher ordering")
290  higherOrd = affineAttr.getValue();
291  } else if (attrName == "posWidth") {
292  Attribute attr;
293  RETURN_ON_FAIL(parser.parseAttribute(attr))
294  auto intAttr = attr.dyn_cast<IntegerAttr>();
295  ERROR_IF(!intAttr, "expected an integral position bitwidth")
296  posWidth = intAttr.getInt();
297  } else if (attrName == "crdWidth") {
298  Attribute attr;
299  RETURN_ON_FAIL(parser.parseAttribute(attr))
300  auto intAttr = attr.dyn_cast<IntegerAttr>();
301  ERROR_IF(!intAttr, "expected an integral index bitwidth")
302  crdWidth = intAttr.getInt();
303  } else if (attrName == "slice") {
304  RETURN_ON_FAIL(parser.parseLSquare())
305  // Dispatches to DimSliceAttr to skip mnemonic
306  bool finished = false;
307  while (auto attr = SparseTensorDimSliceAttr::parse(parser, nullptr)) {
308  auto sliceAttr = attr.cast<SparseTensorDimSliceAttr>();
309  slices.push_back(sliceAttr);
310  if (parser.parseOptionalComma().failed()) {
311  finished = true;
312  break;
313  }
314  }
315  // Wrong when parsing slices
316  if (!finished)
317  return {};
318  RETURN_ON_FAIL(parser.parseRSquare())
319  }
320 
321  // Only the last item can omit the comma
322  if (parser.parseOptionalComma().failed())
323  break;
324  }
325 
326  RETURN_ON_FAIL(parser.parseRBrace())
327  RETURN_ON_FAIL(parser.parseGreater())
328 #undef ERROR_IF
329 #undef RETURN_ON_FAIL
330 
331  // Construct struct-like storage for attribute.
332  return parser.getChecked<SparseTensorEncodingAttr>(
333  parser.getContext(), lvlTypes, dimOrd, higherOrd, posWidth, crdWidth,
334  slices);
335 }

367 LogicalResult SparseTensorEncodingAttr::verify(
368  function_ref<InFlightDiagnostic()> emitError,
369  ArrayRef<DimLevelType> dimLevelType, AffineMap dimOrdering,
370  AffineMap higherOrdering, unsigned posWidth, unsigned crdWidth,
371  ArrayRef<SparseTensorDimSliceAttr> dimSlices) {
372  if (!acceptBitWidth(posWidth))
373  return emitError() << "unexpected position bitwidth: " << posWidth;
374  if (!acceptBitWidth(crdWidth))
375  return emitError() << "unexpected coordinate bitwidth: " << crdWidth;
376  // Before we can check that the level-rank is consistent/coherent
377  // across all fields, we need to define it. The source-of-truth for
378  // the `getLvlRank` method is the length of the level-types array,
379  // since it must always be provided and have full rank; therefore we
380  // use that same source-of-truth here.
381  const Level lvlRank = dimLevelType.size();
382  if (lvlRank == 0)
383  return emitError() << "expected a non-empty array for level types";
384  if (dimOrdering) {
385  if (!dimOrdering.isPermutation())
386  return emitError()
387  << "expected a permutation affine map for dimension ordering";
388  if (dimOrdering.getNumResults() != lvlRank)
389  return emitError() << "unexpected mismatch in ordering and dimension "
390  "level types size";
391  }
392  if (higherOrdering) {
393  if (higherOrdering.getNumDims() >= higherOrdering.getNumResults())
394  return emitError() << "unexpected higher ordering mapping from "
395  << higherOrdering.getNumDims() << " to "
396  << higherOrdering.getNumResults();
397  if (higherOrdering.getNumResults() != lvlRank)
398  return emitError() << "unexpected mismatch in higher ordering and "
399  "dimension level types size";
400  }
401  if (!dimSlices.empty() && dimSlices.size() != lvlRank) {
402  return emitError() << "unexpected mismatch in dimension slices and "
403  "dimension level type size";
404  }
405  return success();
406 }

413 LogicalResult SparseTensorEncodingAttr::verifyEncoding(
414  ArrayRef<DynSize> dimShape, Type elementType,
415  function_ref<InFlightDiagnostic()> emitError) const {
416  // Check structural integrity. In particular, this ensures that the
417  // level-rank is coherent across all the fields.
418  RETURN_FAILURE_IF_FAILED(verify(emitError, getDimLevelType(),
419  getDimOrdering(), getHigherOrdering(),
420  getPosWidth(), getCrdWidth(), getDimSlices()))
421  // Check integrity with tensor type specifics. In particular, we
422  // need only check that the dimension-rank of the tensor agrees with
423  // the dimension-rank of the encoding.
424  const Dimension dimRank = dimShape.size();
425  if (dimRank == 0)
426  return emitError() << "expected non-scalar sparse tensor";
427  if (const auto higherOrdering = getHigherOrdering()) {
428  if (higherOrdering.getNumDims() != dimRank)
429  return emitError() << "expected an affine map with " << dimRank
430  << " dimensions for higher ordering";
431  // TODO: verification of higher ordering contents
432  } else if (dimRank != getLvlRank()) {
433  return emitError() << "expected an array of size " << dimRank
434  << " for dimension level types";
435  }
436  return success();
437 }

453  Level startLvl, bool isUnique) {

456  return false;

483 RankedTensorType sparse_tensor::getCOOFromTypeWithOrdering(RankedTensorType rtt,
484  AffineMap lvlPerm,
485  bool ordered) {
486  const SparseTensorType src(rtt);
487  // The dim-rank of the source `RankedTensorType` is used as the lvl-rank
488  // of the result `RankedTensorType`. This follows from the fact that the
489  // result's encoding has the default higher-ordering (hence the result's
490  // lvl-rank equals its dim-rank). We don't need to assert that `lvlRank`
491  // agrees with the size of `lvlPerm` because that will be verified by
492  // `STEA::get`.
493  const Level lvlRank = src.getDimRank();
494  SmallVector<DimLevelType> lvlTypes;
495 
496  // An unordered and non-unique compressed level at beginning.
497  // If this is also the last level, then it is unique.
498  lvlTypes.push_back(
499  *getDimLevelType(LevelFormat::Compressed, ordered, lvlRank == 1));
500  if (lvlRank > 1) {
501  // TODO: it is actually ordered at the level for ordered input.
502  // Followed by unordered non-unique n-2 singleton levels.
503  std::fill_n(std::back_inserter(lvlTypes), lvlRank - 2,
504  *getDimLevelType(LevelFormat::Singleton, ordered, false));
505  // Ends by a unique singleton level unless the lvlRank is 1.
506  lvlTypes.push_back(*getDimLevelType(LevelFormat::Singleton, ordered, true));
507  }
508 
509  // TODO: Maybe pick the bitwidth based on input/output tensors (probably the
510  // largest one among them) in the original operation instead of using the
511  // default value.
512  unsigned posWidth = src.getPosWidth();
513  unsigned crdWidth = src.getCrdWidth();
514  auto enc = SparseTensorEncodingAttr::get(src.getContext(), lvlTypes, lvlPerm,
515  AffineMap(), posWidth, crdWidth);
516  return RankedTensorType::get(src.getDimShape(), src.getElementType(), enc);
517 }

577 /// ordered/unique DLT such that "compressed-nu-no" and "compressed-nu" (as well

585

592  // value for different bitwidth, it also avoids casting between index and
593  // integer (returned by DimOp)
594  0, 0, enc.getDimSlices());

646 static LogicalResult verifyPackUnPack(Operation *op, bool requiresStaticShape,
647  SparseTensorType tensorTp,
648  RankedTensorType valuesTp,
649  RankedTensorType coordinatesTp,
650  IntegerAttr batchedLvls) {
651  unsigned nBatched = batchedLvls ? batchedLvls.getValue().getZExtValue() : 0;
652  if (requiresStaticShape && !tensorTp.hasStaticDimShape())
653  return op->emitError("the sparse-tensor must have static shape");
654  if (!tensorTp.hasEncoding())
655  return op->emitError("the sparse-tensor must have an encoding attribute");
656  if (!tensorTp.isIdentity())
657  return op->emitError("the sparse-tensor must have the identity mapping");
658  if (!isCOOType(tensorTp.getEncoding(), nBatched, true))
659  return op->emitError("the sparse-tensor must have a COO type");
660 
661  if (coordinatesTp.getRank() != 2 + nBatched)
662  return op->emitError("coordinates must have rank 2 + batched_lvls");
663  if (requiresStaticShape && !coordinatesTp.hasStaticShape())
664  return op->emitError("coordinates must have static shape");
665  if (coordinatesTp.getElementType() != tensorTp.getCrdType())
666  return op->emitError("input/output coordinate-types don't match");
667 
668  if (valuesTp.getRank() != 1 + nBatched)
669  return op->emitError("values must have rank 1 + batched_lvls");
670  if (requiresStaticShape && !valuesTp.hasStaticShape())
671  return op->emitError("values must have static shape");
672  if (valuesTp.getElementType() != tensorTp.getElementType())
673  return op->emitError("input/output element-types don't match");
674 
675  for (unsigned i = 0; i < nBatched; i++) {
676  const auto valBatch = valuesTp.getShape()[i];
677  const auto crdBatch = coordinatesTp.getShape()[i];
678  if (ShapedType::isDynamic(valBatch) || ShapedType::isDynamic(crdBatch) ||
679  crdBatch != valBatch) {
680  return op->emitError(
681  "values/coordinates batched level sizes don't match statically");
682  }
683  }
684 
685  const auto valuesNSE = valuesTp.getShape()[nBatched];
686  const auto coordsNSE = coordinatesTp.getShape()[nBatched];
687  if (!ShapedType::isDynamic(valuesNSE) && !ShapedType::isDynamic(coordsNSE) &&
688  valuesNSE != coordsNSE)
689  return op->emitError("values/coordinates number-of-elements don't match");
690 
691  // NOTE: We use `getLvlRank` because the `coordinatesTp` is for
692  // level-coordinates (cf., the op documentation).
693  const DynSize coordsRank = coordinatesTp.getShape()[1 + nBatched];
694  const Level tensorRank = tensorTp.getLvlRank();
695  // FIXME: replace the `operator!=` with our backported `safelyNE`.
696  if (!ShapedType::isDynamic(coordsRank) &&
697  coordsRank != static_cast<DynSize>(tensorRank) - nBatched)
698  return op->emitError("input/output level-ranks don't match");
699 
700  return success();
701 }

840 static LogicalResult verifyNumBlockArgs(T *op, Region &region,
841  const char *regionName,
842  TypeRange inputTypes, Type outputType) {
843  unsigned numArgs = region.getNumArguments();
844  unsigned expectedNum = inputTypes.size();
845  if (numArgs != expectedNum)
846  return op->emitError() << regionName << " region must have exactly "
847  << expectedNum << " arguments";

856  YieldOp yield = dyn_cast<YieldOp>(term);
857  if (!yield)
858  return op->emitError() << regionName
859  << " region must end with sparse_tensor.yield";
860  if (!yield.getResult() || yield.getResult().getType() != outputType)
861  return op->emitError() << regionName << " region yield type mismatch";
862 
863  return success();
864 }

919 LogicalResult ConcatenateOp::verify() {
920  const auto dstTp = getSparseTensorType(*this);
921  const Dimension concatDim = getDimension();
922  const Dimension dimRank = dstTp.getDimRank();
923 
924  if (getInputs().size() <= 1)
925  return emitError("Need at least two tensors to concatenate.");
926 
927  if (concatDim >= dimRank)
928  return emitError(llvm::formatv(
929  "Concat-dimension is out of bounds for dimension-rank ({0} >= {1})",
930  concatDim, dimRank));
931 
932  for (const auto &it : llvm::enumerate(getInputs())) {
933  const auto i = it.index();
934  const auto srcTp = getSparseTensorType(it.value());
935  if (srcTp.hasDynamicDimShape())
936  return emitError(llvm::formatv("Input tensor ${0} has dynamic shape", i));
937  const Dimension srcDimRank = srcTp.getDimRank();
938  if (srcDimRank != dimRank)
939  return emitError(
940  llvm::formatv("Input tensor ${0} has a different rank (rank={1}) "
941  "from the output tensor (rank={2}).",
942  i, srcDimRank, dimRank));
943  }
944 
945  for (Dimension d = 0; d < dimRank; d++) {
946  const DynSize dstSh = dstTp.getDimShape()[d];
947  if (d == concatDim) {
948  if (!ShapedType::isDynamic(dstSh)) {
949  // If we reach here, then all inputs have static shapes. So we
950  // can use `getDimShape()[d]` instead of `*getDynamicDimSize(d)`
951  // to avoid redundant assertions in the loop.
952  StaticSize sumSz = 0;
953  for (const auto src : getInputs())
954  sumSz += getSparseTensorType(src).getDimShape()[d];
955  // If all dimension are statically known, the sum of all the input
956  // dimensions should be equal to the output dimension.
957  if (sumSz != dstSh)
958  return emitError(
959  "The concatenation dimension of the output tensor should be the "
960  "sum of all the concatenation dimensions of the input tensors.");
961  }
962  } else {
963  DynSize prev = dstSh;
964  for (const auto src : getInputs()) {
965  const auto sh = getSparseTensorType(src).getDimShape()[d];
966  if (!ShapedType::isDynamic(prev) && sh != prev)
967  return emitError("All dimensions (expect for the concatenating one) "
968  "should be equal.");
969  prev = sh;
970  }
971  }
972  }
973 
974  return success();
975 }

1006  OpBuilder &builder, OperationState &result, Value tensor,
1007  ValueRange initArgs, AffineMapAttr order,
1008  function_ref<void(OpBuilder &, Location, ValueRange, Value, ValueRange)>
1009  bodyBuilder) {
1010  build(builder, result, initArgs.getTypes(), tensor, initArgs, order);
1011  // Builds foreach body.
1012  if (!bodyBuilder)
1013  return;
