38 /// Pattern to rewrite a subview op with constant arguments.
39 template <typename OpType, typename ResultTypeFunc, typename CastOpFunc>
40 class OpWithOffsetSizesAndStridesConstantArgumentFolder final
41  : public OpRewritePattern<OpType> {
42 public:
43  using OpRewritePattern<OpType>::OpRewritePattern;
44 
45  LogicalResult matchAndRewrite(OpType op,
46  PatternRewriter &rewriter) const override {
47  SmallVector<OpFoldResult> mixedOffsets(op.getMixedOffsets());
48  SmallVector<OpFoldResult> mixedSizes(op.getMixedSizes());
49  SmallVector<OpFoldResult> mixedStrides(op.getMixedStrides());
50 
51  // No constant operands were folded, just return;
52  if (failed(foldDynamicIndexList(rewriter, mixedOffsets)) &&
53  failed(foldDynamicIndexList(rewriter, mixedSizes)) &&
54  failed(foldDynamicIndexList(rewriter, mixedStrides)))
55  return failure();
56 
57  // Create the new op in canonical form.
58  ResultTypeFunc resultTypeFunc;
59  auto resultType =
60  resultTypeFunc(op, mixedOffsets, mixedSizes, mixedStrides);
61  if (!resultType)
62  return failure();
63  auto newOp =
64  rewriter.create<OpType>(op.getLoc(), resultType, op.getSource(),
65  mixedOffsets, mixedSizes, mixedStrides);
66  CastOpFunc func;
67  func(rewriter, op, newOp);
68 
69  return success();
70  }
71 };
