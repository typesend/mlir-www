25 /// A helper class to be used with `ValueBoundsOpInterface`. This class stores a
26 /// constraint system and mapping of constrained variables to index-typed
27 /// values or dimension sizes of shaped values.
28 ///
29 /// Interface implementations of `ValueBoundsOpInterface` use `addBounds` to
30 /// insert constraints about their results and/or region block arguments into
31 /// the constraint set in the form of an AffineExpr. When a bound should be
32 /// expressed in terms of another value/dimension, `getExpr` can be used to
33 /// retrieve an AffineExpr that represents the specified value/dimension.
34 ///
35 /// When a value/dimension is retrieved for the first time through `getExpr`,
36 /// it is added to an internal worklist. See `computeBound` for more details.
37 ///
38 /// Note: Any modification of existing IR invalides the data stored in this
39 /// class. Adding new operations is allowed.
40 class ValueBoundsConstraintSet {
41 protected:
42  /// Helper class that builds a bound for a shaped value dimension or
43  /// index-typed value.
44  class BoundBuilder {
45  public:
46  /// Specify a dimension, assuming that the underlying value is a shaped
47  /// value.
48  BoundBuilder &operator[](int64_t dim);
49 
50  // These overloaded operators add lower/upper/equality bounds.
51  void operator<(AffineExpr expr);
52  void operator<=(AffineExpr expr);
53  void operator>(AffineExpr expr);
54  void operator>=(AffineExpr expr);
55  void operator==(AffineExpr expr);
56  void operator<(OpFoldResult ofr);
57  void operator<=(OpFoldResult ofr);
58  void operator>(OpFoldResult ofr);
59  void operator>=(OpFoldResult ofr);
60  void operator==(OpFoldResult ofr);
61  void operator<(int64_t i);
62  void operator<=(int64_t i);
63  void operator>(int64_t i);
64  void operator>=(int64_t i);
65  void operator==(int64_t i);
66 
67  protected:
68  friend class ValueBoundsConstraintSet;
69  BoundBuilder(ValueBoundsConstraintSet &cstr, Value value)
70  : cstr(cstr), value(value) {}
71 
72  private:
73  BoundBuilder(const BoundBuilder &) = delete;
74  BoundBuilder &operator=(const BoundBuilder &) = delete;
75  bool operator==(const BoundBuilder &) = delete;
76  bool operator!=(const BoundBuilder &) = delete;
77 
78  ValueBoundsConstraintSet &cstr;
79  Value value;
80  std::optional<int64_t> dim;
81  };
82 
83 public:
84  /// The stop condition when traversing the backward slice of a shaped value/
85  /// index-type value. The traversal continues until the stop condition
86  /// evaluates to "true" for a value.
87  ///
88  /// The first parameter of the function is the shaped value/index-typed
89  /// value. The second parameter is the dimension in case of a shaped value.
90  using StopConditionFn =
91  function_ref<bool(Value, std::optional<int64_t> /*dim*/)>;
92 
93  /// Compute a bound for the given index-typed value or shape dimension size.
94  /// The computed bound is stored in `resultMap`. The operands of the bound are
95  /// stored in `mapOperands`. An operand is either an index-type SSA value
96  /// or a shaped value and a dimension.
97  ///
98  /// `dim` must be `nullopt` if and only if `value` is index-typed. The bound
99  /// is computed in terms of values/dimensions for which `stopCondition`
100  /// evaluates to "true". To that end, the backward slice (reverse use-def
101  /// chain) of the given value is visited in a worklist-driven manner and the
102  /// constraint set is populated according to `ValueBoundsOpInterface` for each
103  /// visited value.
104  ///
105  /// By default, lower/equal bounds are closed and upper bounds are open. If
106  /// `closedUB` is set to "true", upper bounds are also closed.
107  static LogicalResult computeBound(AffineMap &resultMap,
108  ValueDimList &mapOperands,
109  presburger::BoundType type, Value value,
110  std::optional<int64_t> dim,
111  StopConditionFn stopCondition,
112  bool closedUB = false);
113 
114  /// Compute a bound in terms of the values/dimensions in `dependencies`. The
115  /// computed bound consists of only constant terms and dependent values (or
116  /// dimension sizes thereof).
117  static LogicalResult
118  computeDependentBound(AffineMap &resultMap, ValueDimList &mapOperands,
119  presburger::BoundType type, Value value,
120  std::optional<int64_t> dim, ValueDimList dependencies,
121  bool closedUB = false);
122 
123  /// Compute a bound in that is independent of all values in `independencies`.
124  ///
125  /// Independencies are the opposite of dependencies. The computed bound does
126  /// not contain any SSA values that are part of `independencies`. E.g., this
127  /// function can be used to make ops hoistable from loops. To that end, ops
128  /// must be made independent of loop induction variables (in the case of "for"
129  /// loops). Loop induction variables are the independencies; they may not
130  /// appear in the computed bound.
131  static LogicalResult
132  computeIndependentBound(AffineMap &resultMap, ValueDimList &mapOperands,
133  presburger::BoundType type, Value value,
134  std::optional<int64_t> dim, ValueRange independencies,
135  bool closedUB = false);
136 
137  /// Compute a constant bound for the given index-typed value or shape
138  /// dimension size.
139  ///
140  /// `dim` must be `nullopt` if and only if `value` is index-typed. This
141  /// function traverses the backward slice of the given value in a
142  /// worklist-driven manner until `stopCondition` evaluates to "true". The
143  /// constraint set is populated according to `ValueBoundsOpInterface` for each
144  /// visited value. (No constraints are added for values for which the stop
145  /// condition evaluates to "true".)
146  ///
147  /// The stop condition is optional: If none is specified, the backward slice
148  /// is traversed in a breadth-first manner until a constant bound could be
149  /// computed.
150  ///
151  /// By default, lower/equal bounds are closed and upper bounds are open. If
152  /// `closedUB` is set to "true", upper bounds are also closed.
153  static FailureOr<int64_t>
154  computeConstantBound(presburger::BoundType type, Value value,
155  std::optional<int64_t> dim = std::nullopt,
156  StopConditionFn stopCondition = nullptr,
157  bool closedUB = false);
158 
159  /// Add a bound for the given index-typed value or shaped value. This function
160  /// returns a builder that adds the bound.
161  BoundBuilder bound(Value value) { return BoundBuilder(*this, value); }
162 
163  /// Return an expression that represents the given index-typed value or shaped
164  /// value dimension. If this value/dimension was not used so far, it is added
165  /// to the worklist.
166  ///
167  /// `dim` must be `nullopt` if and only if the given value is of index type.
168  AffineExpr getExpr(Value value, std::optional<int64_t> dim = std::nullopt);
169 
170  /// Return an expression that represents a constant or index-typed SSA value.
171  /// In case of a value, if this value was not used so far, it is added to the
172  /// worklist.
173  AffineExpr getExpr(OpFoldResult ofr);
174 
175  /// Return an expression that represents a constant.
176  AffineExpr getExpr(int64_t constant);
177 
178 protected:
179  /// Dimension identifier to indicate a value is index-typed. This is used for
180  /// internal data structures/API only.
181  static constexpr int64_t kIndexValue = -1;
182 
183  /// An index-typed value or the dimension of a shaped-type value.
184  using ValueDim = std::pair<Value, int64_t>;
185 
186  ValueBoundsConstraintSet(Value value, std::optional<int64_t> dim);
187 
188  /// Iteratively process all elements on the worklist until an index-typed
189  /// value or shaped value meets `stopCondition`. Such values are not processed
190  /// any further.
191  void processWorklist(StopConditionFn stopCondition);
192 
193  /// Bound the given column in the underlying constraint set by the given
194  /// expression.
195  void addBound(presburger::BoundType type, int64_t pos, AffineExpr expr);
196 
197  /// Return the column position of the given value/dimension. Asserts that the
198  /// value/dimension exists in the constraint set.
199  int64_t getPos(Value value, std::optional<int64_t> dim = std::nullopt) const;
200 
201  /// Insert a value/dimension into the constraint set. If `isSymbol` is set to
202  /// "false", a dimension is added.
203  ///
204  /// Note: There are certain affine restrictions wrt. dimensions. E.g., they
205  /// cannot be multiplied. Furthermore, bounds can only be queried for
206  /// dimensions but not for symbols.
207  int64_t insert(Value value, std::optional<int64_t> dim, bool isSymbol = true);
208 
209  /// Project out the given column in the constraint set.
210  void projectOut(int64_t pos);
211 
212  /// Project out all columns for which the condition holds.
213  void projectOut(function_ref<bool(ValueDim)> condition);
214 
215  /// Mapping of columns to values/shape dimensions.
216  SmallVector<ValueDim> positionToValueDim;
217  /// Reverse mapping of values/shape dimensions to columns.
218  DenseMap<ValueDim, int64_t> valueDimToPosition;
219 
220  /// Worklist of values/shape dimensions that have not been processed yet.
221  std::queue<int64_t> worklist;
222 
223  /// Constraint system of equalities and inequalities.
224  FlatLinearConstraints cstr;
225 
226  /// Builder for constructing affine expressions.
227  Builder builder;
228 };

236 /// Default implementation for destination style ops: Tied OpResults and
237 /// OpOperands have the same type.
238 template <typename ConcreteOp>
239 struct DstValueBoundsOpInterfaceExternalModel
240  : public ValueBoundsOpInterface::ExternalModel<
241  DstValueBoundsOpInterfaceExternalModel<ConcreteOp>, ConcreteOp> {
242  void populateBoundsForShapedValueDim(Operation *op, Value value, int64_t dim,
243  ValueBoundsConstraintSet &cstr) const {
244  auto dstOp = cast<DestinationStyleOpInterface>(op);
245  assert(value.getDefiningOp() == dstOp);
246 
247  Value tiedOperand = dstOp.getTiedOpOperand(value.cast<OpResult>())->get();
248  cstr.bound(value)[dim] == cstr.getExpr(tiedOperand, dim);
249  }
250 };
