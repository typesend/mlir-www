34 DiagnosedSilenceableFailure transform::MemRefMultiBufferOp::apply(
35  transform::TransformResults &transformResults,
36  transform::TransformState &state) {
37  SmallVector<Operation *> results;
38  ArrayRef<Operation *> payloadOps = state.getPayloadOps(getTarget());
39  IRRewriter rewriter(getContext());
40  for (auto *op : payloadOps) {
41  bool canApplyMultiBuffer = true;
42  auto target = cast<memref::AllocOp>(op);
43  LLVM_DEBUG(DBGS() << "Start multibuffer transform op: " << target << "\n";);
44  // Skip allocations not used in a loop.
45  for (Operation *user : target->getUsers()) {
46  if (isa<memref::DeallocOp>(user))
47  continue;
48  auto loop = user->getParentOfType<LoopLikeOpInterface>();
49  if (!loop) {
50  LLVM_DEBUG(DBGS() << "--allocation not used in a loop\n";
51  DBGS() << "----due to user: " << *user;);
52  canApplyMultiBuffer = false;
53  break;
54  }
55  }
56  if (!canApplyMultiBuffer) {
57  LLVM_DEBUG(DBGS() << "--cannot apply multibuffering -> Skip\n";);
58  continue;
59  }
60 
61  auto newBuffer =
62  memref::multiBuffer(rewriter, target, getFactor(), getSkipAnalysis());
63 
64  if (failed(newBuffer)) {
65  LLVM_DEBUG(DBGS() << "--op failed to multibuffer\n";);
66  return emitSilenceableFailure(target->getLoc())
67  << "op failed to multibuffer";
68  }
69 
70  results.push_back(*newBuffer);
71  }
72  transformResults.set(getResult().cast<OpResult>(), results);
73  return DiagnosedSilenceableFailure::success();
74 }

82  Operation *target, transform::ApplyToEachResultList &results,
83  transform::TransformState &state) {
84  if (!target->hasTrait<OpTrait::IsIsolatedFromAbove>()) {
85  auto diag = this->emitOpError("requires isolated-from-above targets");
86  diag.attachNote(target->getLoc()) << "non-isolated target";
87  return DiagnosedSilenceableFailure::definiteFailure();
88  }
89 
90  MLIRContext *ctx = getContext();
91  RewritePatternSet patterns(ctx);
92  memref::populateExtractAddressComputationsPatterns(patterns);
93

95  return emitDefaultDefiniteFailure(target);
96 
97  results.push_back(target);

104 
105 DiagnosedSilenceableFailure transform::MemRefMakeLoopIndependentOp::applyToOne(
106  Operation *target, transform::ApplyToEachResultList &results,
107  transform::TransformState &state) {
108  // Gather IVs.
109  SmallVector<Value> ivs;
110  Operation *nextOp = target;
111  for (uint64_t i = 0, e = getNumLoops(); i < e; ++i) {
112  nextOp = nextOp->getParentOfType<scf::ForOp>();
113  if (!nextOp) {
114  DiagnosedSilenceableFailure diag = emitSilenceableError()
115  << "could not find " << i
116  << "-th enclosing loop";
117  diag.attachNote(target->getLoc()) << "target op";
118  return diag;
119  }
120  ivs.push_back(cast<scf::ForOp>(nextOp).getInductionVar());
121  }
122 
123  // Rewrite IR.
124  IRRewriter rewriter(target->getContext());
125  FailureOr<Value> replacement = failure();
126  if (auto allocaOp = dyn_cast<memref::AllocaOp>(target)) {
127  replacement = memref::replaceWithIndependentOp(rewriter, allocaOp, ivs);
128  } else {
129  DiagnosedSilenceableFailure diag = emitSilenceableError()
130  << "unsupported target op";
131  diag.attachNote(target->getLoc()) << "target op";
132  return diag;
133  }
134  if (failed(replacement)) {
135  DiagnosedSilenceableFailure diag =
136  emitSilenceableError() << "could not make target op loop-independent";
137  diag.attachNote(target->getLoc()) << "target op";
138  return diag;
139  }
140  results.push_back(replacement->getDefiningOp());
141  return DiagnosedSilenceableFailure::success();
142 }
