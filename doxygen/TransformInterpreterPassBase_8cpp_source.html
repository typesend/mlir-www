53 /// Utility to parse the content of a `transformFileName` MLIR file containing
54 /// a transform dialect specification.
55 static LogicalResult
56 parseTransformModuleFromFile(MLIRContext *context,
57  llvm::StringRef transformFileName,
58  OwningOpRef<ModuleOp> &transformModule) {
59  if (transformFileName.empty()) {
60  LLVM_DEBUG(
61  DBGS() << "no transform file name specified, assuming the transform "
62  "module is embedded in the IR next to the top-level\n");
63  return success();
64  }
65  // Parse transformFileName content into a ModuleOp.
66  std::string errorMessage;
67  auto memoryBuffer = mlir::openInputFile(transformFileName, &errorMessage);
68  if (!memoryBuffer) {
69  return emitError(FileLineColLoc::get(
70  StringAttr::get(context, transformFileName), 0, 0))
71  << "failed to parse transform file";
72  }
73  // Tell sourceMgr about this buffer, the parser will pick it up.
74  llvm::SourceMgr sourceMgr;
75  sourceMgr.AddNewSourceBuffer(std::move(memoryBuffer), llvm::SMLoc());
76  transformModule =
77  OwningOpRef<ModuleOp>(parseSourceFile<ModuleOp>(sourceMgr, context));
78  return success();
79 }

81 /// Finds the single top-level transform operation with `root` as ancestor.
82 /// Reports an error if there is more than one such operation and returns the
83 /// first one found. Reports an error returns nullptr if no such operation
84 /// found.
85 static Operation *findTopLevelTransform(Operation *root,
86  StringRef filenameOption) {
87  ::mlir::transform::TransformOpInterface topLevelTransform = nullptr;
88  WalkResult walkResult = root->walk<WalkOrder::PreOrder>(
89  [&](::mlir::transform::TransformOpInterface transformOp) {
90  if (!transformOp
91  ->hasTrait<transform::PossibleTopLevelTransformOpTrait>())
92  return WalkResult::skip();
93  if (!topLevelTransform) {
94  topLevelTransform = transformOp;
95  return WalkResult::skip();
96  }
97  auto diag = transformOp.emitError()
98  << "more than one top-level transform op";
99  diag.attachNote(topLevelTransform.getLoc())
100  << "previous top-level transform op";
101  return WalkResult::interrupt();
102  });
103  if (walkResult.wasInterrupted())
104  return nullptr;
105  if (!topLevelTransform) {
106  auto diag = root->emitError()
107  << "could not find a nested top-level transform op";
108  diag.attachNote() << "use the '" << filenameOption
109  << "' option to provide transform as external file";
110  return nullptr;
111  }
112  return topLevelTransform;
113 }

115 /// Finds an operation nested in `root` that has the transform dialect tag
116 /// attribute with the value specified as `tag`. Assumes only one operation
117 /// may have the tag. Returns nullptr if there is no such operation.
118 static Operation *findOpWithTag(Operation *root, StringRef tagKey,
119  StringRef tagValue) {
120  Operation *found = nullptr;
121  WalkResult walkResult = root->walk<WalkOrder::PreOrder>(
122  [tagKey, tagValue, &found, root](Operation *op) {
123  auto attr = op->getAttrOfType<StringAttr>(tagKey);
124  if (!attr || attr.getValue() != tagValue)
125  return WalkResult::advance();
126 
127  if (found) {
128  InFlightDiagnostic diag = root->emitError()
129  << "more than one operation with " << tagKey
130  << "=\"" << tagValue << "\" attribute";
131  diag.attachNote(found->getLoc()) << "first operation";
132  diag.attachNote(op->getLoc()) << "other operation";
133  return WalkResult::interrupt();
134  }
135 
136  found = op;
137  return WalkResult::advance();
138  });
139  if (walkResult.wasInterrupted())
140  return nullptr;
141 
142  if (!found) {
143  root->emitError() << "could not find the operation with " << tagKey << "=\""
144  << tagValue << "\" attribute";
145  }
146  return found;
147 }

149 /// Returns the ancestor of `target` that doesn't have a parent.
150 static Operation *getRootOperation(Operation *target) {
151  Operation *root = target;
152  while (root->getParentOp())
153  root = root->getParentOp();
154  return root;
155 }

197 /// Saves the payload and the transform IR into a temporary file and reports
198 /// the file name to `os`.
199 void saveReproToTempFile(
200  llvm::raw_ostream &os, Operation *target, Operation *transform,
201  StringRef passName, const Pass::Option<std::string> &debugPayloadRootTag,
202  const Pass::Option<std::string> &debugTransformRootTag,
203  const Pass::Option<std::string> &transformLibraryFileName,
204  StringRef binaryName) {
205  using llvm::sys::fs::TempFile;
206  Operation *root = getRootOperation(target);
207 
208  SmallVector<char, 128> tmpPath;
209  llvm::sys::path::system_temp_directory(/*erasedOnReboot=*/true, tmpPath);
210  llvm::sys::path::append(tmpPath, "transform_dialect_%%%%%%.mlir");
211  llvm::Expected<TempFile> tempFile = TempFile::create(tmpPath);
212  if (!tempFile) {
213  os << "could not open temporary file to save the repro\n";
214  return;
215  }
216 
217  llvm::raw_fd_ostream fout(tempFile->FD, /*shouldClose=*/false);
218  printModuleForRepro(fout, root, transform);
219  fout.flush();
220  std::string filename = tempFile->TmpName;
221 
222  if (tempFile->keep()) {
223  os << "could not preserve the temporary file with the repro\n";
224  return;
225  }
226 
227  os << "=== Transform Interpreter Repro ===\n";
228  printReproCall(os, root->getName().getStringRef(), passName,
229  debugPayloadRootTag, debugTransformRootTag,
230  transformLibraryFileName, binaryName)
231  << " " << filename << "\n";
232  os << "===================================\n";
233 }

235 // Optionally perform debug actions requested by the user to dump IR and a
236 // repro to stderr and/or a file.
237 static void performOptionalDebugActions(
238  Operation *target, Operation *transform, StringRef passName,
239  const Pass::Option<std::string> &debugPayloadRootTag,
240  const Pass::Option<std::string> &debugTransformRootTag,
241  const Pass::Option<std::string> &transformLibraryFileName,
242  StringRef binaryName) {
243  MLIRContext *context = target->getContext();
244 
245  // If we are not planning to print, bail early.
246  bool hasDebugFlags = false;
247  DEBUG_WITH_TYPE(DEBUG_TYPE_DUMP_STDERR, { hasDebugFlags = true; });
248  DEBUG_WITH_TYPE(DEBUG_TYPE_DUMP_FILE, { hasDebugFlags = true; });
249  if (!hasDebugFlags)
250  return;
251 
252  // We will be mutating the IR to set attributes. If this is running
253  // concurrently on several parts of a container or using a shared transform
254  // script, this would create a race. Bail in multithreaded mode and require
255  // the user to disable threading to dump repros.
256  static llvm::sys::SmartMutex<true> dbgStreamMutex;
257  if (target->getContext()->isMultithreadingEnabled()) {
258  llvm::sys::SmartScopedLock<true> lock(dbgStreamMutex);
259  llvm::dbgs() << "=======================================================\n";
260  llvm::dbgs() << "| Transform reproducers cannot be produced |\n";
261  llvm::dbgs() << "| in multi-threaded mode! |\n";
262  llvm::dbgs() << "=======================================================\n";
263  return;
264  }
265 
266  Operation *root = getRootOperation(target);
267 
268  // Add temporary debug / repro attributes, these must never leak out.
269  if (debugPayloadRootTag.empty()) {
270  target->setAttr(
271  kTransformDialectTagAttrName,
272  StringAttr::get(context, kTransformDialectTagPayloadRootValue));
273  }
274  if (debugTransformRootTag.empty()) {
275  transform->setAttr(
276  kTransformDialectTagAttrName,
277  StringAttr::get(context, kTransformDialectTagTransformContainerValue));
278  }
279 
280  DEBUG_WITH_TYPE(DEBUG_TYPE_DUMP_STDERR, {
281  llvm::dbgs() << "=== Transform Interpreter Repro ===\n";
282  printReproCall(llvm::dbgs() << "cat <<EOF | ",
283  root->getName().getStringRef(), passName,
284  debugPayloadRootTag, debugTransformRootTag,
285  transformLibraryFileName, binaryName)
286  << "\n";
287  printModuleForRepro(llvm::dbgs(), root, transform);
288  llvm::dbgs() << "\nEOF\n";
289  llvm::dbgs() << "===================================\n";
290  });
291  (void)root;
292  DEBUG_WITH_TYPE(DEBUG_TYPE_DUMP_FILE, {
293  saveReproToTempFile(llvm::dbgs(), target, transform, passName,
294  debugPayloadRootTag, debugTransformRootTag,
295  transformLibraryFileName, binaryName);
296  });
297 }

299 /// Replaces external symbols in `block` with their (non-external) definitions
300 /// from the given module.
301 static LogicalResult defineDeclaredSymbols(Block &block, ModuleOp definitions) {
302  MLIRContext &ctx = *definitions->getContext();
303  auto consumedName =
304  StringAttr::get(&ctx, transform::TransformDialect::kArgConsumedAttrName);
305  auto readOnlyName =
306  StringAttr::get(&ctx, transform::TransformDialect::kArgReadOnlyAttrName);
307 
308  for (Operation &op : llvm::make_early_inc_range(block)) {
309  LLVM_DEBUG(DBGS() << op << "\n");
310  auto symbol = dyn_cast<SymbolOpInterface>(op);
311  if (!symbol)
312  continue;
313  if (symbol->getNumRegions() == 1 && !symbol->getRegion(0).empty())
314  continue;
315 
316  LLVM_DEBUG(DBGS() << "looking for definition of symbol "
317  << symbol.getNameAttr() << ":");
318  SymbolTable symbolTable(definitions);
319  Operation *externalSymbol = symbolTable.lookup(symbol.getNameAttr());
320  if (!externalSymbol || externalSymbol->getNumRegions() != 1 ||
321  externalSymbol->getRegion(0).empty()) {
322  LLVM_DEBUG(llvm::dbgs() << "not found\n");
323  continue;
324  }
325 
326  auto symbolFunc = dyn_cast<FunctionOpInterface>(op);
327  auto externalSymbolFunc = dyn_cast<FunctionOpInterface>(externalSymbol);
328  if (!symbolFunc || !externalSymbolFunc) {
329  LLVM_DEBUG(llvm::dbgs() << "cannot compare types\n");
330  continue;
331  }
332 
333  LLVM_DEBUG(llvm::dbgs() << "found @" << externalSymbol << "\n");
334  if (symbolFunc.getFunctionType() != externalSymbolFunc.getFunctionType()) {
335  return symbolFunc.emitError()
336  << "external definition has a mismatching signature ("
337  << externalSymbolFunc.getFunctionType() << ")";
338  }
339 
340  for (unsigned i = 0, e = symbolFunc.getNumArguments(); i < e; ++i) {
341  bool isExternalConsumed =
342  externalSymbolFunc.getArgAttr(i, consumedName) != nullptr;
343  bool isExternalReadonly =
344  externalSymbolFunc.getArgAttr(i, readOnlyName) != nullptr;
345  bool isConsumed = symbolFunc.getArgAttr(i, consumedName) != nullptr;
346  bool isReadonly = symbolFunc.getArgAttr(i, readOnlyName) != nullptr;
347  if (!isExternalConsumed && !isExternalReadonly) {
348  if (isConsumed)
349  externalSymbolFunc.setArgAttr(i, consumedName, UnitAttr::get(&ctx));
350  else if (isReadonly)
351  externalSymbolFunc.setArgAttr(i, readOnlyName, UnitAttr::get(&ctx));
352  continue;
353  }
354 
355  if ((isExternalConsumed && !isConsumed) ||
356  (isExternalReadonly && !isReadonly)) {
357  return symbolFunc.emitError()
358  << "external definition has mismatching consumption annotations "
359  "for argument #"
360  << i;
361  }
362  }
363 
364  OpBuilder builder(&op);
365  builder.setInsertionPoint(&op);
366  builder.clone(*externalSymbol);
367  symbol->erase();
368  }
369 
370  return success();
371 }

373 LogicalResult transform::detail::interpreterBaseRunOnOperationImpl(
374  Operation *target, StringRef passName,
375  const std::shared_ptr<OwningOpRef<ModuleOp>> &sharedTransformModule,
376  const std::shared_ptr<OwningOpRef<ModuleOp>> &libraryModule,
377  const RaggedArray<MappedValue> &extraMappings,
378  const TransformOptions &options,
379  const Pass::Option<std::string> &transformFileName,
380  const Pass::Option<std::string> &transformLibraryFileName,
381  const Pass::Option<std::string> &debugPayloadRootTag,
382  const Pass::Option<std::string> &debugTransformRootTag,
383  StringRef binaryName) {
384 
385  // Step 1
386  // ------
387  // If debugPayloadRootTag was passed, then we are in user-specified selection
388  // of the transformed IR. This corresponds to REPL debug mode. Otherwise, just
389  // apply to `target`.
390  Operation *payloadRoot = target;
391  if (!debugPayloadRootTag.empty()) {
392  payloadRoot = findOpWithTag(target, kTransformDialectTagAttrName,
393  debugPayloadRootTag);
394  if (!payloadRoot)
395  return failure();
396  }
397 
398  // Step 2
399  // ------
400  // If a shared transform was specified separately, use it. Otherwise, the
401  // transform is embedded in the payload IR. If debugTransformRootTag was
402  // passed, then we are in user-specified selection of the transforming IR.
403  // This corresponds to REPL debug mode.
404  bool sharedTransform = (sharedTransformModule && *sharedTransformModule);
405  Operation *transformContainer =
406  sharedTransform ? sharedTransformModule->get() : target;
407  Operation *transformRoot =
408  debugTransformRootTag.empty()
409  ? findTopLevelTransform(transformContainer,
410  transformFileName.getArgStr())
411  : findOpWithTag(transformContainer, kTransformDialectTagAttrName,
412  debugTransformRootTag);
413  if (!transformRoot)
414  return failure();
415 
416  if (!transformRoot->hasTrait<PossibleTopLevelTransformOpTrait>()) {
417  return emitError(transformRoot->getLoc())
418  << "expected the transform entry point to be a top-level transform "
419  "op";
420  }
421 
422  // Step 3
423  // ------
424  // Copy external defintions for symbols if provided. Be aware of potential
425  // concurrent execution (normally, the error shouldn't be triggered unless the
426  // transform IR modifies itself in a pass, which is also forbidden elsewhere).
427  if (!sharedTransform && libraryModule && *libraryModule) {
428  if (!target->isProperAncestor(transformRoot)) {
429  InFlightDiagnostic diag =
430  transformRoot->emitError()
431  << "cannot inject transform definitions next to pass anchor op";
432  diag.attachNote(target->getLoc()) << "pass anchor op";
433  return diag;
434  }
435  if (failed(defineDeclaredSymbols(*transformRoot->getBlock(),
436  libraryModule->get())))
437  return failure();
438  }
439 
440  // Step 4
441  // ------
442  // Optionally perform debug actions requested by the user to dump IR and a
443  // repro to stderr and/or a file.
444  performOptionalDebugActions(target, transformRoot, passName,
445  debugPayloadRootTag, debugTransformRootTag,
446  transformLibraryFileName, binaryName);
447 
448  // Step 5
449  // ------
450  // Apply the transform to the IR
451  return applyTransforms(payloadRoot, cast<TransformOpInterface>(transformRoot),
452  extraMappings, options);
453 }

455 LogicalResult transform::detail::interpreterBaseInitializeImpl(
456  MLIRContext *context, StringRef transformFileName,
457  StringRef transformLibraryFileName,
458  std::shared_ptr<OwningOpRef<ModuleOp>> &module,
459  std::shared_ptr<OwningOpRef<ModuleOp>> &libraryModule) {
460  OwningOpRef<ModuleOp> parsed;
461  if (failed(parseTransformModuleFromFile(context, transformFileName, parsed)))
462  return failure();
463  if (parsed && failed(mlir::verify(*parsed)))
464  return failure();
465 
466  OwningOpRef<ModuleOp> parsedLibrary;
467  if (failed(parseTransformModuleFromFile(context, transformLibraryFileName,
468  parsedLibrary)))
469  return failure();
470  if (parsedLibrary && failed(mlir::verify(*parsedLibrary)))
471  return failure();
472 
473  module = std::make_shared<OwningOpRef<ModuleOp>>(std::move(parsed));
474  if (!parsedLibrary || !*parsedLibrary)
475  return success();
476 
477  if (module && *module) {
478  if (failed(defineDeclaredSymbols(*module->get().getBody(),
479  parsedLibrary.get())))
480  return failure();
481  } else {
482  libraryModule =
483  std::make_shared<OwningOpRef<ModuleOp>>(std::move(parsedLibrary));
484  }
485  return success();
486 }
