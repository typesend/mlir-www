68 /// Generates code to compute the *absolute* offset of the slice based on the
69 /// provide minimum coordinates in the slice.
70 /// E.g., when reducing d0 + d1 + d2, we need two slices to fully reduced the
71 /// expression, i,e, s1 = slice(T, d0), s2 = slice(s1, d1). The *absolute*
72 /// offset is the offset computed relative to the initial tensors T.
73 ///
74 /// When isNonEmpty == true, the computed offset is meaningless and should not
75 /// be used during runtime, the method generates code to return 0 currently in
76 /// that case.
77 ///
78 /// offset = isNonEmpty && minCrd >= size ? minCrd - size + 1 : 0;
79 static Value offsetFromMinCoord(OpBuilder &builder, Location loc, Value minCrd,
80  Value size, Value isNonEmpty) {
81  Value geSize = CMPI(uge, minCrd, size);
82  Value pred = ANDI(isNonEmpty, geSize);
83  // Computes minCrd - size + 1
84  Value mms = SUBI(ADDI(minCrd, C_IDX(1)), size);
85  // This is the absolute offset related to the underly tensor.
86  return SELECT(pred, mms, C_IDX(0));
87 }

89 /// Converts a coordinate relative to the underlying tensor to the coordinate
90 /// relative to the slice, returns a extra reminder value
91 // FIXME: that description says "tensorCrd -> sliceCrd"; but the function
92 // name suggests it should be "sliceCrd -> tensorCrd".
93 static std::pair<Value, Value> fromSliceCrd(OpBuilder &builder, Location loc,
94  Value crd, Value offset,
95  Value stride, Value tensor,
96  Level lvl) {
97  // sliceCrd = (tensorCrd - offset) / stride
98  crd = SUBI(crd, offset);
99  Value rem = builder.create<arith::RemUIOp>(loc, crd, stride);
100  crd = builder.create<arith::DivUIOp>(loc, crd, stride);
101  return std::make_pair(crd, rem);
102 }

104 std::pair<Value, Value>
105 LoopEmitter::genSliceLegitPredicate(OpBuilder &builder, Location loc, Value crd,
106  TensorId tid, Level lvl) {
107  assert(isSparseSlices[tid]);
108  Value slice = tensors[tid];
109  Value offset = sliceOffsets[tid][lvl];
110  Value stride = sliceStrides[tid][lvl];
111  auto enc = getSparseTensorEncoding(slice.getType());
112 
113  const auto [newCrd, crdRem] =
114  fromSliceCrd(builder, loc, crd, offset, stride, slice, lvl);
115 
116  SmallVector<Value, 3> conds; // at most 3 conditions
117 
118  // First, coord >= offset (skip the check if offset is known to be 0).
119  if (auto staticOffset = enc.getStaticLvlSliceOffset(lvl);
120  !(staticOffset.has_value() && *staticOffset == 0)) {
121  auto geOffset = CMPI(uge, crd, offset);
122  conds.push_back(geOffset);
123  }
124 
125  // Second, coord_in_slice < length
126  auto ltLength = CMPI(ult, newCrd, lvlSizes[tid][lvl]);
127  conds.push_back(ltLength);
128 
129  // Third, rem == 0 (skip the check if stride is known to be 1).
130  if (auto staticStride = enc.getStaticLvlSliceStride(lvl);
131  !(staticStride.has_value() && *staticStride == 1)) {
132  auto fitStride = CMPI(eq, crdRem, C_IDX(0));
133  conds.push_back(fitStride);
134  }
135 
136  // Must meet all condition to be a valid coordinate in slice.
137  auto pred = conds.front();
138  for (auto cond : ValueRange(conds).drop_front())
139  pred = ANDI(pred, cond);
140 
141  return {newCrd, pred};
142 }

159 Value LoopEmitter::genSegmentHigh(OpBuilder &builder, Location loc,
160  TensorId tid, Level lvl, Value pLo,
161  Value pHi) {
162  const auto coordinates = coordinatesBuffers[tid][lvl];
163  const auto sameCrd = genIndexLoad(builder, loc, coordinates, pLo);
164  auto whileOp = builder.create<scf::WhileOp>(
165  loc, builder.getIndexType(), pLo,
166  /*beforeBuilder=*/
167  [pHi, coordinates, sameCrd](OpBuilder &builder, Location loc,
168  ValueRange ivs) {
169  const auto pos = ivs[0];
170  Value inBound = builder.create<arith::CmpIOp>(
171  loc, arith::CmpIPredicate::ult, pos, pHi);
172  auto ifInBound =
173  builder.create<scf::IfOp>(loc, builder.getI1Type(), inBound, true);
174  {
175  OpBuilder::InsertionGuard guard(builder);
176  // Load the next coordinates only when inbound (to avoid OOB
177  // acccesses).
178  builder.setInsertionPointToStart(ifInBound.thenBlock());
179  Value crd = genIndexLoad(builder, loc, coordinates, pos);
180  Value isSameCrd = builder.create<arith::CmpIOp>(
181  loc, arith::CmpIPredicate::eq, crd, sameCrd);
182  YIELD(isSameCrd);
183  // Else, the position is out of bound, yield false to terminate the
184  // loop.
185  builder.setInsertionPointToStart(ifInBound.elseBlock());
186  YIELD(constantI1(builder, loc, false));
187  }
188  builder.create<scf::ConditionOp>(loc, ifInBound.getResults()[0], ivs);
189  },
190  /*afterBuilder=*/
191  [](OpBuilder &builder, Location loc, ValueRange ivs) {
192  // pos ++
193  Value nextPos = ADDI(ivs[0], C_IDX(1));
194  YIELD(nextPos);
195  });
196  // Return the segment high.
197  return whileOp.getResult(0);
198 }

200 Value LoopEmitter::genSparseCrd(OpBuilder &builder, Location loc, TensorId tid,
201  Level dstLvl) {
202  Value crd = C_IDX(0);
203  const auto reassoc = getCollapseReassociation(tid, dstLvl);
204  const unsigned reassocSize = reassoc.size();
205  for (unsigned i = 0; i < reassocSize; i++) {
206  const Level srcLvl = reassoc[i];
207  // A load on the coordinates array yields the coordinate.
208  const Value mem = coordinatesBuffers[tid][srcLvl];
209  /// FIXME: See the [CLARIFY_POSITS_LVL] note in the header.
210  const Value pos = posits[tid][dstLvl];
211  const Value off = genIndexLoad(builder, loc, mem, pos);
212  // Linearized the coordinates within the same collapse reassociation.
213  crd = ADDI(crd, off);
214  if (i != reassocSize - 1) {
215  crd = MULI(crd, this->lvlSizes[tid][reassoc[i + 1]]);
216  }
217  }
218  return crd;
219 }

227 void LoopEmitter::initialize(ValueRange ts, StringAttr loopTag, bool hasOutput,
228  bool isSparseOut, ArrayRef<LoopId> topSort,
229  DependentLvlGetter dimGetter) {
230  // First initialize the top-level type of the fields.
231  this->loopTag = loopTag;
232  this->hasOutput = hasOutput;
233  this->isSparseOut = isSparseOut;
234 
235  const unsigned numTensors = ts.size();
236  this->tensors.assign(ts.begin(), ts.end());
237  this->lvlTypes.assign(numTensors, std::vector<DimLevelType>());
238  this->lvlSizes.assign(numTensors, std::vector<Value>());
239  this->highs.assign(numTensors, std::vector<Value>());
240  this->segHi.assign(numTensors, std::vector<Value>());
241  this->posits.assign(numTensors, std::vector<Value>());
242  this->coords.assign(numTensors, std::vector<Value>());
243  this->positionsBuffers.assign(numTensors, std::vector<Value>());
244  this->coordinatesBuffers.assign(numTensors, std::vector<Value>());
245  this->valBuffer.assign(numTensors, nullptr);
246  this->collapseReassoc.assign(numTensors, nullptr);
247  this->isSparseSlices.assign(numTensors, false);
248  this->sliceOffsets.assign(numTensors, std::vector<Value>());
249  this->sliceStrides.assign(numTensors, std::vector<Value>());
250 
251  const LoopOrd numLoops = topSort.size();
252  // These zeros will be overwritten below, but we need to initialize
253  // them to something since we'll need random-access assignment.
254  this->loopIdToOrd.assign(numLoops, 0);
255  this->loopStack.reserve(numLoops);
256  this->loopSeqStack.reserve(numLoops);
257 
258  // Index-reduction related fields.
259  this->dependentLvlMap.assign(
260  numTensors, std::vector<std::vector<std::pair<TensorId, Level>>>());
261  this->slicePosBuffer.assign(numTensors, std::vector<std::vector<Value>>());
262  this->sliceSizes.assign(numTensors, std::vector<std::vector<Value>>());
263  this->sliceStack.assign(numTensors, std::vector<SliceInfo>());
264  this->levelReducedDep.assign(numTensors, std::vector<unsigned>());
265 
266  // Initialize nested types of `TensorId`-indexed fields.
267  for (TensorId tid = 0; tid < numTensors; tid++) {
268  const Value t = tensors[tid];
269  // a scalar or 0-dimension tensors
270  if (isZeroRankedTensorOrScalar(t.getType()))
271  continue;
272 
273  auto rtp = getRankedTensorType(t);
274  if (auto reshape = t.getDefiningOp<tensor::CollapseShapeOp>();
275  isUniqueCOOType(rtp) && reshape) {
276  // TODO: Supports more kinds of sparse tensors.
277  // FIXME: We should instead lower reshape operations on sparse tensors to
278  // view change.
279  collapseReassoc[tid] = reshape.getReassociation();
280  rtp = reshape.getSrcType();
281  // Overwrites the tensor to the source tensor of reshape operations.
282  tensors[tid] = reshape.getSrc();
283  }
284  const SparseTensorType stt(rtp);
285  const Level lvlRank = stt.getLvlRank();
286  // We always treat sparse output tensor as dense so that we always iterate
287  // it based on lvl size.
288  if (stt.hasEncoding() && !(isOutputTensor(tid) && isSparseOut)) {
289  const auto enc = stt.getEncoding();
290  isSparseSlices[tid] = enc.isSlice();
291  for (auto lvlTp : enc.getDimLevelType())
292  lvlTypes[tid].push_back(lvlTp);
293  } else {
294  lvlTypes[tid].assign(lvlRank, DimLevelType::Dense);
295  }
296 
297  // Initialize using empty value.
298  lvlSizes[tid].assign(lvlRank, Value());
299  highs[tid].assign(lvlRank, Value());
300  segHi[tid].assign(lvlRank, Value());
301  posits[tid].assign(lvlRank, Value());
302  coords[tid].assign(lvlRank, Value());
303  positionsBuffers[tid].assign(lvlRank, Value());
304  coordinatesBuffers[tid].assign(lvlRank, Value());
305  sliceOffsets[tid].assign(lvlRank, Value());
306  sliceStrides[tid].assign(lvlRank, Value());
307 
308  // Slice-driven loops related initialization.
309  levelReducedDep[tid].assign(lvlRank, 0);
310  dependentLvlMap[tid].assign(lvlRank,
311  std::vector<std::pair<TensorId, Level>>());
312  slicePosBuffer[tid].assign(lvlRank, std::vector<Value>());
313  sliceSizes[tid].assign(lvlRank, std::vector<Value>());
314  sliceStack[tid].emplace_back(/*minCrd=*/Value(),
315  /*offset=*/Value(), /*isNonEmpty*/ Value(),
316  std::nullopt, 0);
317  if (dimGetter) {
318  auto reassoc = collapseReassoc[tid];
319  Level dstRank = reassoc ? reassoc.size() : lvlRank;
320  for (Level l = 0; l < dstRank; l++) {
321  dependentLvlMap[tid][l] = dimGetter(tid, l);
322  unsigned depends = dependentLvlMap[tid][l].size();
323  if (depends == 0)
324  continue;
325  // TODO: View-base collapse and dependent index reduction are not
326  // compatible right now.
327  assert(!reassoc);
328  // We need `depends - 1` slices to fully the affine expression.
329  sliceSizes[tid][l].assign(depends - 1, nullptr);
330  slicePosBuffer[tid][l].assign(depends - 1, nullptr);
331  }
332  }
333  }
334 
335  // Construct the inverse of the `topSort` from the sparsifier.
336  // This is needed to map `AffineDimExpr`s back to the `LoopOrd`
337  // used in loop emitter.
338  // FIXME: This map should be maintained outside loop emitter.
339  for (LoopOrd n = 0; n < numLoops; n++)
340  loopIdToOrd[topSort[n]] = n;
341 }

343 void LoopEmitter::initializeLoopEmit(OpBuilder &builder, Location loc,
344  LoopEmitter::OutputUpdater updater) {
345  // For every tensor:
346  // * get the values buffer.
347  // * For every level:
348  // * get the positions and coordinates buffers
349  // * get/compute the level-size, which is also used as the upper-bound
350  // on positions.
351  for (TensorId t = 0, numTensors = getNumTensors(); t < numTensors; t++) {
352  const Value tensor = tensors[t];
353  const auto rtp = tensor.getType().dyn_cast<RankedTensorType>();
354  if (!rtp)
355  // Skips only scalar, zero ranked tensor still need to be bufferized and
356  // (probably) filled with zeros by users.
357  continue;
358  // FIXME: the definition of `lvlRank` looks more like a dim-rank;
359  // but the variable is used as a level everywhere below, which
360  // suggests there may be some dim/lvl confusion going on here.
361  const Level lvlRank = rtp.getRank();
362  const auto shape = rtp.getShape();
363  const auto enc = getSparseTensorEncoding(rtp);
364  const Level cooStart = enc ? getCOOStart(enc) : lvlRank;
365  // Scan all levels of current tensor.
366  for (Level l = 0; l < lvlRank; l++) {
367  // This should be called only once at beginning.
368  assert(!positionsBuffers[t][l] && !coordinatesBuffers[t][l] &&
369  !highs[t][l]);
370  const auto lvlTp = lvlTypes[t][l];
371  // Handle sparse storage schemes.
372  if (isCompressedDLT(lvlTp) || isCompressedWithHiDLT(lvlTp)) {
373  // Generate sparse primitives to obtain positions and coordinates.
374  positionsBuffers[t][l] = genToPositions(builder, loc, tensor, l);
375  coordinatesBuffers[t][l] =
376  genToCoordinates(builder, loc, tensor, l, cooStart);
377  } else if (isSingletonDLT(lvlTp)) {
378  // Singleton level, fetch coordinates.
379  coordinatesBuffers[t][l] =
380  genToCoordinates(builder, loc, tensor, l, cooStart);
381  } else {
382  // Dense level, nothing to fetch.
383  assert(isDenseDLT(lvlTp));
384  }
385 
386  // FIXME: `toOrigDim` is deprecated. For now this relies on the
387  // 1:1 mapping between levels and dimensions, since nowhere else
388  // in the code supports HigherOrdering yet either.
389  Value lvlSz = mlir::linalg::createOrFoldDimOp(builder, loc, tensor,
390  toOrigDim(enc, l));
391  // Find upper bound in current dimension.
392  highs[t][l] = lvlSizes[t][l] = lvlSz;
393  if (isSparseSlices[t]) {
394  sliceOffsets[t][l] = genSliceOffset(builder, loc, tensors[t], l);
395  sliceStrides[t][l] = genSliceStride(builder, loc, tensors[t], l);
396  }
397  }
398 
399  // Perform the required bufferization. Dense inputs materialize
400  // from the input tensors. Sparse inputs use sparse primitives to obtain the
401  // values.
402  // Delegates extra output initialization to clients.
403  bool isOutput = isOutputTensor(t);
404  Type elementType = rtp.getElementType();
405  if (!enc) {
406  // Non-annotated dense tensors.
407  BaseMemRefType denseTp = MemRefType::get(shape, elementType);
408 
409  // TODO: if we unconditionally use fully dynamic layout here, it breaks
410  // some vectorization passes which requires static stride = 1.
411  // Is it possible to call vectorization pass after bufferization?
412  if (llvm::isa_and_nonnull<tensor::ExtractSliceOp>(tensor.getDefiningOp()))
413  denseTp = bufferization::getMemRefTypeWithFullyDynamicLayout(rtp);
414 
415  Value denseVal =
416  builder.create<bufferization::ToMemrefOp>(loc, denseTp, tensor);
417  // Dense outputs need special handling.
418  if (isOutput && updater)
419  denseVal = updater(builder, loc, denseVal, tensor);
420 
421  valBuffer[t] = denseVal;
422  } else {
423  // Annotated sparse tensors.
424  // We also need the value buffer for all-dense annotated "sparse" tensors.
425  valBuffer[t] = genToValues(builder, loc, tensor);
426  }
427  // NOTE: we can also prepare for 0 lvl here in advance, this will hoist
428  // some loop preparation from tensor iteration, but will also (undesirably)
429  // hoist the code ouside if-conditions.
430  }
431 
432  Type indexType = builder.getIndexType();
433  Value c0 = constantZero(builder, loc, indexType);
434  for (TensorId t = 0, e = tensors.size(); t < e; t++) {
435  auto rtp = tensors[t].getType().dyn_cast<RankedTensorType>();
436  if (!rtp)
437  continue;
438 
439  Level lvlRank = SparseTensorType(rtp).getLvlRank();
440  for (Level lvl = 0; lvl < lvlRank; lvl++) {
441  if (!dependentLvlMap[t][lvl].empty()) {
442  ArrayRef<std::pair<TensorId, Level>> depLvls = dependentLvlMap[t][lvl];
443  // Needs at least two operands to form a non-trivial affine expression.
444  assert(depLvls.size() > 1);
445 
446  Value size = c0;
447  for (unsigned e = depLvls.size() - 1; e >= 1; e--) {
448  auto [dt, dd] = depLvls[e];
449  size = ADDI(size, lvlSizes[dt][dd]);
450  sliceSizes[t][lvl][e - 1] = size;
451  }
452  }
453  }
454  }
455  localInsertPos = builder.getInsertionPoint()->getPrevNode();
456 }

477 void LoopEmitter::exitCurrentLoopSeq(OpBuilder &builder, Location loc) {
478  assert(loopSeqStack.size() == loopStack.size() + 1);
479 
480  const auto &slicedTids = loopSeqStack.back().second;
481 
482  // Depending on whether the slice is resolved or not at current loop sequence,
483  // end them in different ways.
484  for (auto [tid, lvl, res] : slicedTids) {
485  if (!res) {
486  // If this is a unresolved-slice-driven loop, pops out the slice.
487  assert(sliceStack[tid].back().slicedOnLvl == lvl);
488  sliceStack[tid].pop_back();
489  } else {
490  if (!isDenseDLT(lvlTypes[tid][lvl])) {
491  // Else this is a resolved-slice, and advance posit similar to TACO.
492  Value c1 = C_IDX(1), c2 = C_IDX(2);
493  // pIdx += 2, we finished the current lvl, advance the pointer index of
494  // the previous level by two to skip the [pLo, pHi] for current level.
495  Value sPtrBuf = slicePosBuffer[tid][lvl].back();
496  Value curP = genIndexLoad(builder, loc, sPtrBuf, c1);
497  // TODO: we could probably use an SSA value for it.
498  Value nexP = ADDI(curP, c2);
499  builder.create<memref::StoreOp>(loc, nexP, sPtrBuf, c1);
500  }
501  }
502  }
503  loopSeqStack.pop_back();
504 }

506 Value LoopEmitter::genAffine(OpBuilder &builder, Location loc, AffineExpr a) {
507  switch (a.getKind()) {
508  case AffineExprKind::DimId: {
509  // FIXME: since the one callsite in Sparsification passes in a
510  // level-expression, the `getPosition` must in fact be a `Dimension`.
511  // However, elsewhere we have been lead to expect that `loopIdToOrd`
512  // should be indexed by `LoopId`...
513  const auto loopId = a.cast<AffineDimExpr>().getPosition();
514  assert(loopId < loopIdToOrd.size());
515  return loopStack[loopIdToOrd[loopId]].iv;
516  }
517  case AffineExprKind::Add: {
518  auto binOp = a.cast<AffineBinaryOpExpr>();
519  return ADDI(genAffine(builder, loc, binOp.getLHS()),
520  genAffine(builder, loc, binOp.getRHS()));
521  }
522  case AffineExprKind::Mul: {
523  auto binOp = a.cast<AffineBinaryOpExpr>();
524  return MULI(genAffine(builder, loc, binOp.getLHS()),
525  genAffine(builder, loc, binOp.getRHS()));
526  }
527  case AffineExprKind::Constant: {
528  int64_t c = a.cast<AffineConstantExpr>().getValue();
529  return C_IDX(c);
530  }
531  default:
532  llvm_unreachable("unexpected affine subscript");
533  }

536 Operation *LoopEmitter::emitForLoopOverTensorAtLvl(
537  OpBuilder &builder, Location loc, TensorId tid, Level dstLvl, Value lo,
538  Value hi, MutableArrayRef<Value> reduc, bool isParallel) {
539  bool isSparseCond = isCompressedDLT(lvlTypes[tid][dstLvl]) ||
540  isCompressedWithHiDLT(lvlTypes[tid][dstLvl]) ||
541  isSingletonDLT(lvlTypes[tid][dstLvl]);
542 
543  const auto reassoc = getCollapseReassociation(tid, dstLvl);
544  // TODO: support dynamic slices.
545  // Uses the first dimension here to build the loop bound (which is also the
546  // biggest range).
547  const Level srcLvl = reassoc.front();
548  Value step = C_IDX(1);
549 
550  Operation *loop = nullptr;
551  Value iv;
552  if (isParallel) {
553  assert(collapseReassoc[tid] == nullptr);
554  scf::ParallelOp parOp =
555  builder.create<scf::ParallelOp>(loc, lo, hi, step, reduc);
556  builder.setInsertionPointToStart(parOp.getBody());
557  assert(parOp.getNumReductions() == reduc.size());
558  iv = parOp.getInductionVars()[0];
559 
560  // In-place update on the reduction variable vector.
561  // Note that the init vals is not the actual reduction variables but instead
562  // used as a "special handle" to (temporarily) represent them. The
563  // expression on init vals will be moved into scf.reduce and replaced with
564  // the block arguments when exiting the loop (see exitForLoop). This is
565  // needed as we can not build the actual reduction block and get the actual
566  // reduction varaible before users fill parallel loop body.
567  for (int i = 0, e = reduc.size(); i < e; i++)
568  reduc[i] = parOp.getInitVals()[i];
569  loop = parOp;
570  } else {
571  scf::ForOp forOp = builder.create<scf::ForOp>(loc, lo, hi, step, reduc);
572  builder.setInsertionPointToStart(forOp.getBody());
573  iv = forOp.getInductionVar();
574 
575  // In-place update on the reduction variable vector.
576  assert(forOp.getNumRegionIterArgs() == reduc.size());
577  for (int i = 0, e = reduc.size(); i < e; i++)
578  reduc[i] = forOp.getRegionIterArg(i);
579  loop = forOp;
580  }
581  assert(loop && iv);
582 
583  Value crd;
584  if (isSparseCond) {
585  assert(reassoc.size() == 1 || isUniqueCOOType(tensors[tid].getType()));
586  // For COO, the position is the same across consecutive levels.
587  /// FIXME: See the [CLARIFY_POSITS_LVL] note in the header.
588  llvm::for_each(reassoc,
589  [this, tid, iv](Level srcLvl) { posits[tid][srcLvl] = iv; });
590  crd = genSparseCrd(builder, loc, tid, dstLvl);
591  } else {
592  // Dense tensor, the coordinate is the inducation variable.
593  crd = iv;
594  }
595 
596  if (isSparseSlices[tid] && isSparseCond) {
597  // For sparse level slices, we need to filter out invalid coordinates that
598  // are not included in the slice.
599  SmallVector<Type> types;
600  for (Value red : reduc)
601  types.push_back(red.getType());
602 
603  auto [trans, pred] = genSliceLegitPredicate(builder, loc, crd, tid, srcLvl);
604  bool hasReduc = !types.empty();
605  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, types, pred,
606  /*else*/ hasReduc);
607  if (hasReduc) {
608  // scf.for (a) -> v
609  // %s = scf.if (a) -> v
610  // user-generated code.
611  // else
612  // yield a
613  // yield %s
614  YIELD(ifOp.getResults());
615  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
616  // On mismatch.
617  YIELD(reduc);
618  }
619  // Set the insertion point to matched branch.
620  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
621  crd = trans;
622  }
623 
624  assert(crd);
625  coords[tid][dstLvl] = crd;
626  return loop;
627 }

662 Operation *LoopEmitter::enterLoopOverTensorAtLvl(OpBuilder &builder,
663  Location loc,
664  ArrayRef<TensorLevel> tidLvls,
665  MutableArrayRef<Value> reduc,
666  bool isParallel) {
667  // TODO: support multiple return on parallel for?
668  assert(!isParallel || reduc.size() <= 1);
669  bool isSparseCond = false, isSparseSliceCond = false;
670  auto [tid, lvl] = unpackTensorLevel(tidLvls.front());
671 
672  // Finds out the tensor level that we should use to generate loops. Amongs all
673  // the tensor levels, there is at most one sparse tensor level.
674  for (auto [t, l] : unpackTensorLevelRange(tidLvls)) {
675  assert(lvlTypes[t].size() > l); // Must be a valid tid, dim pair
676  assert(!coords[t][l] || // We cannot re-enter the same level
677  !dependentLvlMap[t][l].empty()); // unless it is a slice-driver loop
678  auto lvlType = lvlTypes[t][l];
679  // Must be a recognizable DLT.
680  assert(isDenseDLT(lvlType) || isCompressedDLT(lvlType) ||
681  isCompressedWithHiDLT(lvlType) || isSingletonDLT(lvlType));
682 
683  // This is a slice-driven loop on sparse level.
684  if (!dependentLvlMap[t][l].empty() && !isDenseDLT(lvlType)) {
685  assert(!isSparseSliceCond && !isSparseCond);
686  isSparseSliceCond = true;
687  tid = t;
688  lvl = l;
689  continue;
690  }
691 
692  bool isSparse = isCompressedDLT(lvlType) || isSingletonDLT(lvlType) ||
693  isCompressedWithHiDLT(lvlType);
694  // We can at most have one sparse input, otherwise, a while loop is
695  // required to co-iterate multiple sparse tensors.
696  assert(!isSparseCond || !isSparse);
697  assert(!isSparseSliceCond || !isSparseCond);
698  if (isSparse) {
699  tid = t;
700  lvl = l;
701  }
702  isSparseCond = isSparseCond || isSparse;
703  }
704 
705  DimLevelType lvlType = lvlTypes[tid][lvl];
706  // TODO: Dense slice driven loop can be generated using for loop as well.
707  assert(!isSparseSliceCond || !isDenseDLT(lvlType));
708  bool isDenseSliceCond =
709  isDenseDLT(lvlType) && !dependentLvlMap[tid][lvl].empty();
710  // if the slice is fully reduced, we can now use TACO-based algorithm to
711  // iterate it.
712 
713  Operation *l = nullptr;
714 
715  // At most one tensor used as condition in for loop;
716  SmallVector<TensorLevel, 1> condTidLvl;
717  // There might be multiple dense slice driven tensor.
718  SmallVector<SliceLoopInfo> sliceDrivenInfo;
719 
720  // Generates loops differently depending on whether we need a slice-driven
721  // loop or a simple level traversal loop.
722  if (isSparseSliceCond) {
723  bool fullyReduced = depFullyReduced(tid, lvl);
724  if (!fullyReduced) {
725  l = emitSliceDrivenLoopOverTensorAtLvl(builder, loc, tid, lvl, reduc);
726  } else {
727  // If the slice is fully reduced, we can now use TACO-based algorithm to
728  // iterate it.
729  l = emitWhileLoopOverSliceAtSparseLvl(
730  builder, loc, posits[tid][lvl], highs[tid][lvl],
731  getFinalSliceOnLvl(tid, lvl).offset, sliceSizes[tid][lvl].back(), tid,
732  lvl, reduc);
733  }
734  levelReducedDep[tid][lvl]++;
735  sliceDrivenInfo.emplace_back(tid, lvl, fullyReduced);
736  } else {
737  Value lo = isSparseCond ? posits[tid][lvl] // current offset
738  : loopSeqStack.back().first; // universal index
739  Value hi = highs[tid][lvl];
740  if (isDenseSliceCond) {
741  bool fullyReduced = depFullyReduced(tid, lvl);
742  Value sliceSz = sliceSizes[tid][lvl][sliceStack[tid].back().depth - 1];
743  // Adjust for loop hi for dense slice-driven loop.
744  if (fullyReduced) {
745  hi = sliceSz;
746  condTidLvl.push_back(makeTensorLevel(tid, lvl));
747  } else {
748  hi = SUBI(lvlSizes[tid][lvl], sliceSz);
749  hi = ADDI(hi, C_IDX(1));
750  }
751  } else {
752  condTidLvl.push_back(makeTensorLevel(tid, lvl));
753  }
754  l = emitForLoopOverTensorAtLvl(builder, loc, tid, lvl, lo, hi, reduc,
755  isParallel);
756  }
757  Value iv = coords[tid][lvl];
758  for (auto [t, l] : unpackTensorLevelRange(tidLvls)) {
759  // We only need to handle slice-driven loops on dense level here.
760  // If it is a slice-driven loop on sparse level, it needs a while loop to
761  // insert break statements, and it must have been handled correctly in L692.
762  if (!dependentLvlMap[t][l].empty() && isDenseDLT(lvlTypes[t][l])) {
763  // Pushes sliced levels to build correct LoopInfo.
764  bool fullyReduc = depFullyReduced(t, l);
765  SliceInfo &info = sliceStack[t].back();
766  if (fullyReduc) {
767  posits[t][l] = genAddress(builder, loc, t, l, ADDI(info.offset, iv));
768  } else {
769  // Puts sliced dense loop into LoopInfo so that LoopEmitter knows how to
770  // exit it.
771  sliceDrivenInfo.emplace_back(t, l, fullyReduc);
772  // Update the slice information as we enter the new loop.
773  assert(*info.slicedOnLvl == l);
774  info.minCrd = info.offset = iv;
775  info.isNonEmpty = constantI1(builder, loc, true);
776  levelReducedDep[t][l]++;
777  }
778  }
779  }
780  // NOTE: we can also prepare for next dim here in advance
781  // Pushes the loop into stack.
782  loopStack.emplace_back(condTidLvl, sliceDrivenInfo, l,
783  builder.getInsertionBlock(), iv, loopTag);
784  // Emit extra locals.
785  emitExtraLocalsForTensorsAtDenseLvls(builder, loc, tidLvls);
786  return l;
787 }

866 Operation *LoopEmitter::enterCoIterationOverTensorsAtLvls(
867  OpBuilder &builder, Location loc, ArrayRef<TensorLevel> tidLvls,
868  bool needsUniv, MutableArrayRef<Value> reduc) {
869  // NOTE: the slice driven tensor-related reduction variable must
870  // appear before normal tensors.
871  SmallVector<Type> types;
872  SmallVector<Value> operands;
873  // Construct the while-loop with a parameter for each coordinate.
874  const Type indexType = builder.getIndexType();
875  for (auto [tid, lvl] : unpackTensorLevelRange(tidLvls)) {
876  // TODO: support coiteration with slice driven tensors.
877  const auto lvlTp = lvlTypes[tid][lvl];
878  assert(dependentLvlMap[tid][lvl].empty() && "TODO: not yet implemented");
879  if (isCompressedDLT(lvlTp) || isSingletonDLT(lvlTp) ||
880  isCompressedWithHiDLT(lvlTp)) {
881  const auto reassoc = getCollapseReassociation(tid, lvl);
882  for (unsigned i = 0, e = reassoc.size() - 1; i < e; i++) {
883  if (!isUniqueDLT(lvlTypes[tid][reassoc[i]])) {
884  // This is the segment high for each non-unique levels.
885  types.push_back(indexType);
886  operands.push_back(C_IDX(0));
887  }
888  }
889  const auto pos = posits[tid][reassoc.front()];
890  assert(pos);
891  types.push_back(indexType);
892  operands.push_back(pos);
893  }
894  }
895  // The position where user-supplied reduction variable starts.
896  for (Value rec : reduc) {
897  types.push_back(rec.getType());
898  operands.push_back(rec);
899  }
900  if (needsUniv) {
901  types.push_back(indexType);
902  // Update universal index.
903  operands.push_back(loopSeqStack.back().first);
904  }
905  assert(types.size() == operands.size());
906  scf::WhileOp whileOp = builder.create<scf::WhileOp>(loc, types, operands);
907 
908  SmallVector<Location> locs(types.size(), loc);
909  Block *before = builder.createBlock(&whileOp.getBefore(), {}, types, locs);
910  Block *after = builder.createBlock(&whileOp.getAfter(), {}, types, locs);
911 
912  // Build the "before" region, which effectively consists
913  // of a conjunction of "i < upper" tests on all induction.
914  builder.setInsertionPointToStart(&whileOp.getBefore().front());
915  Value cond;
916  unsigned o = 0;
917  for (auto [t, lvl] : unpackTensorLevelRange(tidLvls)) {
918  const TensorId tid = t; // Why `t` can not be captured by lambda?
919  const auto lvlTp = lvlTypes[tid][lvl];
920  if (isCompressedDLT(lvlTp) || isSingletonDLT(lvlTp) ||
921  isCompressedWithHiDLT(lvlTp)) {
922  const auto reassoc = getCollapseReassociation(tid, lvl);
923  assert(reassoc.size() == 1 || isUniqueCOOType(tensors[tid].getType()));
924  for (unsigned i = 0, e = reassoc.size() - 1; i < e; i++) {
925  if (!isUniqueDLT(lvlTypes[tid][reassoc[i]])) {
926  // Links the SSA chain for segHi.
927  segHi[tid][reassoc[i]] = after->getArgument(o++);
928  }
929  }
930  Value op1 = before->getArgument(o);
931  // We used the first level bound as the bound the collapsed set of levels.
932  Value op2 = highs[tid][reassoc.front()];
933  Value opc = CMPI(ult, op1, op2);
934  cond = cond ? ANDI(cond, opc) : opc;
935  // Update positions
936  Value pos = after->getArgument(o++);
937  // For COO, the position is the same across consecutive levels.
938  /// FIXME: See the [CLARIFY_POSITS_LVL] note in the header.
939  llvm::for_each(reassoc, [this, tid, pos](Level srcLvl) {
940  posits[tid][srcLvl] = pos;
941  });
942  }
943  }
944  builder.create<scf::ConditionOp>(loc, cond, before->getArguments());
945 
946  // Generates while body.
947  builder.setInsertionPointToStart(&whileOp.getAfter().front());
948 
949  SmallVector<std::pair<Value, unsigned>> slicesPreds;
950  unsigned i = 0;
951  for (auto [tid, lvl] : unpackTensorLevelRange(tidLvls)) {
952  // Prepares for next level.
953  const auto lvlTp = lvlTypes[tid][lvl];
954  if (isCompressedDLT(lvlTp) || isSingletonDLT(lvlTp) ||
955  isCompressedWithHiDLT(lvlTp)) {
956  coords[tid][lvl] = genSparseCrd(builder, loc, tid, lvl);
957  if (isSparseSlices[tid]) {
958  auto [trans, pred] =
959  genSliceLegitPredicate(builder, loc, coords[tid][lvl], tid, lvl);
960  slicesPreds.emplace_back(pred, i);
961  // Updates to the relative coordinate to the slice.
962  coords[tid][lvl] = trans;
963  }
964  i++;
965  }
966  }
967 
968  if (!slicesPreds.empty()) {
969  // Skips invalid loop iteration when slice coordinate is inapplicable.
970  SmallVector<Value> yields(after->getArguments());
971  // Generates a list of if statments
972  // pos = in_slice ? pos : pos + 1
973  // TODO: instead of always picking pos + 1, we should set pos = high to
974  // break to loop if the coordinates are larger than the slice size.
975  //
976  // This "idx" is the index into `llvm::zip(tids, lvls)`
977  for (auto [pred, idx] : slicesPreds) {
978  Value nextPos = ADDI(yields[idx], C_IDX(1));
979  yields[idx] = SELECT(pred, yields[idx], nextPos);
980  }
981 
982  Value pred = slicesPreds.front().first;
983  for (int i = 1, e = slicesPreds.size(); i < e; i++) {
984  pred = ANDI(pred, slicesPreds[i].first);
985  }
986  auto ifOp = builder.create<scf::IfOp>(loc, types, pred, /*else*/ true);
987  ifOp->setAttr(getLoopEmitterLoopAttrName(),
988  StringAttr::get(builder.getContext(), "slice"));
989  YIELD(ifOp->getResults());
990  assert(types.size() == yields.size());
991  // If not all slices are legit
992  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
993  YIELD(yields);
994 
995  // If all slices are legit, start the user generated code.
996  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
997  }
998 
999  Value min;
1000  // Finds the minimum coordinate
1001  if (!needsUniv) {
1002  for (auto [tid, lvl] : unpackTensorLevelRange(tidLvls)) {
1003  const auto lvlTp = lvlTypes[tid][lvl];
1004  if (isCompressedDLT(lvlTp) || isSingletonDLT(lvlTp) ||
1005  isCompressedWithHiDLT(lvlTp)) {
1006  const auto crd = coords[tid][lvl];
1007  if (min) {
1008  Value cmp = CMPI(ult, coords[tid][lvl], min);
1009  min = SELECT(cmp, coords[tid][lvl], min);
1010  } else {
1011  min = crd;
1012  }
1013  }
1014  }
1015  } else {
1016  assert(!min);
1017  // Otherwise, universal index is the minimal pos.
1018  min = after->getArguments().back();
1019  }
1020 
1021  // Sets up the loop stack.
1022  loopStack.emplace_back(tidLvls, ArrayRef<SliceLoopInfo>(), whileOp,
1023  builder.getInsertionBlock(), min, loopTag);
1024  assert(loopStack.size() == loopSeqStack.size());
1025 
1026  for (auto [tid, dstLvl] : unpackTensorLevelRange(tidLvls)) {
1027  const auto reassoc = getCollapseReassociation(tid, dstLvl);
1028  assert(reassoc.size() == 1 || isUniqueCOOType(tensors[tid].getType()));
1029  // TODO: Refactors this into smaller functions.
1030  // NOTE: For all the collapsed level (except for the last one, that is why
1031  // the loop ends with `reassoc.size() - 1`), as each iteration is advanced
1032  // by the segment size of the last level, which does not always invalidate
1033  // the segment size for the previous levels, thus we need to propagate the
1034  // segment sizes across loop iterations and only forward if needed.
1035  //
1036  // E.g., for a COO tensor with the following coordinates array.
1037  // (0, 0, 1),
1038  // (0, 0, 2),
1039  // (1, 1, 1),
1040  // segHi[lvl=0] = segHi[lvl=1] = 2
1041  // segHi[lvl=2] = 1,
1042  // the first iteration does not invalidate segHi[0] and segHi[1]
1043  for (unsigned i = 0, e = reassoc.size() - 1; i < e; i++) {
1044  const Level srcLvl = reassoc[i];
1045  if (!isUniqueDLT(lvlTypes[tid][srcLvl])) {
1046  const Value pos = posits[tid][srcLvl];
1047  const auto oldSegHi = segHi[tid][srcLvl];
1048  assert(oldSegHi);
1049  Value newSegHi = builder.create<arith::CmpIOp>(
1050  loc, arith::CmpIPredicate::uge, pos, oldSegHi);
1051  auto ifNewSegHi = builder.create<scf::IfOp>(loc, builder.getIndexType(),
1052  newSegHi, true);
1053  {
1054  OpBuilder::InsertionGuard guard(builder);

1162 void LoopEmitter::exitForLoop(RewriterBase &rewriter, Location loc,
1163  MutableArrayRef<Value> reduc) {
1164  const LoopInfo &loopInfo = loopStack.back();
1165  rewriter.setInsertionPointToEnd(loopInfo.userCodeBlock);
1166  for (auto [tid, lvl, reduced] : loopInfo.sliceDrivenInfo) {
1167  SliceInfo &info = sliceStack[tid].back();
1168  assert(isDenseDLT(lvlTypes[tid][lvl]));
1169  assert(*info.slicedOnLvl == lvl && !reduced);
1170  (void)reduced;
1171  // Resets slices pointers as the resolved slices are invalidated after we
1172  // moves forward to the next slice.
1173  invalidateSliceIterIdx(rewriter, loc, tid, lvl);
1174  info.minCrd = info.offset = info.isNonEmpty = Value();
1175  levelReducedDep[tid][lvl]--;
1176  }
1177  if (auto forOp = llvm::dyn_cast<scf::ForOp>(loopInfo.loop)) {
1178  if (!reduc.empty()) {
1179  assert(reduc.size() == forOp.getNumResults());
1180  rewriter.create<scf::YieldOp>(loc, reduc);
1181  }
1182  // Exit the loop.
1183  rewriter.setInsertionPointAfter(forOp);
1184  // In-place update reduction variables.
1185  for (unsigned i = 0, e = forOp.getResults().size(); i < e; i++)
1186  reduc[i] = forOp.getResult(i);
1187  } else {
1188  auto parOp = llvm::cast<scf::ParallelOp>(loopInfo.loop);
1189  if (!reduc.empty()) {
1190  assert(reduc.size() == parOp.getInitVals().size() && reduc.size() == 1);
1191  Operation *redExp = reduc.front().getDefiningOp();
1192  // Reduction expression should have no use.
1193  assert(redExp->getUses().empty());
1194  // This must be a binary operation.
1195  // NOTE: This is users' responsibilty to ensure the operation are
1196  // commutative.
1197  assert(redExp->getNumOperands() == 2 && redExp->getNumResults() == 1);
1198 
1199  Value redVal = parOp.getInitVals().front();
1200  Value curVal;
1201  if (redExp->getOperand(0) == redVal)
1202  curVal = redExp->getOperand(1);
1203  else if (redExp->getOperand(1) == redVal)
1204  curVal = redExp->getOperand(0);
1205  // One of the operands must be the init value (which is also the
1206  // previous reduction value).
1207  assert(curVal);
1208 #ifndef NDEBUG
1209  // The reduction expression should be the only user of the reduction val
1210  // inside the parallel for.
1211  unsigned numUsers = 0;
1212  for (Operation *op : redVal.getUsers()) {
1213  if (op->getParentOp() == parOp)
1214  numUsers++;
1215  }
1216  assert(numUsers == 1);
1217 #endif // NDEBUG
1218 
1219  rewriter.setInsertionPointAfter(redExp);
1220  auto redOp = rewriter.create<scf::ReduceOp>(loc, curVal);
1221  // Attach to the reduction op.
1222  Block *redBlock = &redOp.getRegion().getBlocks().front();
1223  rewriter.setInsertionPointToEnd(redBlock);
1224  Operation *newRed = rewriter.clone(*redExp);
1225  // Replaces arguments of the reduction expression by using the block
1226  // arguments from scf.reduce.
1227  rewriter.updateRootInPlace(
1228  newRed, [&]() { newRed->setOperands(redBlock->getArguments()); });
1229  // Erases the out-dated reduction expression.
1230  rewriter.eraseOp(redExp);
1231  rewriter.setInsertionPointToEnd(redBlock);
1232  rewriter.create<scf::ReduceReturnOp>(loc, newRed->getResult(0));
1233  }
1234  rewriter.setInsertionPointAfter(parOp);
1235  // In-place update reduction variables.
1236  for (unsigned i = 0, e = parOp.getResults().size(); i < e; i++)
1237  reduc[i] = parOp.getResult(i);
1238  }
1239 
1240  // Finished iterating a tensor, clean up
1241  // We only do the clean up on for loop as while loops do not necessarily
1242  // finish the iteration on a sparse tensor
1243  for (auto [tid, lvl] : unpackTensorLevelRange(loopInfo.tidLvls)) {
1244  // Reset to null.
1245  coords[tid][lvl] = Value();
1246  posits[tid][lvl] = Value();
1247  // Dense level, high is fixed.
1248  if (!isDenseDLT(lvlTypes[tid][lvl]))
1249  highs[tid][lvl] = Value();
1250  }
1251 }

1253 void LoopEmitter::exitWhileLoop(OpBuilder &builder, Location loc,
1254  MutableArrayRef<Value> reduc) {
1255  const LoopInfo &loopInfo = loopStack.back();
1256  auto whileOp = llvm::cast<scf::WhileOp>(loopInfo.loop);
1257  builder.setInsertionPointToEnd(loopInfo.userCodeBlock);
1258  Value iv = loopInfo.iv;
1259 
1260  // Finalize the induction. Note that the induction could be performed
1261  // in the individual if-branches to avoid re-evaluating the conditions.
1262  // However, that would result in a rather elaborate forest of yield
1263  // instructions during code generation. Moreover, performing the induction
1264  // after the if-statements more closely resembles code generated by TACO.
1265  unsigned o = 0;
1266  SmallVector<Value> operands;
1267  unsigned delta = 0;
1268  for (auto [tid, lvl, resolved] : loopInfo.sliceDrivenInfo) {
1269  // TODO: handle dense.
1270  assert(isCompressedDLT(lvlTypes[tid][lvl]));
1271  levelReducedDep[tid][lvl]--;
1272  if (!resolved) {
1273  genSliceNextInduction(builder, loc, whileOp, tid, lvl, operands, o);
1274  continue;
1275  }
1276  // TODO: We need to distinguish coiterate loop with slice-driven loop and
1277  // fully reduced while op for iterating one slices.
1278  // FIXME: since we didn't implement coiteration, this must be iteration
1279  // just on fully resolved slice.
1280  assert(loopInfo.sliceDrivenInfo.size() == 1 && loopInfo.tidLvls.empty());
1281  // The if guard to filter out out-range coordinates.
1282  assert(llvm::isa<scf::IfOp>(builder.getInsertionBlock()->getParentOp()));
1283  posits[tid][lvl] = whileOp->getResult(o++);
1284  // FIXME: we are not using continue here since we do not support
1285  // coiteration on slices. But it need to be treated similarly as the
1286  // universal index.
1287  o++; // skip continue flag.
1288  // Since we did not push two results from whileOp. The size of the
1289  // operands vector is smaller than the actual number of return values from
1290  // the whileOp.
1291  // It is because we are actually generating yield in the IfOp inside the
1292  // whileOp to only iterates over inbound coordinates within the slices.
1293  delta += 2;
1294  };
1295 
1296  Value one = C_IDX(1);
1297  for (auto [tid, dstLvl] : unpackTensorLevelRange(loopInfo.tidLvls)) {
1298  const auto lvlTp = lvlTypes[tid][dstLvl];
1299  if (isCompressedDLT(lvlTp) || isSingletonDLT(lvlTp) ||
1300  isCompressedWithHiDLT(lvlTp)) {
1301  const auto reassoc = getCollapseReassociation(tid, dstLvl);
1302  assert(reassoc.size() == 1 || isUniqueCOOType(tensors[tid].getType()));
1303  for (unsigned i = 0, e = reassoc.size() - 1; i < e; i++) {
1304  const Level srcLvl = reassoc[i];
1305  if (!isUniqueDLT(lvlTypes[tid][srcLvl])) {
1306  operands.push_back(segHi[tid][srcLvl]);
1307  o++;
1308  }
1309  }
1310  const Value crd = coords[tid][dstLvl];
1311  const Value pos = posits[tid][dstLvl];
1312  Value cmp = CMPI(eq, crd, iv);
1313  // If the loop contains a coiteration with non-unique level, we fast
1314  // forward all the duplicated coords by setting the position to the
1315  // segment high.
1316  Value add = !isUniqueDLT(lvlTypes[tid][reassoc.back()])
1317  ? segHi[tid][reassoc.back()]
1318  : ADDI(pos, one);
1319 
1320  operands.push_back(SELECT(cmp, add, pos));
1321  // Following loops continue iteration from the break point of the
1322  // current while loop.
1323  const Value newPos = whileOp->getResult(o++);
1324  // We need to define a new local variable for `tid` to avoid
1325  // warnings about "captured structured bindings are a C++20 extension".
1326  // FIXME(wrengr): define a helper function to capture this idiom!
1327  const TensorId newTid = tid;
1328  llvm::for_each(reassoc, [this, newTid, newPos](Level srcLvl) {
1329  posits[newTid][srcLvl] = newPos;
1330  });
1331  // The coordinate is invalid now.
1332  coords[tid][dstLvl] = nullptr;
1333  // The segment high is invalid now.
1334  segHi[tid][dstLvl] = nullptr;
1335  // highs remains unchanged.
1336  }
1337  }
1338 
1339  // Reduction value from users.
1340  for (auto &i : reduc) {
1341  operands.push_back(i);
1342  // In place update reduction variable.
1343  i = whileOp->getResult(o++);
1344  }
1345 
1346  // An (optional) universal index.
1347  if (operands.size() + delta < whileOp.getNumResults()) {
1348  assert(operands.size() + delta + 1 == whileOp.getNumResults());
1349  // The last one is the universial index.
1350  operands.push_back(ADDI(iv, one));
1351  // update the loop starting point of current loop sequence
1352  loopSeqStack.back().first = whileOp->getResult(o++);
1353  }
1354 
1355  assert(o == operands.size() + delta);
1356  YIELD(operands);
1357  builder.setInsertionPointAfter(whileOp);
1358 }

1783  Level lvl) {

1785

1787  // Do not need to prepare for slice driven loop on dense level after it is
1788  // fully reduced.
1789  if (isDenseDLT(lvlTypes[tid][lvl]))
1790  return true;

1792  // generates slice begin any more, instead we fall back to TACO-based

1794  Value pLoPtr =

1809  if (isSingletonDLT(lvlType)) {
1810  llvm_unreachable("TODO: dense level should be easy to support, while "
1811  "singleton level requres more efforts");

1815  assert(!sliceStack[tid].empty());

1818  auto baseEnc = getSparseTensorEncoding(tensors[tid].getType());
1819  if (baseEnc.isSlice())
1820  llvm_unreachable("TODO: not yet implemented");

1829  builder.setInsertionPointAfter(localInsertPos);

1832  // Accumlates the size required to cache the pLo for the slice.
1833  // E.g., if we want to cache the pIdx for slice<d0xd1xf64> on the second
1834  // level. We at most need to a memref<d0xindex>.
1835  // NOTE: this is apperantly an over-approximation when the previous
1836  // level is compressed, and we can compute a precise memory size
1837  // inside the loops. But that would also requires us to allocate/free
1838  // memorys in loops.

1843  assert(sliceSizes[tid][lvl].size() >= depth);

1850  bufSize = MULI(bufSize, c2);
1851  // Additional two metadata {memSize, idx} at head.
1852  bufSize = ADDI(bufSize, c2);
1853  llvm::for_each(
1854  slicePosBuffer[tid][lvl], [bufSize, loc, &builder](Value &cache) {
1855  cache = genAlloca(builder, loc, bufSize, builder.getIndexType());
1856  });
1857  }
1858

1860  (lvl >= 1 && lvlFullyResolved(tid, lvl - 1))) {
1861  // First level or previous level has been full resolved.
1862  genResolvedSliceBegin(builder, loc, tid, lvl);
1863  } else {
1864  // The previous level has not been full resolved.
1865  genUnResolvedSliceBegin(builder, loc, tid, lvl);
1866  }
1867  return false;
1868 }
