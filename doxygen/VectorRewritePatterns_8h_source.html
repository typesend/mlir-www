40  std::function<std::optional<SmallVector<int64_t>>(Operation *op)>;
41  /// Function that returns the shape of the vector to unroll to for a given
42  /// operation. The unrolling is aborted if the function returns
43  /// `std::nullopt`.

46  nativeShape = std::move(fn);
47  return *this;

50  /// Set the native shape to use for unrolling.
51  UnrollVectorOptions &setNativeShape(ArrayRef<int64_t> shape) {
52  SmallVector<int64_t> tsShape(shape.begin(), shape.end());
53  nativeShape = [=](Operation *) -> std::optional<SmallVector<int64_t>> {
54  return tsShape;

56  return *this;
57  }

59  /// Function that returns the traversal order (in terms of "for loop order",

66  UnrollVectorOptions &

68  traversalOrderCallback = std::move(traversalOrderFn);
69  return *this;
70  }

103 /// Populate `patterns` with the following patterns.
104 ///
105 /// - VectorTransferFullPartialRewriter
106 ///
107 /// Split a vector.transfer operation into an in-bounds (i.e., no out-of-bounds
108 /// masking) fast path and a slow path.
109 ///
110 /// Example (a 2-D vector.transfer_read):
111 /// ```
112 /// %1 = vector.transfer_read %0[...], %pad : memref<A...>, vector<...>
113 /// ```
114 /// is transformed into:
115 /// ```
116 /// %1:3 = scf.if (%inBounds) {
117 /// // fast path, direct cast
118 /// memref.cast %A: memref<A...> to compatibleMemRefType
119 /// scf.yield %view : compatibleMemRefType, index, index
120 /// } else {
121 /// // slow path, not in-bounds vector.transfer or linalg.copy.
122 /// memref.cast %alloc: memref<B...> to compatibleMemRefType
123 /// scf.yield %4 : compatibleMemRefType, index, index
124 // }
125 /// %0 = vector.transfer_read %1#0[%1#1, %1#2] {in_bounds = [true ... true]}
126 /// ```
127 /// where `alloc` is a top of the function alloca'ed buffer of one vector.
128 ///
129 /// Preconditions:
130 /// 1. `xferOp.permutation_map()` must be a minor identity map
131 /// 2. the rank of the `xferOp.memref()` and the rank of the `xferOp.vector()`
132 /// must be equal. This will be relaxed in the future but requires
133 /// rank-reducing subviews.
134 void populateVectorTransferFullPartialPatterns(
135  RewritePatternSet &patterns, const VectorTransformsOptions &options);

148 /// Populate `patterns` with the following patterns.
149 ///
150 /// [DecomposeDifferentRankInsertStridedSlice]
151 /// ==========================================
152 /// RewritePattern for InsertStridedSliceOp where source and destination vectors
153 /// have different ranks.
154 ///
155 /// When ranks are different, InsertStridedSlice needs to extract a properly
156 /// ranked vector from the destination vector into which to insert. This pattern
157 /// only takes care of this extraction part and forwards the rest to
158 /// [VectorInsertStridedSliceOpSameRankRewritePattern].
159 ///
160 /// For a k-D source and n-D destination vector (k < n), we emit:
161 /// 1. ExtractOp to extract the (unique) (n-1)-D subvector into which to
162 /// insert the k-D source.
163 /// 2. k-D -> (n-1)-D InsertStridedSlice op
164 /// 3. InsertOp that is the reverse of 1.
165 ///
166 /// [DecomposeNDExtractStridedSlice]
167 /// ================================
168 /// For such cases, we can rewrite it to ExtractOp/ExtractElementOp + lower
169 /// rank ExtractStridedSliceOp + InsertOp/InsertElementOp for the n-D case.
170 void populateVectorInsertExtractStridedSliceDecompositionPatterns(
171  RewritePatternSet &patterns, PatternBenefit benefit = 1);

184 /// Populate `patterns` with a pattern to break down 1-D vector.bitcast ops
185 /// based on the destination vector shape. Bitcasts from a lower bitwidth
186 /// element type to a higher bitwidth one are extracted from the lower bitwidth
187 /// based on the native destination vector shape and inserted based on the ratio
188 /// of the bitwidths.
189 ///
190 /// This acts as a last resort way to break down vector.bitcast ops to smaller

192 /// single element of the higher bitwidth, the is an optional control function.
193 /// If `controlFn` is not nullptr, the pattern will only apply to ops where

200 /// Populate `patterns` with the following patterns.
201 ///
202 /// Patterns in populateVectorInsertExtractStridedSliceDecompositionPatterns();

204 /// [ConvertSameRankInsertStridedSliceIntoShuffle]
205 /// ==============================================

211 /// 2. InsertStridedSlice (k-1)-D into (n-1)-D

213 /// 3. InsertOp that is the reverse of 1.

221 /// Collect a set of pattern to unroll vector operations to a smaller shapes.
222 /// `options` structure controls which operations are unrolled and the target
223 /// shape.
224 /// `op` is unrolled to the `targetShape` as follows, for each of its operands:
225 /// 1. the unrolled type `unrolledVectorType` and number of unrolled instances
226 /// `numUnrolledInstances` are computed from the `targetShape`. For now it is
227 /// assumed the unrolling factors divide the vector sizes.
228 /// 2. ExtractStridedSlice are created to break-up the vector operands.
229 /// 3. the original op is cloned `numUnrolledInstances` times, once for each
230 /// result.
231 /// 4. InsertStridedSlice are inserted to re-assemble the slices into the
232 /// original vectore shape.
233 ///
234 /// Example:
235 ///
236 /// opA(operand0, operand1) // numUnrolledInstances = 3
237 ///
238 /// operand0 operand1
239 /// | |
240 /// fork fork
241 /// <----------gather all fork ops --------->
242 /// /|\ /|\
243 /// f00 f01 f02 f10 f11 f12
244 /// <---------- clone op 3 times --------->
245 /// opA0(f00, f10), opA1(f01, f11), opA2(f02, f12)
246 /// \ | /
247 /// <-------------------- join ------------------------->
248 ///
249 /// Other local patterns then kick in iteratively (including DCE) and compose
250 /// to combine the ExtractStridedSlice/InsertStridedSlice.
251 void populateVectorUnrollPatterns(RewritePatternSet &patterns,
252  const UnrollVectorOptions &options,
253  PatternBenefit benefit = 1);
