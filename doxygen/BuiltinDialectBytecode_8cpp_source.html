31 // Returns the bitwidth if known, else return 0.
32 static unsigned getIntegerBitWidth(DialectBytecodeReader &reader, Type type) {
33  if (auto intType = dyn_cast<IntegerType>(type)) {
34  return intType.getWidth();
35  } else if (type.isa<IndexType>()) {
36  return IndexType::kInternalStorageBitWidth;
37  }
38  reader.emitError()
39  << "expected integer or index type for IntegerAttr, but got: " << type;
40  return 0;
41 }

43 static LogicalResult readAPIntWithKnownWidth(DialectBytecodeReader &reader,
44  Type type, FailureOr<APInt> &val) {
45  unsigned bitWidth = getIntegerBitWidth(reader, type);
46  if (bitWidth == 0)
47  return failure();
48  val = reader.readAPIntWithKnownWidth(bitWidth);
