23 struct DepthwiseConv2DIsMul : public OpRewritePattern<tosa::DepthwiseConv2DOp> {
24  explicit DepthwiseConv2DIsMul(MLIRContext *context)
25  : OpRewritePattern(context) {}
26 
27  LogicalResult matchAndRewrite(tosa::DepthwiseConv2DOp op,
28  PatternRewriter &rewriter) const override {
29  Value input = op.getInput();
30  Value weight = op.getWeight();
31  ShapedType inputType = input.getType().cast<ShapedType>();
32  ShapedType weightType = weight.getType().cast<ShapedType>();
33  ShapedType resultType = op.getOutput().getType().cast<ShapedType>();
34 
35  if (!(inputType.hasStaticShape() && weightType.hasStaticShape() &&
36  resultType.hasStaticShape())) {
37  return failure();
38  }
39 
40  if (!llvm::all_of(op.getStride(), [](int64_t v) { return v == 1; }))
41  return failure();
42 
43  // Only works for a 1x1 kernel.
44  ArrayRef<int64_t> weightShape = weightType.getShape();
45  if (weightShape[0] != 1 || weightShape[1] != 1) {
46  return failure();
47  }
48 
49  // Reshape input to [N, H, W, C] -> [N, H, W, C, 1].
50  ArrayRef<int64_t> inputShape = inputType.getShape();
51  llvm::SmallVector<int64_t, 2> revisedInputShape{
52  inputShape[0], inputShape[1], inputShape[2], inputShape[3], 1};
53  inputType = RankedTensorType::get(
54  revisedInputShape,
55  input.getType().dyn_cast<RankedTensorType>().getElementType());
56  input = rewriter
57  .create<tosa::ReshapeOp>(
58  op.getLoc(), inputType, input,
59  rewriter.getDenseI64ArrayAttr(revisedInputShape))
60  .getResult();
61 
62  if (inputType.getElementType() != resultType.getElementType()) {
63  inputType = inputType.clone(resultType.getElementType());
64  input = rewriter.create<tosa::CastOp>(op.getLoc(), inputType, input);
65  }
66 
67  if (weightType.getElementType() != resultType.getElementType()) {
68  weightType = weightType.clone(resultType.getElementType());
69  weight = rewriter.create<tosa::CastOp>(op.getLoc(), weightType, weight);
70  }
71 
72  if (auto quantizationInfo = op.getQuantizationInfo()) {
73  auto iZp = quantizationInfo->getInputZp();
74  auto wZp = quantizationInfo->getWeightZp();
75 
76  auto applyZp = [&](Value val, int64_t zp) -> Value {
77  if (zp == 0)
78  return val;
79  auto ety = val.getType().cast<ShapedType>().getElementType();
80  auto zpTy = RankedTensorType::get({}, ety);
81  auto zpAttr =
82  DenseElementsAttr::get(zpTy, rewriter.getIntegerAttr(ety, zp));
83  auto zpVal = rewriter.create<tosa::ConstOp>(op.getLoc(), zpTy, zpAttr);
84  return rewriter.create<tosa::SubOp>(op.getLoc(), val.getType(), val,
85  zpVal);
86  };
87 
88  input = applyZp(input, iZp);
89  weight = applyZp(weight, wZp);
90  }
91 
92  ArrayRef<int64_t> padAttr = op.getPad();
93  llvm::SmallVector<int64_t> pad(10, 0);
94  for (const auto &it : llvm::enumerate(padAttr))
95  pad[it.index() + 2] = it.value();
96 
97  if (llvm::any_of(pad, [](int64_t p) { return p != 0; })) {
98  Type inputETy = inputType.getElementType();
99  Attribute zeroAttr = rewriter.getZeroAttr(inputETy);
100 
101  llvm::SmallVector<int64_t> newShape(inputType.getShape());
102  for (int i = 0, s = pad.size(); i < s; ++i) {
103  if (newShape[i / 2] != ShapedType::kDynamic) {
104  newShape[i / 2] += pad[i];
105  }
106  }
107 
108  auto padSizeTy = RankedTensorType::get({5, 2}, rewriter.getI64Type());
109  auto padSize =
110  DenseIntElementsAttr::get(padSizeTy, ArrayRef<int64_t>(pad));
111  Value padSizeVal =
112  rewriter.create<tosa::ConstOp>(op->getLoc(), padSizeTy, padSize);
113 
114  auto padTy = RankedTensorType::get({}, inputETy);
115  auto padAttr = DenseElementsAttr::get(padTy, zeroAttr);
116  Value padVal =
117  rewriter.create<tosa::ConstOp>(op->getLoc(), padTy, padAttr);
118  inputType = RankedTensorType::get(newShape, inputETy);
119  input = rewriter.create<tosa::PadOp>(op->getLoc(), inputType, input,
120  padSizeVal, padVal);
121  }
122 
123  // Perform an elementwise mul over the reshaped input and weight.
124  llvm::SmallVector<int64_t, 2> mulShape{
125  inputType.getDimSize(0), inputType.getDimSize(1),
126  inputType.getDimSize(2), inputType.getDimSize(3), weightShape[3]};
127  auto mulShapeType = RankedTensorType::get(
128  mulShape,
129  weight.getType().dyn_cast<RankedTensorType>().getElementType());
130  Value mulValue = rewriter
131  .create<tosa::MulOp>(op.getLoc(), mulShapeType, input,
132  weight, /*shift=*/0)
133  .getResult();
134 
135  // Reshape output to [N, H, W, C * M].
136  auto outputShape = op.getOutput().getType().cast<ShapedType>().getShape();
137  auto outputShapeType = RankedTensorType::get(
138  outputShape,
139  input.getType().dyn_cast<RankedTensorType>().getElementType());
140  auto outputValue = rewriter.create<tosa::ReshapeOp>(
141  op.getLoc(), outputShapeType, mulValue,
142  rewriter.getDenseI64ArrayAttr(outputShape));
143 
144  // Add in the bias.
145  rewriter
146  .replaceOpWithNewOp<tosa::AddOp>(op, outputShapeType, outputValue,
147  op.getBias())
148  .getResult();
149  return success();
150  }
151 };
