51 // Helper method to find zero/uninitialized allocation.
52 static bool isAlloc(OpOperand *op, bool isZero) {
53  Value val = op->get();
54  // Check allocation, with zero alloc when required.
55  if (auto alloc = val.getDefiningOp<AllocTensorOp>()) {
56  Value copy = alloc.getCopy();
57  if (isZero)
58  return copy && isZeroValue(copy);
59  return !copy;

61  // Last resort for zero alloc: the whole value is zero.
62  return isZero && isZeroValue(val);

65 // Helper to detect sampling operation.
66 static bool isSampling(GenericOp op) {
67  auto yieldOp = cast<linalg::YieldOp>(op.getRegion().front().getTerminator());
68  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {
69  if (isa<arith::MulFOp>(def) || isa<arith::MulIOp>(def)) {
70  // Both scalar input arguments used exactly once.
71  Value s1 = op.getBlock()->getArgument(0);
72  Value s2 = op.getBlock()->getArgument(1);
73  return (def->getOperand(0) == s1 && def->getOperand(1) == s2) ||
74  (def->getOperand(1) == s1 && def->getOperand(0) == s2);
75  }
76  }
77  return false;
78 }

80 // Helper to detect chain of multiplications that do not involve x.

92 // Helper to detect x = x + <multiplications>.
93 static bool isSumOfMul(GenericOp op) {
94  auto yieldOp = cast<linalg::YieldOp>(op.getRegion().front().getTerminator());
95  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {

153 static LogicalResult genForeachOnSparseConstant(ForeachOp op,
154  RewriterBase &rewriter,
155  SparseElementsAttr attr) {
156  auto loc = op.getLoc();
157  SmallVector<Value> reduc = op.getInitArgs();
158 
159  // Foreach on constant.
160  foreachInSparseConstant(
161  rewriter, loc, attr, op.getOrder().value_or(AffineMap()),
162  [&reduc, &rewriter, op](ArrayRef<Value> cvs, Value v) mutable {
163  SmallVector<Value> args;
164  args.append(cvs.begin(), cvs.end());
165  args.push_back(v);
166  args.append(reduc);
167  // Clones the foreach op to get a copy of the loop body.
168  auto cloned = cast<ForeachOp>(rewriter.clone(*op.getOperation()));
169  assert(args.size() == cloned.getBody()->getNumArguments());
170  Operation *yield = cloned.getBody()->getTerminator();
171  rewriter.inlineBlockBefore(cloned.getBody(), op, args);
172  // clean up
173  rewriter.eraseOp(cloned);
174  reduc = yield->getOperands();
175  rewriter.eraseOp(yield);
176  });
177 
178  rewriter.replaceOp(op, reduc);
179  return success();
180 }

182 /// Populates the given sizes array for concatenation from types (for static
183 /// sizes) and from the source tensors (for dynamic sizes).
184 static void concatSizesFromInputs(OpBuilder &builder,
185  SmallVectorImpl<Value> &sizes, Location loc,
186  ShapedType dstTp, ValueRange srcs,
187  unsigned dim) {
188  auto dstShape = dstTp.getShape();
189  sizesFromSrc(builder, sizes, loc, srcs[0]);
190 
191  // Sum up on the `dim` if the dimension is dynamic.
192  if (dstShape[dim] != ShapedType::kDynamic) {
193  // Faithfully take the static size.
194  sizes[dim] = constantIndex(builder, loc, dstShape[dim]);
195  } else {
196  // Else, compute the shape dynamically.
197  for (const auto &src : srcs.drop_front()) {
198  Value srcSz = linalg::createOrFoldDimOp(builder, loc, src, dim);
199  // Sum up all the sizes.
200  sizes[dim] = builder.create<arith::AddIOp>(loc, sizes[dim], srcSz);
201  }
202  }
203 }

211 /// Rewriting rule that converts direct yield of zero with initial allocation.
212 struct FoldInvariantYield : public OpRewritePattern<GenericOp> {
213 public:
214  using OpRewritePattern<GenericOp>::OpRewritePattern;
215 
216  LogicalResult matchAndRewrite(GenericOp op,
217  PatternRewriter &rewriter) const override {
218  if (!op.hasTensorSemantics() || op.getNumResults() != 1 ||
219  !isAlloc(op.getDpsInitOperand(0), /*isZero=*/false) ||
220  !isZeroYield(op) || !op.getDpsInitOperand(0)->get().hasOneUse())
221  return failure();
222  auto outputType = getRankedTensorType(op.getResult(0));
223  // Yielding zero on newly allocated (all-zero) sparse tensors can be
224  // optimized out directly (regardless of dynamic or static size).
225  if (getSparseTensorEncoding(outputType)) {
226  rewriter.replaceOp(op, op.getDpsInitOperand(0)->get());
227  return success();
228  }
229  // Incorporate zero value into allocation copy.
230  if (!outputType.hasStaticShape())
231  return failure();
232  Value zero = constantZero(rewriter, op.getLoc(), op.getResult(0).getType());
233  AllocTensorOp a =
234  op.getDpsInitOperand(0)->get().getDefiningOp<AllocTensorOp>();
235  rewriter.updateRootInPlace(a, [&]() { a.getCopyMutable().assign(zero); });
236  rewriter.replaceOp(op, op.getDpsInitOperand(0)->get());
237  return success();
238  }
239 };

241 /// Rewriting rule that converts two kernels:
242 ///
243 /// T(i,j) = SUM(k, A(i,j,k) * B(i,j,k) * ... )
244 /// X(i,j) = S(i,j) * T(i,j)
245 ///
246 /// into a single kernel, using distributive law:
247 ///
248 /// X(i,j) = SUM(k, S(i,j) * A(i,j,k) * B(i,j,k) * ... )
249 ///
250 /// This kind of fusion (merging two ops into one but using arithmetic
251 /// equalities that may not hold for floating-point computations) would
252 /// be undesirable in the dense case, since we distribute the multiplication
253 /// into the reduction loop. However, for sparse sampling tensor S, such
254 /// a fusion may actually reduce the asymptotic complexity of the kernel,
255 /// since intermediate results may be nullified.
256 struct FuseSparseMultiplyOverAdd : public OpRewritePattern<GenericOp> {
257 public:
258  using OpRewritePattern<GenericOp>::OpRewritePattern;
259 
260  LogicalResult matchAndRewrite(GenericOp op,
261  PatternRewriter &rewriter) const override {
262  // Check consumer.
263  if (!op.hasTensorSemantics() || op.getNumDpsInputs() != 2 ||
264  op.getNumResults() != 1 ||
265  op.getNumParallelLoops() != op.getNumLoops() ||
266  !op.getMatchingIndexingMap(op.getDpsInitOperand(0)).isIdentity() ||
267  !op.getMatchingIndexingMap(op.getDpsInputOperand(0)).isIdentity() ||
268  !op.getMatchingIndexingMap(op.getDpsInputOperand(1)).isIdentity())
269  return failure();
270  // Find consuming OP2(sparse, other) or OP2(other, sparse). The other
271  // operand can be sparse or dense, since the point of this rewriting rule
272  // is detecting a situation in which *more* sparsity is introduced into
273  // a computation, be it already sparse or still dense.
274  unsigned other = 0;
275  if (isSparseTensor(op.getDpsInputOperand(0)))
276  other = 1;
277  else if (!isSparseTensor(op.getDpsInputOperand(1)))
278  return failure();
279  // Check producer.
280  auto prod = dyn_cast_or_null<GenericOp>(
281  op.getDpsInputOperand(other)->get().getDefiningOp());
282  if (!prod || !prod.hasTensorSemantics() || prod.getNumResults() != 1 ||
283  !prod.getResult(0).hasOneUse())
284  return failure();
285  // Sampling consumer and sum of multiplication chain producer.
286  if (!isAlloc(op.getDpsInitOperand(0), /*isZero=*/false) ||
287  !isAlloc(prod.getDpsInitOperand(0), /*isZero=*/true) ||
288  !isSampling(op) || !isSumOfMul(prod))
289  return failure();
290  // Modify operand structure of producer and consumer.
291  Location loc = prod.getLoc();
292  SmallVector<Value> inputOps = prod.getInputs();
293  SmallVector<Value> outputOps = op.getOutputs();
294  SmallVector<AffineMap> fusedIndexMaps = prod.getIndexingMapsArray();
295  inputOps.push_back(op.getDpsInputOperand(1 - other)->get());
296  fusedIndexMaps.push_back(fusedIndexMaps.back()); // mimic other
297  // Fuse producer and consumer into a new generic op.
298  auto fusedOp = rewriter.create<GenericOp>(
299  loc, op.getResult(0).getType(), inputOps, outputOps,
300  rewriter.getAffineMapArrayAttr(fusedIndexMaps), prod.getIteratorTypes(),
301  /*doc=*/nullptr, /*library_call=*/nullptr);
302  Block &prodBlock = prod.getRegion().front();
303  Block &consBlock = op.getRegion().front();
304  IRMapping mapper;
305  Block *fusedBlock = new Block();
306  fusedOp.getRegion().push_back(fusedBlock);
307  unsigned num = prodBlock.getNumArguments();
308  for (unsigned i = 0; i < num - 1; i++)
309  addArg(mapper, fusedBlock, prodBlock.getArgument(i));
310  addArg(mapper, fusedBlock, consBlock.getArgument(1 - other));
311  addArg(mapper, fusedBlock, prodBlock.getArgument(num - 1));
312  // Clone bodies of the producer and consumer in new evaluation order.
313  auto *acc = prodBlock.getTerminator()->getOperand(0).getDefiningOp();
314  auto *sampler = consBlock.getTerminator()->getOperand(0).getDefiningOp();
315  rewriter.setInsertionPointToStart(fusedBlock);
316  Value last;
317  for (auto &op : prodBlock.without_terminator())
318  if (&op != acc) {
319  last = op.getResult(0);
320  rewriter.clone(op, mapper);
321  }
322  mapper.map(consBlock.getArgument(other), fusedBlock->back().getResult(0));
323  mapper.map(last, rewriter.clone(*sampler, mapper)->getResult(0));
324  last = rewriter.clone(*acc, mapper)->getResult(0);
325  rewriter.create<linalg::YieldOp>(loc, last);
326  // Force initial value on merged allocation for dense outputs.
327  if (!getSparseTensorEncoding(op.getResult(0).getType())) {
328  Value init = prod.getDpsInitOperand(0)
329  ->get()
330  .getDefiningOp<AllocTensorOp>()
331  .getCopy();
332  AllocTensorOp a =
333  op.getDpsInitOperand(0)->get().getDefiningOp<AllocTensorOp>();
334  rewriter.updateRootInPlace(a, [&]() { a.getCopyMutable().assign(init); });
335  }
336  // Replace consumer with fused operation. Old producer
337  // and consumer ops will be removed by DCE.
338  rewriter.replaceOp(op, fusedOp->getResults());
339  return success();
340  }
341 
342 private:
343  // Helper to add argument and record the mapping.
344  static void addArg(IRMapping &mapper, Block *b, BlockArgument a) {
345  mapper.map(a, b->addArgument(a.getType(), a.getLoc()));
346  }
347 };

349 // Fuse a tensor cast into producing operation. Note that a tensor.cast
350 // should really not be used to convert between sparse encodings. Since
351 // the pattern currently appears as a result of some prior rewriting
352 // we make an attempt to repair very obvious cases.
353 // TODO: audit the pure tensor dialect rewriting rules
354 struct FuseTensorCast : public OpRewritePattern<tensor::CastOp> {
355 public:
356  using OpRewritePattern<tensor::CastOp>::OpRewritePattern;
357 
358  LogicalResult matchAndRewrite(tensor::CastOp op,
359  PatternRewriter &rewriter) const override {
360  Type srcType = op.getSource().getType();
361  Type dstType = op.getDest().getType();
362  // A nop cast simply folds away.
363  if (srcType == dstType) {
364  rewriter.replaceOp(op, op->getResults());
365  return success();
366  }
367  // See if a sparsity changing cast can be fused into producer.
368  if (tensor::isSameTypeWithoutEncoding(srcType, dstType)) {
369  if (Operation *def = op.getSource().getDefiningOp()) {
370  if (def->hasOneUse() && isa<tensor::ExtractSliceOp>(def)) {
371  def->getResult(0).setType(op->getResultTypes()[0]);
372  rewriter.replaceOp(op, def->getResult(0));
373  return success();
374  }
375  }
376  }
377  // Repair tensor casts with at least one sparse operand into the
378  // the properly supported sparse_tensor.convert.
379  if (getSparseTensorEncoding(srcType) || getSparseTensorEncoding(dstType)) {
380  rewriter.replaceOpWithNewOp<ConvertOp>(op, dstType, op.getSource());
381  return success();
382  }
383  // Fail otherwise.
384  return failure();
385  }

388 /// Sparse rewriting rule for sparse-to-sparse reshape operator.
389 template <typename ReshapeOp>
390 struct Sparse2SparseReshapeRewriter : public OpRewritePattern<ReshapeOp> {
391 public:
392  using OpRewritePattern<ReshapeOp>::OpRewritePattern;
393 
394  LogicalResult matchAndRewrite(ReshapeOp op,
395  PatternRewriter &rewriter) const override {
396  Location loc = op.getLoc();
397  Value srcTensor = op.getSrc();
398  const auto srcTp = getSparseTensorType(srcTensor);
399  const auto dstTp = getSparseTensorType(op.getResult());
400  if (!srcTp.hasEncoding() || !dstTp.hasEncoding())
401  return failure();
402 
403  // Generate code to represent the static dimension constants or compute
404  // the dynamic dimension values.
405  SmallVector<Value> srcSizes;
406  sizesForTensor(rewriter, srcSizes, loc, srcTp, srcTensor);
407  SmallVector<Value> dstSizes;
408  SmallVector<Value> dstDynSizes;
409  if (dstTp.hasStaticDimShape()) {
410  for (Dimension d : dstTp.getDimShape())
411  dstSizes.push_back(constantIndex(rewriter, loc, d));
412  } else {
413  ArrayRef<DynSize> dstShape = dstTp.getDimShape();
414  genReshapeDstShape(rewriter, loc, dstSizes, srcSizes, dstShape,
415  op.getReassociationIndices());
416  for (auto [idx, shape] : llvm::enumerate(dstShape)) {
417  if (shape == ShapedType::kDynamic)
418  dstDynSizes.push_back(dstSizes[idx]);
419  }
420  }
421  Value nnz = rewriter.create<NumberOfEntriesOp>(loc, srcTensor);
422  // Only need a unordered COO buffer if input and output are not sorted
423  // in the same way.
424  Type bufferTp =
425  srcTp.isAllOrdered() && srcTp.isIdentity() && dstTp.isIdentity()
426  ? dstTp.getRankedTensorType()
427  : getUnorderedCOOFromType(dstTp);
428 
429  Value buffer =
430  rewriter
431  .create<AllocTensorOp>(loc, bufferTp, dstDynSizes, Value(),
432  /*sizeHint=*/nnz, Attribute())
433  .getResult();
434 
435  // Implement the sparse2sparse reshape as follows:
436  // foreach srcCoords %srcTensor
437  // insert reshapeCvs(srcCoords), %buffer
438  //
439  // followed by an optional
440  // %t = sparse_tensor.cast %tmp
441  // depending on whether the input/output are sorted in the same way.
442  const auto encSrc = srcTp.getEncoding();
443  ForeachOp foreachOp = rewriter.create<ForeachOp>(
444  loc, srcTensor, buffer,
445  [&](OpBuilder &builder, Location loc, ValueRange srcLcvs, Value v,
446  ValueRange reduc) {
447  const Dimension dimRank = srcTp.getDimRank();
448  SmallVector<Value> srcDcvs;
449  srcDcvs.reserve(dimRank);
450  for (Dimension d = 0; d < dimRank; d++) {
451  // FIXME: `toStoredDim` is deprecated
452  Level lvl = toStoredDim(encSrc, d);
453  srcDcvs.push_back(srcLcvs[lvl]);
454  }
455  SmallVector<Value> dstDcvs;
456  reshapeCvs(builder, loc, op.getReassociationIndices(), srcSizes,
457  srcDcvs, dstSizes, dstDcvs);
458  auto t = builder.create<InsertOp>(loc, v, reduc.front(), dstDcvs);
459  builder.create<sparse_tensor::YieldOp>(loc, t);
460  });
461 
462  Value t = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
463  if (bufferTp != dstTp) {
464  auto dstRTT = dstTp.getRankedTensorType();
465  Value converted = rewriter.create<ConvertOp>(loc, dstRTT, t).getResult();
466  rewriter.create<DeallocTensorOp>(loc, t);
467  t = converted;
468  }
469  rewriter.replaceOp(op, t);
470  return success();
471  }

474 /// Sparse rewriting rule for sparse-to-dense and dense-to-sparse reshape
475 /// operator.
476 template <typename ReshapeOp>
477 struct ReshapeRewriter : public OpRewritePattern<ReshapeOp> {
478 public:
479  using OpRewritePattern<ReshapeOp>::OpRewritePattern;
480 
481  LogicalResult matchAndRewrite(ReshapeOp op,
482  PatternRewriter &rewriter) const override {
483  Location loc = op->getLoc();
484  auto encDst = getSparseTensorEncoding(op.getResult().getType());
485  auto encSrc = getSparseTensorEncoding(op.getSrc().getType());
486  // Since a pure dense expansion is very cheap (change of view), for
487  // a sparse2dense or dense2sparse, we can simply unfuse a sparse
488  // conversion from the reshape operation itself.
489  // All other cases are handled elsewhere.
490  if (encDst && encSrc) {
491  return failure();
492  }
493  if (encSrc) {
494  auto rtp = getRankedTensorType(op.getSrc());
495  auto denseTp =
496  RankedTensorType::get(rtp.getShape(), rtp.getElementType());
497  auto convert = rewriter.create<ConvertOp>(loc, denseTp, op.getSrc());
498  rewriter.updateRootInPlace(op, [&]() { op->setOperand(0, convert); });
499  return success();
500  }
501  if (encDst) {
502  auto rtp = getRankedTensorType(op.getResult());
503  auto denseTp =
504  RankedTensorType::get(rtp.getShape(), rtp.getElementType());
505  auto reshape = rewriter.create<ReshapeOp>(loc, denseTp, op.getSrc(),
506  op.getReassociation());
507  Value convert = rewriter.create<ConvertOp>(loc, rtp, reshape);
508  rewriter.replaceOp(op, convert);
509  return success();
510  }
511  return failure();
512  }

515 struct ConcatenateRewriter : public OpRewritePattern<ConcatenateOp> {
516  using OpRewritePattern::OpRewritePattern;
517  LogicalResult matchAndRewrite(ConcatenateOp op,
518  PatternRewriter &rewriter) const override {
519  const Location loc = op.getLoc();
520  const auto dstTp = getSparseTensorType(op);
521  const Dimension dimRank = dstTp.getDimRank();
522  const Dimension conDim = op.getDimension();
523  SmallVector<Value> sizes;
524  concatSizesFromInputs(rewriter, sizes, loc, dstTp, op.getInputs(), conDim);
525 
526  // %t = concatenate %s1, %s2, %s3 {dim = 1}
527  // ==>
528  // if (isSparseDst)
529  // if (allDense)
530  // %tmp = bufferization.alloc_tensor dstTp
531  // else
532  // %tmp = bufferization.alloc_tensor : unordered COO
533  // else
534  // %tmp = memref.alloc : dense tensor
535  // foreach in %s1 : insert d0, d1, %tmp
536  // foreach in %s2 : insert d0, d1 + size(s1), %tmp
537  // foreach in %s3 : insert d0, d1 + size(s1) + size(s2), %tmp
538  // %t = convert_to_dest_tensor(%tmp)
539  //
540  // NOTE: this cannot be `const` because it will be changed when
541  // `needTmpCOO`, but that's buried in the conditional below and
542  // thus not easily extracted.
543  auto encDst = dstTp.getEncoding();
544  Value dst; // Destination tensor for inserting source tensor values.
545  bool needTmpCOO = true;
546  const bool allDense = dstTp.hasEncoding() && dstTp.isAllDense();
547  Value annotatedDenseDst;
548  if (dstTp.hasEncoding()) {
549  bool allOrdered = false;
550  // When concatenating on dimension 0, and all inputs are sorted and have
551  // an identity dimOrdering, the concatenate will generate coords in
552  // lexOrder thus no need for the tmp COO buffer.
553  // TODO: When conDim != 0, as long as conDim is the first dimension
554  // in all input/output buffers, and all input/output buffers have the same
555  // dimOrdering, the tmp COO buffer is still unnecessary (e.g, concatenate
556  // CSC matrices along column).
557  if (!allDense && conDim == 0 && dstTp.isIdentity()) {
558  for (auto i : op.getInputs()) {
559  const auto stt = getSparseTensorType(i);
560  allOrdered = stt.isAllOrdered() && stt.isIdentity();
561  if (!allOrdered)
562  break;
563  }
564  }
565 
566  needTmpCOO = !allDense && !allOrdered;
567  const RankedTensorType tp = needTmpCOO ? getUnorderedCOOFromType(dstTp)
568  : dstTp.getRankedTensorType();
569  encDst = needTmpCOO ? getSparseTensorEncoding(tp) : encDst;
570  SmallVector<Value> dynSizes;
571  getDynamicSizes(dstTp, sizes, dynSizes);
572  dst = rewriter.create<AllocTensorOp>(loc, tp, dynSizes).getResult();
573  if (allDense) {
574  // Create a view of the values buffer to match the unannotated dense
575  // tensor.
576  Value valuesBuffer = genToValues(rewriter, loc, dst);
577  Value dimCoords =
578  genAlloca(rewriter, loc, dimRank, rewriter.getIndexType(),
579  /*staticShape=*/true);
580  annotatedDenseDst = dst;
581  dst = reshapeValuesToLevels(rewriter, loc, encDst, sizes, valuesBuffer,
582  dimCoords);
583  }
584  } else {
585  // TODO: Dense buffers should be allocated/deallocated via the callback
586  // in BufferizationOptions.
587  dst = allocDenseTensor(rewriter, loc, dstTp, sizes);
588  }
589 
590  Value offset = constantIndex(rewriter, loc, 0);
591  SmallVector<Value> initArgs;
592  if (encDst && !allDense)
593  initArgs.push_back(dst);
594  ForeachOp foreachOp;
595  for (Value input : op.getInputs()) {
596  // Build a for op for each input tensor to append new values into the
597  // output tensor.
598  foreachOp = rewriter.create<ForeachOp>(
599  loc, input, initArgs,
600  [&](OpBuilder &builder, Location loc, ValueRange dcvs, Value v,
601  ValueRange reduc) {
602  SmallVector<Value> dstLcvs(dstTp.getLvlRank());
603  for (Dimension d = 0; d < dimRank; d++) {
604  Value crd = dcvs[d];
605  if (d == conDim)
606  // Transform coordinates for the concatenating dim.
607  crd = builder.create<arith::AddIOp>(loc, crd, offset);
608  // FIXME: `toStoredDim` is deprecated
609  dstLcvs[toStoredDim(encDst, d)] = crd;
610  }
611  if (encDst && !allDense) {
612  Value cond = genIsNonzero(rewriter, loc, v);
613  scf::IfOp ifOp = builder.create<scf::IfOp>(
614  loc, TypeRange(reduc.front().getType()), cond, /*else*/ true);
615  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
616  Value t =
617  builder.create<InsertOp>(loc, v, reduc.front(), dstLcvs);
618  rewriter.create<scf::YieldOp>(loc, t);
619  rewriter.setInsertionPointToStart(&ifOp.getElseRegion().front());
620  rewriter.create<scf::YieldOp>(loc, reduc.front());
621  rewriter.setInsertionPointAfter(ifOp);
622  rewriter.create<sparse_tensor::YieldOp>(loc, ifOp.getResult(0));
623  } else {
624  builder.create<memref::StoreOp>(loc, v, dst, dstLcvs);
625  builder.create<sparse_tensor::YieldOp>(loc);
626  }
627  });
628  // Accumulates the offset. Note that only static-shaped inputs are allowed
629  // by concatenate op verifier, which saves us from computing the offset
630  // dynamically.
631  const auto sh = getSparseTensorType(input).getStaticDimSize(conDim);
632  assert(sh.has_value());
633  offset = rewriter.create<arith::AddIOp>(
634  loc, offset, constantIndex(rewriter, loc, *sh));
635  if (encDst && !allDense) {
636  dst = foreachOp.getResult(0);
637  initArgs[0] = dst;
638  }
639  }
640 
641  // Temp variable to avoid needing to call `getRankedTensorType`
642  // in the three use-sites below.
643  const RankedTensorType dstRTT = dstTp;
644  if (!encDst) {
645  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, dstRTT, dst);
646  } else if (allDense) {
647  rewriter.replaceOp(
648  op, rewriter.create<ConvertOp>(loc, dstRTT, annotatedDenseDst)
649  .getResult());
650  } else {
651  dst = rewriter.create<LoadOp>(loc, dst, true);
652  if (needTmpCOO) {
653  Value tmpCoo = dst;
654  dst = rewriter.create<ConvertOp>(loc, dstRTT, tmpCoo).getResult();
655  rewriter.create<DeallocTensorOp>(loc, tmpCoo);
656  }
657  rewriter.replaceOp(op, dst);
658  }
659  return success();
660  }

664 struct ConvertRewriter : public OpRewritePattern<ConvertOp> {
665  using OpRewritePattern::OpRewritePattern;
666  LogicalResult matchAndRewrite(ConvertOp op,
667  PatternRewriter &rewriter) const override {
668  auto encDst = getSparseTensorEncoding(op.getType());
669  auto encSrc = getSparseTensorEncoding(op.getSource().getType());
670  if (encDst && encSrc && !encSrc.isSlice() &&
671  encSrc.withoutBitWidths() == encDst.withoutBitWidths()) {
672  // Trivial tensor conversion and simple element type conversion is handled
673  // in codegen.
674  return failure();
675  }
676  // TODO: Add a cast before generating InsertOp.
677  assert(op.getSource().getType().getElementType() ==
678  op.getDest().getType().getElementType());
679  if (encSrc && encDst)
680  return sparse2SparseRewrite(op, rewriter);
681  if (encSrc && !encDst)
682  return sparse2DenseRewrite(op, rewriter);
683  if (!encSrc && encDst)
684  return dense2SparseRewrite(op, rewriter);

686  // Dense-to-dense convert is a nop and handled by canonicalization.
687  return failure();

690 private:
691  // Handles sparse constant to sparse tensor or dense tensor to sparse tensor
692  // conversion as follows:
693  // t = new sparse COO tensor
694  // fill t using src
695  // dst = convert t
696  //
697  // To fill the COO tensor from a dense tensor:
698  // for i1 in dim1
699  // ..
700  // for ik in dimk
701  // val = a[i1,..,ik]
702  // if val != 0
703  // t->add(val, [i1,..,ik], [p1,..,pk])
704  //
705  // To fill the COO tensor from a sparse constant in COO format:
706  // for i in range(NNZ)
707  // val = values[i]
708  // [i1,..,ik] = coordinates[i]
709  // t->add(val, [i1,..,ik], [p1,..,pk])
710  LogicalResult dense2SparseRewrite(ConvertOp op,
711  PatternRewriter &rewriter) const {
712  Location loc = op.getLoc();
713  Value src = op.getSource();
714  const auto dstTp = getSparseTensorType(op);
715  SmallVector<Value> sizes;
716  sizesFromSrc(rewriter, sizes, loc, src);
717  SmallVector<Value> dynSizes;
718  getDynamicSizes(dstTp, sizes, dynSizes);
719 
720  bool fromSparseConst = false;
721  if (auto constOp = op.getSource().getDefiningOp<arith::ConstantOp>()) {
722  if (constOp.getValue().dyn_cast<SparseElementsAttr>()) {
723  fromSparseConst = true;
724  }
725  }
726 
727  const auto encDst = dstTp.getEncoding();
728  // We don't need a temporary COO tensor if the destination has an identity
729  // ordering. Otherwise, we use the destination ordering for the temporary
730  // COO tensor.
731  // TODO: enhance foreachOp to take ordering to remove the need of a
732  // temporary COO tensor here.
733  const RankedTensorType bufferTp = dstTp.isIdentity() || fromSparseConst
734  ? dstTp.getRankedTensorType()
735  : getUnorderedCOOFromTypeWithOrdering(
736  dstTp, dstTp.getDimToLvlMap());
737  // Only imposes foreach order on dense constant (which will be statically
738  // sorted by the sparse compiler), otherwise the rotated loop sequence
739  // results to bad cache locality.
740  AffineMapAttr foreachOrder = nullptr;
741  if (encDst.getDimOrdering() && fromSparseConst)
742  foreachOrder = AffineMapAttr::get(encDst.getDimOrdering());
743 
744  auto buffer =
745  rewriter.create<AllocTensorOp>(loc, bufferTp, dynSizes).getResult();
746  auto foreachOp = rewriter.create<ForeachOp>(
747  loc, src, buffer, foreachOrder,
748  [&](OpBuilder &builder, Location loc, ValueRange dcvs, Value v,
749  ValueRange reduc) {
750  Value input = reduc.front();
751  const Dimension dimRank = dstTp.getDimRank();
752  const Level lvlRank = dstTp.getLvlRank();
753  SmallVector<Value> lcvs(lvlRank);
754  for (Dimension d = 0; d < dimRank; d++)
755  // FIXME: `toStoredDim` is deprecated
756  lcvs[toStoredDim(encDst, d)] = dcvs[d];
757  if (fromSparseConst) {
758  input = builder.create<InsertOp>(loc, v, input, lcvs);
759  } else {
760  Value cond = genIsNonzero(builder, loc, v);
761  auto ifOp = builder.create<scf::IfOp>(
762  loc, TypeRange(input.getType()), cond, /*else*/ true);
763  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
764  Value insert = builder.create<InsertOp>(loc, v, input, lcvs);
765  builder.create<scf::YieldOp>(loc, insert);
766  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
767  builder.create<scf::YieldOp>(loc, input);
768  builder.setInsertionPointAfter(ifOp);
769  input = ifOp.getResult(0);
770  }
771  builder.create<sparse_tensor::YieldOp>(loc, input);
772  });
773  rewriter.setInsertionPointAfter(op);
774  src = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
775  if (bufferTp != dstTp) {
776  rewriter.replaceOpWithNewOp<ConvertOp>(op, dstTp.getRankedTensorType(),
777  src);
778  rewriter.create<DeallocTensorOp>(loc, src);
779  } else {
780  rewriter.replaceOp(op, src);
781  }
782 
783  return success();
784  }

786  // Handles sparse tensor to dense tensor conversion as follows:
787  // dst = new dense tensor;
788  // foreach elemment in src
789  // dst[element.coords] = element.value
790  LogicalResult sparse2DenseRewrite(ConvertOp op,
791  PatternRewriter &rewriter) const {
792  Location loc = op->getLoc();
793  RankedTensorType dstTp = getRankedTensorType(op);
794  Value src = op.getSource();
795  RankedTensorType srcTp = getRankedTensorType(src);
796 
797  SmallVector<Value> sizes;
798  sizesForTensor(rewriter, sizes, loc, srcTp, src);
799 
800  Value dst = allocDenseTensor(rewriter, loc, dstTp, sizes);
801  Block *insertionBlock = rewriter.getInsertionBlock();
802  bool noEscape = bufferization::allocationDoesNotEscape(op->getOpResult(0));
803 
804  rewriter.create<ForeachOp>(loc, src, std::nullopt,
805  [&](OpBuilder &builder, Location loc,
806  ValueRange args, Value v, ValueRange reduc) {
807  builder.create<memref::StoreOp>(loc, v, dst,
808  args);
809  builder.create<sparse_tensor::YieldOp>(loc);
810  });
811 
812  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, dstTp, dst);
813 
814  // Deallocate the buffer.
815  if (noEscape) {
816  rewriter.setInsertionPoint(insertionBlock->getTerminator());
817  deallocDenseTensor(rewriter, loc, dst);
818  }
819  return success();
820  }

822  // Handles sparse tensor to sparse tensor conversion as follows:
823  // if src is not COO
824  // construct a COO to represent the src
825  // sort the src COO
826  // foreach elemment in the sorted src COO
827  // insert element to dst
828  LogicalResult sparse2SparseRewrite(ConvertOp op,
829  PatternRewriter &rewriter) const {
830  const Location loc = op->getLoc();
831  // These two variables cannot be `const` because they're conditionally
832  // changed below. Ideally we'd use `SparseTensorType` for `srcRTT`;
833  // however that class's copy-ctor is implicitly deleted.
834  Value src = op.getSource();
835  auto srcRTT = getRankedTensorType(src);
836  const auto dstTp = getSparseTensorType(op);
837  const auto encDst = dstTp.getEncoding();
838  const Level dstLvlRank = dstTp.getLvlRank();
839  const Dimension dimRank = dstTp.getDimRank();
840  // This assertion should be guaranteed by validity of the op,
841  // but just for paranoia's sake.
842  assert(static_cast<Dimension>(srcRTT.getRank()) == dimRank);
843 
844  SmallVector<Value> srcSizes;
845  sizesForTensor(rewriter, srcSizes, loc, srcRTT, src);
846  Value tmpCoo = Value();
847  Value nnz = rewriter.create<NumberOfEntriesOp>(loc, src);
848  // We need a tmp COO buffer if and only if
849  // 1. the src tensor is not a COO and
850  // 2. the src tensor is not ordered in the same way as the target
851  // tensor (e.g., src tensor is not ordered or src tensor haves a different
852  // dimOrdering).
853  if (const SparseTensorType srcTp(srcRTT);
854  !(srcTp.isAllOrdered() && srcTp.hasSameDimToLvlMap(dstTp))) {
855  // Construct a COO tensor from the src tensor.
856  // TODO: there may be cases for which more efficiently without
857  // going through an intermediate COO, such as cases that only change
858  // the overhead types.
859  SmallVector<Value> dynSrcSizes;
860  getDynamicSizes(srcRTT, srcSizes, dynSrcSizes);
861  srcRTT =
862  getUnorderedCOOFromTypeWithOrdering(srcRTT, dstTp.getDimToLvlMap());
863  // Ensure that mutating `srcRTT` didn't invalidate `dimRank`.
864  assert(static_cast<Dimension>(srcRTT.getRank()) == dimRank);
865  tmpCoo = rewriter
866  .create<AllocTensorOp>(loc, srcRTT, dynSrcSizes, Value(),
867  /*sizeHint=*/nnz, Attribute())
868  .getResult();
869  auto foreachOp = rewriter.create<ForeachOp>(
870  loc, src, tmpCoo,
871  [&](OpBuilder &builder, Location loc, ValueRange dcvs, Value v,
872  ValueRange reduc) {
873  SmallVector<Value> dstLcvs(dstLvlRank);
874  for (Dimension d = 0; d < dimRank; d++) {
875  // FIXME: `toStoredDim` is deprecated
876  Level l = toStoredDim(encDst, d);
877  dstLcvs[l] = dcvs[d];
878  }
879  auto t = builder.create<InsertOp>(loc, v, reduc.front(), dstLcvs);
880  builder.create<sparse_tensor::YieldOp>(loc, t);
881  });
882  src = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
883  }
884 
885  // Now that the conditional is done, we can use `SparseTensorType`.
886  const SparseTensorType srcTp(srcRTT);
887 
888  // Only need to sort if the srcTp is not already sorted (we faithfully take
889  // the guarantee from the sparse tensor encoding).
890  if (!srcTp.isAllOrdered()) {
891  // Retrieve the values-array.
892  Value y = genToValues(rewriter, loc, src);
893  const auto encSrc = srcTp.getEncoding();
894  // Sort the COO tensor so that its elements are ordered via increasing
895  // coordinates for the storage ordering of the dst tensor. Use SortCoo
896  // if the COO tensor has the same ordering as the dst tensor.
897  if (dimRank > 1 && srcTp.hasSameDimToLvlMap(dstTp)) {
898  MemRefType coordsTp =
899  get1DMemRefType(encSrc.getCrdType(), /*withLayout=*/false);
900  Value xs = rewriter.create<ToCoordinatesBufferOp>(loc, coordsTp, src);
901  rewriter.create<SortCooOp>(
902  loc, nnz, xs, ValueRange{y}, rewriter.getIndexAttr(dimRank),
903  rewriter.getIndexAttr(0), SparseTensorSortKind::HybridQuickSort);
904  } else {
905  // Gather the coordinates-arrays in the dst tensor storage order.
906  SmallVector<Value> xs(dstLvlRank);
907  const Level srcLvlRank = srcTp.getLvlRank();
908  for (Level srcLvl = 0; srcLvl < srcLvlRank; srcLvl++) {
909  // FIXME: `toOrigDim` is deprecated
910  Dimension dim = toOrigDim(encSrc, srcLvl);
911  // FIXME: `toStoredDim` is deprecated
912  Level dstLvl = toStoredDim(encDst, dim);
913  xs[dstLvl] =
914  genToCoordinates(rewriter, loc, src, srcLvl, /*cooStart=*/0);
915  }
916  rewriter.create<SortOp>(loc, nnz, xs, ValueRange{y},
917  SparseTensorSortKind::HybridQuickSort);
918  }
919  }
920 
921  // For each element in the COO tensor, insert the element to the dst tensor.
922  SmallVector<Value> dynDstSizes;
923  getDynamicSizes(dstTp, srcSizes, dynDstSizes);
924  Value dst = rewriter
925  .create<AllocTensorOp>(loc, dstTp.getRankedTensorType(),
926  dynDstSizes, Value(),
927  /*sizeHint=*/nnz, Attribute())
928  .getResult();
929  SmallVector<Value> dstLcvs(dstLvlRank);
930  auto foreachOp = rewriter.create<ForeachOp>(
931  loc, src, dst,
932  [&](OpBuilder &builder, Location loc, ValueRange dcvs, Value v,
933  ValueRange reduc) {
934  for (Dimension d = 0; d < dimRank; d++) {
935  // FIXME: `toStoredDim` is deprecated
936  Level l = toStoredDim(encDst, d);
937  dstLcvs[l] = dcvs[d];
938  }
939  auto t = builder.create<InsertOp>(loc, v, reduc.front(), dstLcvs);
940  builder.create<sparse_tensor::YieldOp>(loc, t);
941  });
942 
943  // Release the temporary COO if it is created. Note that tmpCoo is
944  // invalidated due to foreach and updated to src.
945  if (tmpCoo)
946  rewriter.create<DeallocTensorOp>(loc, src);
947 
948  // Directly replace op with dst results in bufferization error message
949  // "sparse tensor allocation should not escape function".
950  // As such, we insert a trivial tensor convert which will be removed by
951  // codegen.
952  rewriter.setInsertionPointAfter(op);
953  auto t = rewriter.create<LoadOp>(loc, foreachOp.getResult(0), true);
954  rewriter.replaceOpWithNewOp<ConvertOp>(op, dstTp.getRankedTensorType(), t);
955  return success();
956  }

959 /// Sparse rewriting rule for the foreach operator.
960 struct ForeachRewriter : public OpRewritePattern<ForeachOp> {
961 public:
962  using OpRewritePattern::OpRewritePattern;
963 
964  LogicalResult matchAndRewrite(ForeachOp op,
965  PatternRewriter &rewriter) const override {
966 
967  auto loc = op.getLoc();
968  Value input = op.getTensor();
969  SmallVector<Value> reduc = op.getInitArgs();
970  const auto stt = getSparseTensorType(input);
971  const Dimension dimRank = stt.getDimRank();
972  const Level lvlRank = stt.getLvlRank();
973 
974  // Special-case: for each over a sparse constant uses its own rewriting
975  // rule.
976  if (auto constOp = input.getDefiningOp<arith::ConstantOp>()) {
977  if (auto attr = constOp.getValue().dyn_cast<SparseElementsAttr>()) {
978  return genForeachOnSparseConstant(op, rewriter, attr);
979  }
980  }
981 
982  // Otherwise, use loop emitter to generate loops.
983  const auto enc = stt.getEncoding();
984 
985  // 1. Generates loop for the sparse input.
986  LoopEmitter loopEmitter(
987  ValueRange{input},
988  StringAttr::get(getContext(), ForeachOp::getOperationName()));
989  loopEmitter.initializeLoopEmit(rewriter, loc);
990  for (Level l = 0; l < lvlRank; l++) {
991  // TODO: provide utility function for loop sequences that only contains
992  // one for loop?
993  const SmallVector<TensorLevel, 1> tidLvls{
994  loopEmitter.makeTensorLevel(0, l)};
995  loopEmitter.enterNewLoopSeq(rewriter, loc, tidLvls);
996  // Note that reduc will be taken care of by loop emitter and get updated
997  // in place.
998  loopEmitter.enterLoopOverTensorAtLvl(rewriter, loc, tidLvls, reduc);
999  }
1000 
1001  SmallVector<Value> lcvs;
1002  lcvs.reserve(lvlRank);
1003  loopEmitter.getLoopIVs(lcvs);
1004 
1005  if (op.getOrder()) {
1006  // FIXME: There is some dim/lvl confusion here since `dimRank != lvlRank`
1007  SmallVector<Value> dcvs = lcvs; // keep a copy
1008  for (Dimension d = 0; d < dimRank; d++) {
1009  auto l = op.getOrder()->getDimPosition(d);
1010  lcvs[l] = dcvs[d];
1011  }
1012  }
1013  Value vals = loopEmitter.getValBuffer()[0];
1014  Value pos = loopEmitter.getPosits()[0].back();
1015  // Loads the value from sparse tensor using position-index;
1016  // loads the value from dense tensor using coords.
1017  Value val = enc ? rewriter.create<memref::LoadOp>(loc, vals, pos)
1018  : rewriter.create<memref::LoadOp>(loc, vals, lcvs);
1019 
1020  // 2. Inline the block in the foreach operator.
1021  Block *srcBlock = op.getBody();
1022 
1023  // Remap coordinates.
1024  SmallVector<Value> args;
1025  for (Dimension d = 0; d < dimRank; d++) {
1026  // FIXME: `toStoredDim` is deprecated
1027  Value dimCrd = lcvs[toStoredDim(enc, d)];
1028  args.push_back(dimCrd);
1029  }
1030  // Remap value.
1031  args.push_back(val);
1032  // Remap reduction variables.
1033  args.append(reduc);
1034 
1035  // Remove sparse_tensor.yield.
1036  SmallVector<Value> reducValue = srcBlock->getTerminator()->getOperands();
1037  rewriter.eraseOp(srcBlock->getTerminator());
1038 
1039  // Inline body.
1040  if (!reducValue.empty()) {
1041  rewriter.mergeBlocks(srcBlock, rewriter.getBlock(), args);
1042  } else {
1043  // This is annoying, since scf.for inserts a implicit yield op when
1044  // there is no reduction variable upon creation, in this case we need to
1045  // merge the block *before* the yield op.
1046  rewriter.inlineBlockBefore(srcBlock, &*rewriter.getInsertionPoint(),
1047  args);
1048  }
1049 
1050  for (Dimension d = 0; d < dimRank; d++) {
1051  // Link the reduction chain. Note that loop emitter update the reducValue
1052  // in place.
1053  loopEmitter.exitCurrentLoop(rewriter, loc, reducValue);
1054  loopEmitter.exitCurrentLoopSeq(rewriter, loc);
1055  }
1056 
1057  // Replace the foreach operator with the value returned by the outtermost
1058  // for loop.
1059  rewriter.replaceOp(op, reducValue);
1060  return success();
1061  }
1062 };

1092 struct OutRewriter : public OpRewritePattern<OutOp> {

1094  LogicalResult matchAndRewrite(OutOp op,

1096  Location loc = op.getLoc();
1097  // Calculate NNZ.
