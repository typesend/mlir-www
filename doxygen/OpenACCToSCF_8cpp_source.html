30 /// Pattern to transform the `getIfCond` on operation without region into a
31 /// scf.if and move the operation into the `then` region.
32 template <typename OpTy>
33 class ExpandIfCondition : public OpRewritePattern<OpTy> {
34  using OpRewritePattern<OpTy>::OpRewritePattern;
35 
36  LogicalResult matchAndRewrite(OpTy op,
37  PatternRewriter &rewriter) const override {
38  // Early exit if there is no condition.
39  if (!op.getIfCond())
40  return failure();
41 
42  IntegerAttr constAttr;
43  if (!matchPattern(op.getIfCond(), m_Constant(&constAttr))) {
44  auto ifOp = rewriter.create<scf::IfOp>(op.getLoc(), TypeRange(),
45  op.getIfCond(), false);
46  rewriter.updateRootInPlace(op, [&]() { op.getIfCondMutable().erase(0); });
47  auto thenBodyBuilder = ifOp.getThenBodyBuilder();
48  thenBodyBuilder.setListener(rewriter.getListener());
49  thenBodyBuilder.clone(*op.getOperation());
50  rewriter.eraseOp(op);
51  } else {
52  if (constAttr.getInt())
53  rewriter.updateRootInPlace(op,
54  [&]() { op.getIfCondMutable().erase(0); });
55  else
56  rewriter.eraseOp(op);
57  }
58  return success();
59  }
60 };
61 } // namespace

76 void ConvertOpenACCToSCFPass::runOnOperation() {
77  auto op = getOperation();
78  auto *context = op.getContext();
79 
80  RewritePatternSet patterns(context);
81  ConversionTarget target(*context);
82  populateOpenACCToSCFConversionPatterns(patterns);
83 
84  target.addLegalDialect<scf::SCFDialect>();
85  target.addLegalDialect<acc::OpenACCDialect>();
86 
87  target.addDynamicallyLegalOp<acc::EnterDataOp>(
88  [](acc::EnterDataOp op) { return !op.getIfCond(); });
89 
90  target.addDynamicallyLegalOp<acc::ExitDataOp>(
91  [](acc::ExitDataOp op) { return !op.getIfCond(); });
92 
93  target.addDynamicallyLegalOp<acc::UpdateOp>(
94  [](acc::UpdateOp op) { return !op.getIfCond(); });
95 
96  if (failed(applyPartialConversion(op, target, std::move(patterns))))
97  signalPassFailure();
98 }
