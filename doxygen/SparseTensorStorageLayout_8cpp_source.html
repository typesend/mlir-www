32 // This is only ever called from `SparseTensorTypeToBufferConverter`,
33 // which is why the first argument is `RankedTensorType` rather than
34 // `SparseTensorType`.
35 static std::optional<LogicalResult>
36 convertSparseTensorType(RankedTensorType rtp, SmallVectorImpl<Type> &fields) {
37  const SparseTensorType stt(rtp);
38  if (!stt.hasEncoding())
39  return std::nullopt;
40 
41  foreachFieldAndTypeInSparseTensor(
42  stt,
43  [&fields](Type fieldType, FieldIndex fieldIdx,
44  SparseTensorFieldKind /*fieldKind*/, Level /*lvl*/,
45  DimLevelType /*dlt*/) -> bool {
46  assert(fieldIdx == fields.size());
47  fields.push_back(fieldType);
48  return true;
49  });
50  return success();
51 }

104 Value sparse_tensor::SparseTensorDescriptor::getCrdMemRefOrView(
105  OpBuilder &builder, Location loc, Level lvl) const {
106  const Level cooStart = getCOOStart(rType.getEncoding());
107  if (lvl < cooStart)
108  return getMemRefField(SparseTensorFieldKind::CrdMemRef, lvl);
109 
110  Value stride = constantIndex(builder, loc, rType.getLvlRank() - cooStart);
111  Value size = getCrdMemSize(builder, loc, cooStart);
112  size = builder.create<arith::DivUIOp>(loc, size, stride);
113  return builder.create<memref::SubViewOp>(
114  loc, getMemRefField(SparseTensorFieldKind::CrdMemRef, cooStart),
115  /*offset=*/ValueRange{constantIndex(builder, loc, lvl - cooStart)},
116  /*size=*/ValueRange{size},
117  /*step=*/ValueRange{stride});
118 }

124 constexpr FieldIndex kDataFieldStartingIdx = 0;
125 
126 void sparse_tensor::foreachFieldInSparseTensor(
127  const SparseTensorEncodingAttr enc,
128  llvm::function_ref<bool(FieldIndex, SparseTensorFieldKind, Level,
129  DimLevelType)>
130  callback) {
131  assert(enc);
132 
133 #define RETURN_ON_FALSE(fidx, kind, dim, dlt) \
134  if (!(callback(fidx, kind, dim, dlt))) \
135  return;
136 
137  const auto lvlTypes = enc.getDimLevelType();
138  const Level lvlRank = enc.getLvlRank();
139  const Level cooStart = getCOOStart(enc);
140  const Level end = cooStart == lvlRank ? cooStart : cooStart + 1;
141  FieldIndex fieldIdx = kDataFieldStartingIdx;
142  // Per-dimension storage.
143  for (Level l = 0; l < end; l++) {
144  // Dimension level types apply in order to the reordered dimension.
145  // As a result, the compound type can be constructed directly in the given
146  // order.
147  const auto dlt = lvlTypes[l];
148  if (isCompressedDLT(dlt) || isCompressedWithHiDLT(dlt)) {
149  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::PosMemRef, l, dlt);
150  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::CrdMemRef, l, dlt);
151  } else if (isSingletonDLT(dlt)) {
152  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::CrdMemRef, l, dlt);
153  } else {
154  assert(isDenseDLT(dlt)); // no fields
155  }
156  }
157  // The values array.
158  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::ValMemRef, -1u,
159  DimLevelType::Undef);
160 
161  // Put metadata at the end.
162  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::StorageSpec, -1u,
163  DimLevelType::Undef);
164 
165 #undef RETURN_ON_FALSE
166 }

168 void sparse_tensor::foreachFieldAndTypeInSparseTensor(
169  SparseTensorType stt,
170  llvm::function_ref<bool(Type, FieldIndex, SparseTensorFieldKind, Level,
171  DimLevelType)>
172  callback) {
173  assert(stt.hasEncoding());
174  // Construct the basic types.
175  const Type crdType = stt.getCrdType();
176  const Type posType = stt.getPosType();
177  const Type eltType = stt.getElementType();
178 
179  const Type metaDataType = StorageSpecifierType::get(stt.getEncoding());
180  // memref<? x pos> positions
181  const Type posMemType = MemRefType::get({ShapedType::kDynamic}, posType);
182  // memref<? x crd> coordinates
183  const Type crdMemType = MemRefType::get({ShapedType::kDynamic}, crdType);
184  // memref<? x eltType> values
185  const Type valMemType = MemRefType::get({ShapedType::kDynamic}, eltType);
186 
187  foreachFieldInSparseTensor(
188  stt.getEncoding(),
189  [metaDataType, posMemType, crdMemType, valMemType,
190  callback](FieldIndex fieldIdx, SparseTensorFieldKind fieldKind,
191  Level lvl, DimLevelType dlt) -> bool {
192  switch (fieldKind) {
193  case SparseTensorFieldKind::StorageSpec:
194  return callback(metaDataType, fieldIdx, fieldKind, lvl, dlt);
195  case SparseTensorFieldKind::PosMemRef:
196  return callback(posMemType, fieldIdx, fieldKind, lvl, dlt);
197  case SparseTensorFieldKind::CrdMemRef:
198  return callback(crdMemType, fieldIdx, fieldKind, lvl, dlt);
199  case SparseTensorFieldKind::ValMemRef:
200  return callback(valMemType, fieldIdx, fieldKind, lvl, dlt);
201  };
202  llvm_unreachable("unrecognized field kind");
203  });
204 }

217 unsigned
218 sparse_tensor::getNumDataFieldsFromEncoding(SparseTensorEncodingAttr enc) {
219  unsigned numFields = 0; // one value memref
220  foreachFieldInSparseTensor(enc,
221  [&numFields](FieldIndex fidx,
222  SparseTensorFieldKind, Level,
223  DimLevelType) -> bool {
224  if (fidx >= kDataFieldStartingIdx)
225  numFields++;
226  return true;
227  });
228  numFields -= 1; // the last field is MetaData field
229  assert(numFields ==
230  getNumFieldsFromEncoding(enc) - kDataFieldStartingIdx - 1);
231  return numFields;
232 }
