24  /// Populate bounds of values/dimensions for iter_args/OpResults.
25  static void populateIterArgBounds(scf::ForOp forOp, Value value,
26  std::optional<int64_t> dim,
27  ValueBoundsConstraintSet &cstr) {
28  // `value` is an iter_arg or an OpResult.
29  int64_t iterArgIdx;
30  if (auto iterArg = value.dyn_cast<BlockArgument>()) {
31  iterArgIdx = iterArg.getArgNumber() - forOp.getNumInductionVars();
32  } else {
33  iterArgIdx = value.cast<OpResult>().getResultNumber();
34  }
35 
36  // An EQ constraint can be added if the yielded value (dimension size)
37  // equals the corresponding block argument (dimension size).
38  assert(forOp.getLoopBody().hasOneBlock() &&
39  "multiple blocks not supported");
40  Value yieldedValue =
41  cast<scf::YieldOp>(forOp.getLoopBody().front().getTerminator())
42  .getOperand(iterArgIdx);
43  Value iterArg = forOp.getRegionIterArg(iterArgIdx);
44  Value initArg = forOp.getInitArgs()[iterArgIdx];
45 
46  auto addEqBound = [&]() {
47  if (dim.has_value()) {
48  cstr.bound(value)[*dim] == cstr.getExpr(initArg, dim);
49  } else {
50  cstr.bound(value) == initArg;
51  }
52  };
53 
54  if (yieldedValue == iterArg) {
55  addEqBound();
56  return;
57  }
58 
59  // Compute EQ bound for yielded value.
60  AffineMap bound;
61  ValueDimList boundOperands;
62  LogicalResult status = ValueBoundsConstraintSet::computeBound(
63  bound, boundOperands, BoundType::EQ, yieldedValue, dim,
64  [&](Value v, std::optional<int64_t> d) {
65  // Stop when reaching a block argument of the loop body.
66  if (auto bbArg = v.dyn_cast<BlockArgument>())
67  return bbArg.getOwner()->getParentOp() == forOp;
68  // Stop when reaching a value that is defined outside of the loop. It
69  // is impossible to reach an iter_arg from there.
70  Operation *op = v.getDefiningOp();
71  return forOp.getLoopBody().findAncestorOpInRegion(*op) == nullptr;
72  });
73  if (failed(status))
74  return;
75  if (bound.getNumResults() != 1)
76  return;
77 
78  // Check if computed bound equals the corresponding iter_arg.
79  Value singleValue = nullptr;
80  std::optional<int64_t> singleDim;
81  if (auto dimExpr = bound.getResult(0).dyn_cast<AffineDimExpr>()) {
82  int64_t idx = dimExpr.getPosition();
83  singleValue = boundOperands[idx].first;
84  singleDim = boundOperands[idx].second;
85  } else if (auto symExpr = bound.getResult(0).dyn_cast<AffineSymbolExpr>()) {
86  int64_t idx = symExpr.getPosition() + bound.getNumDims();
87  singleValue = boundOperands[idx].first;
88  singleDim = boundOperands[idx].second;
89  }
90  if (singleValue == iterArg && singleDim == dim)
91  addEqBound();
92  }

94  void populateBoundsForIndexValue(Operation *op, Value value,
95  ValueBoundsConstraintSet &cstr) const {
96  auto forOp = cast<ForOp>(op);
97 
98  if (value == forOp.getInductionVar()) {
99  // TODO: Take into account step size.
100  cstr.bound(value) >= forOp.getLowerBound();
101  cstr.bound(value) < forOp.getUpperBound();
102  return;
103  }
104 
105  // Handle iter_args and OpResults.
106  populateIterArgBounds(forOp, value, std::nullopt, cstr);
107  }
108

111  auto forOp = cast<ForOp>(op);
112  // Handle iter_args and OpResults.
113  populateIterArgBounds(forOp, value, dim, cstr);
114  }
