132 /// Returns true iff MLIR operand has any sparse operand or result.
133 inline bool hasAnySparseOperandOrResult(Operation *op) {
134  bool anySparseIn = llvm::any_of(op->getOperands().getTypes(), [](Type t) {
135  return getSparseTensorEncoding(t) != nullptr;
136  });
137  bool anySparseOut = llvm::any_of(op->getResults().getTypes(), [](Type t) {
138  return getSparseTensorEncoding(t) != nullptr;
139  });
140  return anySparseIn || anySparseOut;
141 }
