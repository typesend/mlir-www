9 // Typedefs and enums shared between MLIR code for manipulating the
10 // IR, and the lightweight runtime support library for sparse tensor
11 // manipulations. That is, all the enums are used to define the API
12 // of the runtime library and hence are also needed when generating
13 // calls into the runtime library. Moveover, the `DimLevelType` enum
14 // is also used as the internal IR encoding of dimension level types,
15 // to avoid code duplication (e.g., for the predicates).
16 //
17 // This file also defines x-macros <https://en.wikipedia.org/wiki/X_Macro>
18 // so that we can generate variations of the public functions for each
19 // supported primary- and/or overhead-type.
20 //
21 // Because this file defines a library which is a dependency of the
22 // runtime library itself, this file must not depend on any MLIR internals
23 // (e.g., operators, attributes, ArrayRefs, etc) lest the runtime library
24 // inherit those dependencies.
25 //
26 //===----------------------------------------------------------------------===//

161 /// This enum defines all the sparse representations supportable by
162 /// the SparseTensor dialect. We use a lightweight encoding to encode
163 /// both the "format" per se (dense, compressed, singleton) as well as
164 /// the "properties" (ordered, unique). The encoding is chosen for
165 /// performance of the runtime library, and thus may change in future
166 /// versions; consequently, client code should use the predicate functions
167 /// defined below, rather than relying on knowledge about the particular
168 /// binary encoding.
169 ///
170 /// The `Undef` "format" is a special value used internally for cases
171 /// where we need to store an undefined or indeterminate `DimLevelType`.
172 /// It should not be used externally, since it does not indicate an
173 /// actual/representable format.
174 enum class DimLevelType : uint8_t {
175  Undef = 0, // 0b0000_00
176  Dense = 4, // 0b0001_00
177  Compressed = 8, // 0b0010_00
178  CompressedNu = 9, // 0b0010_01
179  CompressedNo = 10, // 0b0010_10
180  CompressedNuNo = 11, // 0b0010_11
181  Singleton = 16, // 0b0100_00
182  SingletonNu = 17, // 0b0100_01
183  SingletonNo = 18, // 0b0100_10
184  SingletonNuNo = 19, // 0b0100_11
185  CompressedWithHi = 32, // 0b1000_00
186  CompressedWithHiNu = 33, // 0b1000_01
187  CompressedWithHiNo = 34, // 0b1000_10
188  CompressedWithHiNuNo = 35, // 0b1000_11
189 };
190 
191 /// This enum defines all the storage formats supported by the sparse compiler,
192 /// without the level properties.
193 enum class LevelFormat : uint8_t {
194  Dense = 4, // 0b0001_00
195  Compressed = 8, // 0b0010_00
196  Singleton = 16, // 0b0100_00
197  CompressedWithHi = 32, // 0b1000_00
198 };

236 /// Check that the `DimLevelType` contains a valid (possibly undefined) value.
237 constexpr bool isValidDLT(DimLevelType dlt) {
238  const uint8_t formatBits = static_cast<uint8_t>(dlt) >> 2;
239  const uint8_t propertyBits = static_cast<uint8_t>(dlt) & 3;
240  // If undefined or dense, then must be unique and ordered.
241  // Otherwise, the format must be one of the known ones.
242  return (formatBits <= 1)
243  ? (propertyBits == 0)
244  : (formatBits == 2 || formatBits == 4 || formatBits == 8);
245 }

247 /// Check if the `DimLevelType` is the special undefined value.
248 constexpr bool isUndefDLT(DimLevelType dlt) {
249  return dlt == DimLevelType::Undef;
250 }

252 /// Check if the `DimLevelType` is dense.
253 constexpr bool isDenseDLT(DimLevelType dlt) {
254  return dlt == DimLevelType::Dense;
255 }
256 
257 // We use the idiom `(dlt & ~3) == format` in order to only return true
258 // for valid DLTs. Whereas the `dlt & format` idiom is a bit faster but
259 // can return false-positives on invalid DLTs.
260 
261 /// Check if the `DimLevelType` is compressed (regardless of properties).
262 constexpr bool isCompressedDLT(DimLevelType dlt) {
263  return (static_cast<uint8_t>(dlt) & ~3) ==
264  static_cast<uint8_t>(DimLevelType::Compressed);
265 }
266 
267 /// Check if the `DimLevelType` is compressed (regardless of properties).
268 constexpr bool isCompressedWithHiDLT(DimLevelType dlt) {
269  return (static_cast<uint8_t>(dlt) & ~3) ==
270  static_cast<uint8_t>(DimLevelType::CompressedWithHi);
271 }
272 
273 /// Check if the `DimLevelType` is singleton (regardless of properties).
274 constexpr bool isSingletonDLT(DimLevelType dlt) {
275  return (static_cast<uint8_t>(dlt) & ~3) ==
276  static_cast<uint8_t>(DimLevelType::Singleton);
277 }

279 /// Check if the `DimLevelType` is ordered (regardless of storage format).
280 constexpr bool isOrderedDLT(DimLevelType dlt) {
281  return !(static_cast<uint8_t>(dlt) & 2);
282 }

284 /// Check if the `DimLevelType` is unique (regardless of storage format).
285 constexpr bool isUniqueDLT(DimLevelType dlt) {
286  return !(static_cast<uint8_t>(dlt) & 1);
287 }

289 /// Convert a DimLevelType to its corresponding LevelFormat.
290 /// Returns std::nullopt when input dlt is Undef.
291 constexpr std::optional<LevelFormat> getLevelFormat(DimLevelType dlt) {
292  if (dlt == DimLevelType::Undef)
293  return std::nullopt;
294  return static_cast<LevelFormat>(static_cast<uint8_t>(dlt) & ~3);
295 }
296 
297 /// Convert a LevelFormat to its corresponding DimLevelType with the given
298 /// properties. Returns std::nullopt when the properties are not applicable for
299 /// the input level format.
300 /// TODO: factor out a new LevelProperties type so we can add new properties
301 /// without changing this function's signature
302 constexpr std::optional<DimLevelType>
303 getDimLevelType(LevelFormat lf, bool ordered, bool unique) {
304  auto dlt = static_cast<DimLevelType>(static_cast<uint8_t>(lf) |
305  (ordered ? 0 : 2) | (unique ? 0 : 1));
306  return isValidDLT(dlt) ? std::optional(dlt) : std::nullopt;
307 }
