43 /// Constructs a new GPU module (for GPU kernels) inside the given top module,
44 /// or returns an existing GPU module if one was built previously.
45 static gpu::GPUModuleOp genGPUModule(OpBuilder &builder, ModuleOp topModule) {
46  for (auto op : topModule.getBodyRegion().getOps<gpu::GPUModuleOp>())
47  return op; // existing
48  markAsGPUContainer(topModule);
49  builder.setInsertionPointToStart(&topModule.getBodyRegion().front());
50  return builder.create<gpu::GPUModuleOp>(topModule->getLoc(),
51  "sparse_kernels");
52 }

54 /// Constructs a new GPU kernel in the given GPU module.
55 static gpu::GPUFuncOp genGPUFunc(OpBuilder &builder, gpu::GPUModuleOp gpuModule,
56  SmallVectorImpl<Value> &args) {
57  // Get a unique kernel name. Not very creative,
58  // but we simply try kernel0, kernel1, etc.
59  unsigned kernelNumber = 0;
60  SmallString<16> kernelName;
61  do {
62  kernelName.clear();
63  ("kernel" + Twine(kernelNumber++)).toStringRef(kernelName);
64  } while (gpuModule.lookupSymbol(kernelName));
65  // Then we insert a new kernel with given arguments into the module.
66  builder.setInsertionPointToStart(&gpuModule.getBodyRegion().front());
67  SmallVector<Type> argsTp;
68  for (unsigned i = 0, e = args.size(); i < e; i++)
69  argsTp.push_back(args[i].getType());
70  FunctionType type = FunctionType::get(gpuModule->getContext(), argsTp, {});
71  auto gpuFunc =
72  builder.create<gpu::GPUFuncOp>(gpuModule->getLoc(), kernelName, type);
73  gpuFunc->setAttr(gpu::GPUDialect::getKernelFuncAttrName(),
74  builder.getUnitAttr());
75  return gpuFunc;
76 }

78 /// Constructs code to launch GPU kernel.
79 static Value genLaunchGPUFunc(OpBuilder &builder, gpu::GPUFuncOp gpuFunc,
80  SmallVectorImpl<Value> &args,
81  SmallVectorImpl<Value> &tokens,
82  unsigned numThreads) {
83  Location loc = gpuFunc->getLoc();
84  Value none = TypedValue<::mlir::IntegerType>{};
85  Value one = constantIndex(builder, loc, 1);
86  Value numT = constantIndex(builder, loc, numThreads);
87  gpu::KernelDim3 gridSize = {one, one, one};
88  gpu::KernelDim3 blckSize = {numT, one, one};
89  return builder
90  .create<gpu::LaunchFuncOp>(loc, gpuFunc, gridSize, blckSize,
91  /*dynSharedMemSz*/ none, args,
92  builder.getType<gpu::AsyncTokenType>(), tokens)
93  .getAsyncToken();
94 }

96 /// Maps the provided ranked host buffer into the device address space.
97 /// Writes from the host are guaranteed to be visible to device kernels
98 /// that are launched afterwards. Writes from the device are guaranteed
99 /// to be visible on the host after synchronizing with the device kernel
100 /// completion. Needs to cast the buffer to a unranked buffer.
101 static Value genHostRegisterMemref(OpBuilder &builder, Location loc,
102  Value mem) {
103  MemRefType memTp = mem.getType().cast<MemRefType>();
104  UnrankedMemRefType resTp =
105  UnrankedMemRefType::get(memTp.getElementType(), /*memorySpace=*/0);
106  Value cast = builder.create<memref::CastOp>(loc, resTp, mem);
107  builder.create<gpu::HostRegisterOp>(loc, cast);
108  return cast;
109 }

111 /// Unmaps the provided buffer, expecting the casted buffer.
112 static void genHostUnregisterMemref(OpBuilder &builder, Location loc,
113  Value cast) {
114  builder.create<gpu::HostUnregisterOp>(loc, cast);
115 }

130 /// Allocates memory on the device.
131 /// TODO: A `host_shared` attribute could be used to indicate that
132 /// the buffer is visible by both host and device, but lowering
133 /// that feature does not seem to be fully supported yet.
134 static gpu::AllocOp genAllocMemRef(OpBuilder &builder, Location loc, Value mem,
135  Value token) {
136  auto tp = mem.getType().cast<ShapedType>();
137  auto elemTp = tp.getElementType();
138  auto shape = tp.getShape();
139  auto memTp = MemRefType::get(shape, elemTp);
140  SmallVector<Value> dynamicSizes;
141  for (unsigned r = 0, rank = tp.getRank(); r < rank; r++) {
142  if (shape[r] == ShapedType::kDynamic) {
143  Value dim = constantIndex(builder, loc, r);
144  Value dimOp = builder.create<memref::DimOp>(loc, mem, dim);
145  dynamicSizes.push_back(dimOp);
146  }
147  }
148  return builder.create<gpu::AllocOp>(loc, TypeRange({memTp, token.getType()}),
149  token, dynamicSizes, ValueRange());
150 }

155  return builder.create<gpu::DeallocOp>(loc, token.getType(), token, mem)

162  return builder.create<gpu::MemcpyOp>(loc, token.getType(), token, dst, src)

166 /// Prepares the outlined arguments, passing scalars and buffers in. Here we
167 /// assume that the first buffer is the one allocated for output. We create
168 /// a set of properly chained asynchronous allocation/copy pairs to increase
169 /// overlap before launching the kernel.
170 /// TODO: the output assumption may be a bit too brittle
171 static Value genParametersIn(OpBuilder &builder, Location loc,
172  SmallVectorImpl<Value> &scalars,
173  SmallVectorImpl<Value> &buffers,
174  SmallVectorImpl<Value> &args,
175  SmallVectorImpl<Value> &tokens,
176  bool useHostRegistrationForOut) {
177  Value out;
178  // Scalars are passed by value.
179  for (Value s : scalars)
180  args.push_back(s);
181  // Buffers are need to be made visible on device.
182  for (Value b : buffers) {
183  if (useHostRegistrationForOut) {
184  out = genHostRegisterMemref(builder, loc, b);
185  args.push_back(b);
186  useHostRegistrationForOut = false;
187  continue;
188  }
189  Value firstToken = genFirstWait(builder, loc);
190  auto alloc = genAllocMemRef(builder, loc, b, firstToken);
191  Value devMem = alloc.getResult(0);
192  Value depToken = alloc.getAsyncToken(); // copy-after-alloc
193  args.push_back(devMem);
194  tokens.push_back(genCopyMemRef(builder, loc, devMem, b, depToken));
195  }
196  return out;
197 }

199 /// Finalizes the outlined arguments. The output buffer is copied depending
200 /// on the kernel token and then deallocated. All other buffers are simply
201 /// deallocated. Then we wait for all operations to complete.
202 static void genParametersOut(OpBuilder &builder, Location loc, Value out,
203  Value kernelToken, SmallVectorImpl<Value> &scalars,
204  SmallVectorImpl<Value> &buffers,
205  SmallVectorImpl<Value> &args,
206  SmallVectorImpl<Value> &tokens) {
207  unsigned base = scalars.size();
208  for (unsigned i = base, e = args.size(); i < e; i++) {
209  Value firstToken;
210  if (i == base) {
211  // Assumed output parameter: unregister or copy-out.
212  if (out) {
213  genHostUnregisterMemref(builder, loc, out);
214  out = Value();
215  continue;
216  }
217  firstToken =
218  genCopyMemRef(builder, loc, buffers[0], args[i], kernelToken);
219  } else {
220  firstToken = genFirstWait(builder, loc);
221  }
222  tokens.push_back(genDeallocMemRef(builder, loc, args[i], firstToken));
223  }
224 }

226 /// Constructs code for new GPU kernel.
227 static void genGPUCode(PatternRewriter &rewriter, gpu::GPUFuncOp gpuFunc,
228  scf::ParallelOp forallOp,
229  SmallVectorImpl<Value> &constants,
230  SmallVectorImpl<Value> &scalars,
231  SmallVectorImpl<Value> &buffers) {
232  Location loc = gpuFunc->getLoc();
233  Block &block = gpuFunc.getBody().front();
234  rewriter.setInsertionPointToStart(&block);
235 
236  // Re-generate the constants, recapture all arguments.
237  unsigned arg = 0;
238  IRMapping irMap;
239  for (Value c : constants)
240  irMap.map(c, rewriter.clone(*c.getDefiningOp())->getResult(0));
241  for (Value s : scalars)
242  irMap.map(s, block.getArgument(arg++));
243  for (Value b : buffers)
244  irMap.map(b, block.getArgument(arg++));
245 
246  // Assume 1-dimensional grid/block configuration (only x dimension),
247  // so that:
248  // row = blockIdx.x * blockDim.x + threadIdx.x
249  // inc = blockDim.x * gridDim.x
250  Value bid = rewriter.create<gpu::BlockIdOp>(loc, gpu::Dimension::x);
251  Value bsz = rewriter.create<gpu::BlockDimOp>(loc, gpu::Dimension::x);
252  Value tid = rewriter.create<gpu::ThreadIdOp>(loc, gpu::Dimension::x);
253  Value gsz = rewriter.create<gpu::GridDimOp>(loc, gpu::Dimension::x);
254  Value mul = rewriter.create<arith::MulIOp>(loc, bid, bsz);
255  Value row = rewriter.create<arith::AddIOp>(loc, mul, tid);
256  Value inc = rewriter.create<arith::MulIOp>(loc, bsz, gsz);
257 
258  // Construct the iteration over the computational space that
259  // accounts for the fact that the total number of threads and
260  // the amount of work to be done usually do not match precisely.
261  // for (r = row; r < N; r += inc) {
262  // <loop-body>
263  // }
264  Value upper = irMap.lookup(forallOp.getUpperBound()[0]);
265  scf::ForOp forOp = rewriter.create<scf::ForOp>(loc, row, upper, inc);
266  rewriter.cloneRegionBefore(forallOp.getLoopBody(), forOp.getLoopBody(),
267  forOp.getLoopBody().begin(), irMap);
268 
269  // Done.
270  rewriter.setInsertionPointAfter(forOp);
271  rewriter.create<gpu::ReturnOp>(gpuFunc->getLoc());
272 }

278 /// Proof-of-concept rewriter. This rule generates a CUDA implementation
279 /// for each outermost forall loop generated by the sparse compiler.
280 /// TODO: right works with parallelization-strategy=dense-outer-loop
281 /// but give this its own flags in the future
282 struct ForallRewriter : public OpRewritePattern<scf::ParallelOp> {
283  using OpRewritePattern<scf::ParallelOp>::OpRewritePattern;
284 
285  ForallRewriter(MLIRContext *context, unsigned nT)
286  : OpRewritePattern(context), numThreads(nT){};
287 
288  LogicalResult matchAndRewrite(scf::ParallelOp forallOp,
289  PatternRewriter &rewriter) const override {
290  // Reject inadmissible loop form.
291  // Essentially only accept a loop, generated by the sparse compiler,
292  // of the form
293  // forall (i = 0; i < N; i++)
294  // so that cyclic scheduling over the threads is easy.
295  if (!forallOp->hasAttr(LoopEmitter::getLoopEmitterLoopAttrName()) ||
296  forallOp.getNumReductions() != 0 || forallOp.getNumLoops() != 1 ||
297  !matchPattern(forallOp.getLowerBound()[0], m_Zero()) ||
298  !matchPattern(forallOp.getStep()[0], m_One()))
299  return failure();
300  // Collect every value that is computed outside the parallel loop.
301  SetVector<Value> invariants; // stable iteration!
302  forallOp->walk([&](Operation *op) {
303  // Collect all values of admissible ops.
304  for (OpOperand &o : op->getOpOperands()) {
305  Value val = o.get();
306  Block *block;
307  if (auto arg = val.dyn_cast<BlockArgument>())
308  block = arg.getOwner();
309  else
310  block = val.getDefiningOp()->getBlock();
311  if (!isNestedIn(block, forallOp))
312  invariants.insert(val);
313  }
314  });
315  // Outline the outside values as proper parameters. Fail when sharing
316  // value between host and device is not straightforward.
317  SmallVector<Value> constants;
318  SmallVector<Value> scalars;
319  SmallVector<Value> buffers;
320  for (Value val : invariants) {
321  Type tp = val.getType();
322  if (val.getDefiningOp<arith::ConstantOp>())
323  constants.push_back(val);
324  else if (tp.isa<FloatType>() || tp.isIntOrIndex())
325  scalars.push_back(val);
326  else if (isa<MemRefType>(tp))
327  buffers.push_back(val);
328  else
329  return failure(); // don't know how to share
330  }
331  // Pass outlined non-constant values.
332  // TODO: Experiment with `useHostRegistrationForOut` to see if we want to
333  // keep the feature at all (either through a heuristic or compiler
334  // option for gpu codegen).
335  Location loc = forallOp->getLoc();
336  SmallVector<Value> args;
337  SmallVector<Value> tokens;
338  Value out = genParametersIn(rewriter, loc, scalars, buffers, args, tokens,
339  /*useHostRegistrationForOut=*/false);
340  // Set up GPU module and construct GPU function.
341  auto saveIp = rewriter.saveInsertionPoint();
342  ModuleOp topModule = forallOp->getParentOfType<ModuleOp>();
343  auto gpuModule = genGPUModule(rewriter, topModule);
344  auto gpuFunc = genGPUFunc(rewriter, gpuModule, args);
345  genGPUCode(rewriter, gpuFunc, forallOp, constants, scalars, buffers);
346  // Generate code that launches the kernel asynchronously, blocking on all
347  // opens tokens and yielding a new token for the output.
348  // TODO: Passing in tokens to launch up does not seem to be properly lowered
349  // by cubin yet, hence the current blocking wait.
350  rewriter.restoreInsertionPoint(saveIp);
351  genBlockingWait(rewriter, loc, tokens);
352  tokens.clear();
353  Value kernelToken =
354  genLaunchGPUFunc(rewriter, gpuFunc, args, tokens, numThreads);
355  // Finalize the outlined arguments.
356  genParametersOut(rewriter, loc, out, kernelToken, scalars, buffers, args,
357  tokens);
358  genBlockingWait(rewriter, loc, tokens);
359  rewriter.eraseOp(forallOp);
360  return success();
361  }
362 
363 private:
364  // Helper method to see if block appears in given loop.
365  static bool isNestedIn(Block *block, scf::ParallelOp forallOp) {
366  for (Operation *o = block->getParentOp(); o; o = o->getParentOp()) {
367  if (o == forallOp)
368  return true;
369  }
370  return false;
371  }
372 
373  unsigned numThreads;
374 };
