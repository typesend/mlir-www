135 /// Determines if affine expression is invariant.
136 static bool isInvariantAffine(AffineExpr a, ArrayRef<LoopId> loopStack,
137  LoopId ldx, bool &isAtLoop) {
138  switch (a.getKind()) {
139  case AffineExprKind::DimId: {
140  const LoopId i = a.cast<AffineDimExpr>().getPosition();
141  if (i == ldx) {
142  isAtLoop = true;
143  // Must be invariant if we are at the given loop.
144  return true;
145  }
146  bool isInvariant = false;
147  for (LoopId l : loopStack) {
148  isInvariant = (l == i);
149  if (isInvariant)
150  break;
151  }
152  return isInvariant;
153  }
154  case AffineExprKind::Add:
155  case AffineExprKind::Mul: {
156  auto binOp = a.cast<AffineBinaryOpExpr>();
157  return isInvariantAffine(binOp.getLHS(), loopStack, ldx, isAtLoop) &&
158  isInvariantAffine(binOp.getRHS(), loopStack, ldx, isAtLoop);
159  }
160  default: {
161  assert(a.isa<AffineConstantExpr>());
162  return true;
163  }
164  }
165 }

173 /// Helper method to construct a permuted dimension ordering
174 /// that adheres to the given topological sort.
175 //
176 // FIXME: does the above actually mean "dimensions", or should it say
177 // "level ordering"? The same dim/lvl confusion applies to all the code
178 // and comments in the definition below.
179 static AffineMap permute(CodegenEnv &env, AffineMap m) {
180  assert(m.getNumDims() + env.merger().getNumFilterLoops() ==
181  env.topSortSize() &&
182  "size mismatch");
183  // Construct the inverse of `m`; to avoid the asymptotic complexity
184  // of calling `m.getPermutedPosition` repeatedly.
185  //
186  // The variable `perm` must use `unsigned` rather than `Dimension`/`Level`,
187  // because that's what `AffineMap::getPermutationMap` requires.
188  // TODO: however, `perm` should be renamed to make clear what exactly
189  // it's storing a permutation of.
190  SmallVector<unsigned> perm;
191  const unsigned numResults = m.getNumResults();
192  BitVector worklist(numResults, true);
193  LoopOrd loopDepth = 1;
194 
195  // Construct the permutation.
196  while (worklist.any() && loopDepth <= env.topSortSize()) {
197  const unsigned preSize = perm.size();
198  for (unsigned dim : worklist.set_bits()) {
199  bool isAtLoop = false;
200  if (m.getResult(dim).isa<AffineConstantExpr>() ||
201  (isInvariantAffine(m.getResult(dim), env.getLoopStackUpTo(loopDepth),
202  env.topSortAt(loopDepth - 1), isAtLoop) &&
203  isAtLoop)) {
204  // If the matching affine is constant expression or just become
205  // invariant. We can visit the dimension now without breaking the
206  // topSort constraint.
207  perm.push_back(dim);
208  }
209  }
210 
211  // Removes resolved dimension.
212  for (unsigned i = preSize, e = perm.size(); i < e; i++)
213  worklist.reset(perm[i]);
214 
215  // Try entering the next loop in the stack.
216  loopDepth++;
217  }
218 
219  assert(perm.size() == numResults);
220  return AffineMap::getPermutationMap(perm, env.op().getContext());
221 }

223 /// Helper method to inspect affine expressions. Rejects cases where the
224 /// same index is used more than once. Also rejects compound affine
225 /// expressions in sparse dimensions.
226 /// filterIdx stores the current filter loop idx should be used for the next
227 /// compound affine sparse level, and it will be incremented by one when
228 /// used.
229 static bool findAffine(Merger &merger, TensorId tid, Level lvl, AffineExpr a,
230  DimLevelType dlt, LoopId &filterLdx,
231  bool setLvlFormat = true) {
232  switch (a.getKind()) {
233  case AffineExprKind::DimId: {
234  const LoopId idx = merger.makeLoopId(a.cast<AffineDimExpr>().getPosition());
235  if (!isUndefDLT(merger.getDimLevelType(tid, idx)))
236  return false; // used more than once
237 
238  if (setLvlFormat)
239  merger.setLevelAndType(tid, idx, lvl, dlt);
240  return true;
241  }
242  case AffineExprKind::Add:
243  case AffineExprKind::Mul:
244  case AffineExprKind::Constant: {
245  if (!isDenseDLT(dlt) && setLvlFormat) {
246  assert(isUndefDLT(merger.getDimLevelType(tid, filterLdx)));
247  // Use a filter loop for sparse affine expression.
248  merger.setLevelAndType(tid, filterLdx, lvl, dlt);
249  ++filterLdx;
250  }
251 
252  if (auto binOp = a.dyn_cast<AffineBinaryOpExpr>()) {
253  // We do not set dim level format for affine expresssion like d0 + d1 on
254  // either loop index at d0 or d1.
255  // We continue the recursion merely to check whether current affine is
256  // admissible or not.
257  return findAffine(merger, tid, lvl, binOp.getLHS(), dlt, filterLdx,
258  false) &&
259  findAffine(merger, tid, lvl, binOp.getRHS(), dlt, filterLdx,
260  false);
261  }
262  // Falls through when it is a constant Affine
263  return true;
264  }
265  default:
266  return false;
267  }
268 }

270 /// Helper method to inspect affine expressions for index variable reduction
271 /// based codegen. It finds the dependent index set for all tensor levels in the
272 /// current expression we are generating.
273 ///
274 /// For example, when handling A[i+j][j+k], we build the two way mapping in
275 /// merger between (tensor, level) pairs and their dependent index variable set:
276 /// A_0 <=> [i, j] and A_1 <=> [j, k]
277 ///
278 /// It rejects cases (returns false)
279 /// 1st, when the same index is used more than once, e.g., A[i+j][i]
280 /// 2nd, when multiplication is used in the non-trivial index expression.
281 /// 3rd, when a constant operand is used in the non-trivial index expression.
282 ///
283 /// TODO: constant should be easy to handle.
284 static bool findDepIdxSet(Merger &merger, TensorId tensor, Level lvl,
285  AffineExpr a, DimLevelType dlt,
286  bool isSubExp = false) {
287  switch (a.getKind()) {
288  case AffineExprKind::DimId: {
289  const LoopId ldx = merger.makeLoopId(a.cast<AffineDimExpr>().getPosition());
290  if (!isUndefDLT(merger.getDimLevelType(tensor, ldx)))
291  return false; // used more than once, e.g., A[i][i]
292 
293  // TODO: Generalizes the following two cases. A[i] (with trivial index
294  // expression) can be treated as a special affine index expression. We do
295  // not necessarily need to differentiate them.
296  if (!isSubExp)
297  merger.setLevelAndType(tensor, ldx, lvl, dlt);
298 
299  if (isSubExp) {
300  // The current loops appears in more than one affine expressions on the
301  // same tensor. We can not handle this case. e.g., A[i+j][i+k], `i` is
302  // used twice.
303  if (merger.hasDependentLvl(ldx, tensor)) {
304  // TODO: This can be supported by coiterate slices if the loop idx is
305  // appeared on affine index for different tensor, or take slice on
306  // mulitple dimensions when it is on the same tensor.
307  // E.g.,
308  // `d0 + d1` for indexing t0[lvl0] and `d0 + d2` for indexing t1[lvl0]
309  // d0_1 = getNextSliceOffset t0 along lvl0
310  // d0_2 = getNextSliceOffset t1 along lvl0
311  // if d0_1 == d0_2 then d0 = d0_1 = d0_1
312  // else increase min(d0_1, d0_2).
313  return false;
314  }
315  merger.setLoopDependentTensorLevel(ldx, tensor, lvl, dlt);
316  }
317  return true;
318  }
319  case AffineExprKind::Constant:
320  case AffineExprKind::Mul:
321  // TODO: Support Mul and Constant AffineExp for slice-based codegen
322  return false;
323  case AffineExprKind::Add: {
324  auto binOp = a.cast<AffineBinaryOpExpr>();
325  return findDepIdxSet(merger, tensor, lvl, binOp.getLHS(), dlt, true) &&
326  findDepIdxSet(merger, tensor, lvl, binOp.getRHS(), dlt, true);
327  }
328  default:
329  return false;
330  }
331 }

333 /// Get the total number of compound affine expressions in the
334 /// `getMatchingIndexingMap` for the given tensor. For the following inputs:
335 ///
336 /// map = (d0, d1, d2) => (d0 + d1, d2)
337 /// lvlTypes = ["compressed", "compressed"]
338 ///
339 /// Returns 1 (because the first level is compressed and its corresponding
340 /// indexing-expression is `d0 + d1`)
341 static unsigned getNumNonTrivialIdxExpOnSparseLvls(AffineMap map,
342  Value tensor) {
343  // The `tensor` is not guaranted to have `RankedTensorType`, therefore
344  // we can't use `getRankedTensorType`/`getSparseTensorType` here.
345  // However, we don't need to handle `StorageSpecifierType`, so we
346  // can use `SparseTensorType` once we guard against non-tensors.
347  const auto rtp = tensor.getType().dyn_cast<RankedTensorType>();
348  if (!rtp)
349  return 0;
350  const SparseTensorType stt(rtp);
351 
352  // FIXME: There's some dim/lvl confusion here. The previous version of
353  // the code asserted that there are `lvlRank`-many expressions, but then
354  // the `exprs[d]` expression assumes there are in fact `dimRank`-many
355  // expressions. Even though `ArrayRef::operator[]` will check for OOB,
356  // the mismatch between the assertion and the usage belies that this code
357  // cannot support non-permutations.
358  //
359  // Elsewhere in this file the maps returned by
360  // `linalg::GenericOp::getMatchingIndexingMap` are inconsistent about
361  // whether they're expected to have `lvlRank`-many or `dimRank`-many
362  // expressions (cf., `genSubscript` vs `findSparseAnnotations`);
363  // so those are no help in determining which is actually intended.
364  //
365  // For now we work around this problem by asserting the two ranks agree.
366  const Dimension dimRank = stt.getDimRank();
367  const Level lvlRank = stt.getLvlRank();
368  assert(dimRank == lvlRank && "Non-permutations not currently supported");
369  const auto exprs = map.getResults();
370  assert(static_cast<Dimension>(exprs.size()) == dimRank &&
371  "AffineMap does not have dimension-rank many results");
372  (void)dimRank;
373  unsigned num = 0;
374  for (Level l = 0; l < lvlRank; l++) {
375  // FIXME: `toOrigDim` is deprecated.
376  const Dimension d = toOrigDim(stt.getEncoding(), l);
377  if (!exprs[d].isa<AffineDimExpr>() && !stt.isDenseLvl(l))
378  num++;
379  }
380  return num;
381 }

401 /// Helper method to inspect sparse encodings in the tensor types.
402 /// Fills the per-dimension sparsity information for all tensors.
403 /// Returns true if the sparse annotations and affine subscript
404 /// expressions of all tensors are admissible. Returns false if
405 /// no annotations are found or inadmissible constructs occur.
406 /// We currently support two different ways to handle non-trivial index
407 /// expression on sparse tensors, and they accept different affine expressions.
408 /// When using filter-loop-based approach, it accept (almost) arbitrary affine
409 /// index expression on sparse tensor but it is much less efficient, and will be
410 /// gradually removed from the codebase.
411 /// When using dependent index reducton-based approach, it currently only
412 /// supports affine addition index expression.
413 static bool findSparseAnnotations(CodegenEnv &env, bool idxReducBased) {
414  bool annotated = false;
415  // `filterLdx` may be mutated by `findAffine`.
416  LoopId filterLdx = env.merger().getStartingFilterLoopId();
417  for (OpOperand &t : env.op()->getOpOperands()) {
418  const TensorId tid = env.makeTensorId(t.getOperandNumber());
419  const auto map = env.op().getMatchingIndexingMap(&t);
420  const auto enc = getSparseTensorEncoding(t.get().getType());
421  if (enc)
422  annotated = true;
423 
424  const Level lvlRank = map.getNumResults();
425  assert(!enc || lvlRank == enc.getLvlRank());
426  assert(static_cast<Level>(env.op().getRank(&t)) == lvlRank);
427 
428  // We only need to do index reduction if there is at least one non-trivial
429  // index expression on sparse levels.
430  // If all non-trivial index expression is on dense levels, we can
431  // efficiently rely on the random access to locate the element.
432  bool needIdxReduc =
433  enc && getNumNonTrivialIdxExpOnSparseLvls(map, t.get()) != 0;
434  // If then current tensor being inspected requires affine index, it need
435  // to be sliced.
436  for (Level l = 0; l < lvlRank; l++) {
437  // FIXME: `toOrigDim` is deprecated.
438  const AffineExpr a = map.getResult(toOrigDim(enc, l));
439  const DimLevelType dlt = enc.getLvlType(l);
440  if (idxReducBased && needIdxReduc) {
441  if (!findDepIdxSet(env.merger(), tid, l, a, dlt))
442  return false; // inadmissible affine expression
443  } else {
444  if (!findAffine(env.merger(), tid, l, a, dlt, filterLdx))
445  return false; // inadmissible affine expression
446  }
447  }
448  }
449  assert(filterLdx == env.merger().getNumLoops());
450  return annotated;
451 }

453 /// A helper to compute a topological sort. O(n^2) time complexity
454 /// as we use adj matrix for the graph.
455 /// The sorted result will put the first Reduction iterator to the
456 /// latest possible `LoopOrd`.
457 ///
458 /// The `inDegree` is indexed by `LoopId`, and the `adjM` is indexed by
459 /// `(LoopId,LoopId)`.
460 static bool topSortOptimal(CodegenEnv &env,
461  ArrayRef<utils::IteratorType> iteratorTypes,
462  std::vector<unsigned> &inDegree,
463  std::vector<std::vector<bool>> &adjM) {
464  std::vector<LoopId> redIt; // reduce iterator with 0 degree
465  std::vector<LoopId> parIt; // parallel iterator with 0 degree
466  std::vector<LoopId> filterIt; // filter loop with 0 degree
467  const LoopId numLoops = env.merger().getNumLoops();
468  for (LoopId i = 0; i < numLoops; i++) {
469  if (inDegree[i] == 0) {
470  if (env.merger().isFilterLoop(i))
471  filterIt.push_back(i);
472  else if (linalg::isReductionIterator(iteratorTypes[i]))
473  redIt.push_back(i);
474  else
475  parIt.push_back(i);
476  }
477  }
478 
479  while (!redIt.empty() || !parIt.empty() || !filterIt.empty()) {
480  // We always choose in order of filter loop -> parallel loop -> reduction
481  // loop because
482  // 1. Putting reduction loop early might make the loop sequence
483  // inadmissible.
484  // 2. Filter loops should be put as early as possible for better
485  // performance, since only one (if any) iteration will carry the
486  // computation. E.g., for (1 to N)
487  // for (1 to M)
488  // for (1 to K)
489  // if (xxx)
490  // O(X) computation => O(NMK+NMX) time complexity
491  //
492  // By putting the filter loop one level up, we got
493  //
494  // for (1 to N)
495  // for (1 to K)
496  // if (xxx)
497  // for (1 to M)
498  // O(X) computation => O(NK+NMX) time complexity
499  auto &it = !filterIt.empty() ? filterIt : (!parIt.empty() ? parIt : redIt);
500  auto src = it.back();
501  env.topSortPushBack(src);
502  it.pop_back();
503  // Update in-degree, and push 0-degree node into worklist.
504  for (LoopId dst = 0; dst < numLoops; dst++) {
505  if (adjM[src][dst] && --inDegree[dst] == 0) {
506  if (env.merger().isFilterLoop(dst))
507  filterIt.push_back(dst);
508  else if (linalg::isReductionIterator(iteratorTypes[dst]))
509  redIt.push_back(dst);
510  else
511  parIt.push_back(dst);
512  }
513  }
514  }
515  return env.topSortSize() == numLoops;
516 }

518 /// Helper method to add all constraints from the indices in one affine
519 /// expression before all indices in the other affine expression. For
520 /// example i0+i1 < i2+i3+1 yields i0<i2, i0<i3, i1<i2, and i1<i3.
521 /// The affine expression `a` is empty iff `fidx` have a value, leading to
522 /// b = (i0 + i1) < fidx => i0 < fidx, i1 < fidx.
523 /// The affine expression `b` is empty iff `tidx` have a value, leading to
524 /// tidx < a = (i0 + i1) => tidx < i0, tidx < i1.
525 ///
526 /// The `inDegree` is indexed by `LoopId`, and the `adjM` is indexed by
527 /// `(LoopId,LoopId)`.
528 static void addAffineOrderings(std::vector<std::vector<bool>> &adjM,
529  std::vector<unsigned> &inDegree, AffineExpr a,
530  AffineExpr b, std::optional<LoopId> fidx,
531  std::optional<LoopId> tidx) {
532  if (!a && !b) {
533  // Recursion leaf.
534  assert(fidx && tidx);
535  const LoopId f = *fidx, t = *tidx;
536  if (!adjM[f][t]) {
537  adjM[f][t] = true;
538  inDegree[t]++;
539  }
540  return;
541  }
542  // Picks an affine expression and expand (recurse into) it.
543  const auto toExpand = a ? a : b;
544  switch (toExpand.getKind()) {
545  case AffineExprKind::DimId: {
546  const std::optional<LoopId> idx{
547  toExpand.cast<AffineDimExpr>().getPosition()};
548  if (toExpand == a)
549  addAffineOrderings(adjM, inDegree, AffineExpr(), b, idx, tidx);
550  else // toExpand == b
551  addAffineOrderings(adjM, inDegree, a, AffineExpr(), fidx, idx);
552  break;
553  }
554  case AffineExprKind::Add:
555  case AffineExprKind::Mul: {
556  auto binOp = toExpand.cast<AffineBinaryOpExpr>();
557  if (toExpand == a) {
558  addAffineOrderings(adjM, inDegree, binOp.getLHS(), b, fidx, tidx);
559  addAffineOrderings(adjM, inDegree, binOp.getRHS(), b, fidx, tidx);
560  } else {
561  addAffineOrderings(adjM, inDegree, a, binOp.getLHS(), fidx, tidx);
562  addAffineOrderings(adjM, inDegree, a, binOp.getRHS(), fidx, tidx);
563  }

571 static void tryRelaxAffineConstraints(linalg::GenericOp op,
572  std::optional<LoopId> &fldx,
573  AffineExpr &fa,
574  std::optional<LoopId> &tldx,
575  AffineExpr &ta) {
576  // We use a heuristic here to only pick one dim expression from each
577  // compound affine expression to establish the order between two dense
578  // dimensions.
579  if (!tldx) {
580  AffineDimFinder finder(op);
581  // NOTE: The ordering can only be loosen when the destination level is
582  // dense (when !tldx), for [dense, sparse] -> (d0 + d1, d2), we still
583  // require both d0 < d2 and d1 < d2 to ensure correct ordering (i.e.,
584  // no ordering like d0->d2->d1).
585  // TODO: this is obviously a sub optimal solution.
586  if (!fldx && !fa.isa<AffineConstantExpr>()) {
587  // Heuristic: we prefer parallel loop for lhs to reduce the chance
588  // we add reduce < parallel ordering.
589  finder.setPickedIterType(utils::IteratorType::parallel);
590  finder.walkPostOrder(fa);
591  fa = finder.getDimExpr();
592  fldx = finder.getDimExpr().getPosition();
593  }
594  if (!ta.isa<AffineConstantExpr>()) {
595  // Heuristic: we prefer reduction loop for rhs to reduce the chance
596  // adding reduce < parallel ordering.
597  finder.setPickedIterType(utils::IteratorType::reduction);
598  finder.walkPostOrder(ta);
599  ta = finder.getDimExpr();
600  tldx = finder.getDimExpr().getPosition();
601  }
602  }

605 static void addFilterLoopBasedConstraints(CodegenEnv &env, OpOperand &t,
606  OpOperand *skip, SortMask mask,
607  std::vector<std::vector<bool>> &adjM,
608  std::vector<unsigned> &inDegree) {
609  // Get map, encoding, and tensor-identifier.
610  const auto map = env.op().getMatchingIndexingMap(&t);
611  const auto enc = getSparseTensorEncoding(t.get().getType());
612  const TensorId tid = env.makeTensorId(t.getOperandNumber());
613 
614  // Each tensor expression and optional dimension ordering (row-major
615  // by default) puts an ordering constraint on the loop indices. For
616  // example, the tensor expresion A_ijk forces the ordering i < j < k
617  // on the loop indices if no explicit dimension ordering is given.
618  const Level lvlRank = map.getNumResults();
619  assert(!enc || lvlRank == enc.getLvlRank());
620  for (Level lvl = 0; lvl < lvlRank; lvl++) {
621  // FIXME: `toOrigDim` is deprecated.
622  AffineExpr ta = map.getResult(toOrigDim(enc, lvl));
623  std::optional<LoopId> tldx = env.merger().getLoopId(tid, lvl);
624  // Filter loops should be constructed after all the dependent loops,
625  // i.e., d0 + d1 < filter_loop(d0 + d1)
626  if (tldx && env.merger().isFilterLoop(*tldx)) {
627  assert(!ta.isa<AffineDimExpr>() &&
628  !isDenseDLT(enc.getDimLevelType()[lvl]));
629  addAffineOrderings(adjM, inDegree, ta, AffineExpr(), std::nullopt, tldx);
630  // Now that the ordering of affine expression is captured by filter
631  // loop idx, we only need to ensure the affine ordering against filter
632  // loop. Thus, we reset the affine express to nil here to mark it as
633  // resolved.
634  ta = AffineExpr();
635  }
636 
637  // Skip tensor during cycle resolution, though order between filter loop
638  // and dependent loops need to be guaranteed unconditionally.
639  if (&t == skip)
640  continue;
641 
642  if (lvl > 0) {
643  // FIXME: `toOrigDim` is deprecated.
644  AffineExpr fa = map.getResult(toOrigDim(enc, lvl - 1));
645  std::optional<LoopId> fldx = env.merger().getLoopId(tid, lvl - 1);
646 
647  // Applying order constraints on every pair of dimExpr between two
648  // compound affine expressions can sometime too strict:
649  // E.g, for [dense, dense] -> (d0 + d1, d2 + d3).
650  // It is totally fine to have loop sequence d0->d2->d1->d3 instead of
651  // requiring d0 < d2, d1 < d2, d0 < d3, d1 < d3.
652  // We also relax the affine constraint when use slice-based algorithm
653  // as there is no filter loop for affine index on sparse dimension.
654  // TODO: do we really need the condition?
655  if (!includesDense(mask))
656  tryRelaxAffineConstraints(env.op(), fldx, fa, tldx, ta);
657 
658  // (d0 + d1) < (d2 + d3), or
659  // filter_loop_d-1 < (d2 + d3), or
660  // (d0 + d1) < filter_loop_d, or
661  // filter_loop_d-1 < filter_loop_d depending on whether fa/ta is reset
662  // above.
663  addAffineOrderings(adjM, inDegree, fa, ta, fldx, tldx);
664  }
665  }

668 static void addSliceBasedConstraints(CodegenEnv &env, OpOperand &t,
669  OpOperand *skip, SortMask mask,
670  std::vector<std::vector<bool>> &adjM,
671  std::vector<unsigned> &inDegree) {
672  // Get map and encoding.
673  const auto map = env.op().getMatchingIndexingMap(&t);
674  const auto enc = getSparseTensorEncoding(t.get().getType());
675 
676  // No special treatment for simple indices.
677  if (getNumNonTrivialIdxExpOnSparseLvls(map, t.get()) == 0)
678  return addFilterLoopBasedConstraints(env, t, skip, mask, adjM, inDegree);
679 
680  // Skip tensor during cycle resolution, though order between filter loop
681  // and dependent loops need to be guaranteed unconditionally.
682  if (&t == skip)
683  return;
684 
685  AffineDimFinder finder(env.op());
686  finder.setPickedIterType(utils::IteratorType::reduction);
687  // To compute iteration graph for tensor[d0 + d1 + d3, d4 + d5 + d6],
688  // we requires there exist d_x \in {d0, d1, d3} and d_y \in {d4, d5, d6},
689  // and d_x > d_y && {d0, d1, d3} - d_x > {d4, d5, d6} - d_y
690  const Level lvlRank = map.getNumResults();
691  assert(!enc || lvlRank == enc.getLvlRank());
692  for (Level lvl = 1; lvl < lvlRank; lvl++) {
693  // FIXME: `toOrigDim` is deprecated.
694  const AffineExpr fa = map.getResult(toOrigDim(enc, lvl - 1));
695  const AffineExpr ta = map.getResult(toOrigDim(enc, lvl));
696 
697  // This is a heuristic, we pick an abitrary reduction loop from lhs and
698  // rhs and use them as d_x and d_y.
699  finder.walkPostOrder(fa);
700  const AffineDimExpr fexp = finder.getDimExpr();
701  const LoopId fldx = env.makeLoopId(fexp.getPosition());
702 
703  finder.walkPostOrder(ta);
704  const AffineDimExpr texp = finder.getDimExpr();
705  const LoopId tldx = env.makeLoopId(texp.getPosition());
706 
707  // d_x > d_y
708  if (!adjM[fldx][tldx]) {
709  adjM[fldx][tldx] = true;
710  inDegree[tldx]++;
711  }
712 
713  AffineDimCollector fCollector;
714  fCollector.walkPostOrder(fa);
715  AffineDimCollector tCollector;
716  tCollector.walkPostOrder(ta);
717 
718  // make sure dx and dy is the last;
719  for (auto fd : fCollector.dims) {
720  const LoopId f = env.makeLoopId(fd.getPosition());
721  if (f == fldx)
722  continue;
723  if (!adjM[f][fldx]) {
724  adjM[f][fldx] = true;
725  inDegree[fldx]++;
726  }
727  }
728  for (auto td : tCollector.dims) {
729  const LoopId t = env.makeLoopId(td.getPosition());
730  if (t == tldx)
731  continue;
732  if (!adjM[t][tldx]) {
733  adjM[t][tldx] = true;
734  inDegree[tldx]++;
735  }
736  }
737  // Since we only support affine addition, the order between two dim
738  // expression does not really matters.
739  // {d0, d1, d3} - d_x > {d4, d5, d6} - d_y
740  // This is to ensure that the affine expressions are reduced in sparse
741  // tensor level ordering.
742  // TODO: this ordering could probably be loosen if we support out-of-order
743  // reduction.
744  // TODO: the evaluation order need to be ensure to
745  // support affine multiplication.
746  for (auto fd : fCollector.dims) {
747  const LoopId f = env.makeLoopId(fd.getPosition());
748  if (f == fldx) // skip d_x
749  continue;
750 
751  for (auto td : tCollector.dims) {
752  const LoopId t = env.makeLoopId(td.getPosition());
753  if (t == tldx) // skip d_y
754  continue;
755  if (!adjM[f][t]) {
756  adjM[f][t] = true;
757  inDegree[t]++;
758  }
759  }
760  }
761  }
762 }

764 /// Computes a topologically sorted iteration graph for the linalg operation.
765 /// Ensures all tensors are visited in natural index order. This is
766 /// essential for sparse storage formats since these only support access
767 /// along fixed dimensions. Even for dense storage formats, however, the natural
768 /// index order yields innermost unit-stride access with better spatial
769 /// locality.
770 static bool computeIterationGraph(CodegenEnv &env, SortMask mask,
771  OpOperand *skip, bool idxReducBased = false) {
772  // Set up an n x n from/to adjacency matrix of the iteration graph
773  // for the implicit loop indices i_0 .. i_n-1.
774  const unsigned numLoops = env.merger().getNumLoops();
775  std::vector<std::vector<bool>> adjM(numLoops,
776  std::vector<bool>(numLoops, false));
777  std::vector<unsigned> inDegree(numLoops, 0); // in-degree of each node.
778  const auto iteratorTypes = env.op().getIteratorTypesArray();
779  // Iterate over the indexing maps of every tensor in the tensor expression.
780  for (OpOperand &t : env.op()->getOpOperands()) {
781  // Get map and encoding.
782  const auto enc = getSparseTensorEncoding(t.get().getType());
783  // Skips dense inputs/outputs when not requested.
784  const bool isDenseInput = !enc && env.op().isDpsInput(&t);
785  const bool isDenseOutput = !enc && !isDenseInput;
786  if ((isDenseInput && !includesDenseInput(mask)) ||
787  (isDenseOutput && !includesDenseOutput(mask)))
788  continue;
789 
790  // Push unrelated loops into sparse iteration space, so these
791  // will be skipped more often.
792  // TODO: Do we really need this?
793  if (includesUndef(mask)) {
794  const TensorId tid = env.makeTensorId(t.getOperandNumber());
795  for (LoopId i = 0; i < numLoops; i++) {
796  const auto dltI = env.dlt(tid, i);
797  if (isCompressedDLT(dltI) || isCompressedWithHiDLT(dltI) ||
798  isSingletonDLT(dltI)) {
799  for (LoopId j = 0; j < numLoops; j++)
800  if (isUndefDLT(env.dlt(tid, j))) {
801  adjM[i][j] = true;
802  inDegree[j]++;
803  }
804  } else {
805  assert(isDenseDLT(dltI) || isUndefDLT(dltI));
806  }
807  }
808  }
809  // Push unrelated loops into sparse iteration space, so these
810  // will be skipped more often.
811  if (idxReducBased)
812  addSliceBasedConstraints(env, t, skip, mask, adjM, inDegree);
813  else
814  addFilterLoopBasedConstraints(env, t, skip, mask, adjM, inDegree);
815  }
816  // Topologically sort the iteration graph to determine loop order.
817  // Report failure for a cyclic iteration graph.
818  env.topSortClear(numLoops);
819  return topSortOptimal(env, iteratorTypes, inDegree, adjM);
820 }

826 /// Local bufferization of all dense and sparse data structures.
827 static void genBuffers(CodegenEnv &env, OpBuilder &builder) {
828  linalg::GenericOp op = env.op();
829  Location loc = op.getLoc();
830  assert(op.getNumOperands() == op.getNumDpsInputs() + 1);
831 
832  env.emitter().initializeLoopEmit(
833  builder, loc,
834  /// Generates buffer for the output tensor.
835  /// Note that all sparse kernels assume that when all elements are written
836  /// to (viz. x(i) = y(i) * z(i)), the output buffer is already initialized
837  /// to all zeroes and only nonzeroes values are computed and written out.
838  /// For updates (viz. x(i) += y(i) * z(i)), only nonzeroes values are used
839  /// for the updates and no assumption on the original contents of the
840  /// output buffer is necessary.
841  [&op](OpBuilder &builder, Location loc, Value memref,
842  Value tensor) -> Value {
843  // Must not be a sparse tensor.
844  assert(!getSparseTensorEncoding(tensor.getType()));
845  // Two output tensor references should point to the same object.
846  OpOperand *lhs = op.getDpsInitOperand(0);
847  assert(lhs->get() == tensor);
848  // An output tensor can simply materialize from the buffer of the tensor
849  // that appears in the outs() clause. For updates, this has the
850  // advantage that only the nonzero value are involved in the
851  // computation, keeping the operation O(nnz). In all other cases, we are
852  // forced to zero out the buffer to enforce the assumption above, which
853  // may negatively impact running complexity (viz. O(n^2 + nnz) vs.
854  // O(nnz) for matrices).
855  // TODO: use better analysis to avoid zeroing out the buffer?
856  bool isInit = op.isInitTensor(lhs);
857  Value init = memref;
858  if (!isInit) {
859  Value zero = constantZero(builder, loc,
860  getElementTypeOrSelf(tensor.getType()));
861  builder.create<linalg::FillOp>(loc, ValueRange{zero},
862  ValueRange{init});
863  }
864  return init;
865  });
866 }

887 static Value genSubscript(CodegenEnv &env, OpBuilder &builder, OpOperand *t,

892  const auto stt = getSparseTensorType(t->get());

944 /// Generates insertion code to implement dynamic tensor store.
945 static void genInsertionStore(CodegenEnv &env, OpBuilder &builder, OpOperand *t,
946  Value rhs) {
947  linalg::GenericOp op = env.op();
948  Location loc = op.getLoc();
949  // Direct insertion in lexicographic coordinate order.
950  if (!env.isExpand()) {
951  const LoopOrd numLoops = op.getRank(t);
952  // TODO: rewrite this to use `env.emitter().getLoopIVs(ivs)`
953  // instead. We just need to either assert that `numLoops ==
954  // env.emitter().getCurrentDepth()`, or else update the `getLoopIVs`
955  // method to take an optional parameter to restrict to a smaller depth.
956  SmallVector<Value> ivs;
957  ivs.reserve(numLoops);
958  for (LoopOrd n = 0; n < numLoops; n++) {
959  const auto iv = env.emitter().getLoopIV(n);
960  assert(iv);
961  ivs.push_back(iv);
962  }
963  Value chain = env.getInsertionChain();
964  if (!env.getValidLexInsert()) {
965  env.updateInsertionChain(builder.create<InsertOp>(loc, rhs, chain, ivs));
966  } else {
967  // Generates runtime check for a valid lex during reduction,
968  // to avoid inserting the identity value for empty reductions.
969  // if (validLexInsert) then
970  // insert(rhs) into chain
971  // return updated chain
972  // else
973  // return unmodified chain
974  scf::IfOp ifValidLexInsert = builder.create<scf::IfOp>(
975  loc, chain.getType(), env.getValidLexInsert(),
976  /*else=*/true);
977  // True branch.
978  builder.setInsertionPointToStart(ifValidLexInsert.thenBlock());
979  Value res = builder.create<InsertOp>(loc, rhs, chain, ivs);
980  builder.create<scf::YieldOp>(loc, res);
981  // False branch.
982  builder.setInsertionPointToStart(ifValidLexInsert.elseBlock());
983  builder.create<scf::YieldOp>(loc, chain);
984  // Value assignment.
985  builder.setInsertionPointAfter(ifValidLexInsert);
986  env.updateInsertionChain(ifValidLexInsert.getResult(0));
987  }
988  return;
989  }
990  // Generates insertion code along expanded access pattern.
991  // if (!expFilled[i]) then
992  // expFilled[i] = true
993  // expAdded[inserts++] = i
994  // endif
995  // values[i] = rhs
996  Value values = env.getExpandValues();
997  Value filled = env.getExpandFilled();
998  Value added = env.getExpandAdded();
999  Value count = env.getExpandCount();
1000  Value index = genIndex(env, t);
1001  Value fval = constantI1(builder, loc, false);
1002  Value tval = constantI1(builder, loc, true);
1003  // If statement.
1004  Value isFilled = builder.create<memref::LoadOp>(loc, filled, index);
1005  Value cond = builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::eq,
1006  isFilled, fval);
1007  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, builder.getIndexType(), cond,
1008  /*else=*/true);
1009  // True branch.
1010  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
1011  builder.create<memref::StoreOp>(loc, tval, filled, index);
1012  builder.create<memref::StoreOp>(loc, index, added, count);
1013  Value one = constantIndex(builder, loc, 1);
1014  Value add = builder.create<arith::AddIOp>(loc, count, one);
1015  builder.create<scf::YieldOp>(loc, add);
1016  // False branch.
1017  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
1018  builder.create<scf::YieldOp>(loc, count);
1019  builder.setInsertionPointAfter(ifOp);
1020  // Value assignment.
1021  env.updateExpandCount(ifOp.getResult(0));
1022  builder.create<memref::StoreOp>(loc, rhs, values, index);
1023 }

1046 /// Generates a store on a dense or sparse tensor.
1047 static void genTensorStore(CodegenEnv &env, OpBuilder &builder, ExprId exp,
1048  Value rhs) {
1049  linalg::GenericOp op = env.op();
1050  Location loc = op.getLoc();
1051  // Test if this is a scalarized reduction.
1052  if (env.isReduc()) {
1053  env.updateReduc(rhs);
1054  return;
1055  }
1056  // Store during insertion.
1057  OpOperand *t = op.getDpsInitOperand(0);
1058  if (env.isSparseOutput(t)) {
1059  if (!rhs) {
1060  // Only unary and binary are allowed to return uninitialized rhs
1061  // to indicate missing output.
1062  assert(env.exp(exp).kind == TensorExp::Kind::kUnary ||
1063  env.exp(exp).kind == TensorExp::Kind::kBinary);
1064  } else if (env.exp(exp).kind == TensorExp::Kind::kSelect) {
1065  // Select operation insertion.
1066  Value chain = env.getInsertionChain();
1067  scf::IfOp ifOp =
1068  builder.create<scf::IfOp>(loc, chain.getType(), rhs, /*else=*/true);
1069  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
1070  // Existing value was preserved to be used here.
1071  assert(env.exp(exp).val);
1072  Value v0 = env.exp(exp).val;
1073  genInsertionStore(env, builder, t, v0);
1074  env.merger().clearExprValue(exp);
1075  // Yield modified insertion chain along true branch.
1076  Value mchain = env.getInsertionChain();
1077  builder.create<scf::YieldOp>(op.getLoc(), mchain);
1078  // Yield original insertion chain along false branch.
1079  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
1080  builder.create<scf::YieldOp>(loc, chain);
1081  // Done with if statement.
1082  env.updateInsertionChain(ifOp->getResult(0));
1083  builder.setInsertionPointAfter(ifOp);
1084  } else {
1085  genInsertionStore(env, builder, t, rhs);
1086  }
1087  return;
1088  }
1089  // Actual store.
1090  SmallVector<Value> args;
1091  Value ptr = genSubscript(env, builder, t, args);
1092  builder.create<memref::StoreOp>(loc, rhs, ptr, args);
1093 }

1122 /// Recursively generates tensor expression.
1123 static Value genExp(CodegenEnv &env, RewriterBase &rewriter, ExprId e,
1124  LoopId ldx) {
1125  linalg::GenericOp op = env.op();

1129  return Value();

1131  const auto kind = exp.kind;

1133  return genTensorLoad(env, rewriter, e);
1134  if (kind == TensorExp::Kind::kInvariant)
1135  return genInvariantValue(env, e);
1136  if (kind == TensorExp::Kind::kLoopVar)

1146  (kind == TensorExp::Kind::kUnary || kind == TensorExp::Kind::kBinary ||

1148  kind == TensorExp::Kind::kReduce || kind == TensorExp::Kind::kSelect))

1155  env.merger().setExprValue(e, v0); // Preserve value for later use.

1160 /// Hoists loop invariant tensor loads for which indices have been exhausted.
1161 static void genInvariants(CodegenEnv &env, OpBuilder &builder, ExprId exp,
1162  LoopId ldx, bool atStart) {
1163  if (exp == ::mlir::sparse_tensor::detail::kInvalidId)
1164  return;
1165  if (env.exp(exp).kind == TensorExp::Kind::kTensor) {
1166  // Inspect tensor indices.
1167  bool isAtLoop = ldx == ::mlir::sparse_tensor::detail::kInvalidId;
1168  linalg::GenericOp op = env.op();
1169  OpOperand &t = op->getOpOperand(env.exp(exp).tensor);
1170  const auto map = op.getMatchingIndexingMap(&t);
1171  const auto stt = getSparseTensorType(t.get());
1172  const Level lvlRank = stt.getLvlRank();
1173  assert(static_cast<Level>(map.getNumResults()) == lvlRank);
1174  for (Level l = 0; l < lvlRank; l++) {
1175  // FIXME: `toOrigDim` is deprecated.
1176  // FIXME: above we asserted that there are `lvlRank` many results,
1177  // but this is assuming there are in fact `dimRank` many results instead.
1178  const AffineExpr a = map.getResult(toOrigDim(stt.getEncoding(), l));
1179  const auto sldx =
1180  env.merger().getLoopId(env.makeTensorId(t.getOperandNumber()), l);
1181  if (sldx && env.merger().isFilterLoop(*sldx)) {
1182  if (!env.getLoopVar(*sldx))
1183  // The filter loops has not been constructed.
1184  return;
1185  if (*sldx == ldx)
1186  isAtLoop = true;
1187  } else if (!isInvariantAffine(env, a, ldx, isAtLoop))
1188  return; // still in play
1189  }
1190  // All exhausted at this level (isAtLoop denotes exactly at this LoopId).
1191  if (!isAtLoop)
1192  return;
1193  OpOperand *lhs = op.getDpsInitOperand(0);
1194  if (lhs == &t) {
1195  // Start or end a scalarized reduction.
1196  if (atStart) {
1197  Value load = env.isCustomReduc() ? env.getCustomRedId()
1198  : genTensorLoad(env, builder, exp);
1199  env.startReduc(exp, load);
1200  if (env.hasSparseOutput())
1201  env.setValidLexInsert(constantI1(builder, env.op().getLoc(), false));
1202  } else {
1203  genTensorStore(env, builder, exp, env.endReduc());
1204  env.clearValidLexInsert();
1205  }
1206  } else {
1207  // Start or end loop invariant hoisting of a tensor load.
1208  if (atStart)
1209  env.merger().setExprValue(exp, genTensorLoad(env, builder, exp));
1210  else
1211  env.merger().clearExprValue(exp);
1212  }
1213  } else if (env.exp(exp).kind != TensorExp::Kind::kInvariant &&
1214  env.exp(exp).kind != TensorExp::Kind::kLoopVar) {
1215  // Traverse into the binary operations. Note that we only hoist
1216  // tensor loads, since subsequent MLIR/LLVM passes know how to
1217  // deal with all other kinds of derived loop invariants.
1218  if (env.exp(exp).kind == TensorExp::Kind::kReduce)
1219  env.startCustomReduc(exp); // enter custom
1220  const ExprId e0 = env.exp(exp).children.e0;
1221  const ExprId e1 = env.exp(exp).children.e1;
1222  genInvariants(env, builder, e0, ldx, atStart);
1223  genInvariants(env, builder, e1, ldx, atStart);
1224  if (env.exp(exp).kind == TensorExp::Kind::kReduce)
1225  env.endCustomReduc(); // exit custom
1226  }
1227 }

1229 /// Generates an expanded access pattern in innermost dimension.
1230 static void genExpand(CodegenEnv &env, OpBuilder &builder, LoopOrd at,
1231  bool atStart) {
1232  linalg::GenericOp op = env.op();
1233  OpOperand *lhs = op.getDpsInitOperand(0);
1234  if (!env.atExpandLevel(lhs, op.getRank(lhs), at))
1235  return; // not needed at this level
1236  assert(!env.isReduc());
1237  // Generate start or end of an expanded access pattern. Note that because
1238  // an expension does not rely on the ongoing contents of the sparse storage
1239  // scheme, we can use the original tensor as incoming SSA value (which
1240  // simplifies codegen a bit). If expansion on the actual contents is ever
1241  // needed, we will need to use the SSA value in the insertion chain instead.
1242  Value tensor = lhs->get();
1243  Location loc = op.getLoc();
1244  if (atStart) {
1245  auto dynShape = {ShapedType::kDynamic};
1246  Type etp = tensor.getType().cast<ShapedType>().getElementType();
1247  Type t1 = MemRefType::get(dynShape, etp);
1248  Type t2 = MemRefType::get(dynShape, builder.getI1Type());
1249  Type t3 = MemRefType::get(dynShape, builder.getIndexType());
1250  Type t4 = builder.getIndexType();
1251  auto r = builder.create<ExpandOp>(loc, TypeRange({t1, t2, t3, t4}), tensor);
1252  assert(r.getNumResults() == 4);
1253  env.startExpand(r.getResult(0), r.getResult(1), r.getResult(2),
1254  r.getResult(3));
1255  } else {
1256  SmallVector<Value> indices;
1257  for (LoopOrd i = 0; i < at; i++)
1258  indices.push_back(env.emitter().getLoopIV(i));
1259  Value values = env.getExpandValues();
1260  Value filled = env.getExpandFilled();
1261  Value added = env.getExpandAdded();
1262  Value count = env.getExpandCount();
1263  Value chain = env.getInsertionChain();
1264  Value compress = builder.create<CompressOp>(loc, values, filled, added,
1265  count, chain, indices);
1266  env.updateInsertionChain(compress);
1267  env.endExpand();
1268  }

1271 /// Returns parallelization strategy. Any implicit loop in the Linalg
1272 /// operation that is marked "parallel" is a candidate. Whether it is actually
1273 /// converted to a parallel operation depends on the requested strategy.
1274 static bool isParallelFor(CodegenEnv &env, bool isOuter, bool isSparse) {
1275  // Reject parallelization of sparse output.
1276  if (env.hasSparseOutput())
1277  return false;
1278  // Parallel loops on tensor expansion can cause data races.
1279  if (env.isExpand())
1280  return false;
1281  // Inspect strategy.
1282  switch (env.options().parallelizationStrategy) {
1283  case SparseParallelizationStrategy::kNone:
1284  return false;
1285  case SparseParallelizationStrategy::kDenseOuterLoop:
1286  return isOuter && !isSparse;
1287  case SparseParallelizationStrategy::kAnyStorageOuterLoop:
1288  return isOuter;
1289  case SparseParallelizationStrategy::kDenseAnyLoop:
1290  return !isSparse;
1291  case SparseParallelizationStrategy::kAnyStorageAnyLoop:
1292  return true;
1293  }
1294  llvm_unreachable("unexpected parallelization strategy");
1295 }

1633  // needed for linearized env.

1646  LatPointId li, bool needsUniv) {

1650  // becomes invariant and the address shall now be generated at the current
1651  // level.

1712 /// Recursively generates code while computing iteration lattices in order
1713 /// to manage the complexity of implementing co-iteration over unions
1714 /// and intersections of sparse iterations spaces.
1715 static void genStmt(CodegenEnv &env, RewriterBase &rewriter, ExprId exp,
1716  LoopOrd at) {
1717  // At each leaf, assign remaining tensor (sub)expression to output tensor.
1718  if (at == env.topSortSize()) {
1719  const LoopId ldx = env.topSortAt(at - 1);
1720  Value rhs = genExp(env, rewriter, exp, ldx);
1721  genTensorStore(env, rewriter, exp, rhs);
1722  return;
1723  }
1724 
1725  // Construct iteration lattices for current loop index, with L0 at top.
1726  const LoopId idx = env.topSortAt(at);
1727  const LoopId ldx = at == 0 ? ::mlir::sparse_tensor::detail::kInvalidId
1728  : env.topSortAt(at - 1);
1729  const LatSetId lts =
1730  env.merger().optimizeSet(env.merger().buildLattices(exp, idx));
1731 
1732  // Start a loop sequence.
1733  bool needsUniv = startLoopSeq(env, rewriter, exp, at, idx, ldx, lts);
1734 
1735  // Emit a loop for every lattice point L0 >= Li in this loop sequence.
1736  //
1737  // NOTE: We cannot change this to `for (const LatPointId li : env.set(lts))`
1738  // because the loop body causes data-movement which invalidates
1739  // the iterator.
1740  const unsigned lsize = env.set(lts).size();
1741  for (unsigned i = 0; i < lsize; i++) {
1742  const LatPointId li = env.set(lts)[i];
1743  // Start a loop.
1744  auto [loop, isSingleCond] = startLoop(env, rewriter, at, li, needsUniv);
1745 
1746  // Visit all lattices points with Li >= Lj to generate the
1747  // loop-body, possibly with if statements for coiteration.
1748  Value redInput = env.getReduc();
1749  Value cntInput = env.getExpandCount();
1750  Value insInput = env.getInsertionChain();
1751  // NOTE: We cannot change this to `for (const LatPointId lj : env.set(lts))`
1752  // because the loop body causes data-movement which invalidates the
1753  // iterator.
1754  for (unsigned j = 0; j < lsize; j++) {
1755  const LatPointId lj = env.set(lts)[j];
1756  const ExprId ej = env.lat(lj).exp;
1757  if (li == lj || env.merger().latGT(li, lj)) {
1758  // Recurse into body of each branch.
1759  if (!isSingleCond) {
1760  scf::IfOp ifOp = genIf(env, rewriter, idx, lj);
1761  genStmt(env, rewriter, ej, at + 1);
1762  endIf(env, rewriter, ifOp, loop, redInput, cntInput, insInput);
1763  } else {
1764  genStmt(env, rewriter, ej, at + 1);
1765  }
1766  }
1767  }
1768 
1769  // End a loop.
1770  needsUniv = endLoop(env, rewriter, loop, idx, li, needsUniv, isSingleCond);
1771  }
1772 
1773  // End a loop sequence.
1774  endLoopSeq(env, rewriter, exp, at, idx, ldx);
1775 }

1777 /// Converts the result computed by the sparse kernel into the required form.
1778 static void genResult(CodegenEnv &env, RewriterBase &rewriter) {
1779  linalg::GenericOp op = env.op();
1780  OpOperand *lhs = op.getDpsInitOperand(0);
1781  Value tensor = lhs->get();
1782  Type resType = tensor.getType();
1783  if (getSparseTensorEncoding(resType)) {
1784  // The sparse tensor rematerializes from the original sparse tensor's
1785  // underlying sparse storage format. For an insertion chain, the
1786  // tensor materializes from the chain with 'hasInserts' enabled.
1787  bool hasInserts = false;
1788  if (Value chain = env.getInsertionChain()) {
1789  hasInserts = true;
1790  tensor = chain;
1791  }
1792  rewriter.replaceOpWithNewOp<LoadOp>(op, resType, tensor, hasInserts);
1793  } else {
1794  // To rematerialize an non-annotated tensor, simply load it
1795  // from the bufferized value.
1796  Value val = env.emitter().getValBuffer().back(); // value array
1797  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, resType, val);
1798  }

1807 /// Sparse rewriting rule for generic Lingalg operation.
1808 struct GenericOpSparsifier : public OpRewritePattern<linalg::GenericOp> {
1809 public:
1810  GenericOpSparsifier(MLIRContext *context, SparsificationOptions o)
1811  : OpRewritePattern<linalg::GenericOp>(context), options(o) {}
1812 
1813  LogicalResult matchAndRewrite(linalg::GenericOp op,
1814  PatternRewriter &rewriter) const override {
1815  // Only accept single output operations without affine index on sparse
1816  // output.
1817  if (op.getNumDpsInits() != 1 || hasNonTrivialAffineOnSparseOut(op))
1818  return failure();
1819 
1820  // Sets up a code generation environment.
1821  const unsigned numTensors = op->getNumOperands();
1822  const unsigned numLoops = op.getNumLoops();
1823  const unsigned numFilterLoops = getNumNonTrivialIdxExpOnSparseLvls(op);
1824  // TODO: we should probably always use slice-based codegen whenever
1825  // possible, we can even intermix slice-based and filter-loop based codegen.
1826  bool idxReducBased = options.enableIndexReduction && numFilterLoops != 0;
1827 
1828  // If we have indexing map like (d0) -> (0, d0), there might be more
1829  // levels then loops because of the constant index, that means we can not
1830  // use numLoops as the upper bound for ranks of all tensors.
1831  // TODO: Constant indices are currently not support on sparse tensor, but
1832  // are allowed in non-annotated dense tensor. Support it, it would be
1833  // required for sparse tensor slice rank reducing too.
1834  Level maxLvlRank = 0;
1835  for (auto operand : op.getOperands()) {
1836  if (auto rtp = operand.getType().dyn_cast<RankedTensorType>()) {
1837  maxLvlRank = std::max(maxLvlRank, SparseTensorType(rtp).getLvlRank());
1838  }
1839  }
1840 
1841  // If we uses slice based algorithm for affine index, we do not need filter
1842  // loop.
1843  CodegenEnv env(op, options, numTensors, numLoops,
1844  /*numFilterLoops=*/idxReducBased ? 0 : numFilterLoops,
1845  maxLvlRank);
1846 
1847  // Detects sparse annotations and translates the per-level sparsity
1848  // information for all tensors to loop indices in the kernel.
1849  if (!findSparseAnnotations(env, idxReducBased))
1850  return failure();
1851 
1852  // Constructs the tensor expressions tree from `op`, returns failure if the
1853  // tree can not be built or the tensor expression is inadmissible.
1854  if (failed(env.initTensorExp()))
1855  return failure();
1856 
1857  // Computes a topologically sorted iteration graph to ensure tensors
1858  // are visited in natural index order. Gradually relaxes the considered
1859  // constraints until an acyclic iteration graph results, such that sparse
1860  // code generation can proceed. As a last resort, an attempt is made
1861  // to resolve cycles by inserting a conversion.
1862  bool isAdmissible = false;
1863  bool hasCycle = true;
1864 
1865  // An const list of all masks that we used for interation graph
1866  // computation. Must be ordered from more strict to less strict.
1867  // Ideally (though might not be guaranteed), the eariler a constraint mask
1868  // can be satisfied, the faster the generated kernel will be.
1869  const auto allMasks = {
1870  SortMask::kIncludeAll, SortMask::kIncludeDense,
1871  SortMask::kIncludeDenseInput, SortMask::kIncludeDenseOutput,
1872  SortMask::kIncludeUndef, SortMask::kSparseOnly};
1873  for (const SortMask mask : allMasks) {
1874  if (computeIterationGraph(env, mask, nullptr, idxReducBased)) {
1875  hasCycle = false;
1876  if (env.isAdmissibleTopoOrder()) {
1877  isAdmissible = true;
1878  break;
1879  }
1880  // else try a set of less strict constraints.
1881  }
1882  }
1883  if (hasCycle) {
1884  return idxReducBased
1885  ? failure() // TODO: should cycle be resolved differently?
1886  : resolveCycle(env, rewriter); // one last shot
1887  }
1888 
1889  if (!isAdmissible)
1890  return failure(); // inadmissible expression, reject
1891 
1892  // Recursively generates code if admissible.
1893  env.startEmit();
1894  genBuffers(env, rewriter);
1895  // TODO: Constant affine expression should be handled differently when using
1896  // slice-based codegen, it does not matter now becasue we already reject the
1897  // constant expression at a earlier stage.
1898  genInitConstantDenseAddress(env, rewriter);
1899  genStmt(env, rewriter, env.getExprId(), 0);
1900  genResult(env, rewriter);
1901  return success();
1902  }
1903 
1904 private:
1905  // Last resort cycle resolution.
1906  LogicalResult resolveCycle(CodegenEnv &env, PatternRewriter &rewriter) const {
1907  // Compute topological sort while leaving out every
1908  // sparse input tensor in succession until an acylic
1909  // iteration graph results.
1910  for (OpOperand *t : env.op().getDpsInputOperands()) {
1911  const TensorId tid = env.makeTensorId(t->getOperandNumber());
1912  Value tval = t->get();
1913  auto srcEnc = getSparseTensorEncoding(tval.getType());
1914  if (!srcEnc || !computeIterationGraph(env, SortMask::kSparseOnly, t))
1915  continue;
1916  // Found an input tensor that resolves the cycle by inserting a
1917  // conversion into a sparse tensor that adheres to the iteration
1918  // graph order. Also releases the temporary sparse tensor.
1919  //
1920  // TODO: investigate fusing the conversion with computation,
1921  // especially if it is a direct yield!
1922  //
1923  auto srcTp = getRankedTensorType(tval);
1924  auto dstEnc = SparseTensorEncodingAttr::get(
1925  getContext(), srcEnc.getDimLevelType(),
1926  permute(env, env.op().getMatchingIndexingMap(t)), // new order
1927  srcEnc.getHigherOrdering(), srcEnc.getPosWidth(),
1928  srcEnc.getCrdWidth());
1929  auto dstTp = RankedTensorType::get(srcTp.getShape(),
1930  srcTp.getElementType(), dstEnc);
1931  auto convert = rewriter.create<ConvertOp>(tval.getLoc(), dstTp, tval);
1932  rewriter.updateRootInPlace(env.op(),
1933  [&]() { env.op()->setOperand(tid, convert); });
1934  rewriter.setInsertionPointAfter(env.op());
1935  rewriter.create<bufferization::DeallocTensorOp>(tval.getLoc(), convert);
1936  return success();
1937  }
1938  // Cannot be resolved with a single conversion.
1939  // TODO: convert more than one?
1940  return failure();
1941  }

1944  SparsificationOptions options;
