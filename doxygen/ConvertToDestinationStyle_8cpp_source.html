32 // Implements backtracking to traverse indices of the output buffer while
33 // iterating over op.elements().
34 static Value createInserts(RewriterBase &rewriter, Location loc, int dim,
35  Value destination, ArrayRef<int64_t> shape,
36  ArrayRef<Value> constants,
37  OperandRange::iterator &elementIt,
38  SmallVectorImpl<Value> &indices) {
39  if (dim == static_cast<int>(shape.size()) - 1) {
40  for (int i = 0; i < shape.back(); ++i) {
41  indices.back() = constants[i];
42  destination = rewriter.create<tensor::InsertOp>(loc, *elementIt,
43  destination, indices);
44  ++elementIt;
45  }
46  return destination;
47  }
48  for (int i = 0; i < shape[dim]; ++i) {
49  indices[dim] = constants[i];
50  destination = createInserts(rewriter, loc, dim + 1, destination, shape,
51  constants, elementIt, indices);
52  }
53  return destination;
54 }

56 static Operation *movePaddingToFillOrGenericOp(RewriterBase &rewriter,
57  Location loc, PadOp padOp,
58  Value dest) {
59  OpBuilder::InsertionGuard g(rewriter);
60  RankedTensorType resultType = padOp.getResultType();
61 
62  // Examine the yielded value to decide if a linalg.generic is neede or a
63  // linalg.fill is sufficient.
64  Value yieldedValue =
65  cast<tensor::YieldOp>(padOp.getBody()->getTerminator()).getValue();
66  Attribute constYieldedValue;
67  // Is the yielded value a bbArg defined outside of the PadOp?
68  bool outsideBbArg =
69  yieldedValue.isa<BlockArgument>() &&
70  yieldedValue.cast<BlockArgument>().getOwner()->getParentOp() !=
71  padOp.getOperation();
72  // Is the yielded value an OpResult defined outside of the PadOp?
73  bool outsideOpResult =
74  yieldedValue.isa<OpResult>() &&
75  yieldedValue.getDefiningOp()->getParentOp() != padOp.getOperation();
76  bool invariantYieldedValue = outsideBbArg || outsideOpResult;
77  if (matchPattern(yieldedValue, m_Constant(&constYieldedValue))) {
78  // Padding with a constant: Create linalg.fill.
79  Dialect *arithDialect =
80  rewriter.getContext()->getLoadedDialect<arith::ArithDialect>();
81  Value fillValue =
82  arithDialect
83  ->materializeConstant(rewriter, constYieldedValue,
84  yieldedValue.getType(), yieldedValue.getLoc())
85  ->getResult(0);
86  auto fillOp = rewriter.create<linalg::FillOp>(loc, ValueRange(fillValue),
87  ValueRange(dest));
88  return fillOp;
89  }
90 
91  if (invariantYieldedValue) {
92  // Padding with an invariant value.
93  auto fillOp = rewriter.create<linalg::FillOp>(loc, ValueRange(yieldedValue),
94  ValueRange(dest));
95  return fillOp;
96  }
97 
98  // Create linalg.generic.
99  SmallVector<utils::IteratorType> iteratorTypes(resultType.getRank(),
100  utils::IteratorType::parallel);
101  SmallVector<AffineMap> indexingMaps(
102  1, rewriter.getMultiDimIdentityMap(resultType.getRank()));
103  auto genericOp = rewriter.create<linalg::GenericOp>(
104  loc, resultType, /*inputs=*/ValueRange(),
105  /*outputs=*/ValueRange{dest}, /*indexingMaps=*/
106  indexingMaps, iteratorTypes);
107  Block *body = rewriter.createBlock(&genericOp->getRegion(0), {},
108  resultType.getElementType(), loc);
109  rewriter.setInsertionPointToStart(body);
110  SmallVector<Value> bbArgReplacements;
111  for (int64_t i = 0; i < resultType.getRank(); ++i)
112  bbArgReplacements.push_back(rewriter.create<linalg::IndexOp>(loc, i));
113  rewriter.mergeBlocks(padOp.getBody(), body, bbArgReplacements);
114 
115  // Update terminator.
116  auto yieldOp = cast<tensor::YieldOp>(body->getTerminator());
117  rewriter.replaceOpWithNewOp<linalg::YieldOp>(yieldOp, yieldOp.getValue());
118  return genericOp;
119 }

121 static SmallVector<Value> reifyOrComputeDynamicSizes(OpBuilder &b,
122  Value value) {
123  auto tensorType = value.getType().cast<RankedTensorType>();
124  if (tensorType.hasStaticShape())
125  return {};
126 
127  // Try to reify dynamic sizes.
128  ReifiedRankedShapedTypeDims reifiedShape;
129  if (value.isa<OpResult>() &&
130  succeeded(reifyResultShapes(b, value.getDefiningOp(), reifiedShape))) {
131  SmallVector<Value> dynSizes;
132  for (int64_t i = 0; i < tensorType.getRank(); ++i) {
133  if (tensorType.isDynamicDim(i))
134  dynSizes.push_back(
135  reifiedShape[value.cast<OpResult>().getResultNumber()][i]
136  .get<Value>());
137  }
138  return dynSizes;
139  }
140 
141  // Create tensor.dim ops.
142  SmallVector<Value> dynSizes;
143  for (int64_t i = 0; i < tensorType.getRank(); ++i) {
144  if (tensorType.isDynamicDim(i))
145  dynSizes.push_back(
146  b.create<DimOp>(value.getLoc(), value,
147  b.create<arith::ConstantIndexOp>(value.getLoc(), i)));
148  }
149  return dynSizes;
150 }

152 static Value createAllocationForTensor(RewriterBase &rewriter, Location loc,
153  Value value,
154  Attribute memorySpace = {}) {
155  OpBuilder::InsertionGuard g(rewriter);
156  auto tensorType = value.getType().cast<RankedTensorType>();
157 
158  // Create buffer allocation.
159  auto memrefType = bufferization::getMemRefTypeWithStaticIdentityLayout(
160  tensorType, memorySpace)
161  .cast<MemRefType>();
162  SmallVector<Value> dynamicSizes = reifyOrComputeDynamicSizes(rewriter, value);
163  Value alloc = rewriter.create<memref::AllocOp>(loc, memrefType, dynamicSizes);
164 
165  // Place deallocation at the end of the block.
166  rewriter.setInsertionPoint(rewriter.getInsertionBlock()->getTerminator());
167  rewriter.create<memref::DeallocOp>(loc, alloc);
168 
169  return alloc;
170 }

172 Value linalg::bufferizeToAllocation(RewriterBase &rewriter, PadOp padOp,
173  Attribute memorySpace) {
174  OpBuilder::InsertionGuard g(rewriter);
175  rewriter.setInsertionPoint(padOp);
176  Location loc = padOp.getLoc();
177 
178  // Create buffer allocation.
179  Value alloc =
180  createAllocationForTensor(rewriter, loc, padOp.getResult(), memorySpace);
181  rewriter.setInsertionPointAfter(alloc.getDefiningOp());
182 
183  // Create linalg.fill or linalg.generic.
184  Operation *fillOp = movePaddingToFillOrGenericOp(rewriter, loc, padOp, alloc);
185  rewriter.setInsertionPointAfter(fillOp);
186 
187  // Create memref.tensor_store.
188  SmallVector<OpFoldResult> sizes =
189  getMixedSizes(rewriter, loc, padOp.getSource());
190  SmallVector<OpFoldResult> strides(padOp.getResultType().getRank(),
191  rewriter.getIndexAttr(1));
192  Value subview = rewriter.create<memref::SubViewOp>(
193  loc, alloc, /*offsets=*/padOp.getMixedLowPad(), sizes, strides);
194  rewriter.create<memref::TensorStoreOp>(loc, padOp.getSource(), subview);
195 
196  // Create bufferization.to_tensor with "restrict" and "writable". The returned
197  // tensor is a new buffer allocation, so it does not alias with any buffer.
198  Value toTensorOp = rewriter.create<bufferization::ToTensorOp>(
199  loc, alloc, /*restrict=*/true, /*writable=*/true);
200  rewriter.replaceOp(padOp, toTensorOp);
201  return toTensorOp;
202 }

204 /// Lower tensor.from_elements to a sequence of chained tensor.insert.
205 FailureOr<Operation *> mlir::linalg::rewriteInDestinationPassingStyle(
206  RewriterBase &rewriter, tensor::FromElementsOp fromElementsOp) {
207  Location loc = fromElementsOp.getLoc();
208  RankedTensorType tensorType =
209  fromElementsOp.getType().cast<RankedTensorType>();
210  auto shape = tensorType.getShape();
211 
212  // Create tensor.empty.
213  auto emptyOp = rewriter.create<EmptyOp>(loc, tensorType, ValueRange());
214 
215  // Case: tensor<elem_type>.
216  if (shape.empty()) {
217  Operation *res = rewriter.replaceOpWithNewOp<tensor::InsertOp>(
218  fromElementsOp, fromElementsOp.getElements().front(),
219  emptyOp.getResult(), ValueRange());
220  return res;
221  }
222 
223  // Create constants for the range of possible indices [0, max{shape_i}).
224  auto maxDim = *std::max_element(shape.begin(), shape.end());
225  SmallVector<Value, 2> constants;
226  constants.reserve(maxDim);
227  for (int i = 0; i < maxDim; ++i)
228  constants.push_back(rewriter.create<arith::ConstantIndexOp>(loc, i));
229 
230  // Traverse all elements and create tensor.insert ops.
231  auto elementIt = fromElementsOp.getElements().begin();
232  SmallVector<Value, 2> indices(tensorType.getRank(), constants[0]);
233  Value result = createInserts(rewriter, loc, /*dim=*/0, emptyOp.getResult(),
234  shape, constants, elementIt, indices);
235 
236  // Replace tensor.from_elements.
237  rewriter.replaceOp(fromElementsOp, result);
238  return result.getDefiningOp();
239 }

241 /// Lower tensor.generate to linalg.generic.
242 FailureOr<Operation *>
243 mlir::linalg::rewriteInDestinationPassingStyle(RewriterBase &rewriter,
244  tensor::GenerateOp generateOp) {
245  // Only ops with exactly one block are supported.
246  if (!generateOp.getBody().hasOneBlock())
247  return failure();
248 
249  Location loc = generateOp.getLoc();
250  RankedTensorType tensorType = generateOp.getType().cast<RankedTensorType>();
251 
252  // Create tensor.empty.
253  auto emptyOp =
254  rewriter.create<EmptyOp>(loc, tensorType, generateOp.getDynamicExtents());
255 
256  // Create linalg.generic.
257  SmallVector<utils::IteratorType> iteratorTypes(tensorType.getRank(),
258  utils::IteratorType::parallel);
259  SmallVector<AffineMap> indexingMaps(
260  1, rewriter.getMultiDimIdentityMap(tensorType.getRank()));
261  auto genericOp = rewriter.create<linalg::GenericOp>(
262  loc, tensorType, /*inputs=*/ValueRange(),
263  /*outputs=*/ValueRange{emptyOp.getResult()}, /*indexingMaps=*/
264  indexingMaps, iteratorTypes);
265  Block *body = rewriter.createBlock(&genericOp->getRegion(0), {},
266  tensorType.getElementType(), loc);
267  rewriter.setInsertionPointToStart(body);
268  SmallVector<Value> bbArgReplacements;
269  for (int64_t i = 0; i < tensorType.getRank(); ++i)
270  bbArgReplacements.push_back(rewriter.create<linalg::IndexOp>(loc, i));
271  rewriter.mergeBlocks(&generateOp.getBody().front(), body, bbArgReplacements);
272 
273  // Update terminator.
274  auto yieldOp = cast<tensor::YieldOp>(body->getTerminator());
275  rewriter.replaceOpWithNewOp<linalg::YieldOp>(yieldOp, yieldOp.getValue());
276 
277  // Replace tensor.generate.
278  rewriter.replaceOp(generateOp, genericOp->getResult(0));
279  return genericOp.getOperation();
280 }

282 /// Lower tensor.pad to linalg.generic + tensor.insert_slice.
283 FailureOr<Operation *>
284 mlir::linalg::rewriteInDestinationPassingStyle(RewriterBase &rewriter,
285  tensor::PadOp padOp) {
286  // Only ops with exactly one block are supported.
287  if (!padOp.getBodyRegion().hasOneBlock())
288  return failure();
289 
290  // Create tensor.empty.
291  Location loc = padOp.getLoc();
292  RankedTensorType resultType = padOp.getResultType();
293  ReifiedRankedShapedTypeDims reifiedShape;
294  if (failed(reifyResultShapes(rewriter, padOp, reifiedShape)))
295  return rewriter.notifyMatchFailure(
296  padOp, "failed to reify tensor.pad op result shape");
297  SmallVector<Value> dynamicSizes;
298  for (int64_t i = 0; i < resultType.getRank(); ++i)
299  if (resultType.isDynamicDim(i))
300  dynamicSizes.push_back(reifiedShape[0][i].get<Value>());
301 
302  // If the `padOp` has a nofold attribute and all paddings are known to be 0,
303  // explicitly insert a `linalg.copy`.
304  if (padOp.getNofoldAttr() &&
305  llvm::all_of(padOp.getMixedLowPad(), isZeroIndex) &&
306  llvm::all_of(padOp.getMixedHighPad(), isZeroIndex)) {
307  using bufferization::AllocTensorOp;
308  Value allocated =
309  rewriter.create<AllocTensorOp>(loc, resultType, dynamicSizes);
310  auto copyOp = rewriter.replaceOpWithNewOp<linalg::CopyOp>(
311  padOp, padOp.getSource(), allocated);
312  return copyOp.getOperation();
313  }
314 
315  Value empty = rewriter.create<EmptyOp>(loc, resultType, dynamicSizes);
316  // Create linalg.fill or linalg.generic.
317  Operation *fillOp = movePaddingToFillOrGenericOp(rewriter, loc, padOp, empty);
318  rewriter.setInsertionPointAfter(fillOp);
319 
320  // Create tensor::InsertSliceOp.
321  SmallVector<OpFoldResult> sliceSizes =
322  getMixedSizes(rewriter, loc, padOp.getSource());
323  SmallVector<OpFoldResult> sliceStrides(resultType.getRank(),
324  rewriter.getIndexAttr(1));
325  auto insertSliceOp = rewriter.replaceOpWithNewOp<tensor::InsertSliceOp>(
326  padOp, padOp.getSource(), fillOp->getResult(0),
327  /*offsets=*/padOp.getMixedLowPad(), sliceSizes, sliceStrides);
328  return insertSliceOp.getOperation();
329 }

331 Value linalg::bufferizeToAllocation(RewriterBase &rewriter, Value value,
332  Attribute memorySpace) {
333  // Call specialized overload for certain ops.
334  if (auto padOp = value.getDefiningOp<PadOp>())
335  return bufferizeToAllocation(rewriter, padOp, memorySpace);
336 
337  // Collect all uses.
338  SmallVector<OpOperand *> uses = llvm::to_vector(
339  llvm::map_range(value.getUses(), [](OpOperand &use) { return &use; }));
340 
341  OpBuilder::InsertionGuard g(rewriter);
342  if (auto bbArg = value.dyn_cast<BlockArgument>()) {
343  rewriter.setInsertionPointToStart(bbArg.getOwner());
344  } else {
345  rewriter.setInsertionPointAfter(value.getDefiningOp());
346  }
347  Location loc = value.getLoc();
348 
349  // Create buffer allocation.
350  Value alloc = createAllocationForTensor(rewriter, loc, value, memorySpace);
351 
352  // Create memref.tensor_store.
353  rewriter.setInsertionPointAfter(alloc.getDefiningOp());
354  rewriter.create<memref::TensorStoreOp>(loc, value, alloc);
355 
356  // Create bufferization.to_tensor with "restrict" and "writable". The returned
357  // tensor is a new buffer allocation, so it does not alias with any buffer.
358  Value toTensorOp = rewriter.create<bufferization::ToTensorOp>(
359  loc, alloc, /*restrict=*/true, /*writable=*/true);
360  for (OpOperand *use : uses) {
361  rewriter.updateRootInPlace(use->getOwner(),
362  [&]() { use->set(toTensorOp); });
363  }
364 
365  return toTensorOp;
366 }
