155 // Similar to `composeMap` except that no Values need be associated with the
156 // constraint system nor are they looked at -- the dimensions and symbols of
157 // `other` are expected to correspond 1:1 to `this` system.
158 LogicalResult FlatLinearConstraints::composeMatchingMap(AffineMap other) {
159  assert(other.getNumDims() == getNumDimVars() && "dim mismatch");
160  assert(other.getNumSymbols() == getNumSymbolVars() && "symbol mismatch");
161 
162  std::vector<SmallVector<int64_t, 8>> flatExprs;
163  if (failed(flattenAlignedMapAndMergeLocals(other, &flatExprs)))
164  return failure();
165  assert(flatExprs.size() == other.getNumResults());
166 
167  // Add dimensions corresponding to the map's results.
168  insertDimVar(/*pos=*/0, /*num=*/other.getNumResults());
169 
170  // We add one equality for each result connecting the result dim of the map to
171  // the other variables.
172  // E.g.: if the expression is 16*i0 + i1, and this is the r^th
173  // iteration/result of the value map, we are adding the equality:
174  // d_r - 16*i0 - i1 = 0. Similarly, when flattening (i0 + 1, i0 + 8*i2), we
175  // add two equalities: d_0 - i0 - 1 == 0, d1 - i0 - 8*i2 == 0.
176  for (unsigned r = 0, e = flatExprs.size(); r < e; r++) {
177  const auto &flatExpr = flatExprs[r];
178  assert(flatExpr.size() >= other.getNumInputs() + 1);
179 
180  SmallVector<int64_t, 8> eqToAdd(getNumCols(), 0);
181  // Set the coefficient for this result to one.
182  eqToAdd[r] = 1;
183 
184  // Dims and symbols.
185  for (unsigned i = 0, f = other.getNumInputs(); i < f; i++) {
186  // Negate `eq[r]` since the newly added dimension will be set to this one.
187  eqToAdd[e + i] = -flatExpr[i];
188  }
189  // Local columns of `eq` are at the beginning.
190  unsigned j = getNumDimVars() + getNumSymbolVars();
191  unsigned end = flatExpr.size() - 1;
192  for (unsigned i = other.getNumInputs(); i < end; i++, j++) {
193  eqToAdd[j] = -flatExpr[i];
194  }
195 
196  // Constant term.
197  eqToAdd[getNumCols() - 1] = -flatExpr[flatExpr.size() - 1];
198 
199  // Add the equality connecting the result of the map to this constraint set.
200  addEquality(eqToAdd);
201  }
202 
203  return success();
204 }

206 // Determine whether the variable at 'pos' (say var_r) can be expressed as
207 // modulo of another known variable (say var_n) w.r.t a constant. For example,
208 // if the following constraints hold true:
209 // ```
210 // 0 <= var_r <= divisor - 1
211 // var_n - (divisor * q_expr) = var_r
212 // ```
213 // where `var_n` is a known variable (called dividend), and `q_expr` is an
214 // `AffineExpr` (called the quotient expression), `var_r` can be written as:
215 //
216 // `var_r = var_n mod divisor`.
217 //
218 // Additionally, in a special case of the above constaints where `q_expr` is an
219 // variable itself that is not yet known (say `var_q`), it can be written as a
220 // floordiv in the following way:
221 //
222 // `var_q = var_n floordiv divisor`.
223 //
224 // Returns true if the above mod or floordiv are detected, updating 'memo' with
225 // these new expressions. Returns false otherwise.
226 static bool detectAsMod(const FlatLinearConstraints &cst, unsigned pos,
227  int64_t lbConst, int64_t ubConst,
228  SmallVectorImpl<AffineExpr> &memo,
229  MLIRContext *context) {
230  assert(pos < cst.getNumVars() && "invalid position");
231 
232  // Check if a divisor satisfying the condition `0 <= var_r <= divisor - 1` can
233  // be determined.
234  if (lbConst != 0 || ubConst < 1)
235  return false;
236  int64_t divisor = ubConst + 1;
237 
238  // Check for the aforementioned conditions in each equality.
239  for (unsigned curEquality = 0, numEqualities = cst.getNumEqualities();
240  curEquality < numEqualities; curEquality++) {
241  int64_t coefficientAtPos = cst.atEq64(curEquality, pos);
242  // If current equality does not involve `var_r`, continue to the next
243  // equality.
244  if (coefficientAtPos == 0)
245  continue;
246 
247  // Constant term should be 0 in this equality.
248  if (cst.atEq64(curEquality, cst.getNumCols() - 1) != 0)
249  continue;
250 
251  // Traverse through the equality and construct the dividend expression
252  // `dividendExpr`, to contain all the variables which are known and are
253  // not divisible by `(coefficientAtPos * divisor)`. Hope here is that the
254  // `dividendExpr` gets simplified into a single variable `var_n` discussed
255  // above.
256  auto dividendExpr = getAffineConstantExpr(0, context);
257 
258  // Track the terms that go into quotient expression, later used to detect
259  // additional floordiv.
260  unsigned quotientCount = 0;
261  int quotientPosition = -1;
262  int quotientSign = 1;
263 
264  // Consider each term in the current equality.
265  unsigned curVar, e;
266  for (curVar = 0, e = cst.getNumDimAndSymbolVars(); curVar < e; ++curVar) {
267  // Ignore var_r.
268  if (curVar == pos)
269  continue;
270  int64_t coefficientOfCurVar = cst.atEq64(curEquality, curVar);
271  // Ignore vars that do not contribute to the current equality.
272  if (coefficientOfCurVar == 0)
273  continue;
274  // Check if the current var goes into the quotient expression.
275  if (coefficientOfCurVar % (divisor * coefficientAtPos) == 0) {
276  quotientCount++;
277  quotientPosition = curVar;
278  quotientSign = (coefficientOfCurVar * coefficientAtPos) > 0 ? 1 : -1;
279  continue;
280  }
281  // Variables that are part of dividendExpr should be known.
282  if (!memo[curVar])
283  break;
284  // Append the current variable to the dividend expression.
285  dividendExpr = dividendExpr + memo[curVar] * coefficientOfCurVar;
286  }
287 
288  // Can't construct expression as it depends on a yet uncomputed var.
289  if (curVar < e)
290  continue;
291 
292  // Express `var_r` in terms of the other vars collected so far.
293  if (coefficientAtPos > 0)
294  dividendExpr = (-dividendExpr).floorDiv(coefficientAtPos);
295  else
296  dividendExpr = dividendExpr.floorDiv(-coefficientAtPos);
297 
298  // Simplify the expression.
299  dividendExpr = simplifyAffineExpr(dividendExpr, cst.getNumDimVars(),
300  cst.getNumSymbolVars());
301  // Only if the final dividend expression is just a single var (which we call
302  // `var_n`), we can proceed.
303  // TODO: Handle AffineSymbolExpr as well. There is no reason to restrict it
304  // to dims themselves.
305  auto dimExpr = dividendExpr.dyn_cast<AffineDimExpr>();
306  if (!dimExpr)
307  continue;
308 
309  // Express `var_r` as `var_n % divisor` and store the expression in `memo`.
310  if (quotientCount >= 1) {
311  auto ub = cst.getConstantBound64(BoundType::UB, dimExpr.getPosition());
312  // If `var_n` has an upperbound that is less than the divisor, mod can be
313  // eliminated altogether.
314  if (ub && *ub < divisor)
315  memo[pos] = dimExpr;
316  else
317  memo[pos] = dimExpr % divisor;
318  // If a unique quotient `var_q` was seen, it can be expressed as
319  // `var_n floordiv divisor`.
320  if (quotientCount == 1 && !memo[quotientPosition])
321  memo[quotientPosition] = dimExpr.floorDiv(divisor) * quotientSign;
322 
323  return true;
324  }
325  }
326  return false;
327 }

329 /// Check if the pos^th variable can be expressed as a floordiv of an affine
330 /// function of other variables (where the divisor is a positive constant)
331 /// given the initial set of expressions in `exprs`. If it can be, the
332 /// corresponding position in `exprs` is set as the detected affine expr. For
333 /// eg: 4q <= i + j <= 4q + 3 <=> q = (i + j) floordiv 4. An equality can
334 /// also yield a floordiv: eg. 4q = i + j <=> q = (i + j) floordiv 4. 32q + 28
335 /// <= i <= 32q + 31 => q = i floordiv 32.
336 static bool detectAsFloorDiv(const FlatLinearConstraints &cst, unsigned pos,
337  MLIRContext *context,
338  SmallVectorImpl<AffineExpr> &exprs) {
339  assert(pos < cst.getNumVars() && "invalid position");
340 
341  // Get upper-lower bound pair for this variable.
342  SmallVector<bool, 8> foundRepr(cst.getNumVars(), false);
343  for (unsigned i = 0, e = cst.getNumVars(); i < e; ++i)
344  if (exprs[i])
345  foundRepr[i] = true;
346 
347  SmallVector<int64_t, 8> dividend(cst.getNumCols());
348  unsigned divisor;
349  auto ulPair = computeSingleVarRepr(cst, foundRepr, pos, dividend, divisor);
350 
351  // No upper-lower bound pair found for this var.
352  if (ulPair.kind == ReprKind::None || ulPair.kind == ReprKind::Equality)
353  return false;
354 
355  // Construct the dividend expression.
356  auto dividendExpr = getAffineConstantExpr(dividend.back(), context);
357  for (unsigned c = 0, f = cst.getNumVars(); c < f; c++)
358  if (dividend[c] != 0)
359  dividendExpr = dividendExpr + dividend[c] * exprs[c];
360 
361  // Successfully detected the floordiv.
362  exprs[pos] = dividendExpr.floorDiv(divisor);
363  return true;
364 }

366 std::pair<AffineMap, AffineMap> FlatLinearConstraints::getLowerAndUpperBound(
367  unsigned pos, unsigned offset, unsigned num, unsigned symStartPos,
368  ArrayRef<AffineExpr> localExprs, MLIRContext *context,
369  bool closedUB) const {
370  assert(pos + offset < getNumDimVars() && "invalid dim start pos");
371  assert(symStartPos >= (pos + offset) && "invalid sym start pos");
372  assert(getNumLocalVars() == localExprs.size() &&
373  "incorrect local exprs count");
374 
375  SmallVector<unsigned, 4> lbIndices, ubIndices, eqIndices;
376  getLowerAndUpperBoundIndices(pos + offset, &lbIndices, &ubIndices, &eqIndices,
377  offset, num);
378 
379  /// Add to 'b' from 'a' in set [0, offset) U [offset + num, symbStartPos).
380  auto addCoeffs = [&](ArrayRef<int64_t> a, SmallVectorImpl<int64_t> &b) {
381  b.clear();
382  for (unsigned i = 0, e = a.size(); i < e; ++i) {
383  if (i < offset || i >= offset + num)
384  b.push_back(a[i]);
385  }
386  };
387 
388  SmallVector<int64_t, 8> lb, ub;
389  SmallVector<AffineExpr, 4> lbExprs;
390  unsigned dimCount = symStartPos - num;
391  unsigned symCount = getNumDimAndSymbolVars() - symStartPos;
392  lbExprs.reserve(lbIndices.size() + eqIndices.size());
393  // Lower bound expressions.
394  for (auto idx : lbIndices) {
395  auto ineq = getInequality64(idx);
396  // Extract the lower bound (in terms of other coeff's + const), i.e., if
397  // i - j + 1 >= 0 is the constraint, 'pos' is for i the lower bound is j
398  // - 1.
399  addCoeffs(ineq, lb);
400  std::transform(lb.begin(), lb.end(), lb.begin(), std::negate<int64_t>());
401  auto expr =
402  getAffineExprFromFlatForm(lb, dimCount, symCount, localExprs, context);
403  // expr ceildiv divisor is (expr + divisor - 1) floordiv divisor
404  int64_t divisor = std::abs(ineq[pos + offset]);
405  expr = (expr + divisor - 1).floorDiv(divisor);
406  lbExprs.push_back(expr);
407  }
408 
409  SmallVector<AffineExpr, 4> ubExprs;
410  ubExprs.reserve(ubIndices.size() + eqIndices.size());
411  // Upper bound expressions.
412  for (auto idx : ubIndices) {
413  auto ineq = getInequality64(idx);
414  // Extract the upper bound (in terms of other coeff's + const).
415  addCoeffs(ineq, ub);
416  auto expr =
417  getAffineExprFromFlatForm(ub, dimCount, symCount, localExprs, context);
418  expr = expr.floorDiv(std::abs(ineq[pos + offset]));
419  int64_t ubAdjustment = closedUB ? 0 : 1;
420  ubExprs.push_back(expr + ubAdjustment);
421  }
422 
423  // Equalities. It's both a lower and a upper bound.
424  SmallVector<int64_t, 4> b;
425  for (auto idx : eqIndices) {
426  auto eq = getEquality64(idx);
427  addCoeffs(eq, b);
428  if (eq[pos + offset] > 0)
429  std::transform(b.begin(), b.end(), b.begin(), std::negate<int64_t>());
430 
431  // Extract the upper bound (in terms of other coeff's + const).
432  auto expr =
433  getAffineExprFromFlatForm(b, dimCount, symCount, localExprs, context);
434  expr = expr.floorDiv(std::abs(eq[pos + offset]));
435  // Upper bound is exclusive.
436  ubExprs.push_back(expr + 1);
437  // Lower bound.
438  expr =
439  getAffineExprFromFlatForm(b, dimCount, symCount, localExprs, context);
440  expr = expr.ceilDiv(std::abs(eq[pos + offset]));
441  lbExprs.push_back(expr);
442  }
443 
444  auto lbMap = AffineMap::get(dimCount, symCount, lbExprs, context);
445  auto ubMap = AffineMap::get(dimCount, symCount, ubExprs, context);
446 
447  return {lbMap, ubMap};
448 }

450 /// Computes the lower and upper bounds of the first 'num' dimensional
451 /// variables (starting at 'offset') as affine maps of the remaining
452 /// variables (dimensional and symbolic variables). Local variables are
453 /// themselves explicitly computed as affine functions of other variables in
454 /// this process if needed.
455 void FlatLinearConstraints::getSliceBounds(unsigned offset, unsigned num,
456  MLIRContext *context,
457  SmallVectorImpl<AffineMap> *lbMaps,
458  SmallVectorImpl<AffineMap> *ubMaps,
459  bool closedUB) {
460  assert(offset + num <= getNumDimVars() && "invalid range");
461 
462  // Basic simplification.
463  normalizeConstraintsByGCD();
464 
465  LLVM_DEBUG(llvm::dbgs() << "getSliceBounds for first " << num
466  << " variables\n");
467  LLVM_DEBUG(dump());
468 
469  // Record computed/detected variables.
470  SmallVector<AffineExpr, 8> memo(getNumVars());
471  // Initialize dimensional and symbolic variables.
472  for (unsigned i = 0, e = getNumDimVars(); i < e; i++) {
473  if (i < offset)
474  memo[i] = getAffineDimExpr(i, context);
475  else if (i >= offset + num)
476  memo[i] = getAffineDimExpr(i - num, context);
477  }
478  for (unsigned i = getNumDimVars(), e = getNumDimAndSymbolVars(); i < e; i++)
479  memo[i] = getAffineSymbolExpr(i - getNumDimVars(), context);
480 
481  bool changed;
482  do {
483  changed = false;
484  // Identify yet unknown variables as constants or mod's / floordiv's of
485  // other variables if possible.
486  for (unsigned pos = 0; pos < getNumVars(); pos++) {
487  if (memo[pos])
488  continue;
489 
490  auto lbConst = getConstantBound64(BoundType::LB, pos);
491  auto ubConst = getConstantBound64(BoundType::UB, pos);
492  if (lbConst.has_value() && ubConst.has_value()) {
493  // Detect equality to a constant.
494  if (*lbConst == *ubConst) {
495  memo[pos] = getAffineConstantExpr(*lbConst, context);
496  changed = true;
497  continue;
498  }
499 
500  // Detect a variable as modulo of another variable w.r.t a
501  // constant.
502  if (detectAsMod(*this, pos, *lbConst, *ubConst, memo, context)) {
503  changed = true;
504  continue;
505  }
506  }
507 
508  // Detect a variable as a floordiv of an affine function of other
509  // variables (divisor is a positive constant).
510  if (detectAsFloorDiv(*this, pos, context, memo)) {
511  changed = true;
512  continue;
513  }
514 
515  // Detect a variable as an expression of other variables.
516  unsigned idx;
517  if (!findConstraintWithNonZeroAt(pos, /*isEq=*/true, &idx)) {
518  continue;
519  }
520 
521  // Build AffineExpr solving for variable 'pos' in terms of all others.
522  auto expr = getAffineConstantExpr(0, context);
523  unsigned j, e;
524  for (j = 0, e = getNumVars(); j < e; ++j) {
525  if (j == pos)
526  continue;
527  int64_t c = atEq64(idx, j);
528  if (c == 0)
529  continue;
530  // If any of the involved IDs hasn't been found yet, we can't proceed.
531  if (!memo[j])
532  break;
533  expr = expr + memo[j] * c;
534  }
535  if (j < e)
536  // Can't construct expression as it depends on a yet uncomputed
537  // variable.
538  continue;
539 
540  // Add constant term to AffineExpr.
541  expr = expr + atEq64(idx, getNumVars());
542  int64_t vPos = atEq64(idx, pos);
543  assert(vPos != 0 && "expected non-zero here");
544  if (vPos > 0)
545  expr = (-expr).floorDiv(vPos);
546  else
547  // vPos < 0.
548  expr = expr.floorDiv(-vPos);
549  // Successfully constructed expression.
550  memo[pos] = expr;
551  changed = true;
552  }
553  // This loop is guaranteed to reach a fixed point - since once an
554  // variable's explicit form is computed (in memo[pos]), it's not updated
555  // again.
556  } while (changed);
557 
558  int64_t ubAdjustment = closedUB ? 0 : 1;
559 
560  // Set the lower and upper bound maps for all the variables that were
561  // computed as affine expressions of the rest as the "detected expr" and
562  // "detected expr + 1" respectively; set the undetected ones to null.
563  std::optional<FlatLinearConstraints> tmpClone;
564  for (unsigned pos = 0; pos < num; pos++) {
565  unsigned numMapDims = getNumDimVars() - num;
566  unsigned numMapSymbols = getNumSymbolVars();
567  AffineExpr expr = memo[pos + offset];
568  if (expr)
569  expr = simplifyAffineExpr(expr, numMapDims, numMapSymbols);
570 
571  AffineMap &lbMap = (*lbMaps)[pos];
572  AffineMap &ubMap = (*ubMaps)[pos];
573 
574  if (expr) {
575  lbMap = AffineMap::get(numMapDims, numMapSymbols, expr);
576  ubMap = AffineMap::get(numMapDims, numMapSymbols, expr + ubAdjustment);
577  } else {
578  // TODO: Whenever there are local variables in the dependence
579  // constraints, we'll conservatively over-approximate, since we don't
580  // always explicitly compute them above (in the while loop).
581  if (getNumLocalVars() == 0) {
582  // Work on a copy so that we don't update this constraint system.
583  if (!tmpClone) {
584  tmpClone.emplace(FlatLinearConstraints(*this));
585  // Removing redundant inequalities is necessary so that we don't get
586  // redundant loop bounds.
587  tmpClone->removeRedundantInequalities();
588  }
589  std::tie(lbMap, ubMap) = tmpClone->getLowerAndUpperBound(
590  pos, offset, num, getNumDimVars(), /*localExprs=*/{}, context,
591  closedUB);
592  }
593 
594  // If the above fails, we'll just use the constant lower bound and the
595  // constant upper bound (if they exist) as the slice bounds.
596  // TODO: being conservative for the moment in cases that
597  // lead to multiple bounds - until getConstDifference in LoopFusion.cpp is
598  // fixed (b/126426796).
599  if (!lbMap || lbMap.getNumResults() > 1) {
600  LLVM_DEBUG(llvm::dbgs()
601  << "WARNING: Potentially over-approximating slice lb\n");
602  auto lbConst = getConstantBound64(BoundType::LB, pos + offset);
603  if (lbConst.has_value()) {
604  lbMap = AffineMap::get(numMapDims, numMapSymbols,
605  getAffineConstantExpr(*lbConst, context));
606  }
607  }
608  if (!ubMap || ubMap.getNumResults() > 1) {
609  LLVM_DEBUG(llvm::dbgs()
610  << "WARNING: Potentially over-approximating slice ub\n");
611  auto ubConst = getConstantBound64(BoundType::UB, pos + offset);
612  if (ubConst.has_value()) {
613  ubMap = AffineMap::get(
614  numMapDims, numMapSymbols,
615  getAffineConstantExpr(*ubConst + ubAdjustment, context));
616  }
617  }
618  }
619  LLVM_DEBUG(llvm::dbgs()
620  << "lb map for pos = " << Twine(pos + offset) << ", expr: ");
621  LLVM_DEBUG(lbMap.dump(););
622  LLVM_DEBUG(llvm::dbgs()
623  << "ub map for pos = " << Twine(pos + offset) << ", expr: ");
624  LLVM_DEBUG(ubMap.dump(););

652 LogicalResult FlatLinearConstraints::addBound(BoundType type, unsigned pos,
653  AffineMap boundMap,
654  bool isClosedBound) {
655  assert(boundMap.getNumDims() == getNumDimVars() && "dim mismatch");
656  assert(boundMap.getNumSymbols() == getNumSymbolVars() && "symbol mismatch");
657  assert(pos < getNumDimAndSymbolVars() && "invalid position");
658  assert((type != BoundType::EQ || isClosedBound) &&
659  "EQ bound must be closed.");
660 
661  // Equality follows the logic of lower bound except that we add an equality
662  // instead of an inequality.
663  assert((type != BoundType::EQ || boundMap.getNumResults() == 1) &&
664  "single result expected");
665  bool lower = type == BoundType::LB || type == BoundType::EQ;
666 
667  std::vector<SmallVector<int64_t, 8>> flatExprs;
668  if (failed(flattenAlignedMapAndMergeLocals(boundMap, &flatExprs)))
669  return failure();
670  assert(flatExprs.size() == boundMap.getNumResults());
671 
672  // Add one (in)equality for each result.
673  for (const auto &flatExpr : flatExprs) {
674  SmallVector<int64_t> ineq(getNumCols(), 0);
675  // Dims and symbols.
676  for (unsigned j = 0, e = boundMap.getNumInputs(); j < e; j++) {
677  ineq[j] = lower ? -flatExpr[j] : flatExpr[j];
678  }
679  // Invalid bound: pos appears in `boundMap`.
680  // TODO: This should be an assertion. Fix `addDomainFromSliceMaps` and/or
681  // its callers to prevent invalid bounds from being added.
682  if (ineq[pos] != 0)
683  continue;
684  ineq[pos] = lower ? 1 : -1;
685  // Local columns of `ineq` are at the beginning.
686  unsigned j = getNumDimVars() + getNumSymbolVars();
687  unsigned end = flatExpr.size() - 1;
688  for (unsigned i = boundMap.getNumInputs(); i < end; i++, j++) {
689  ineq[j] = lower ? -flatExpr[i] : flatExpr[i];
690  }
691  // Make the bound closed in if flatExpr is open. The inequality is always
692  // created in the upper bound form, so the adjustment is -1.
693  int64_t boundAdjustment = (isClosedBound || type == BoundType::EQ) ? 0 : -1;
694  // Constant term.
695  ineq[getNumCols() - 1] = (lower ? -flatExpr[flatExpr.size() - 1]
696  : flatExpr[flatExpr.size() - 1]) +
697  boundAdjustment;
698  type == BoundType::EQ ? addEquality(ineq) : addInequality(ineq);
699  }
700 
701  return success();
702 }

709 /// Compute an explicit representation for local vars. For all systems coming
710 /// from MLIR integer sets, maps, or expressions where local vars were
711 /// introduced to model floordivs and mods, this always succeeds.
712 LogicalResult
713 FlatLinearConstraints::computeLocalVars(SmallVectorImpl<AffineExpr> &memo,
714  MLIRContext *context) const {
715  unsigned numDims = getNumDimVars();
716  unsigned numSyms = getNumSymbolVars();
717 
718  // Initialize dimensional and symbolic variables.
719  for (unsigned i = 0; i < numDims; i++)
720  memo[i] = getAffineDimExpr(i, context);
721  for (unsigned i = numDims, e = numDims + numSyms; i < e; i++)
722  memo[i] = getAffineSymbolExpr(i - numDims, context);
723 
724  bool changed;
725  do {
726  // Each time `changed` is true at the end of this iteration, one or more
727  // local vars would have been detected as floordivs and set in memo; so the
728  // number of null entries in memo[...] strictly reduces; so this converges.
729  changed = false;
730  for (unsigned i = 0, e = getNumLocalVars(); i < e; ++i)
731  if (!memo[numDims + numSyms + i] &&
732  detectAsFloorDiv(*this, /*pos=*/numDims + numSyms + i, context, memo))
733  changed = true;
734  } while (changed);
735 
736  ArrayRef<AffineExpr> localExprs =
737  ArrayRef<AffineExpr>(memo).take_back(getNumLocalVars());
738  return success(
739  llvm::all_of(localExprs, [](AffineExpr expr) { return expr; }));

838 // Construct a hyperrectangular constraint set from ValueRanges that represent
839 // induction variables, lower and upper bounds. `ivs`, `lbs` and `ubs` are
840 // expected to match one to one. The order of variables and constraints is:
841 //
842 // ivs | lbs | ubs | eq/ineq
843 // ----+-----+-----+---------
844 // 1 -1 0 >= 0
845 // ----+-----+-----+---------
846 // -1 0 1 >= 0
847 //
848 // All dimensions as set as VarKind::SetDim.
849 FlatLinearValueConstraints
850 FlatLinearValueConstraints::getHyperrectangular(ValueRange ivs, ValueRange lbs,
851  ValueRange ubs) {
852  FlatLinearValueConstraints res;
853  unsigned nIvs = ivs.size();
854  assert(nIvs == lbs.size() && "expected as many lower bounds as ivs");
855  assert(nIvs == ubs.size() && "expected as many upper bounds as ivs");
856 
857  if (nIvs == 0)
858  return res;
859 
860  res.appendDimVar(ivs);
861  unsigned lbsStart = res.appendDimVar(lbs);
862  unsigned ubsStart = res.appendDimVar(ubs);
863 
864  MLIRContext *ctx = ivs.front().getContext();
865  for (int ivIdx = 0, e = nIvs; ivIdx < e; ++ivIdx) {
866  // iv - lb >= 0
867  AffineMap lb = AffineMap::get(/*dimCount=*/3 * nIvs, /*symbolCount=*/0,
868  getAffineDimExpr(lbsStart + ivIdx, ctx));
869  if (failed(res.addBound(BoundType::LB, ivIdx, lb)))
870  llvm_unreachable("Unexpected FlatLinearValueConstraints creation error");
871  // -iv + ub >= 0
872  AffineMap ub = AffineMap::get(/*dimCount=*/3 * nIvs, /*symbolCount=*/0,
873  getAffineDimExpr(ubsStart + ivIdx, ctx));
874  if (failed(res.addBound(BoundType::UB, ivIdx, ub)))
875  llvm_unreachable("Unexpected FlatLinearValueConstraints creation error");
876  }
877  return res;
878 }

992 /// Merge and align the variables of A and B starting at 'offset', so that
993 /// both constraint systems get the union of the contained variables that is
994 /// dimension-wise and symbol-wise unique; both constraint systems are updated
995 /// so that they have the union of all variables, with A's original
996 /// variables appearing first followed by any of B's variables that didn't
997 /// appear in A. Local variables in B that have the same division
998 /// representation as local variables in A are merged into one.
999 // E.g.: Input: A has ((%i, %j) [%M, %N]) and B has (%k, %j) [%P, %N, %M])
1000 // Output: both A, B have (%i, %j, %k) [%M, %N, %P]
1001 static void mergeAndAlignVars(unsigned offset, FlatLinearValueConstraints *a,
1002  FlatLinearValueConstraints *b) {
1003  assert(offset <= a->getNumDimVars() && offset <= b->getNumDimVars());
1004  // A merge/align isn't meaningful if a cst's vars aren't distinct.
1005  assert(areVarsUnique(*a) && "A's values aren't unique");
1006  assert(areVarsUnique(*b) && "B's values aren't unique");
1007 
1008  assert(llvm::all_of(
1009  llvm::drop_begin(a->getMaybeValues(), offset),
1010  [](const std::optional<Value> &var) { return var.has_value(); }));
1011 
1012  assert(llvm::all_of(
1013  llvm::drop_begin(b->getMaybeValues(), offset),
1014  [](const std::optional<Value> &var) { return var.has_value(); }));
1015 
1016  SmallVector<Value, 4> aDimValues;
1017  a->getValues(offset, a->getNumDimVars(), &aDimValues);
1018 
1019  {
1020  // Merge dims from A into B.
1021  unsigned d = offset;
1022  for (auto aDimValue : aDimValues) {
1023  unsigned loc;
1024  if (b->findVar(aDimValue, &loc)) {
1025  assert(loc >= offset && "A's dim appears in B's aligned range");
1026  assert(loc < b->getNumDimVars() &&
1027  "A's dim appears in B's non-dim position");
1028  b->swapVar(d, loc);
1029  } else {
1030  b->insertDimVar(d, aDimValue);
1031  }
1032  d++;
1033  }
1034  // Dimensions that are in B, but not in A, are added at the end.
1035  for (unsigned t = a->getNumDimVars(), e = b->getNumDimVars(); t < e; t++) {
1036  a->appendDimVar(b->getValue(t));
1037  }
1038  assert(a->getNumDimVars() == b->getNumDimVars() &&
1039  "expected same number of dims");
1040  }
1041 
1042  // Merge and align symbols of A and B
1043  a->mergeSymbolVars(*b);
1044  // Merge and align locals of A and B
1045  a->mergeLocalVars(*b);
1046 
1047  assert(areVarsAligned(*a, *b) && "IDs expected to be aligned");
1048 }

1056 /// Merge and align symbols of `this` and `other` such that both get union of
1057 /// of symbols that are unique. Symbols in `this` and `other` should be
1058 /// unique. Symbols with Value as `None` are considered to be inequal to all
1059 /// other symbols.
1060 void FlatLinearValueConstraints::mergeSymbolVars(
1061  FlatLinearValueConstraints &other) {
1062 
1063  assert(areVarsUnique(*this, VarKind::Symbol) && "Symbol vars are not unique");
1064  assert(areVarsUnique(other, VarKind::Symbol) && "Symbol vars are not unique");
1065 
1066  SmallVector<Value, 4> aSymValues;
1067  getValues(getNumDimVars(), getNumDimAndSymbolVars(), &aSymValues);
1068 
1069  // Merge symbols: merge symbols into `other` first from `this`.
1070  unsigned s = other.getNumDimVars();
1071  for (Value aSymValue : aSymValues) {
1072  unsigned loc;
1073  // If the var is a symbol in `other`, then align it, otherwise assume that
1074  // it is a new symbol
1075  if (other.findVar(aSymValue, &loc) && loc >= other.getNumDimVars() &&
1076  loc < other.getNumDimAndSymbolVars())
1077  other.swapVar(s, loc);
1078  else
1079  other.insertSymbolVar(s - other.getNumDimVars(), aSymValue);
1080  s++;
1081  }
1082 
1083  // Symbols that are in other, but not in this, are added at the end.
1084  for (unsigned t = other.getNumDimVars() + getNumSymbolVars(),
1085  e = other.getNumDimAndSymbolVars();
1086  t < e; t++)
1087  insertSymbolVar(getNumSymbolVars(), other.getValue(t));
1088 
1089  assert(getNumSymbolVars() == other.getNumSymbolVars() &&
1090  "expected same number of symbols");
1091  assert(areVarsUnique(*this, VarKind::Symbol) && "Symbol vars are not unique");
1092  assert(areVarsUnique(other, VarKind::Symbol) && "Symbol vars are not unique");
1093 }

1273  ValueRange dims, ValueRange syms,

1283  newSyms->clear();

1285  }

1297  builder.getAffineSymbolExpr(std::distance(syms.begin(), symIt));

1300  // symbol.
