26 void DebugImporter::translate(llvm::Function *func, LLVMFuncOp funcOp) {
27  if (!func->getSubprogram())
28  return;
29 
30  // Add a fused location to link the subprogram information.
31  StringAttr name = StringAttr::get(context, func->getSubprogram()->getName());
32  funcOp->setLoc(FusedLocWith<DISubprogramAttr>::get(
33  {NameLoc::get(name)}, translate(func->getSubprogram()), context));
34 }

54 DICompositeTypeAttr DebugImporter::translateImpl(llvm::DICompositeType *node) {
55  std::optional<DIFlags> flags = symbolizeDIFlags(node->getFlags());
56  SmallVector<DINodeAttr> elements;
57  for (llvm::DINode *element : node->getElements()) {
58  assert(element && "expected a non-null element type");
59  elements.push_back(translate(element));
60  }
61  // Drop the elements parameter if a cyclic dependency is detected. We
62  // currently cannot model these cycles and thus drop the parameter if
63  // required. A cyclic dependency is detected if one of the element nodes
64  // translates to a nullptr since the node is already on the translation stack.
65  // TODO: Support debug metadata with cyclic dependencies.
66  if (llvm::is_contained(elements, nullptr))
67  elements.clear();
68  return DICompositeTypeAttr::get(
69  context, node->getTag(), getStringAttrOrNull(node->getRawName()),
70  translate(node->getFile()), node->getLine(), translate(node->getScope()),
71  translate(node->getBaseType()), flags.value_or(DIFlags::Zero),
72  node->getSizeInBits(), node->getAlignInBits(), elements);
73 }

121 DISubprogramAttr DebugImporter::translateImpl(llvm::DISubprogram *node) {
122  std::optional<DISubprogramFlags> subprogramFlags =
123  symbolizeDISubprogramFlags(node->getSubprogram()->getSPFlags());
124  // Return nullptr if the scope or type is a cyclic dependency.
125  DIScopeAttr scope = translate(node->getScope());
126  if (node->getScope() && !scope)
127  return nullptr;
128  DISubroutineTypeAttr type = translate(node->getType());
129  if (node->getType() && !type)
130  return nullptr;
131  return DISubprogramAttr::get(context, translate(node->getUnit()), scope,
132  getStringAttrOrNull(node->getRawName()),
133  getStringAttrOrNull(node->getRawLinkageName()),
134  translate(node->getFile()), node->getLine(),
135  node->getScopeLine(), subprogramFlags.value(),
136  type);
137 }

139 DISubrangeAttr DebugImporter::translateImpl(llvm::DISubrange *node) {
140  auto getIntegerAttrOrNull = [&](llvm::DISubrange::BoundType data) {
141  if (auto *constInt = llvm::dyn_cast_or_null<llvm::ConstantInt *>(data))
142  return IntegerAttr::get(IntegerType::get(context, 64),
143  constInt->getSExtValue());
144  return IntegerAttr();
145  };
146  return DISubrangeAttr::get(context, getIntegerAttrOrNull(node->getCount()),
147  getIntegerAttrOrNull(node->getLowerBound()),
148  getIntegerAttrOrNull(node->getUpperBound()),
149  getIntegerAttrOrNull(node->getStride()));

152 DISubroutineTypeAttr
153 DebugImporter::translateImpl(llvm::DISubroutineType *node) {
154  SmallVector<DITypeAttr> types;
155  for (llvm::DIType *type : node->getTypeArray()) {
156  if (!type) {
157  // A nullptr entry may appear at the beginning or the end of the
158  // subroutine types list modeling either a void result type or the type of
159  // a variadic argument. Translate the nullptr to an explicit
160  // DINullTypeAttr since the attribute list cannot contain a nullptr entry.
161  types.push_back(DINullTypeAttr::get(context));
162  continue;
163  }
164  types.push_back(translate(type));
165  }
166  // Return nullptr if any of the types is a cyclic dependency.
167  if (llvm::is_contained(types, nullptr))
168  return nullptr;
169  return DISubroutineTypeAttr::get(context, node->getCC(), types);
170 }

176 DINodeAttr DebugImporter::translate(llvm::DINode *node) {
177  if (!node)
178  return nullptr;
179 
180  // Check for a cached instance.
181  if (DINodeAttr attr = nodeToAttr.lookup(node))
182  return attr;
183 
184  // Return nullptr if a cyclic dependency is detected since the same node is
185  // being traversed twice. This check avoids infinite recursion if the debug
186  // metadata contains cycles.
187  if (!translationStack.insert(node))
188  return nullptr;
189  auto guard = llvm::make_scope_exit([&]() { translationStack.pop_back(); });
190 
191  // Convert the debug metadata if possible.
192  auto translateNode = [this](llvm::DINode *node) -> DINodeAttr {
193  if (auto *casted = dyn_cast<llvm::DIBasicType>(node))
194  return translateImpl(casted);
195  if (auto *casted = dyn_cast<llvm::DICompileUnit>(node))
196  return translateImpl(casted);
197  if (auto *casted = dyn_cast<llvm::DICompositeType>(node))
198  return translateImpl(casted);
199  if (auto *casted = dyn_cast<llvm::DIDerivedType>(node))
200  return translateImpl(casted);
201  if (auto *casted = dyn_cast<llvm::DIFile>(node))
202  return translateImpl(casted);
203  if (auto *casted = dyn_cast<llvm::DILexicalBlock>(node))
204  return translateImpl(casted);
205  if (auto *casted = dyn_cast<llvm::DILexicalBlockFile>(node))
206  return translateImpl(casted);
207  if (auto *casted = dyn_cast<llvm::DILocalVariable>(node))
208  return translateImpl(casted);
209  if (auto *casted = dyn_cast<llvm::DISubprogram>(node))
210  return translateImpl(casted);
211  if (auto *casted = dyn_cast<llvm::DINamespace>(node))
212  return translateImpl(casted);
213  if (auto *casted = dyn_cast<llvm::DISubrange>(node))
214  return translateImpl(casted);
215  if (auto *casted = dyn_cast<llvm::DISubroutineType>(node))
216  return translateImpl(casted);
217  return nullptr;
218  };
219  if (DINodeAttr attr = translateNode(node)) {
220  nodeToAttr.insert({node, attr});
221  return attr;
222  }
223  return nullptr;
224 }

230 Location DebugImporter::translateLoc(llvm::DILocation *loc) {
231  if (!loc)
232  return mlirModule.getLoc();
233 
234  // Get the file location of the instruction.
235  Location result = FileLineColLoc::get(context, loc->getFilename(),
236  loc->getLine(), loc->getColumn());
237 
238  // Add call site information, if available.
239  if (llvm::DILocation *inlinedAt = loc->getInlinedAt())
240  result = CallSiteLoc::get(result, translateLoc(inlinedAt));
241 
242  // Add scope information.
243  assert(loc->getScope() && "expected non-null scope");
244  result = FusedLocWith<DIScopeAttr>::get({result}, translate(loc->getScope()),
245  context);
246  return result;
247 }
