70 /// Only retain those attributes that are not constructed by
71 /// `LLVMFuncOp::build`. If `filterArgAttrs` is set, also filter out argument
72 /// attributes.
73 static void filterFuncAttributes(func::FuncOp func, bool filterArgAndResAttrs,
74  SmallVectorImpl<NamedAttribute> &result) {
75  for (const NamedAttribute &attr : func->getAttrs()) {
76  if (attr.getName() == SymbolTable::getSymbolAttrName() ||
77  attr.getName() == func.getFunctionTypeAttrName() ||
78  attr.getName() == linkageAttrName ||
79  attr.getName() == varargsAttrName ||
80  attr.getName() == LLVM::LLVMDialect::getReadnoneAttrName() ||
81  (filterArgAndResAttrs &&
82  (attr.getName() == func.getArgAttrsAttrName() ||
83  attr.getName() == func.getResAttrsAttrName())))
84  continue;
85  result.push_back(attr);
86  }
87 }

89 /// Adds a an empty set of argument attributes for the newly added argument in
90 /// front of the existing ones.
91 static void prependEmptyArgAttr(OpBuilder &builder,
92  SmallVectorImpl<NamedAttribute> &newFuncAttrs,
93  func::FuncOp func) {
94  auto argAttrs = func.getArgAttrs();
95  // Nothing to do when there were no arg attrs beforehand.
96  if (!argAttrs)
97  return;
98 
99  size_t numArguments = func.getNumArguments();
100  SmallVector<Attribute> newArgAttrs;
101  newArgAttrs.reserve(numArguments + 1);
102  // Insert empty dictionary for the new argument.
103  newArgAttrs.push_back(builder.getDictionaryAttr({}));
104 
105  llvm::append_range(newArgAttrs, *argAttrs);
106  auto newNamedAttr = builder.getNamedAttr(func.getArgAttrsAttrName(),
107  builder.getArrayAttr(newArgAttrs));
108  newFuncAttrs.push_back(newNamedAttr);
109 }

111 /// Creates an auxiliary function with pointer-to-memref-descriptor-struct
112 /// arguments instead of unpacked arguments. This function can be called from C
113 /// by passing a pointer to a C struct corresponding to a memref descriptor.
114 /// Similarly, returned memrefs are passed via pointers to a C struct that is
115 /// passed as additional argument.
116 /// Internally, the auxiliary function unpacks the descriptor into individual
117 /// components and forwards them to `newFuncOp` and forwards the results to
118 /// the extra arguments.
119 static void wrapForExternalCallers(OpBuilder &rewriter, Location loc,
120  LLVMTypeConverter &typeConverter,
121  func::FuncOp funcOp,
122  LLVM::LLVMFuncOp newFuncOp) {
123  auto type = funcOp.getFunctionType();
124  auto [wrapperFuncType, resultStructType] =
125  typeConverter.convertFunctionTypeCWrapper(type);
126 
127  SmallVector<NamedAttribute, 4> attributes;
128  // Only modify the argument and result attributes when the result is now an
129  // argument.
130  if (resultStructType)
131  prependEmptyArgAttr(rewriter, attributes, funcOp);
132  filterFuncAttributes(
133  funcOp, /*filterArgAndResAttrs=*/static_cast<bool>(resultStructType),
134  attributes);
135  auto wrapperFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
136  loc, llvm::formatv("_mlir_ciface_{0}", funcOp.getName()).str(),
137  wrapperFuncType, LLVM::Linkage::External, /*dsoLocal*/ false,
138  /*cconv*/ LLVM::CConv::C, attributes);
139 
140  OpBuilder::InsertionGuard guard(rewriter);
141  rewriter.setInsertionPointToStart(wrapperFuncOp.addEntryBlock());
142 
143  SmallVector<Value, 8> args;
144  size_t argOffset = resultStructType ? 1 : 0;
145  for (auto [index, argType] : llvm::enumerate(type.getInputs())) {
146  Value arg = wrapperFuncOp.getArgument(index + argOffset);
147  if (auto memrefType = argType.dyn_cast<MemRefType>()) {
148  Value loaded = rewriter.create<LLVM::LoadOp>(
149  loc, typeConverter.convertType(memrefType), arg);
150  MemRefDescriptor::unpack(rewriter, loc, loaded, memrefType, args);
151  continue;
152  }
153  if (argType.isa<UnrankedMemRefType>()) {
154  Value loaded = rewriter.create<LLVM::LoadOp>(
155  loc, typeConverter.convertType(argType), arg);
156  UnrankedMemRefDescriptor::unpack(rewriter, loc, loaded, args);
157  continue;
158  }
159 
160  args.push_back(arg);
161  }
162 
163  auto call = rewriter.create<LLVM::CallOp>(loc, newFuncOp, args);
164 
165  if (resultStructType) {
166  rewriter.create<LLVM::StoreOp>(loc, call.getResult(),
167  wrapperFuncOp.getArgument(0));
168  rewriter.create<LLVM::ReturnOp>(loc, ValueRange{});
169  } else {
170  rewriter.create<LLVM::ReturnOp>(loc, call.getResults());
171  }
172 }

174 /// Creates an auxiliary function with pointer-to-memref-descriptor-struct
175 /// arguments instead of unpacked arguments. Creates a body for the (external)
176 /// `newFuncOp` that allocates a memref descriptor on stack, packs the
177 /// individual arguments into this descriptor and passes a pointer to it into
178 /// the auxiliary function. If the result of the function cannot be directly
179 /// returned, we write it to a special first argument that provides a pointer
180 /// to a corresponding struct. This auxiliary external function is now
181 /// compatible with functions defined in C using pointers to C structs
182 /// corresponding to a memref descriptor.
183 static void wrapExternalFunction(OpBuilder &builder, Location loc,
184  LLVMTypeConverter &typeConverter,
185  func::FuncOp funcOp,
186  LLVM::LLVMFuncOp newFuncOp) {
187  OpBuilder::InsertionGuard guard(builder);
188 
189  auto [wrapperType, resultStructType] =
190  typeConverter.convertFunctionTypeCWrapper(funcOp.getFunctionType());
191  // This conversion can only fail if it could not convert one of the argument
192  // types. But since it has been applied to a non-wrapper function before, it
193  // should have failed earlier and not reach this point at all.
194  assert(wrapperType && "unexpected type conversion failure");
195 
196  SmallVector<NamedAttribute, 4> attributes;
197  // Only modify the argument and result attributes when the result is now an
198  // argument.
199  if (resultStructType)
200  prependEmptyArgAttr(builder, attributes, funcOp);
201  filterFuncAttributes(
202  funcOp, /*filterArgAndResAttrs=*/static_cast<bool>(resultStructType),
203  attributes);
204 
205  // Create the auxiliary function.
206  auto wrapperFunc = builder.create<LLVM::LLVMFuncOp>(
207  loc, llvm::formatv("_mlir_ciface_{0}", funcOp.getName()).str(),
208  wrapperType, LLVM::Linkage::External, /*dsoLocal*/ false,
209  /*cconv*/ LLVM::CConv::C, attributes);
210 
211  builder.setInsertionPointToStart(newFuncOp.addEntryBlock());
212 
213  // Get a ValueRange containing arguments.
214  FunctionType type = funcOp.getFunctionType();
215  SmallVector<Value, 8> args;
216  args.reserve(type.getNumInputs());
217  ValueRange wrapperArgsRange(newFuncOp.getArguments());
218 
219  if (resultStructType) {
220  // Allocate the struct on the stack and pass the pointer.
221  Type resultType =
222  wrapperType.cast<LLVM::LLVMFunctionType>().getParamType(0);
223  Value one = builder.create<LLVM::ConstantOp>(
224  loc, typeConverter.convertType(builder.getIndexType()),
225  builder.getIntegerAttr(builder.getIndexType(), 1));
226  Value result =
227  builder.create<LLVM::AllocaOp>(loc, resultType, resultStructType, one);
228  args.push_back(result);
229  }
230 
231  // Iterate over the inputs of the original function and pack values into
232  // memref descriptors if the original type is a memref.
233  for (Type input : type.getInputs()) {
234  Value arg;
235  int numToDrop = 1;
236  auto memRefType = input.dyn_cast<MemRefType>();
237  auto unrankedMemRefType = input.dyn_cast<UnrankedMemRefType>();
238  if (memRefType || unrankedMemRefType) {
239  numToDrop = memRefType
240  ? MemRefDescriptor::getNumUnpackedValues(memRefType)
241  : UnrankedMemRefDescriptor::getNumUnpackedValues();
242  Value packed =
243  memRefType
244  ? MemRefDescriptor::pack(builder, loc, typeConverter, memRefType,
245  wrapperArgsRange.take_front(numToDrop))
246  : UnrankedMemRefDescriptor::pack(
247  builder, loc, typeConverter, unrankedMemRefType,
248  wrapperArgsRange.take_front(numToDrop));
249 
250  auto ptrTy = typeConverter.getPointerType(packed.getType());
251  Value one = builder.create<LLVM::ConstantOp>(
252  loc, typeConverter.convertType(builder.getIndexType()),
253  builder.getIntegerAttr(builder.getIndexType(), 1));
254  Value allocated = builder.create<LLVM::AllocaOp>(
255  loc, ptrTy, packed.getType(), one, /*alignment=*/0);
256  builder.create<LLVM::StoreOp>(loc, packed, allocated);
257  arg = allocated;
258  } else {
259  arg = wrapperArgsRange[0];
260  }
261 
262  args.push_back(arg);
263  wrapperArgsRange = wrapperArgsRange.drop_front(numToDrop);
264  }
265  assert(wrapperArgsRange.empty() && "did not map some of the arguments");
266 
267  auto call = builder.create<LLVM::CallOp>(loc, wrapperFunc, args);
268 
269  if (resultStructType) {
270  Value result =
271  builder.create<LLVM::LoadOp>(loc, resultStructType, args.front());
272  builder.create<LLVM::ReturnOp>(loc, result);
273  } else {
274  builder.create<LLVM::ReturnOp>(loc, call.getResults());
275  }
276 }

278 /// Modifies the body of the function to construct the `MemRefDescriptor` from
279 /// the bare pointer calling convention lowering of `memref` types.
280 static void modifyFuncOpToUseBarePtrCallingConv(
281  ConversionPatternRewriter &rewriter, Location loc,
282  LLVMTypeConverter &typeConverter, LLVM::LLVMFuncOp funcOp,
283  TypeRange oldArgTypes) {
284  if (funcOp.getBody().empty())
285  return;
286 
287  // Promote bare pointers from memref arguments to memref descriptors at the
288  // beginning of the function so that all the memrefs in the function have a
289  // uniform representation.
290  Block *entryBlock = &funcOp.getBody().front();
291  auto blockArgs = entryBlock->getArguments();
292  assert(blockArgs.size() == oldArgTypes.size() &&
293  "The number of arguments and types doesn't match");
294 
295  OpBuilder::InsertionGuard guard(rewriter);
296  rewriter.setInsertionPointToStart(entryBlock);
297  for (auto it : llvm::zip(blockArgs, oldArgTypes)) {
298  BlockArgument arg = std::get<0>(it);
299  Type argTy = std::get<1>(it);
300 
301  // Unranked memrefs are not supported in the bare pointer calling
302  // convention. We should have bailed out before in the presence of
303  // unranked memrefs.
304  assert(!argTy.isa<UnrankedMemRefType>() &&
305  "Unranked memref is not supported");
306  auto memrefTy = argTy.dyn_cast<MemRefType>();
307  if (!memrefTy)
308  continue;
309 
310  // Replace barePtr with a placeholder (undef), promote barePtr to a ranked
311  // or unranked memref descriptor and replace placeholder with the last
312  // instruction of the memref descriptor.
313  // TODO: The placeholder is needed to avoid replacing barePtr uses in the
314  // MemRef descriptor instructions. We may want to have a utility in the
315  // rewriter to properly handle this use case.
316  Location loc = funcOp.getLoc();
317  auto placeholder = rewriter.create<LLVM::UndefOp>(
318  loc, typeConverter.convertType(memrefTy));
319  rewriter.replaceUsesOfBlockArgument(arg, placeholder);
320 
321  Value desc = MemRefDescriptor::fromStaticShape(rewriter, loc, typeConverter,
322  memrefTy, arg);
323  rewriter.replaceOp(placeholder, {desc});
324  }
325 }

329 struct FuncOpConversionBase : public ConvertOpToLLVMPattern<func::FuncOp> {
330 protected:
331  using ConvertOpToLLVMPattern<func::FuncOp>::ConvertOpToLLVMPattern;
332 
333  // Convert input FuncOp to LLVMFuncOp by using the LLVMTypeConverter provided
334  // to this legalization pattern.
335  LLVM::LLVMFuncOp
336  convertFuncOpToLLVMFuncOp(func::FuncOp funcOp,
337  ConversionPatternRewriter &rewriter) const {
338  // Convert the original function arguments. They are converted using the
339  // LLVMTypeConverter provided to this legalization pattern.
340  auto varargsAttr = funcOp->getAttrOfType<BoolAttr>(varargsAttrName);
341  TypeConverter::SignatureConversion result(funcOp.getNumArguments());
342  auto llvmType = getTypeConverter()->convertFunctionSignature(
343  funcOp.getFunctionType(), varargsAttr && varargsAttr.getValue(),
344  shouldUseBarePtrCallConv(funcOp, getTypeConverter()), result);
345  if (!llvmType)
346  return nullptr;
347 
348  // Propagate argument/result attributes to all converted arguments/result
349  // obtained after converting a given original argument/result.
350  SmallVector<NamedAttribute, 4> attributes;
351  filterFuncAttributes(funcOp, /*filterArgAndResAttrs=*/true, attributes);
352  if (ArrayAttr resAttrDicts = funcOp.getAllResultAttrs()) {
353  assert(!resAttrDicts.empty() && "expected array to be non-empty");
354  auto newResAttrDicts =
355  (funcOp.getNumResults() == 1)
356  ? resAttrDicts
357  : rewriter.getArrayAttr(rewriter.getDictionaryAttr({}));
358  attributes.push_back(
359  rewriter.getNamedAttr(funcOp.getResAttrsAttrName(), newResAttrDicts));
360  }
361  if (ArrayAttr argAttrDicts = funcOp.getAllArgAttrs()) {
362  SmallVector<Attribute, 4> newArgAttrs(
363  llvmType.cast<LLVM::LLVMFunctionType>().getNumParams());
364  for (unsigned i = 0, e = funcOp.getNumArguments(); i < e; ++i) {
365  // Some LLVM IR attribute have a type attached to them. During FuncOp ->
366  // LLVMFuncOp conversion these types may have changed. Account for that
367  // change by converting attributes' types as well.
368  SmallVector<NamedAttribute, 4> convertedAttrs;
369  auto attrsDict = argAttrDicts[i].cast<DictionaryAttr>();
370  convertedAttrs.reserve(attrsDict.size());
371  for (const NamedAttribute &attr : attrsDict) {
372  const auto convert = [&](const NamedAttribute &attr) {
373  return TypeAttr::get(getTypeConverter()->convertType(
374  attr.getValue().cast<TypeAttr>().getValue()));
375  };
376  if (attr.getName().getValue() ==
377  LLVM::LLVMDialect::getByValAttrName()) {
378  convertedAttrs.push_back(rewriter.getNamedAttr(
379  LLVM::LLVMDialect::getByValAttrName(), convert(attr)));
380  } else if (attr.getName().getValue() ==
381  LLVM::LLVMDialect::getByRefAttrName()) {
382  convertedAttrs.push_back(rewriter.getNamedAttr(
383  LLVM::LLVMDialect::getByRefAttrName(), convert(attr)));
384  } else if (attr.getName().getValue() ==
385  LLVM::LLVMDialect::getStructRetAttrName()) {
386  convertedAttrs.push_back(rewriter.getNamedAttr(
387  LLVM::LLVMDialect::getStructRetAttrName(), convert(attr)));
388  } else if (attr.getName().getValue() ==
389  LLVM::LLVMDialect::getInAllocaAttrName()) {
390  convertedAttrs.push_back(rewriter.getNamedAttr(
391  LLVM::LLVMDialect::getInAllocaAttrName(), convert(attr)));
392  } else {
393  convertedAttrs.push_back(attr);
394  }
395  }
396  auto mapping = result.getInputMapping(i);
397  assert(mapping && "unexpected deletion of function argument");
398  // Only attach the new argument attributes if there is a one-to-one
399  // mapping from old to new types. Otherwise, attributes might be
400  // attached to types that they do not support.
401  if (mapping->size == 1) {
402  newArgAttrs[mapping->inputNo] =
403  DictionaryAttr::get(rewriter.getContext(), convertedAttrs);
404  continue;
405  }
406  // TODO: Implement custom handling for types that expand to multiple
407  // function arguments.
408  for (size_t j = 0; j < mapping->size; ++j)
409  newArgAttrs[mapping->inputNo + j] =
410  DictionaryAttr::get(rewriter.getContext(), {});
411  }
412  attributes.push_back(rewriter.getNamedAttr(
413  funcOp.getArgAttrsAttrName(), rewriter.getArrayAttr(newArgAttrs)));
414  }
415 
416  // Create an LLVM function, use external linkage by default until MLIR
417  // functions have linkage.
418  LLVM::Linkage linkage = LLVM::Linkage::External;
419  if (funcOp->hasAttr(linkageAttrName)) {
420  auto attr =
421  funcOp->getAttr(linkageAttrName).dyn_cast<mlir::LLVM::LinkageAttr>();
422  if (!attr) {
423  funcOp->emitError() << "Contains " << linkageAttrName
424  << " attribute not of type LLVM::LinkageAttr";
425  return nullptr;
426  }
427  linkage = attr.getLinkage();
428  }
429 
430  // Create a memory effect attribute corresponding to readnone.
431  StringRef readnoneAttrName = LLVM::LLVMDialect::getReadnoneAttrName();
432  LLVM::MemoryEffectsAttr memoryAttr = {};
433  if (funcOp->hasAttr(readnoneAttrName)) {
434  auto attr = funcOp->getAttrOfType<UnitAttr>(readnoneAttrName);
435  if (!attr) {
436  funcOp->emitError() << "Contains " << readnoneAttrName
437  << " attribute not of type UnitAttr";
438  return nullptr;
439  }
440  memoryAttr = LLVM::MemoryEffectsAttr::get(rewriter.getContext(),
441  {LLVM::ModRefInfo::NoModRef,
442  LLVM::ModRefInfo::NoModRef,
443  LLVM::ModRefInfo::NoModRef});
444  }
445  auto newFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
446  funcOp.getLoc(), funcOp.getName(), llvmType, linkage,
447  /*dsoLocal*/ false, /*cconv*/ LLVM::CConv::C, attributes);
448  // If the memory attribute was created, add it to the function.
449  if (memoryAttr)
450  newFuncOp.setMemoryAttr(memoryAttr);
451  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
452  newFuncOp.end());
453  if (failed(rewriter.convertRegionTypes(&newFuncOp.getBody(), *typeConverter,
454  &result)))
455  return nullptr;
456 
457  return newFuncOp;
458  }
459 };

461 /// FuncOp legalization pattern that converts MemRef arguments to pointers to
462 /// MemRef descriptors (LLVM struct data types) containing all the MemRef type
463 /// information.
464 struct FuncOpConversion : public FuncOpConversionBase {
465  FuncOpConversion(LLVMTypeConverter &converter)
466  : FuncOpConversionBase(converter) {}
467 
468  LogicalResult
469  matchAndRewrite(func::FuncOp funcOp, OpAdaptor adaptor,
470  ConversionPatternRewriter &rewriter) const override {
471  auto newFuncOp = convertFuncOpToLLVMFuncOp(funcOp, rewriter);
472  if (!newFuncOp)
473  return failure();
474 
475  if (!shouldUseBarePtrCallConv(funcOp, this->getTypeConverter())) {
476  if (funcOp->getAttrOfType<UnitAttr>(
477  LLVM::LLVMDialect::getEmitCWrapperAttrName())) {
478  if (newFuncOp.isVarArg())
479  return funcOp->emitError("C interface for variadic functions is not "
480  "supported yet.");
481 
482  if (newFuncOp.isExternal())
483  wrapExternalFunction(rewriter, funcOp.getLoc(), *getTypeConverter(),
484  funcOp, newFuncOp);
485  else
486  wrapForExternalCallers(rewriter, funcOp.getLoc(), *getTypeConverter(),
487  funcOp, newFuncOp);
488  }
489  } else {
490  modifyFuncOpToUseBarePtrCallingConv(rewriter, funcOp.getLoc(),
491  *getTypeConverter(), newFuncOp,
492  funcOp.getFunctionType().getInputs());
493  }
494 
495  rewriter.eraseOp(funcOp);
496  return success();
497  }
498 };

500 struct ConstantOpLowering : public ConvertOpToLLVMPattern<func::ConstantOp> {

503  LogicalResult

522 // A CallOp automatically promotes MemRefType to a sequence of alloca/store and
523 // passes the pointer to the MemRef across function boundaries.
524 template <typename CallOpType>
525 struct CallOpInterfaceLowering : public ConvertOpToLLVMPattern<CallOpType> {
526  using ConvertOpToLLVMPattern<CallOpType>::ConvertOpToLLVMPattern;
527  using Super = CallOpInterfaceLowering<CallOpType>;
528  using Base = ConvertOpToLLVMPattern<CallOpType>;
529 
530  LogicalResult matchAndRewriteImpl(CallOpType callOp,
531  typename CallOpType::Adaptor adaptor,
532  ConversionPatternRewriter &rewriter,
533  bool useBarePtrCallConv = false) const {
534  // Pack the result types into a struct.
535  Type packedResult = nullptr;
536  unsigned numResults = callOp.getNumResults();
537  auto resultTypes = llvm::to_vector<4>(callOp.getResultTypes());
538 
539  if (numResults != 0) {
540  if (!(packedResult = this->getTypeConverter()->packFunctionResults(
541  resultTypes, useBarePtrCallConv)))
542  return failure();
543  }
544 
545  if (useBarePtrCallConv) {
546  for (auto it : callOp->getOperands()) {
547  Type operandType = it.getType();
548  if (operandType.isa<UnrankedMemRefType>()) {
549  // Unranked memref is not supported in the bare pointer calling
550  // convention.
551  return failure();
552  }
553  }
554  }
555  auto promoted = this->getTypeConverter()->promoteOperands(
556  callOp.getLoc(), /*opOperands=*/callOp->getOperands(),
557  adaptor.getOperands(), rewriter, useBarePtrCallConv);
558  auto newOp = rewriter.create<LLVM::CallOp>(
559  callOp.getLoc(), packedResult ? TypeRange(packedResult) : TypeRange(),
560  promoted, callOp->getAttrs());
561 
562  SmallVector<Value, 4> results;
563  if (numResults < 2) {
564  // If < 2 results, packing did not do anything and we can just return.
565  results.append(newOp.result_begin(), newOp.result_end());
566  } else {
567  // Otherwise, it had been converted to an operation producing a structure.
568  // Extract individual results from the structure and return them as list.
569  results.reserve(numResults);
570  for (unsigned i = 0; i < numResults; ++i) {
571  results.push_back(rewriter.create<LLVM::ExtractValueOp>(
572  callOp.getLoc(), newOp->getResult(0), i));
573  }
574  }
575 
576  if (useBarePtrCallConv) {
577  // For the bare-ptr calling convention, promote memref results to
578  // descriptors.
579  assert(results.size() == resultTypes.size() &&
580  "The number of arguments and types doesn't match");
581  this->getTypeConverter()->promoteBarePtrsToDescriptors(
582  rewriter, callOp.getLoc(), resultTypes, results);
583  } else if (failed(this->copyUnrankedDescriptors(rewriter, callOp.getLoc(),
584  resultTypes, results,
585  /*toDynamic=*/false))) {
586  return failure();
587  }
588 
589  rewriter.replaceOp(callOp, results);
590  return success();
591  }
592 };

647 // Special lowering pattern for `ReturnOps`. Unlike all other operations,
648 // `ReturnOp` interacts with the function signature and must have as many
649 // operands as the function has return values. Because in LLVM IR, functions
650 // can only return 0 or 1 value, we pack multiple values into a structure type.
651 // Emit `UndefOp` followed by `InsertValueOp`s to create such structure if
652 // necessary before returning it
653 struct ReturnOpLowering : public ConvertOpToLLVMPattern<func::ReturnOp> {
654  using ConvertOpToLLVMPattern<func::ReturnOp>::ConvertOpToLLVMPattern;
655 
656  LogicalResult
657  matchAndRewrite(func::ReturnOp op, OpAdaptor adaptor,
658  ConversionPatternRewriter &rewriter) const override {
659  Location loc = op.getLoc();
660  unsigned numArguments = op.getNumOperands();
661  SmallVector<Value, 4> updatedOperands;
662 
663  auto funcOp = op->getParentOfType<LLVM::LLVMFuncOp>();
664  bool useBarePtrCallConv =
665  shouldUseBarePtrCallConv(funcOp, this->getTypeConverter());
666  if (useBarePtrCallConv) {
667  // For the bare-ptr calling convention, extract the aligned pointer to
668  // be returned from the memref descriptor.
669  for (auto it : llvm::zip(op->getOperands(), adaptor.getOperands())) {
670  Type oldTy = std::get<0>(it).getType();
671  Value newOperand = std::get<1>(it);
672  if (oldTy.isa<MemRefType>() && getTypeConverter()->canConvertToBarePtr(
673  oldTy.cast<BaseMemRefType>())) {
674  MemRefDescriptor memrefDesc(newOperand);
675  newOperand = memrefDesc.allocatedPtr(rewriter, loc);
676  } else if (oldTy.isa<UnrankedMemRefType>()) {
677  // Unranked memref is not supported in the bare pointer calling
678  // convention.
679  return failure();
680  }
681  updatedOperands.push_back(newOperand);
682  }
683  } else {
684  updatedOperands = llvm::to_vector<4>(adaptor.getOperands());
685  (void)copyUnrankedDescriptors(rewriter, loc, op.getOperands().getTypes(),
686  updatedOperands,
687  /*toDynamic=*/true);
688  }
689 
690  // If ReturnOp has 0 or 1 operand, create it and return immediately.
691  if (numArguments <= 1) {
692  rewriter.replaceOpWithNewOp<LLVM::ReturnOp>(
693  op, TypeRange(), updatedOperands, op->getAttrs());
694  return success();
695  }
696 
697  // Otherwise, we need to pack the arguments into an LLVM struct type before
698  // returning.
699  auto packedType = getTypeConverter()->packFunctionResults(
700  op.getOperandTypes(), useBarePtrCallConv);
701  if (!packedType) {
702  return rewriter.notifyMatchFailure(op, "could not convert result types");
703  }
704 
705  Value packed = rewriter.create<LLVM::UndefOp>(loc, packedType);
706  for (auto [idx, operand] : llvm::enumerate(updatedOperands)) {
707  packed = rewriter.create<LLVM::InsertValueOp>(loc, packed, operand, idx);
708  }
709  rewriter.replaceOpWithNewOp<LLVM::ReturnOp>(op, TypeRange(), packed,
710  op->getAttrs());
711  return success();
712  }
713 };
