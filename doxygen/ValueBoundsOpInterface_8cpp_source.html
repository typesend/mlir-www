27 static std::optional<int64_t> getConstantIntValue(OpFoldResult ofr) {
28  // Case 1: Check for Constant integer.
29  if (auto val = ofr.dyn_cast<Value>()) {
30  APSInt intVal;

62 void ValueBoundsConstraintSet::addBound(BoundType type, int64_t pos,
63  AffineExpr expr) {
64  LogicalResult status = cstr.addBound(
65  type, pos,
66  AffineMap::get(cstr.getNumDimVars(), cstr.getNumSymbolVars(), expr));
67  if (failed(status)) {
68  // Non-pure (e.g., semi-affine) expressions are not yet supported by
69  // FlatLinearConstraints. However, we can just ignore such failures here.
70  // Even without this bound, there may be enough information in the
71  // constraint system to compute the requested bound. In case this bound is
72  // actually needed, `computeBound` will return `failure`.
73  LLVM_DEBUG(llvm::dbgs() << "Failed to add bound: " << expr << "\n");
74  }
75 }

77 AffineExpr ValueBoundsConstraintSet::getExpr(Value value,
78  std::optional<int64_t> dim) {
79 #ifndef NDEBUG
80  assertValidValueDim(value, dim);
81 #endif // NDEBUG
82 
83  auto shapedType = dyn_cast<ShapedType>(value.getType());
84  if (shapedType) {
85  // Static dimension: return constant directly.
86  if (shapedType.hasRank() && !shapedType.isDynamicDim(*dim))
87  return builder.getAffineConstantExpr(shapedType.getDimSize(*dim));
88  } else {
89  // Constant index value: return directly.
90  if (auto constInt = getConstantIntValue(value))
91  return builder.getAffineConstantExpr(*constInt);
92  }
93 
94  // Dynamic value: add to constraint set.
95  ValueDim valueDim = std::make_pair(value, dim.value_or(kIndexValue));
96  if (!valueDimToPosition.contains(valueDim))
97  (void)insert(value, dim);
98  int64_t pos = getPos(value, dim);
99  return pos < cstr.getNumDimVars()
100  ? builder.getAffineDimExpr(pos)
101  : builder.getAffineSymbolExpr(pos - cstr.getNumDimVars());
102 }

116 int64_t ValueBoundsConstraintSet::insert(Value value,
117  std::optional<int64_t> dim,
118  bool isSymbol) {
119 #ifndef NDEBUG
120  assertValidValueDim(value, dim);
121 #endif // NDEBUG
122 
123  ValueDim valueDim = std::make_pair(value, dim.value_or(kIndexValue));
124  assert(!valueDimToPosition.contains(valueDim) && "already mapped");
125  int64_t pos = isSymbol ? cstr.appendVar(VarKind::Symbol)
126  : cstr.appendVar(VarKind::SetDim);
127  positionToValueDim.insert(positionToValueDim.begin() + pos, valueDim);
128  // Update reverse mapping.
129  for (int64_t i = pos, e = positionToValueDim.size(); i < e; ++i)
130  valueDimToPosition[positionToValueDim[i]] = i;
131 
132  worklist.push(pos);
133  return pos;
134 }

157 void ValueBoundsConstraintSet::processWorklist(StopConditionFn stopCondition) {
158  while (!worklist.empty()) {
159  int64_t pos = worklist.front();
160  worklist.pop();
161  ValueDim valueDim = positionToValueDim[pos];
162  Value value = valueDim.first;
163  int64_t dim = valueDim.second;
164 
165  // Check for static dim size.
166  if (dim != kIndexValue) {
167  auto shapedType = cast<ShapedType>(value.getType());
168  if (shapedType.hasRank() && !shapedType.isDynamicDim(dim)) {
169  bound(value)[dim] == getExpr(shapedType.getDimSize(dim));
170  continue;
171  }
172  }
173 
174  // Do not process any further if the stop condition is met.
175  auto maybeDim = dim == kIndexValue ? std::nullopt : std::make_optional(dim);
176  if (stopCondition(value, maybeDim))
177  continue;
178 
179  // Query `ValueBoundsOpInterface` for constraints. New items may be added to
180  // the worklist.
181  auto valueBoundsOp =
182  dyn_cast<ValueBoundsOpInterface>(getOwnerOfValue(value));
183  if (!valueBoundsOp)
184  continue;
185  if (dim == kIndexValue) {
186  valueBoundsOp.populateBoundsForIndexValue(value, *this);
187  } else {
188  valueBoundsOp.populateBoundsForShapedValueDim(value, dim, *this);
189  }
190  }
191 }

193 void ValueBoundsConstraintSet::projectOut(int64_t pos) {
194  assert(pos >= 0 && pos < static_cast<int64_t>(positionToValueDim.size()) &&
195  "invalid position");
196  cstr.projectOut(pos);
197  bool erased = valueDimToPosition.erase(positionToValueDim[pos]);
198  (void)erased;
199  assert(erased && "inconsistent reverse mapping");
200  positionToValueDim.erase(positionToValueDim.begin() + pos);
201  // Update reverse mapping.
202  for (int64_t i = pos, e = positionToValueDim.size(); i < e; ++i)
203  valueDimToPosition[positionToValueDim[i]] = i;
204 }

206 void ValueBoundsConstraintSet::projectOut(
207  function_ref<bool(ValueDim)> condition) {
208  int64_t nextPos = 0;
209  while (nextPos < static_cast<int64_t>(positionToValueDim.size())) {
210  if (condition(positionToValueDim[nextPos])) {
211  projectOut(nextPos);
212  // The column was projected out so another column is now at that position.
213  // Do not increase the counter.
214  } else {
215  ++nextPos;
216  }
217  }

220 LogicalResult ValueBoundsConstraintSet::computeBound(
221  AffineMap &resultMap, ValueDimList &mapOperands, presburger::BoundType type,
222  Value value, std::optional<int64_t> dim, StopConditionFn stopCondition,
223  bool closedUB) {
224 #ifndef NDEBUG
225  assertValidValueDim(value, dim);
226  assert(!stopCondition(value, dim) &&
227  "stop condition should not be satisfied for starting point");
228 #endif // NDEBUG
229 
230  int64_t ubAdjustment = closedUB ? 0 : 1;
231  Builder b(value.getContext());
232  mapOperands.clear();
233 
234  if (stopCondition(value, dim)) {
235  // Special case: If the stop condition is satisfied for the input
236  // value/dimension, directly return it.
237  mapOperands.push_back(std::make_pair(value, dim));
238  AffineExpr bound = b.getAffineDimExpr(0);
239  if (type == BoundType::UB)
240  bound = bound + ubAdjustment;
241  resultMap = AffineMap::get(/*dimCount=*/1, /*symbolCount=*/0,
242  b.getAffineDimExpr(0));
243  return success();
244  }
245 
246  // Process the backward slice of `value` (i.e., reverse use-def chain) until
247  // `stopCondition` is met.
248  ValueDim valueDim = std::make_pair(value, dim.value_or(kIndexValue));
249  ValueBoundsConstraintSet cstr(value, dim);
250  cstr.processWorklist(stopCondition);
251 
252  // Project out all variables (apart from `valueDim`) that do not match the
253  // stop condition.
254  cstr.projectOut([&](ValueDim p) {
255  // Do not project out `valueDim`.
256  if (valueDim == p)
257  return false;
258  auto maybeDim =
259  p.second == kIndexValue ? std::nullopt : std::make_optional(p.second);
260  return !stopCondition(p.first, maybeDim);
261  });
262 
263  // Compute lower and upper bounds for `valueDim`.
264  int64_t pos = cstr.getPos(value, dim);
265  SmallVector<AffineMap> lb(1), ub(1);
266  cstr.cstr.getSliceBounds(pos, 1, value.getContext(), &lb, &ub,
267  /*getClosedUB=*/true);
268 
269  // Note: There are TODOs in the implementation of `getSliceBounds`. In such a
270  // case, no lower/upper bound can be computed at the moment.
271  // EQ, UB bounds: upper bound is needed.
272  if ((type != BoundType::LB) &&
273  (ub.empty() || !ub[0] || ub[0].getNumResults() == 0))
274  return failure();
275  // EQ, LB bounds: lower bound is needed.
276  if ((type != BoundType::UB) &&
277  (lb.empty() || !lb[0] || lb[0].getNumResults() == 0))
278  return failure();
279 
280  // TODO: Generate an affine map with multiple results.
281  if (type != BoundType::LB)
282  assert(ub.size() == 1 && ub[0].getNumResults() == 1 &&
283  "multiple bounds not supported");
284  if (type != BoundType::UB)
285  assert(lb.size() == 1 && lb[0].getNumResults() == 1 &&
286  "multiple bounds not supported");
287 
288  // EQ bound: lower and upper bound must match.
289  if (type == BoundType::EQ && ub[0] != lb[0])
290  return failure();
291 
292  AffineMap bound;
293  if (type == BoundType::EQ || type == BoundType::LB) {
294  bound = lb[0];
295  } else {
296  // Computed UB is a closed bound.
297  bound = AffineMap::get(ub[0].getNumDims(), ub[0].getNumSymbols(),
298  ub[0].getResult(0) + ubAdjustment);
299  }
300 
301  // Gather all SSA values that are used in the computed bound.
302  assert(cstr.cstr.getNumDimAndSymbolVars() == cstr.positionToValueDim.size() &&
303  "inconsistent mapping state");
304  SmallVector<AffineExpr> replacementDims, replacementSymbols;
305  int64_t numDims = 0, numSymbols = 0;
306  for (int64_t i = 0; i < cstr.cstr.getNumDimAndSymbolVars(); ++i) {
307  // Skip `value`.
308  if (i == pos)
309  continue;
310  // Check if the position `i` is used in the generated bound. If so, it must
311  // be included in the generated affine.apply op.
312  bool used = false;
313  bool isDim = i < cstr.cstr.getNumDimVars();
314  if (isDim) {
315  if (bound.isFunctionOfDim(i))
316  used = true;
317  } else {
318  if (bound.isFunctionOfSymbol(i - cstr.cstr.getNumDimVars()))
319  used = true;
320  }
321 
322  if (!used) {
323  // Not used: Remove dim/symbol from the result.
324  if (isDim) {
325  replacementDims.push_back(b.getAffineConstantExpr(0));
326  } else {
327  replacementSymbols.push_back(b.getAffineConstantExpr(0));
328  }
329  continue;
330  }
331 
332  if (isDim) {
333  replacementDims.push_back(b.getAffineDimExpr(numDims++));
334  } else {
335  replacementSymbols.push_back(b.getAffineSymbolExpr(numSymbols++));
336  }
337 
338  ValueBoundsConstraintSet::ValueDim valueDim = cstr.positionToValueDim[i];
339  Value value = valueDim.first;
340  int64_t dim = valueDim.second;
341  if (dim == ValueBoundsConstraintSet::kIndexValue) {
342  // An index-type value is used: can be used directly in the affine.apply
343  // op.
344  assert(value.getType().isIndex() && "expected index type");
345  mapOperands.push_back(std::make_pair(value, std::nullopt));
346  continue;
347  }
348 
349  assert(cast<ShapedType>(value.getType()).isDynamicDim(dim) &&
350  "expected dynamic dim");
351  mapOperands.push_back(std::make_pair(value, dim));
352  }
353 
354  resultMap = bound.replaceDimsAndSymbols(replacementDims, replacementSymbols,
355  numDims, numSymbols);
356  return success();
357 }

371 LogicalResult ValueBoundsConstraintSet::computeIndependentBound(
372  AffineMap &resultMap, ValueDimList &mapOperands, presburger::BoundType type,
373  Value value, std::optional<int64_t> dim, ValueRange independencies,
374  bool closedUB) {
375  // Return "true" if the given value is independent of all values in
376  // `independencies`. I.e., neither the value itself nor any value in the
377  // backward slice (reverse use-def chain) is contained in `independencies`.
378  auto isIndependent = [&](Value v) {
379  SmallVector<Value> worklist;
380  DenseSet<Value> visited;
381  worklist.push_back(v);
382  while (!worklist.empty()) {
383  Value next = worklist.pop_back_val();
384  if (visited.contains(next))
385  continue;
386  visited.insert(next);
387  if (llvm::is_contained(independencies, next))
388  return false;
389  // TODO: DominanceInfo could be used to stop the traversal early.
390  Operation *op = next.getDefiningOp();
391  if (!op)
392  continue;
393  worklist.append(op->getOperands().begin(), op->getOperands().end());
394  }
395  return true;
396  };
397 
398  // Reify bounds in terms of any independent values.
399  return computeBound(
400  resultMap, mapOperands, type, value, dim,
401  [&](Value v, std::optional<int64_t> d) { return isIndependent(v); },
402  closedUB);
403 }

405 FailureOr<int64_t> ValueBoundsConstraintSet::computeConstantBound(
406  presburger::BoundType type, Value value, std::optional<int64_t> dim,
407  StopConditionFn stopCondition, bool closedUB) {
408 #ifndef NDEBUG
409  assertValidValueDim(value, dim);
410 #endif // NDEBUG
411 
412  // Process the backward slice of `value` (i.e., reverse use-def chain) until
413  // `stopCondition` is met.
414  ValueBoundsConstraintSet cstr(value, dim);
415  int64_t pos = cstr.getPos(value, dim);
416  if (stopCondition) {
417  cstr.processWorklist(stopCondition);
418  } else {
419  // No stop condition specified: Keep adding constraints until a bound could
420  // be computed.
421  cstr.processWorklist(
422  /*stopCondition=*/[&](Value v, std::optional<int64_t> dim) {
423  return cstr.cstr.getConstantBound64(type, pos).has_value();
424  });
425  }
426 
427  // Compute constant bound for `valueDim`.
428  int64_t ubAdjustment = closedUB ? 0 : 1;
429  if (auto bound = cstr.cstr.getConstantBound64(type, pos))
430  return type == BoundType::UB ? *bound + ubAdjustment : *bound;
431  return failure();
432 }
