31 SmallVector<MemorySlot> memref::AllocaOp::getPromotableSlots() {
32  MemRefType type = getType();
33  if (!isSupportedElementType(type.getElementType()))
34  return {};
35  if (!type.hasStaticShape())
36  return {};
37  // Make sure the memref contains only a single element.
38  if (any_of(type.getShape(), [](uint64_t dim) { return dim != 1; }))
39  return {};
40 
41  return {MemorySlot{getResult(), type.getElementType()}};
42 }
