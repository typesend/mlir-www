60 /// A TransformState extension that keeps track of compiled PDL pattern sets.
61 /// This is intended to be used along the WithPDLPatterns op. The extension
62 /// can be constructed given an operation that has a SymbolTable trait and
63 /// contains pdl::PatternOp instances. The patterns are compiled lazily and one
64 /// by one when requested; this behavior is subject to change.
65 class PatternApplicatorExtension : public transform::TransformState::Extension {
66 public:
67  MLIR_DEFINE_EXPLICIT_INTERNAL_INLINE_TYPE_ID(PatternApplicatorExtension)
68 
69  /// Creates the extension for patterns contained in `patternContainer`.
70  explicit PatternApplicatorExtension(transform::TransformState &state,
71  Operation *patternContainer)
72  : Extension(state), patterns(patternContainer) {}
73 
74  /// Appends to `results` the operations contained in `root` that matched the
75  /// PDL pattern with the given name. Note that `root` may or may not be the
76  /// operation that contains PDL patterns. Reports an error if the pattern
77  /// cannot be found. Note that when no operations are matched, this still
78  /// succeeds as long as the pattern exists.
79  LogicalResult findAllMatches(StringRef patternName, Operation *root,
80  SmallVectorImpl<Operation *> &results);
81 
82 private:
83  /// Map from the pattern name to a singleton set of rewrite patterns that only
84  /// contains the pattern with this name. Populated when the pattern is first
85  /// requested.
86  // TODO: reconsider the efficiency of this storage when more usage data is
87  // available. Storing individual patterns in a set and triggering compilation
88  // for each of them has overhead. So does compiling a large set of patterns
89  // only to apply a handlful of them.
90  llvm::StringMap<FrozenRewritePatternSet> compiledPatterns;
91 
92  /// A symbol table operation containing the relevant PDL patterns.
93  SymbolTable patterns;
94 };

96 LogicalResult PatternApplicatorExtension::findAllMatches(
97  StringRef patternName, Operation *root,
98  SmallVectorImpl<Operation *> &results) {
99  auto it = compiledPatterns.find(patternName);
100  if (it == compiledPatterns.end()) {
101  auto patternOp = patterns.lookup<pdl::PatternOp>(patternName);
102  if (!patternOp)
103  return failure();
104 
105  // Copy the pattern operation into a new module that is compiled and
106  // consumed by the PDL interpreter.
107  OwningOpRef<ModuleOp> pdlModuleOp = ModuleOp::create(patternOp.getLoc());
108  auto builder = OpBuilder::atBlockEnd(pdlModuleOp->getBody());
109  builder.clone(*patternOp);
110  PDLPatternModule patternModule(std::move(pdlModuleOp));
111 
112  // Merge in the hooks owned by the dialect. Make a copy as they may be
113  // also used by the following operations.
114  auto *dialect =
115  root->getContext()->getLoadedDialect<transform::TransformDialect>();
116  for (const auto &[name, constraintFn] : dialect->getPDLConstraintHooks())
117  patternModule.registerConstraintFunction(name, constraintFn);
118 
119  // Register a noop rewriter because PDL requires patterns to end with some
120  // rewrite call.
121  patternModule.registerRewriteFunction(
122  "transform.dialect", [](PatternRewriter &, Operation *) {});
123 
124  it = compiledPatterns
125  .try_emplace(patternOp.getName(), std::move(patternModule))
126  .first;
127  }
128 
129  PatternApplicator applicator(it->second);
130  // We want to discourage direct use of PatternRewriter in APIs but In this
131  // very specific case, an IRRewriter is not enough.
132  struct TrivialPatternRewriter : public PatternRewriter {
133  public:
134  explicit TrivialPatternRewriter(MLIRContext *context)
135  : PatternRewriter(context) {}
136  };
137  TrivialPatternRewriter rewriter(root->getContext());
138  applicator.applyDefaultCostModel();
139  root->walk([&](Operation *op) {
140  if (succeeded(applicator.matchAndRewrite(op, rewriter)))
141  results.push_back(op);
142  });
143 
144  return success();
145 }

226 /// Return true if `a` happens before `b`, i.e., `a` or one of its ancestors
227 /// properly dominates `b` and `b` is not inside `a`.
228 static bool happensBefore(Operation *a, Operation *b) {
229  do {
230  if (a->isProperAncestor(b))
231  return false;
232  if (Operation *bAncestor = a->getBlock()->findAncestorOpInBlock(*b)) {
233  return a->isBeforeInBlock(bAncestor);
234  }
235  } while ((a = a->getParentOp()));
236  return false;

239 void transform::TrackingListener::notifyOperationReplaced(
240  Operation *op, ValueRange newValues) {
241  assert(op->getNumResults() == newValues.size() &&
242  "invalid number of replacement values");
243 
244  // Replace value handles.
245  for (auto [oldValue, newValue] : llvm::zip(op->getResults(), newValues))
246  (void)replacePayloadValue(oldValue, newValue);
247 
248  // Replace op handle.
249  SmallVector<Value> opHandles;
250  if (failed(getTransformState().getHandlesForPayloadOp(op, opHandles))) {
251  // Op is not tracked.
252  return;
253  }
254  auto hasAliveUser = [&]() {
255  for (Value v : opHandles)
256  for (Operation *user : v.getUsers())
257  if (!happensBefore(user, transformOp))
258  return true;
259  return false;
260  };
261  if (!hasAliveUser()) {
262  // The op is tracked but the corresponding handles are dead.
263  (void)replacePayloadOp(op, nullptr);
264  return;
265  }
266 
267  Operation *replacement = findReplacementOp(op, newValues);
268  // If the op is tracked but no replacement op was found, send a
269  // notification.
270  if (!replacement)
271  notifyPayloadReplacementNotFound(op, newValues);
272  (void)replacePayloadOp(op, replacement);
273 }

316 DiagnosedSilenceableFailure
317 transform::AlternativesOp::apply(transform::TransformResults &results,
318  transform::TransformState &state) {
319  SmallVector<Operation *> originals;
320  if (Value scopeHandle = getScope())
321  llvm::append_range(originals, state.getPayloadOps(scopeHandle));
322  else
323  originals.push_back(state.getTopLevel());
324 
325  for (Operation *original : originals) {
326  if (original->isAncestor(getOperation())) {
327  auto diag = emitDefiniteFailure()
328  << "scope must not contain the transforms being applied";
329  diag.attachNote(original->getLoc()) << "scope";
330  return diag;
331  }
332  if (!original->hasTrait<OpTrait::IsIsolatedFromAbove>()) {
333  auto diag = emitDefiniteFailure()
334  << "only isolated-from-above ops can be alternative scopes";
335  diag.attachNote(original->getLoc()) << "scope";
336  return diag;
337  }
338  }
339 
340  for (Region &reg : getAlternatives()) {
341  // Clone the scope operations and make the transforms in this alternative
342  // region apply to them by virtue of mapping the block argument (the only
343  // visible handle) to the cloned scope operations. This effectively prevents
344  // the transformation from accessing any IR outside the scope.
345  auto scope = state.make_region_scope(reg);
346  auto clones = llvm::to_vector(
347  llvm::map_range(originals, [](Operation *op) { return op->clone(); }));
348  auto deleteClones = llvm::make_scope_exit([&] {
349  for (Operation *clone : clones)
350  clone->erase();
351  });
352  if (failed(state.mapBlockArguments(reg.front().getArgument(0), clones)))
353  return DiagnosedSilenceableFailure::definiteFailure();
354 
355  bool failed = false;
356  for (Operation &transform : reg.front().without_terminator()) {
357  DiagnosedSilenceableFailure result =
358  state.applyTransform(cast<TransformOpInterface>(transform));
359  if (result.isSilenceableFailure()) {
360  LLVM_DEBUG(DBGS() << "alternative failed: " << result.getMessage()
361  << "\n");
362  failed = true;
363  break;
364  }
365 
366  if (::mlir::failed(result.silence()))
367  return DiagnosedSilenceableFailure::definiteFailure();
368  }
369 
370  // If all operations in the given alternative succeeded, no need to consider
371  // the rest. Replace the original scoping operation with the clone on which
372  // the transformations were performed.
373  if (!failed) {
374  // We will be using the clones, so cancel their scheduled deletion.
375  deleteClones.release();
376  TrackingListener listener(state, *this);
377  IRRewriter rewriter(getContext(), &listener);
378  for (const auto &kvp : llvm::zip(originals, clones)) {
379  Operation *original = std::get<0>(kvp);
380  Operation *clone = std::get<1>(kvp);
381  original->getBlock()->getOperations().insert(original->getIterator(),
382  clone);
383  rewriter.replaceOp(original, clone->getResults());
384  }
385  detail::forwardTerminatorOperands(&reg.front(), state, results);
386  return DiagnosedSilenceableFailure::success();
387  }
388  }
389  return emitSilenceableError() << "all alternatives failed";
390 }

452 /// Applies matcher operations from the given `block` assigning `op` as the
453 /// payload of the block's first argument. Updates `state` accordingly. If any
454 /// of the matcher produces a silenceable failure, discards it (printing the
455 /// content to the debug output stream) and returns failure. If any of the
456 /// matchers produces a definite failure, reports it and returns failure. If all
457 /// matchers in the block succeed, populates `mappings` with the payload
458 /// entities associated with the block terminator operands.
459 static DiagnosedSilenceableFailure
460 matchBlock(Block &block, Operation *op, transform::TransformState &state,
461  SmallVectorImpl<SmallVector<transform::MappedValue>> &mappings) {
462  assert(block.getParent() && "cannot match using a detached block");
463  auto matchScope = state.make_isolated_region_scope(*block.getParent());
464  if (failed(state.mapBlockArgument(block.getArgument(0), {op})))
465  return DiagnosedSilenceableFailure::definiteFailure();
466 
467  for (Operation &match : block.without_terminator()) {
468  if (!isa<transform::MatchOpInterface>(match)) {
469  return emitDefiniteFailure(match.getLoc())
470  << "expected operations in the match part to "
471  "implement MatchOpInterface";
472  }
473  DiagnosedSilenceableFailure diag =
474  state.applyTransform(cast<transform::TransformOpInterface>(match));
475  if (diag.succeeded())
476  continue;
477 
478  return diag;
479  }
480 
481  // Remember the values mapped to the terminator operands so we can
482  // forward them to the action.
483  ValueRange yieldedValues = block.getTerminator()->getOperands();
484  transform::detail::prepareValueMappings(mappings, yieldedValues, state);
485  return DiagnosedSilenceableFailure::success();
486 }

488 DiagnosedSilenceableFailure
489 transform::ForeachMatchOp::apply(transform::TransformResults &results,
490  transform::TransformState &state) {
491  SmallVector<std::pair<FunctionOpInterface, FunctionOpInterface>>
492  matchActionPairs;
493  matchActionPairs.reserve(getMatchers().size());
494  SymbolTableCollection symbolTable;
495  for (auto &&[matcher, action] :
496  llvm::zip_equal(getMatchers(), getActions())) {
497  auto matcherSymbol =
498  symbolTable.lookupNearestSymbolFrom<FunctionOpInterface>(
499  getOperation(), cast<SymbolRefAttr>(matcher));
500  auto actionSymbol =
501  symbolTable.lookupNearestSymbolFrom<FunctionOpInterface>(
502  getOperation(), cast<SymbolRefAttr>(action));
503  assert(matcherSymbol && actionSymbol &&
504  "unresolved symbols not caught by the verifier");
505 
506  if (matcherSymbol.isExternal())
507  return emitDefiniteFailure() << "unresolved external symbol " << matcher;
508  if (actionSymbol.isExternal())
509  return emitDefiniteFailure() << "unresolved external symbol " << action;
510 
511  matchActionPairs.emplace_back(matcherSymbol, actionSymbol);
512  }
513 
514  for (Operation *root : state.getPayloadOps(getRoot())) {
515  WalkResult walkResult = root->walk([&](Operation *op) {
516  // Skip over the root op itself so we don't invalidate it.
517  if (op == root)
518  return WalkResult::advance();
519 
520  DEBUG_MATCHER({
521  DBGS_MATCHER() << "matching ";
522  op->print(llvm::dbgs(),
523  OpPrintingFlags().assumeVerified().skipRegions());
524  llvm::dbgs() << " @" << op << "\n";
525  });
526 
527  // Try all the match/action pairs until the first successful match.
528  for (auto [matcher, action] : matchActionPairs) {
529  SmallVector<SmallVector<MappedValue>> mappings;
530  DiagnosedSilenceableFailure diag =
531  matchBlock(matcher.getFunctionBody().front(), op, state, mappings);
532  if (diag.isDefiniteFailure())
533  return WalkResult::interrupt();
534  if (diag.isSilenceableFailure()) {
535  DEBUG_MATCHER(DBGS_MATCHER() << "matcher " << matcher.getName()
536  << " failed: " << diag.getMessage());
537  continue;
538  }
539 
540  auto scope = state.make_isolated_region_scope(action.getFunctionBody());
541  for (auto &&[arg, map] : llvm::zip_equal(
542  action.getFunctionBody().front().getArguments(), mappings)) {
543  if (failed(state.mapBlockArgument(arg, map)))
544  return WalkResult::interrupt();
545  }
546 
547  for (Operation &transform :
548  action.getFunctionBody().front().without_terminator()) {
549  DiagnosedSilenceableFailure result =
550  state.applyTransform(cast<TransformOpInterface>(transform));
551  if (failed(result.checkAndReport()))
552  return WalkResult::interrupt();
553  }
554  break;
555  }
556  return WalkResult::advance();
557  });
558  if (walkResult.wasInterrupted())
559  return DiagnosedSilenceableFailure::definiteFailure();
560  }
561 
562  // The root operation should not have been affected, so we can just reassign
563  // the payload to the result. Note that we need to consume the root handle to
564  // make sure any handles to operations inside, that could have been affected
565  // by actions, are invalidated.
566  results.set(getUpdated().cast<OpResult>(), state.getPayloadOps(getRoot()));
567  return DiagnosedSilenceableFailure::success();
568 }

583 /// Parses the comma-separated list of symbol reference pairs of the format
584 /// `@matcher -> @action`.
585 static ParseResult parseForeachMatchSymbols(OpAsmParser &parser,
586  ArrayAttr &matchers,
587  ArrayAttr &actions) {
588  StringAttr matcher;
589  StringAttr action;
590  SmallVector<Attribute> matcherList;
591  SmallVector<Attribute> actionList;
592  do {
593  if (parser.parseSymbolName(matcher) || parser.parseArrow() ||
594  parser.parseSymbolName(action)) {
595  return failure();
596  }
597  matcherList.push_back(SymbolRefAttr::get(matcher));
598  actionList.push_back(SymbolRefAttr::get(action));
599  } while (parser.parseOptionalComma().succeeded());
600 
601  matchers = parser.getBuilder().getArrayAttr(matcherList);
602  actions = parser.getBuilder().getArrayAttr(actionList);
603  return success();
604 }

657 /// Checks that the attributes of the function-like operation have correct
658 /// consumption effect annotations. If `alsoVerifyInternal`, checks for
659 /// annotations being present even if they can be inferred from the body.
660 static DiagnosedSilenceableFailure
661 verifyFunctionLikeConsumeAnnotations(FunctionOpInterface op,
662  bool alsoVerifyInternal = false) {
663  auto transformOp = cast<transform::TransformOpInterface>(op.getOperation());
664  llvm::SmallDenseSet<unsigned> consumedArguments;
665  if (!op.isExternal()) {
666  transform::getConsumedBlockArguments(op.getFunctionBody().front(),
667  consumedArguments);
668  }
669  for (unsigned i = 0, e = op.getNumArguments(); i < e; ++i) {
670  bool isConsumed =
671  op.getArgAttr(i, transform::TransformDialect::kArgConsumedAttrName) !=
672  nullptr;
673  bool isReadOnly =
674  op.getArgAttr(i, transform::TransformDialect::kArgReadOnlyAttrName) !=
675  nullptr;
676  if (isConsumed && isReadOnly) {
677  return transformOp.emitSilenceableError()
678  << "argument #" << i << " cannot be both readonly and consumed";
679  }
680  if ((op.isExternal() || alsoVerifyInternal) && !isConsumed && !isReadOnly) {
681  return transformOp.emitSilenceableError()
682  << "must provide consumed/readonly status for arguments of "
683  "external or called ops";
684  }
685  if (op.isExternal())
686  continue;
687 
688  if (consumedArguments.contains(i) && !isConsumed && isReadOnly) {
689  return transformOp.emitSilenceableError()
690  << "argument #" << i
691  << " is consumed in the body but is not marked as such";
692  }
693  if (!consumedArguments.contains(i) && isConsumed) {
694  Diagnostic warning(op->getLoc(), DiagnosticSeverity::Warning);
695  warning << "argument #" << i
696  << " is not consumed in the body but is marked as consumed";
697  return DiagnosedSilenceableFailure::silenceableFailure(
698  std::move(warning));
699  }
700  }
701  return DiagnosedSilenceableFailure::success();

704 LogicalResult transform::ForeachMatchOp::verifySymbolUses(
705  SymbolTableCollection &symbolTable) {
706  assert(getMatchers().size() == getActions().size());
707  auto consumedAttr =
708  StringAttr::get(getContext(), TransformDialect::kArgConsumedAttrName);
709  for (auto &&[matcher, action] :
710  llvm::zip_equal(getMatchers(), getActions())) {
711  auto matcherSymbol = dyn_cast_or_null<FunctionOpInterface>(
712  symbolTable.lookupNearestSymbolFrom(getOperation(),
713  cast<SymbolRefAttr>(matcher)));
714  auto actionSymbol = dyn_cast_or_null<FunctionOpInterface>(
715  symbolTable.lookupNearestSymbolFrom(getOperation(),
716  cast<SymbolRefAttr>(action)));
717  if (!matcherSymbol ||
718  !isa<TransformOpInterface>(matcherSymbol.getOperation()))
719  return emitError() << "unresolved matcher symbol " << matcher;

732  return failure();

756  diag.attachNote(actionSymbol->getLoc()) << "symbol declaration";
757  return diag;
758  }
759 
760  if (matcherSymbol.getArgumentTypes().size() != 1 ||
761  !implementSameTransformInterface(matcherSymbol.getArgumentTypes()[0],

763  InFlightDiagnostic diag =
764  emitOpError() << "expects matcher symbol to have one argument with "
765  "the same transform interface as the first operand";
766  diag.attachNote(matcherSymbol->getLoc()) << "symbol declaration";

769

772  emitOpError()
773  << "does not expect matcher symbol to consume its operand";
774  diag.attachNote(matcherSymbol->getLoc()) << "symbol declaration";

776  }
777  }
778  return success();

995 /// Applies the transform ops contained in `block`. Maps `results` to the same
996 /// values as the operands of the block terminator.
997 static DiagnosedSilenceableFailure
998 applySequenceBlock(Block &block, transform::FailurePropagationMode mode,
999  transform::TransformState &state,
1000  transform::TransformResults &results) {
1001  // Apply the sequenced ops one by one.
1002  for (Operation &transform : block.without_terminator()) {
1003  DiagnosedSilenceableFailure result =
1004  state.applyTransform(cast<transform::TransformOpInterface>(transform));
1005  if (result.isDefiniteFailure())
1006  return result;
1007 
1008  if (result.isSilenceableFailure()) {
1009  if (mode == transform::FailurePropagationMode::Propagate) {
1010  // Propagate empty results in case of early exit.
1011  forwardEmptyOperands(&block, state, results);
1012  return result;
1013  }
1014  (void)result.silence();
1015  }
1016  }
1017 
1018  // Forward the operation mapping for values yielded from the sequence to the
1019  // values produced by the sequence op.
1020  transform::detail::forwardTerminatorOperands(&block, state, results);
1021  return DiagnosedSilenceableFailure::success();

1088 
1089 void transform::IncludeOp::getEffects(
1090  SmallVectorImpl<MemoryEffects::EffectInstance> &effects) {
1091  // Always mark as modifying the payload.
1092  // TODO: a mechanism to annotate effects on payload. Even when all handles are
1093  // only read, the payload may still be modified, so we currently stay on the

1102  // preconditions fail so the trait verifier doesn't complain about missing
1103  // effects and the real precondition failure is reported later on.
1104  auto defaultEffects = [&] { onlyReadsHandle(getOperands(), effects); };

1115  return defaultEffects();
1116  DiagnosedSilenceableFailure earlyVerifierResult =
1117  verifyNamedSequenceOp(callee);
1118  if (!earlyVerifierResult.succeeded()) {
1119  (void)earlyVerifierResult.silence();
1120  return defaultEffects();
1121  }
1122 
1123  for (unsigned i = 0, e = getNumOperands(); i < e; ++i) {
1124  if (callee.getArgAttr(i, TransformDialect::kArgConsumedAttrName))

1126  else

1198  std::string str;
1199  llvm::raw_string_ostream os(str);
1200  value.print(os, /*isSigned=*/true);
1201  return os.str();

1205  ArrayRef<Attribute> references = state.getParams(getReference());
1206 
1207  if (params.size() != references.size()) {
1208  return emitSilenceableError()
1209  << "parameters have different payload lengths (" << params.size()
1210  << " vs " << references.size() << ")";
1211  }
1212 
1213  for (auto &&[i, param, reference] : llvm::enumerate(params, references)) {
1214  auto intAttr = param.dyn_cast<IntegerAttr>();
1215  auto refAttr = reference.dyn_cast<IntegerAttr>();
1216  if (!intAttr || !refAttr) {
1217  return emitDefiniteFailure()
1218  << "non-integer parameter value not expected";
1219  }
1220  if (intAttr.getType() != refAttr.getType()) {
1221  return emitDefiniteFailure()
1222  << "mismatching integer attribute types in parameter #" << i;
1223  }
1224  APInt value = intAttr.getValue();
1225  APInt refValue = refAttr.getValue();
1226 
1227  // TODO: this copy will not be necessary in C++20.
1228  int64_t position = i;
1229  auto reportError = [&](StringRef direction) {
1230  DiagnosedSilenceableFailure diag =
1231  emitSilenceableError() << "expected parameter to be " << direction
1232  << " " << signedAPIntAsString(refValue)
1233  << ", got " << signedAPIntAsString(value);
1234  diag.attachNote(getParam().getLoc())
1235  << "value # " << position
1236  << " associated with the parameter defined here";
1237  return diag;
1238  };

1240  switch (getPredicate()) {
1241  case MatchCmpIPredicate::eq:
1242  if (value.eq(refValue))
1243  break;
1244  return reportError("equal to");
1245  case MatchCmpIPredicate::ne:
1246  if (value.ne(refValue))
1247  break;
1248  return reportError("not equal to");
1249  case MatchCmpIPredicate::lt:
1250  if (value.slt(refValue))
1251  break;
1252  return reportError("less than");
1253  case MatchCmpIPredicate::le:
1254  if (value.sle(refValue))
1255  break;
1256  return reportError("less than or equal to");
1257  case MatchCmpIPredicate::gt:
1258  if (value.sgt(refValue))
1259  break;

1364 /// implementing the same transform dialect type interface. If `allowExternal`
