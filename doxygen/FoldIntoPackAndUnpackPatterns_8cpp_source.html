23 /// Fold a `pad` -> `pack` into `pack` if they have the same padding values and
24 /// the pad op has zero low paddings, or if `pack` has no padding values.
25 struct FoldPadWithPackOp : public OpRewritePattern<PackOp> {
26  using OpRewritePattern<PackOp>::OpRewritePattern;
27 
28  LogicalResult matchAndRewrite(PackOp packOp,
29  PatternRewriter &rewriter) const override {
30  auto padOp = packOp.getSource().getDefiningOp<PadOp>();
31 
32  if (!padOp || padOp.getNofold() || !padOp.hasZeroLowPad())
33  return failure();
34 
35  Value constantPaddingValue = padOp.getConstantPaddingValue();
36  if (!constantPaddingValue)
37  return failure();
38 
39  if (auto paddingValue = packOp.getPaddingValue())
40  if (!isEqualConstantIntOrValue(paddingValue, constantPaddingValue))
41  return failure();
42 
43  rewriter.replaceOpWithNewOp<PackOp>(
44  packOp, padOp.getSource(), packOp.getDest(), packOp.getInnerDimsPos(),
45  packOp.getMixedTiles(), constantPaddingValue,
46  packOp.getOuterDimsPerm());
47  return success();
48  }
49 };

51 /// Fold a `unpack` -> `extract_slice` into the `unpack` since it already
52 /// has extract_slice semantics.
53 struct FoldUnpackWithExtractSliceOp : public OpRewritePattern<ExtractSliceOp> {
54  using OpRewritePattern<ExtractSliceOp>::OpRewritePattern;
55 
56  LogicalResult matchAndRewrite(ExtractSliceOp sliceOp,
57  PatternRewriter &rewriter) const override {
58  auto unpackOp = sliceOp.getSource().getDefiningOp<UnPackOp>();
59  if (!unpackOp)
60  return failure();
61 
62  // Check all offsets are zeros, and all strides are ones.
63  if (!areAllConstantIntValue(sliceOp.getMixedOffsets(), 0) ||
64  !areAllConstantIntValue(sliceOp.getMixedStrides(), 1)) {
65  return rewriter.notifyMatchFailure(
66  sliceOp, "expects offsets to be 0s and strides to be 1s");
67  }
68 
69  // Create a new empty output tensor.
70  Type elementType = unpackOp.getDestType().getElementType();
71  Value output = rewriter.create<EmptyOp>(
72  sliceOp.getLoc(), sliceOp.getMixedSizes(), elementType);
73  rewriter.replaceOpWithNewOp<UnPackOp>(
74  sliceOp, unpackOp.getSource(), output, unpackOp.getInnerDimsPos(),
75  unpackOp.getMixedTiles(), unpackOp.getOuterDimsPerm());
76  return success();
77  }
78 };
