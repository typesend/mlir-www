51 /// Flatten a list of operands that may contain sparse tensors.
52 static void flattenOperands(ValueRange operands,
53  SmallVectorImpl<Value> &flattened) {
54  // In case of
55  // sparse_tensor, c, sparse_tensor
56  // ==>
57  // memref ..., c, memref ...
58  for (auto operand : operands) {
59  if (getSparseTensorEncoding(operand.getType())) {
60  auto tuple = getTuple(operand);
61  // An unrealized_conversion_cast will be inserted by type converter to
62  // inter-mix the gap between 1:N conversion between sparse tensors and
63  // fields. In this case, take the operands in the cast and replace the
64  // sparse tensor output with the flattened type array.
65  flattened.append(tuple.getOperands().begin(), tuple.getOperands().end());
66  } else {
67  flattened.push_back(operand);
68  }
69  }
70 }

145 /// Generates code that allocates a sparse storage scheme for given rank.
146 static void allocSchemeForRank(OpBuilder &builder, Location loc,
147  MutSparseTensorDescriptor desc, Level startLvl) {
148  const SparseTensorType stt(desc.getRankedTensorType());
149  Value linear = constantIndex(builder, loc, 1);
150  const Level lvlRank = stt.getLvlRank();
151  for (Level l = startLvl; l < lvlRank; l++) {
152  const auto dlt = stt.getLvlType(l);
153  if (isCompressedDLT(dlt)) {
154  // Append linear x positions, initialized to zero. Since each compressed
155  // dimension initially already has a single zero entry, this maintains
156  // the desired "linear + 1" length property at all times.
157  Value posZero = constantZero(builder, loc, stt.getPosType());
158  createPushback(builder, loc, desc, SparseTensorFieldKind::PosMemRef, l,
159  posZero, linear);
160  return;
161  }
162  if (isSingletonDLT(dlt)) {
163  return; // nothing to do
164  }
165  // Keep compounding the size, but nothing needs to be initialized
166  // at this level. We will eventually reach a compressed level or
167  // otherwise the values array for the from-here "all-dense" case.
168  assert(isDenseDLT(dlt));
169  Value size = sizeFromTensorAtLvl(builder, loc, desc, l);
170  linear = builder.create<arith::MulIOp>(loc, linear, size);
171  }
172  // Reached values array so prepare for an insertion.
173  Value valZero = constantZero(builder, loc, stt.getElementType());
174  createPushback(builder, loc, desc, SparseTensorFieldKind::ValMemRef,
175  std::nullopt, valZero, linear);
176 }

191 /// Creates allocation for each field in sparse tensor type. Note that
192 /// for all dynamic memrefs, the memory size is really the capacity of
193 /// the "vector", while the actual size resides in the sizes array.
194 ///
195 /// TODO: for efficiency, we will need heuristics to make educated guesses
196 /// on the required capacities (see heuristic variable).
197 ///
198 static void createAllocFields(OpBuilder &builder, Location loc,
199  SparseTensorType stt, ValueRange dynSizes,
200  bool enableInit, SmallVectorImpl<Value> &fields,
201  Value sizeHint) {
202  // Build original sizes.
203  assert((dynSizes.size() == static_cast<size_t>(stt.getNumDynamicDims())) &&
204  "Got wrong number of dynamic sizes");
205  const Dimension dimRank = stt.getDimRank();
206  SmallVector<Value> dimSizes;
207  dimSizes.reserve(dimRank);
208  unsigned i = 0; // cumulative index into `dynSizes`.
209  for (const DynSize sh : stt.getDimShape())
210  dimSizes.push_back(ShapedType::isDynamic(sh)
211  ? dynSizes[i++]
212  : constantIndex(builder, loc, sh));
213 
214  // Set up some heuristic sizes. We try to set the initial
215  // size based on available information. Otherwise we just
216  // initialize a few elements to start the reallocation chain.
217  // TODO: refine this
218  Value posHeuristic, crdHeuristic, valHeuristic;
219  if (stt.isAllDense()) {
220  valHeuristic = dimSizes[0];
221  for (const Value sz : ArrayRef<Value>{dimSizes}.drop_front())
222  valHeuristic = builder.create<arith::MulIOp>(loc, valHeuristic, sz);
223  } else if (sizeHint) {
224  if (getCOOStart(stt.getEncoding()) == 0) {
225  posHeuristic = constantIndex(builder, loc, 2);
226  crdHeuristic = builder.create<arith::MulIOp>(
227  loc, constantIndex(builder, loc, dimRank), sizeHint); // AOS
228  } else if (dimRank == 2 && stt.isDenseLvl(0) && stt.isCompressedLvl(1)) {
229  posHeuristic = builder.create<arith::AddIOp>(
230  loc, sizeHint, constantIndex(builder, loc, 1));
231  crdHeuristic = sizeHint;
232  } else {
233  posHeuristic = crdHeuristic = constantIndex(builder, loc, 16);
234  }
235  valHeuristic = sizeHint;
236  } else {
237  posHeuristic = crdHeuristic = valHeuristic =
238  constantIndex(builder, loc, 16);
239  }

242  stt,
243  [&builder, &fields, stt, loc, posHeuristic, crdHeuristic, valHeuristic,
244  enableInit](Type fType, FieldIndex fIdx, SparseTensorFieldKind fKind,
245  Level /*lvl*/, DimLevelType /*dlt*/) -> bool {
246  assert(fields.size() == fIdx);
247  Value field;

265  // Returns true to continue the iteration.
266  return true;

271  // Initialize the storage scheme to an empty tensor. Initialized memSizes
272  // to all zeros, sets the dimSizes to known values and gives all position
273  // fields an initial zero entry, so that it is easier to maintain the
274  // "linear + 1" length property.
275  Value posZero = constantZero(builder, loc, stt.getPosType());

279  desc.setLvlSize(builder, loc, l, dimSizes[toOrigDim(stt, l)]);

282  createPushback(builder, loc, desc, SparseTensorFieldKind::PosMemRef, l,

284  }

288 /// Helper method that generates block specific to compressed case:
289 ///
290 /// // given: parentPos = posCursor[lvl-1]
291 /// pstart = desc.positions[lvl][parentPos]
292 /// pstop = desc.positions[lvl][parentPos+1]
293 /// plast = pstop - 1
294 /// msz = desc.coordinates[lvl].size()
295 /// if (pstart < pstop) {
296 /// isPresent = (desc.coordinates[lvl][plast] == lvlCoords[lvl])
297 /// } else { // first insertion
298 /// isPresent = false
299 /// desc.positions[lvl][parentPos] = msz
300 /// }
301 /// if (isPresent) { // coordinate is already present
302 /// pnext = plast
303 /// } else {
304 /// desc.coordinates[lvl].push_back(lvlCoords[lvl])
305 /// desc.positions[lvl][parentPos+1] = msz+1
306 /// pnext = msz
307 /// <prepare level lvl+1>
308 /// }
309 /// posCursor[lvl] = pnext
310 static Value genCompressed(OpBuilder &builder, Location loc,
311  MutSparseTensorDescriptor desc, ValueRange lvlCoords,
312  Value /*unused*/, Value parentPos, Level lvl) {
313  const SparseTensorType stt(desc.getRankedTensorType());
314  const Level lvlRank = stt.getLvlRank();
315  assert(lvl < lvlRank && "Level is out of bounds");
316  assert(lvlCoords.size() == static_cast<size_t>(lvlRank) &&
317  "Level-rank mismatch");
318  SmallVector<Type> types;
319  Type indexType = builder.getIndexType();
320  Type boolType = builder.getIntegerType(1);
321  unsigned crdFidx;
322  unsigned crdStride;
323  std::tie(crdFidx, crdStride) = desc.getCrdMemRefIndexAndStride(lvl);
324  const Value one = constantIndex(builder, loc, 1);
325  const Value pp1 = builder.create<arith::AddIOp>(loc, parentPos, one);
326  const Value positionsAtLvl = desc.getPosMemRef(lvl);
327  const Value pstart = genLoad(builder, loc, positionsAtLvl, parentPos);
328  const Value pstop = genLoad(builder, loc, positionsAtLvl, pp1);
329  const Value crdMsz = desc.getCrdMemSize(builder, loc, lvl);
330  const Value crdStrideC =
331  crdStride > 1 ? constantIndex(builder, loc, crdStride) : Value();
332  const Value msz =
333  crdStrideC ? builder.create<arith::DivUIOp>(loc, crdMsz, crdStrideC)
334  : crdMsz;
335  const Value plast = builder.create<arith::SubIOp>(
336  loc, genCast(builder, loc, pstop, indexType), one);
337  // Conditional expression.
338  Value lt = builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::ult,
339  pstart, pstop);
340  types.push_back(boolType);
341  scf::IfOp ifOp1 = builder.create<scf::IfOp>(loc, types, lt, /*else*/ true);
342  types.pop_back();
343  builder.setInsertionPointToStart(&ifOp1.getThenRegion().front());
344  Value crd =
345  genLoad(builder, loc, desc.getMemRefField(crdFidx),
346  crdStrideC ? builder.create<arith::MulIOp>(loc, plast, crdStrideC)
347  : plast);
348  Value eq = builder.create<arith::CmpIOp>(
349  loc, arith::CmpIPredicate::eq, genCast(builder, loc, crd, indexType),
350  lvlCoords[lvl]);
351  builder.create<scf::YieldOp>(loc, eq);
352  builder.setInsertionPointToStart(&ifOp1.getElseRegion().front());
353  if (lvl > 0)
354  genStore(builder, loc, msz, positionsAtLvl, parentPos);
355  builder.create<scf::YieldOp>(loc, constantI1(builder, loc, false));
356  builder.setInsertionPointAfter(ifOp1);
357  // If present construct. Note that for a non-unique dimension level, we
358  // simply set the condition to false and rely on CSE/DCE to clean up the IR.
359  //
360  // TODO: generate less temporary IR?
361  //
362  for (unsigned i = 0, e = desc.getNumFields(); i < e; i++)
363  types.push_back(desc.getField(i).getType());
364  types.push_back(indexType);
365  const Value p = stt.isUniqueLvl(lvl) ? ifOp1.getResult(0)
366  : constantI1(builder, loc, false);
367  scf::IfOp ifOp2 = builder.create<scf::IfOp>(loc, types, p, /*else*/ true);
368  // If present (fields unaffected, update pnext to plast).
369  builder.setInsertionPointToStart(&ifOp2.getThenRegion().front());
370 
371  // FIXME: This does not looks like a clean way, but probably the most
372  // efficient way.
373  desc.getFields().push_back(plast);
374  builder.create<scf::YieldOp>(loc, desc.getFields());
375  desc.getFields().pop_back();
376 
377  // If !present (changes fields, update pnext).
378  builder.setInsertionPointToStart(&ifOp2.getElseRegion().front());
379  Value mszp1 = builder.create<arith::AddIOp>(loc, msz, one);
380  genStore(builder, loc, mszp1, positionsAtLvl, pp1);
381  createPushback(builder, loc, desc, SparseTensorFieldKind::CrdMemRef, lvl,
382  lvlCoords[lvl]);
383  // Prepare the next level "as needed".
384  if ((lvl + 1) < lvlRank)
385  allocSchemeForRank(builder, loc, desc, lvl + 1);
386 
387  desc.getFields().push_back(msz);
388  builder.create<scf::YieldOp>(loc, desc.getFields());
389  desc.getFields().pop_back();
390 
391  // Update fields and return next pos.
392  builder.setInsertionPointAfter(ifOp2);
393  unsigned o = 0;
394  for (unsigned i = 0, e = desc.getNumFields(); i < e; i++)
395  desc.setField(i, ifOp2.getResult(o++));
396  return ifOp2.getResult(o);
397 }

399 /// Generates code along an insertion path without the need for a "cursor".
400 /// This current insertion strategy comes at the expense of some testing
401 /// overhead for each insertion. The strategy will be optimized later for
402 /// common insertion patterns. The current insertion strategy also assumes
403 /// insertions occur in "a reasonable order" that enables building the
404 /// storage scheme in an appending/inserting kind of fashion (i.e. no
405 /// in-between insertions that need data movement). The implementation
406 /// relies on CSE/DCE to clean up all bookkeeping that is not needed.
407 ///
408 /// TODO: better unord/not-unique; also generalize, optimize, specialize!
409 ///
410 static void genInsertBody(OpBuilder &builder, ModuleOp module,
411  func::FuncOp func, RankedTensorType rtp) {
412  const OpBuilder::InsertionGuard insertionGuard(builder);
413  Block *const entryBlock = func.addEntryBlock();
414  builder.setInsertionPointToStart(entryBlock);
415  const ValueRange args = entryBlock->getArguments();
416  const Location loc = func.getLoc();
417  const SparseTensorType stt(rtp);
418  const Level lvlRank = stt.getLvlRank();
419 
420  // Extract fields and coordinates from args.
421  SmallVector<Value> fields = llvm::to_vector(args.drop_back(lvlRank + 1));
422  MutSparseTensorDescriptor desc(rtp, fields);
423  const SmallVector<Value> coords =
424  llvm::to_vector(args.take_back(lvlRank + 1).drop_back());
425  Value value = args.back();
426  Value parentPos = constantZero(builder, loc, builder.getIndexType());
427  // Generate code for every level.
428  for (Level l = 0; l < lvlRank; l++) {
429  const auto dlt = stt.getLvlType(l);
430  if (isCompressedDLT(dlt)) {
431  // Create:
432  // if (!present) {
433  // coordinates[l].push_back(coords[l])
434  // <update positions and prepare level l + 1>
435  // }
436  // positions[l] = coordinates.size() - 1
437  // <insert @ positions[l] at next level l + 1>
438  parentPos =
439  genCompressed(builder, loc, desc, coords, value, parentPos, l);
440  } else if (isSingletonDLT(dlt)) {
441  // Create:
442  // coordinates[l].push_back(coords[l])
443  // positions[l] = positions[l-1]
444  // <insert @ positions[l] at next level l + 1>
445  createPushback(builder, loc, desc, SparseTensorFieldKind::CrdMemRef, l,
446  coords[l]);
447  } else {
448  assert(isDenseDLT(dlt));
449  // Construct the new position as:
450  // positions[l] = size * positions[l-1] + coords[l]
451  // <insert @ positions[l] at next level l + 1>
452  Value size = sizeFromTensorAtLvl(builder, loc, desc, l);
453  Value mult = builder.create<arith::MulIOp>(loc, size, parentPos);
454  parentPos = builder.create<arith::AddIOp>(loc, mult, coords[l]);
455  }
456  }
457  // Reached the actual value append/insert.
458  if (!stt.isDenseLvl(lvlRank - 1))
459  createPushback(builder, loc, desc, SparseTensorFieldKind::ValMemRef,
460  std::nullopt, value);
461  else
462  genStore(builder, loc, value, desc.getValMemRef(), parentPos);
463  builder.create<func::ReturnOp>(loc, fields);
464 }

466 /// Generates a call to a function to perform an insertion operation. If the
467 /// function doesn't exist yet, call `createFunc` to generate the function.
468 static void genInsertionCallHelper(OpBuilder &builder,
469  MutSparseTensorDescriptor desc,
470  SmallVectorImpl<Value> &lcvs, Value value,
471  func::FuncOp insertPoint,
472  StringRef namePrefix,
473  FuncGeneratorType createFunc) {
474  // The mangled name of the function has this format:
475  // <namePrefix>_<DLT>_<shape>_<ordering>_<eltType>_<crdWidth>_<posWidth>
476  const SparseTensorType stt(desc.getRankedTensorType());
477  SmallString<32> nameBuffer;
478  llvm::raw_svector_ostream nameOstream(nameBuffer);
479  nameOstream << namePrefix;
480  const Level lvlRank = stt.getLvlRank();
481  assert(lcvs.size() == static_cast<size_t>(lvlRank));
482  for (Level l = 0; l < lvlRank; l++)
483  nameOstream << toMLIRString(stt.getLvlType(l)) << "_";
484  // Static dim sizes are used in the generated code while dynamic sizes are
485  // loaded from the dimSizes buffer. This is the reason for adding the shape
486  // to the function name.
487  for (const auto sh : stt.getDimShape())
488  nameOstream << sh << "_";
489  // Permutation information is also used in generating insertion.
490  if (!stt.isIdentity())
491  nameOstream << stt.getDimToLvlMap() << "_";
492  nameOstream << stt.getElementType() << "_";
493  nameOstream << stt.getCrdWidth() << "_" << stt.getPosWidth();
494 
495  // Look up the function.
496  ModuleOp module = insertPoint->getParentOfType<ModuleOp>();
497  MLIRContext *context = module.getContext();
498  auto result = SymbolRefAttr::get(context, nameOstream.str());
499  auto func = module.lookupSymbol<func::FuncOp>(result.getAttr());
500 
501  // Construct operands: fields, coords, and value.
502  SmallVector<Value> operands = llvm::to_vector(desc.getFields());
503  operands.append(lcvs);
504  operands.push_back(value);
505  Location loc = insertPoint.getLoc();
506 
507  if (!func) {
508  // Create the function.
509  OpBuilder::InsertionGuard insertionGuard(builder);
510  builder.setInsertionPoint(insertPoint);
511 
512  func = builder.create<func::FuncOp>(
513  loc, nameOstream.str(),
514  FunctionType::get(context, ValueRange(operands).getTypes(),
515  ValueRange(desc.getFields()).getTypes()));
516  func.setPrivate();
517  createFunc(builder, module, func, stt);
518  }
519 
520  // Generate a call to perform the insertion and update `fields` with values
521  // returned from the call.
522  func::CallOp call = builder.create<func::CallOp>(loc, func, operands);
523  for (size_t i = 0, e = desc.getNumFields(); i < e; i++) {
524  desc.getFields()[i] = call.getResult(i);
525  }
526 }

528 /// Generations insertion finalization code.
529 static void genEndInsert(OpBuilder &builder, Location loc,
530  SparseTensorDescriptor desc) {
531  const SparseTensorType stt(desc.getRankedTensorType());
532  const Level lvlRank = stt.getLvlRank();
533  for (Level l = 0; l < lvlRank; l++) {
534  const auto dlt = stt.getLvlType(l);
535  if (isCompressedWithHiDLT(dlt))
536  llvm_unreachable("TODO: Not yet implemented");
537  if (isCompressedDLT(dlt)) {
538  // Compressed dimensions need a position cleanup for all entries
539  // that were not visited during the insertion pass.
540  //
541  // TODO: avoid cleanup and keep compressed scheme consistent at all
542  // times?
543  //
544  if (l > 0) {
545  Type posType = stt.getPosType();
546  Value posMemRef = desc.getPosMemRef(l);
547  Value hi = desc.getPosMemSize(builder, loc, l);
548  Value zero = constantIndex(builder, loc, 0);
549  Value one = constantIndex(builder, loc, 1);
550  // Vector of only one, but needed by createFor's prototype.
551  SmallVector<Value, 1> inits{genLoad(builder, loc, posMemRef, zero)};
552  scf::ForOp loop = createFor(builder, loc, hi, inits, one);
553  Value i = loop.getInductionVar();
554  Value oldv = loop.getRegionIterArg(0);
555  Value newv = genLoad(builder, loc, posMemRef, i);
556  Value posZero = constantZero(builder, loc, posType);
557  Value cond = builder.create<arith::CmpIOp>(
558  loc, arith::CmpIPredicate::eq, newv, posZero);
559  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, TypeRange(posType),
560  cond, /*else*/ true);
561  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
562  genStore(builder, loc, oldv, posMemRef, i);
563  builder.create<scf::YieldOp>(loc, oldv);
564  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
565  builder.create<scf::YieldOp>(loc, newv);
566  builder.setInsertionPointAfter(ifOp);
567  builder.create<scf::YieldOp>(loc, ifOp.getResult(0));
568  builder.setInsertionPointAfter(loop);
569  }
570  } else {
571  assert(isDenseDLT(dlt) || isSingletonDLT(dlt));
572  }
573  }

576 /// Returns a memref that fits the requested length (reallocates if requested
577 /// length is larger, or creates a subview if it is smaller).
578 static Value reallocOrSubView(OpBuilder &builder, Location loc, int64_t len,
579  Value buffer) {
580  MemRefType memTp = getMemRefType(buffer);
581  auto retTp = MemRefType::get(ArrayRef{len}, memTp.getElementType());
582 
583  Value targetLen = constantIndex(builder, loc, len);
584  Value bufferLen = linalg::createOrFoldDimOp(builder, loc, buffer, 0);
585  // Reallocates if target length is greater than the actual buffer len.
586  Value reallocP = builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::ugt,
587  targetLen, bufferLen);
588  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, retTp, reallocP, true);
589  // If targetLen > bufferLen, reallocate to get enough sparse to return.
590  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
591  Value reallocBuf = builder.create<memref::ReallocOp>(loc, retTp, buffer);
592  builder.create<scf::YieldOp>(loc, reallocBuf);
593  // Else, return a subview to fit the size.
594  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
595  Value subViewBuf = builder.create<memref::SubViewOp>(
596  loc, retTp, buffer, /*offset=*/ArrayRef<int64_t>{0},
597  /*size=*/ArrayRef<int64_t>{len},
598  /*stride=*/ArrayRef<int64_t>{1});
599  builder.create<scf::YieldOp>(loc, subViewBuf);
600  // Resets insertion point.
601  builder.setInsertionPointAfter(ifOp);
602  return ifOp.getResult(0);
603 }

605 static Value linearize(OpBuilder &builder, Location loc, ValueRange ivs,
606  ValueRange bounds) {
607  assert(ivs.size() == bounds.size());
608  Value crd = constantIndex(builder, loc, 0);
609  for (unsigned i = 0, e = ivs.size(); i < e; i++) {
610  crd = builder.create<arith::AddIOp>(loc, crd, ivs[i]);
611  if (i != ivs.size() - 1)
612  crd = builder.create<arith::MulIOp>(loc, crd, bounds[i + 1]);
613  }
614  return crd;

643 /// Sparse tensor storage conversion rule for calls.
644 class SparseCallConverter : public OpConversionPattern<func::CallOp> {
645 public:
646  // The default CallOp converter can not handle 1:N type conversion.
647  using OpConversionPattern::OpConversionPattern;
648  LogicalResult
649  matchAndRewrite(func::CallOp op, OpAdaptor adaptor,
650  ConversionPatternRewriter &rewriter) const override {
651  Location loc = op.getLoc();
652  // In case of:
653  // sparse_tensor, f, sparse_tensor = call @foo(...)
654  // ==>
655  // memref..., f, memref = call @foo(...) replace with
656  // cast(memref...)->sparse_tensor, f, cast(memref...)->sparse_tensor
657  SmallVector<Type> finalRetTy;
658  if (failed(typeConverter->convertTypes(op.getResultTypes(), finalRetTy)))
659  return failure();
660 
661  // (1) Genereates new call with flattened return value.
662  SmallVector<Value> flattened;
663  flattenOperands(adaptor.getOperands(), flattened);
664  auto newCall = rewriter.create<func::CallOp>(loc, op.getCallee(),
665  finalRetTy, flattened);
666  // (2) Create cast operation for sparse tensor returns.
667  SmallVector<Value> castedRet;
668  // Tracks the offset of current return value (of the orignal call)
669  // relative to the new call (after sparse tensor flattening);
670  unsigned retOffset = 0;
671  // Temporal buffer to hold the flattened list of type for
672  // a sparse tensor.
673  SmallVector<Type> sparseFlat;
674  for (auto ret : op.getResults()) {
675  assert(retOffset < newCall.getNumResults());
676  auto retType = ret.getType();
677  if (failed(typeConverter->convertType(retType, sparseFlat)))
678  // This should never happen.
679  llvm_unreachable("Failed to convert type in sparse tensor codegen");
680 
681  // Converted types can not be empty when the type conversion succeed.
682  assert(!sparseFlat.empty());
683  if (sparseFlat.size() > 1) {
684  auto flatSize = sparseFlat.size();
685  ValueRange fields(iterator_range<ResultRange::iterator>(
686  newCall.result_begin() + retOffset,
687  newCall.result_begin() + retOffset + flatSize));
688  castedRet.push_back(genTuple(rewriter, loc, retType, fields));
689  retOffset += flatSize;
690  } else {
691  // If this is an 1:1 conversion, no need for casting.
692  castedRet.push_back(newCall.getResult(retOffset));
693  retOffset++;
694  }
695  sparseFlat.clear();
696  }
697 
698  assert(castedRet.size() == op.getNumResults());
699  rewriter.replaceOp(op, castedRet);
700  return success();
701  }

870 /// Sparse codegen rule for the expand op.
871 class SparseExpandConverter : public OpConversionPattern<ExpandOp> {
872 public:
873  using OpConversionPattern::OpConversionPattern;
874  LogicalResult
875  matchAndRewrite(ExpandOp op, OpAdaptor adaptor,
876  ConversionPatternRewriter &rewriter) const override {
877  if (!getSparseTensorEncoding(op.getTensor().getType()))
878  return failure();
879  Location loc = op->getLoc();
880  auto desc = getDescriptorFromTensorTuple(adaptor.getTensor());
881  const auto srcType = getSparseTensorType(op.getTensor());
882  Type eltType = srcType.getElementType();
883  Type boolType = rewriter.getIntegerType(1);
884  Type idxType = rewriter.getIndexType();
885  // All initialization should be done on entry of the loop nest.
886  rewriter.setInsertionPointAfter(op.getTensor().getDefiningOp());
887  // Determine the size for access expansion (always the innermost stored
888  // level size, translated back to original dimension). Note that we
889  // recursively rewrite the new DimOp on the **original** tensor.
890  // FIXME: `toOrigDim` is deprecated.
891  const Dimension innerDim = toOrigDim(srcType, srcType.getLvlRank() - 1);
892  const auto sz = sizeFromTensorAtDim(rewriter, loc, desc, innerDim);
893  // Generate a memref for `sz` elements of type `t`.
894  const auto genAlloc = [&](Type t) {
895  const auto memTp = MemRefType::get({ShapedType::kDynamic}, t);
896  return rewriter.create<memref::AllocOp>(loc, memTp, ValueRange{sz});
897  };
898  // Allocate temporary buffers for values/filled-switch and added.
899  // We do not use stack buffers for this, since the expanded size may
900  // be rather large (as it envelops a single expanded dense dimension).
901  Value values = genAlloc(eltType);
902  Value filled = genAlloc(boolType);
903  Value added = genAlloc(idxType);
904  Value zero = constantZero(rewriter, loc, idxType);
905  // Reset the values/filled-switch to all-zero/false. Note that this
906  // introduces an O(N) operation into the computation, but this reset
907  // operation is amortized over the innermost loops for the access
908  // pattern expansion. As noted in the operation doc, we would like
909  // to amortize this setup cost even between kernels.
910  rewriter.create<linalg::FillOp>(
911  loc, ValueRange{constantZero(rewriter, loc, eltType)},
912  ValueRange{values});
913  rewriter.create<linalg::FillOp>(
914  loc, ValueRange{constantZero(rewriter, loc, boolType)},
915  ValueRange{filled});
916  // Replace expansion op with these buffers and initial coordinate.
917  assert(op.getNumResults() == 4);
918  rewriter.replaceOp(op, {values, filled, added, zero});
919  return success();
920  }

925 public:
926  using OpConversionPattern::OpConversionPattern;
927  LogicalResult
928  matchAndRewrite(CompressOp op, OpAdaptor adaptor,
929  ConversionPatternRewriter &rewriter) const override {
930  Location loc = op->getLoc();
931  SmallVector<Value> fields;
932  auto desc = getMutDescriptorFromTensorTuple(adaptor.getTensor(), fields);
933  Value values = adaptor.getValues();
934  Value filled = adaptor.getFilled();
935  Value added = adaptor.getAdded();
936  Value count = adaptor.getCount();
937  const SparseTensorType dstType(desc.getRankedTensorType());
938  Type eltType = dstType.getElementType();
939  // Prepare level-coords.
940  SmallVector<Value> lcvs(adaptor.getLvlCoords());
941  // If the innermost level is ordered, we need to sort the coordinates
942  // in the "added" array prior to applying the compression.
943  if (dstType.isOrderedLvl(dstType.getLvlRank() - 1))
944  rewriter.create<SortOp>(loc, count, ValueRange{added}, ValueRange{},
945  SparseTensorSortKind::HybridQuickSort);
946  // While performing the insertions, we also need to reset the elements
947  // of the values/filled-switch by only iterating over the set elements,
948  // to ensure that the runtime complexity remains proportional to the
949  // sparsity of the expanded access pattern.
950  //
951  // Generate
952  // out_memrefs = for (i = 0; i < count; i++)(in_memrefs) {
953  // crd = added[i];
954  // value = values[crd];
955  // insert({lvlCoords, crd}, value);
956  // new_memrefs = insert(in_memrefs, {lvlCoords, crd}, value);
957  // values[crd] = 0;
958  // filled[crd] = false;
959  // yield new_memrefs
960  // }
961  scf::ForOp loop = createFor(rewriter, loc, count, desc.getFields());
962  Value i = loop.getInductionVar();
963  Value crd = genLoad(rewriter, loc, added, i);
964  Value value = genLoad(rewriter, loc, values, crd);
965  lcvs.push_back(crd);
966  // TODO: faster for subsequent insertions?
967  auto insertPoint = op->template getParentOfType<func::FuncOp>();
968  genInsertionCallHelper(rewriter, desc, lcvs, value, insertPoint,
969  kInsertFuncNamePrefix, genInsertBody);
970  genStore(rewriter, loc, constantZero(rewriter, loc, eltType), values, crd);
971  genStore(rewriter, loc, constantI1(rewriter, loc, false), filled, crd);
972  rewriter.create<scf::YieldOp>(loc, desc.getFields());
973  rewriter.setInsertionPointAfter(loop);
974  Value result = genTuple(rewriter, loc, dstType, loop->getResults());
975  // Deallocate the buffers on exit of the full loop nest.
976  Operation *parent = getTop(op);
977  rewriter.setInsertionPointAfter(parent);
978  rewriter.create<memref::DeallocOp>(loc, values);
979  rewriter.create<memref::DeallocOp>(loc, filled);
980  rewriter.create<memref::DeallocOp>(loc, added);
981  // Replace operation with resulting memrefs.
982  rewriter.replaceOp(op, result);
983  return success();
984  }

1102  assert(!encDst.isSlice() && "Cannot convert to a sparse tensor slices.");

1104  // rewriting.

1122  // dst = memref.alloc

1124  // for every dst[i] = cast(src[i])

1126  // dst = memref.copy(src)

1137  fields.push_back(srcDesc.getSpecifier());

1142  // would require a subViewOp to avoid overflow when copying

1171  op, genTuple(rewriter, loc, op.getResult().getType(), fields));

1251 static void populateCompressedWithHiPosArray(OpBuilder &builder, Location loc,
1252  ArrayRef<unsigned> batchDimSzs,
1253  Value posMemRef, unsigned nse,
1254  PackOp op) {
1255  SmallVector<Value> lbs, ubs, steps;
1256  Value c0 = constantIndex(builder, loc, 0);
1257  Value c1 = constantIndex(builder, loc, 1);
1258  Value c2 = constantIndex(builder, loc, 2);
1259  for (unsigned dimSz : batchDimSzs) {
1260  lbs.push_back(c0);
1261  ubs.push_back(constantIndex(builder, loc, dimSz));
1262  steps.push_back(c1);
1263  }
1264  auto tensorType = op.getValues().getType();
1265  auto memrefType =
1266  MemRefType::get(tensorType.getShape(), tensorType.getElementType());
1267  Value batV = builder.create<bufferization::ToMemrefOp>(loc, memrefType,
1268  op.getValues());
1269  scf::buildLoopNest(
1270  builder, loc, lbs, ubs, steps,
1271  [&ubs, c0, c1, c2, nse, batV, posMemRef](OpBuilder &builder, Location loc,
1272  ValueRange ivs) {
1273  // Linearize index variables
1274  Value crd = linearize(builder, loc, ivs, ubs);
1275  Value len = constantIndex(builder, loc, nse);
1276  Value pLo = builder.create<arith::MulIOp>(loc, crd, len);
1277  SmallVector<Value> indices(ivs.begin(), ivs.end());
1278  auto whileOp = builder.create<scf::WhileOp>(
1279  loc, TypeRange{builder.getIndexType()}, ValueRange{len},
1280  [&indices, c0, c1, batV](OpBuilder &builder, Location loc,
1281  ValueRange vs) {
1282  Value curLen = vs.front();
1283  Value pred = builder.create<arith::CmpIOp>(
1284  loc, arith::CmpIPredicate::eq, curLen, c0);
1285  auto ifOp = builder.create<scf::IfOp>(
1286  loc, TypeRange{builder.getI1Type()}, pred, true);
1287  {
1288  OpBuilder::InsertionGuard guard(builder);
1289  // if len == 0.
1290  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
1291  builder.create<scf::YieldOp>(loc,
1292  constantI1(builder, loc, false));
1293  // Else branch.
1294  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
1295  indices.push_back(
1296  builder.create<arith::SubIOp>(loc, curLen, c1));
1297  Value val = builder.create<memref::LoadOp>(loc, batV, indices);
1298  indices.pop_back();
1299  Value cont = builder.create<arith::CmpFOp>(
1300  loc, arith::CmpFPredicate::OEQ, val,
1301  constantZero(builder, loc, val.getType()));
1302  builder.create<scf::YieldOp>(loc, cont);
1303  }
1304  builder.create<scf::ConditionOp>(loc, ifOp.getResults()[0], vs);
1305  },
1306  [c1](OpBuilder &builder, Location loc, ValueRange vs) {
1307  // len --;
1308  Value nxLen = builder.create<arith::SubIOp>(loc, vs.front(), c1);
1309  builder.create<scf::YieldOp>(loc, nxLen);
1310  });
1311  len = whileOp.getResults()[0];
1312  Value pHi = builder.create<arith::AddIOp>(loc, pLo, len);
1313  // Stores position lower bound.
1314  Value idx = builder.create<arith::MulIOp>(loc, crd, c2);
1315  genStore(builder, loc, pLo, posMemRef, idx);
1316  // Stores position upper bound.
1317  idx = builder.create<arith::AddIOp>(loc, idx, c1);
1318  genStore(builder, loc, pHi, posMemRef, idx);
1319  });
1320 }

1333  SmallVector<unsigned> batchDimSzs;
1334  batchDimSzs.reserve(batchedLvls);
1335  for (unsigned i = 0; i < batchedLvls; i++) {
1336  // Should already be guaranteed by verifier.
1337  assert(!ShapedType::isDynamic(stt.getDimShape()[i]));
1338  batchedCount *= stt.getDimShape()[i];
1339  batchDimSzs.push_back(stt.getDimShape()[i]);
1340  }
1341

1343  Location loc = op.getLoc();
1344 
1345  foreachFieldAndTypeInSparseTensor(
1346  stt,
1347  [&rewriter, &fields, &op, &batchDimSzs, nse, batchedCount, stt,
1348  loc](Type fType, FieldIndex fIdx, SparseTensorFieldKind fKind,
1349  Level /*lvl*/, DimLevelType dlt) -> bool {
1350  assert(fields.size() == fIdx);

1353  case SparseTensorFieldKind::StorageSpec:

1366  // [0, nse, 2 * nse, ..., batchedCount * nse]
1367  Value idx = constantIndex(rewriter, loc, i);
1368  Value val = constantIndex(rewriter, loc, nse * i);
1369  genStore(rewriter, loc, val, field, idx);
1370  }
1371  } else {
1372  assert(isCompressedWithHiDLT(dlt) && !batchDimSzs.empty());
1373  MemRefType posMemTp = MemRefType::get({batchedCount * 2}, posTp);
1374  field = rewriter.create<memref::AllocOp>(loc, posMemTp);
1375  populateCompressedWithHiPosArray(rewriter, loc, batchDimSzs,
1376  field, nse, op);
1377  }
1378  break;
1379  }
1380  case SparseTensorFieldKind::CrdMemRef: {
1381  auto tensorType = op.getCoordinates().getType();
1382  auto memrefType = MemRefType::get(tensorType.getShape(),
1383  tensorType.getElementType());
1384  field = rewriter.create<bufferization::ToMemrefOp>(
1385  op->getLoc(), memrefType, op.getCoordinates());
1386 
1387  break;
1388  }
1389  case SparseTensorFieldKind::ValMemRef: {
1390  auto tensorType = op.getValues().getType();
1391  auto memrefType = MemRefType::get(tensorType.getShape(),
1392  tensorType.getElementType());
1393  field = rewriter.create<bufferization::ToMemrefOp>(
1394  op->getLoc(), memrefType, op.getValues());
1395  break;
1396  }
1397  }
1398 
1399  assert(field);
1400  if (auto memrefTp = field.getType().dyn_cast<MemRefType>();
1401  memrefTp && memrefTp.getRank() > 1) {
1402  ReassociationIndices reassociation;
1403  for (int i = 0, e = memrefTp.getRank(); i < e; i++)
1404  reassociation.push_back(i);
1405  // Flattens the buffer to rank 1. The value buffer might need be
1406  // collapsed as well due to batching.
1407  field = rewriter.create<memref::CollapseShapeOp>(
1408  loc, field, ArrayRef<ReassociationIndices>(reassociation));
1409  }
1410 
1411  if (fType != field.getType())
1412  field = rewriter.create<memref::CastOp>(loc, fType, field);
1413  fields.push_back(field);
1414  // Returns true to continue the iteration.
1415  return true;
1416  });
1417 
1418  MutSparseTensorDescriptor desc(stt, fields);
1419  auto noe = linalg::createOrFoldDimOp(rewriter, loc, op.getValues(), 0);
1420  for (Level lvl = 0, lvlRank = stt.getLvlRank(); lvl < lvlRank; lvl++) {
1421  // FIXME: dim/lvl confusion!
1422  const auto sh = stt.getDimShape()[lvl];
1423  assert(!ShapedType::isDynamic(sh));
1424  desc.setLvlSize(rewriter, loc, lvl, constantIndex(rewriter, loc, sh));
1425  if (lvl == 0)
1426  desc.setPosMemSize(rewriter, loc, lvl, constantIndex(rewriter, loc, 2));
1427 
1428  desc.setCrdMemSize(rewriter, loc, lvl, noe);
1429  }
1430  desc.setValMemSize(rewriter, loc, noe);
1431 
1432  rewriter.replaceOp(op, genTuple(rewriter, loc, desc));
1433  return success();
1434  }

1481 static LogicalResult genBatchedUnpackOp(UnpackOp op, unsigned nBatched,
1482  SparseTensorDescriptor desc,
1483  ConversionPatternRewriter &rewriter) {
1484  assert(nBatched != 0);
1485  Location loc = op.getLoc();
1486  Value c0 = constantIndex(rewriter, loc, 0);
1487  Value c1 = constantIndex(rewriter, loc, 1);
1488  Value c2 = constantIndex(rewriter, loc, 2);
1489 
1490  auto genZeroedAlloc = [loc,
1491  &rewriter](TensorType tt) -> TypedValue<MemRefType> {
1492  auto mem = rewriter
1493  .create<memref::AllocOp>(
1494  loc, MemRefType::get(tt.getShape(), tt.getElementType()))
1495  .getMemref();
1496  // TODO: Instead of filling the entire buffer, we can only fill the
1497  // trailing zeros.
1498  rewriter.create<linalg::FillOp>(
1499  loc, ValueRange{constantZero(rewriter, loc, tt.getElementType())}, mem);
1500  return mem;
1501  };
1502  SparseTensorType stt = getSparseTensorType(op.getTensor());
1503  TensorType valTensorTp = op.getValues().getType();
1504  TensorType crdTensorTp = op.getCoordinates().getType();
1505  TypedValue<MemRefType> valMemref = genZeroedAlloc(valTensorTp);
1506  TypedValue<MemRefType> crdMemref = genZeroedAlloc(crdTensorTp);
1507  assert(valTensorTp.hasStaticShape() && crdTensorTp.hasStaticShape());
1508 
1509  SmallVector<Value> lbs(nBatched, c0), steps(nBatched, c1);
1510  SmallVector<Value> ubs;
1511  for (unsigned i = 0; i < nBatched; i++) {
1512  assert(!ShapedType::isDynamic(stt.getDimShape()[i]));
1513  ubs.push_back(constantIndex(rewriter, loc, stt.getDimShape()[i]));
1514  }
1515 
1516  DimLevelType dlt = stt.getLvlType(nBatched);
1517  assert(isCompressedDLT(dlt) || isCompressedWithHiDLT(dlt));
1518  Value posStep = isCompressedDLT(dlt) ? c1 // forward position index by 1
1519  : c2; // forward position index by 2
1520  auto loopNest = scf::buildLoopNest(
1521  rewriter, loc, lbs, ubs, steps, {c0 /*maximum nse*/},
1522  [&ubs, c0, c1, posStep, desc, nBatched, &valMemref,
1523  &crdMemref](OpBuilder &builder, Location loc, ValueRange ivs,
1524  ValueRange args) -> scf::ValueVector {
1525  // crdMemref has shape: <... x nse x rank>
1526  unsigned unBatchedRank = crdMemref.getType().getShape().back();
1527  Value values = desc.getValMemRef();
1528  Value flatCrds = unBatchedRank == 1
1529  ? desc.getCrdMemRefOrView(builder, loc, 0)
1530  : desc.getAOSMemRef();
1531 
1532  Value positions = desc.getPosMemRef(nBatched);
1533  Value positLo = builder.create<arith::MulIOp>(
1534  loc, linearize(builder, loc, ivs, ubs), posStep);
1535  Value positHi = builder.create<arith::AddIOp>(loc, positLo, c1);
1536 
1537  Value pLo = genIndexLoad(builder, loc, positions, positLo);
1538  Value pHi = genIndexLoad(builder, loc, positions, positHi);
1539  Value nse = builder.create<arith::SubIOp>(loc, pHi, pLo);
1540 
1541  Value crdLo = builder.create<arith::MulIOp>(
1542  loc, pLo, constantIndex(builder, loc, unBatchedRank));
1543  Value nCrd = builder.create<arith::MulIOp>(
1544  loc, nse, constantIndex(builder, loc, unBatchedRank));
1545 
1546  SmallVector<Value> offsets, sizes, strides;
1547  for (unsigned i = 0; i < nBatched; i++) {
1548  offsets.push_back(ivs[i]);
1549  sizes.push_back(c1);
1550  strides.push_back(c1);
1551  }
1552  // [0, nse, 1].
1553  offsets.push_back(c0);
1554  sizes.push_back(nse);
1555  strides.push_back(c1);
1556

1558  loc, valMemref, offsets, sizes, strides);
1559  auto valReass = getReassociationForFlattening(valView.getType());
1560  Value valDst =
1561  builder.create<memref::CollapseShapeOp>(loc, valView, valReass);
1562  Value valSrc =
1563  builder.create<memref::SubViewOp>(loc, values, pLo, nse, c1);
1564  builder.create<memref::CopyOp>(loc, valSrc, valDst);
1565 
1566  // [0, rank, 1].
1567  offsets.push_back(c0);
1568  sizes.push_back(constantIndex(builder, loc, unBatchedRank));
1569  strides.push_back(c1);
1570 
1571  auto crdView = builder.create<memref::SubViewOp>(
1572  loc, crdMemref, offsets, sizes, strides);
1573  auto crdReass = getReassociationForFlattening(crdView.getType());
1574  Value crdDst =
1575  builder.create<memref::CollapseShapeOp>(loc, crdView, crdReass);
1576  Value crdSrc =
1577  builder.create<memref::SubViewOp>(loc, flatCrds, crdLo, nCrd, c1);
1578  builder.create<memref::CopyOp>(loc, crdSrc, crdDst);
1579 
1580  Value pred = builder.create<arith::CmpIOp>(
1581  loc, arith::CmpIPredicate::ugt, nse, args[0]);
1582  // Choose the larger NSE
1583  return {builder.create<arith::SelectOp>(loc, pred, nse, args[0])};
1584  });
1585 
1586  // Converts MemRefs back to Tensors.
1587  Value values = rewriter.create<bufferization::ToTensorOp>(loc, valMemref);
1588  Value coordinates =
1589  rewriter.create<bufferization::ToTensorOp>(loc, crdMemref);
1590  Value nse =
1591  genCast(rewriter, loc, loopNest.results.front(), op.getNse().getType());
1592 
1593  rewriter.replaceOp(op, {values, coordinates, nse});
1594  return success();
1595 }

1637  LogicalResult
1638  matchAndRewrite(NewOp op, OpAdaptor adaptor,
1639  ConversionPatternRewriter &rewriter) const override {
1640  Location loc = op.getLoc();
1641  const auto dstTp = getSparseTensorType(op.getResult());
1642  // Creating COO with NewOp is handled by direct IR codegen. All other cases
1643  // are handled by rewriting.
1644  if (!dstTp.hasEncoding() || getCOOStart(dstTp.getEncoding()) != 0)
1645  return failure();
1646 
1647  // Implement the NewOp(filename) as follows:
1648  // %reader = @getSparseTensorReader(%filename)
1649  // %nse = @getSparseTensorNSE(%reader)
1650  // %coo = bufferization.alloc_tensor an ordered COO with
1651  // dst dim ordering, size_hint = %nse
1652  // %coordinates = sparse_tensor.coordinates_buffer(%coo)
1653  // %values = sparse_tensor.values(%coo)
1654  // %isSorted = @sparseTensorReaderReadToBuffers(%coordinates, %values)
1655  // if (! %isSorted) sparse_tensor.sort_coo(%nse, %coordinates, %values)
1656  // update storage specifier
1657  // @delSparseTensorReader(%reader)
1658 
1659  // Create a sparse tensor reader.
1660  const Value fileName = op.getSource();
1661  const Type opaqueTp = getOpaquePointerType(rewriter);
1662  // FIXME: use `createCheckedSparseTensorReader` instead, because
1663  // `createSparseTensorReader` is unsafe.
1664  Value reader = createFuncCall(rewriter, loc, "createSparseTensorReader",
1665  {opaqueTp}, {fileName}, EmitCInterface::Off)
1666  .getResult(0);
1667 
1668  const Type indexTp = rewriter.getIndexType();
1669  const Dimension dimRank = dstTp.getDimRank();
1670  const Level lvlRank = dstTp.getLvlRank();
1671 
1672  // If the result tensor has dynamic dimensions, get the dynamic sizes from
1673  // the sparse tensor reader.
1674  SmallVector<Value> dynSizes;
1675  if (dstTp.hasDynamicDimShape()) {
1676  // FIXME: call `getSparseTensorReaderDimSizes` instead, because
1677  // `copySparseTensorReaderDimSizes` copies the memref over,
1678  // instead of just accessing the reader's memory directly.
1679  Value dimSizes = genAlloca(rewriter, loc, dimRank, indexTp);
1680  createFuncCall(rewriter, loc, "copySparseTensorReaderDimSizes", {},
1681  {reader, dimSizes}, EmitCInterface::On);
1682  for (const auto &d : llvm::enumerate(dstTp.getDimShape()))
1683  if (ShapedType::isDynamic(d.value()))
1684  dynSizes.push_back(rewriter.create<memref::LoadOp>(
1685  loc, dimSizes, constantIndex(rewriter, loc, d.index())));
1686  }
1687 
1688  Value nse = createFuncCall(rewriter, loc, "getSparseTensorReaderNSE",
1689  {indexTp}, {reader}, EmitCInterface::Off)
1690  .getResult(0);
1691  // Construct allocation for each field.
1692  SmallVector<Value> fields;
1693  createAllocFields(rewriter, loc, dstTp, dynSizes, /*enableInit=*/false,
1694  fields, nse);
1695  MutSparseTensorDescriptor desc(dstTp, fields);
1696 
1697  // Construct the `dim2lvl` buffer for handing off to the runtime library.
1698  // FIXME: This code is (mostly) copied from the SparseTensorConversion.cpp
1699  // handling of `NewOp`, and only handles permutations. Fixing this
1700  // requires waiting for wrengr to finish redoing the CL that handles
1701  // all dim<->lvl stuff more robustly.
1702  SmallVector<Value> dim2lvlValues(dimRank);
1703  if (!dstTp.isIdentity()) {
1704  const auto dimOrder = dstTp.getDimToLvlMap();
1705  assert(dimOrder.isPermutation() && "Got non-permutation");
1706  for (Level l = 0; l < lvlRank; l++) {
1707  const Dimension d = dimOrder.getDimPosition(l);
1708  dim2lvlValues[d] = constantIndex(rewriter, loc, l);
1709  }
1710  } else {
1711  // The `SparseTensorType` ctor already ensures `dimRank == lvlRank`
1712  // when `isIdentity`; so no need to re-assert it here.
1713  for (Dimension d = 0; d < dimRank; d++)
1714  dim2lvlValues[d] = constantIndex(rewriter, loc, d);
1715  }
1716  Value dim2lvl = allocaBuffer(rewriter, loc, dim2lvlValues);
1717 
1718  // Read the COO tensor data.
1719  Value xs = desc.getAOSMemRef();
1720  Value ys = desc.getValMemRef();
1721 
1722  const Type boolTp = rewriter.getIntegerType(1);
1723  const Type elemTp = dstTp.getElementType();
1724  const Type crdTp = dstTp.getCrdType();
1725  // FIXME: This function name is weird; should rename to
1726  // "sparseTensorReaderReadToBuffers".
1727  SmallString<32> readToBuffersFuncName{"getSparseTensorReaderRead",
1728  overheadTypeFunctionSuffix(crdTp),
1729  primaryTypeFunctionSuffix(elemTp)};
1730  Value isSorted =
1731  createFuncCall(rewriter, loc, readToBuffersFuncName, {boolTp},
1732  {reader, dim2lvl, xs, ys}, EmitCInterface::On)
1733  .getResult(0);
1734 
1735  // If the destination tensor is a sorted COO, we need to sort the COO tensor
1736  // data if the input elements aren't sorted yet.
1737  if (dstTp.isOrderedLvl(lvlRank - 1)) {
1738  Value kFalse = constantI1(rewriter, loc, false);
1739  Value notSorted = rewriter.create<arith::CmpIOp>(
1740  loc, arith::CmpIPredicate::eq, isSorted, kFalse);
1741  scf::IfOp ifOp =
1742  rewriter.create<scf::IfOp>(loc, notSorted, /*else*/ false);
1743  rewriter.setInsertionPointToStart(&ifOp.getThenRegion().front());
1744  rewriter.create<SortCooOp>(
1745  loc, nse, xs, ValueRange{ys}, rewriter.getIndexAttr(lvlRank),
1746  rewriter.getIndexAttr(0), SparseTensorSortKind::HybridQuickSort);
1747  rewriter.setInsertionPointAfter(ifOp);
1748  }
1749 
1750  // Set PosMemRef0[1] = nse.
1751  const Value c1 = constantIndex(rewriter, loc, 1);
1752  const Value posMemref0 = desc.getPosMemRef(0);
1753  const Type posTp = dstTp.getPosType();
1754  const Value posNse = genCast(rewriter, loc, nse, posTp);
1755  rewriter.create<memref::StoreOp>(loc, posNse, posMemref0, c1);
1756 
1757  // Update storage specifier.
1758  Value coordinatesSize = rewriter.create<arith::MulIOp>(
1759  loc, nse, constantIndex(rewriter, loc, lvlRank));
1760  desc.setSpecifierField(rewriter, loc, StorageSpecifierKind::CrdMemSize, 0,
1761  coordinatesSize);
1762  desc.setSpecifierField(rewriter, loc, StorageSpecifierKind::ValMemSize,
1763  std::nullopt, nse);
1764 
1765  // Release the sparse tensor reader.
1766  createFuncCall(rewriter, loc, "delSparseTensorReader", {}, {reader},
1767  EmitCInterface::Off);
1768 
1769  // Replace operation with resulting memrefs.
1770  rewriter.replaceOp(op, genTuple(rewriter, loc, dstTp, fields));
1771  return success();
1772  }
