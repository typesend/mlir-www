42 /// Simplify affine.min / affine.max ops with the given constraints. They are
43 /// either rewritten to affine.apply or left unchanged.
44 template <typename OpTy>
45 struct SimplifyAffineMinMaxOp : public OpRewritePattern<OpTy> {
46  using OpRewritePattern<OpTy>::OpRewritePattern;
47  SimplifyAffineMinMaxOp(MLIRContext *ctx,
48  const FlatAffineValueConstraints &constraints,
49  PatternBenefit benefit = 1)
50  : OpRewritePattern<OpTy>(ctx, benefit), constraints(constraints) {}
51 
52  LogicalResult matchAndRewrite(OpTy op,
53  PatternRewriter &rewriter) const override {
54  FailureOr<AffineValueMap> simplified =
55  simplifyConstrainedMinMaxOp(op, constraints);
56  if (failed(simplified))
57  return failure();
58  rewriter.replaceOpWithNewOp<AffineApplyOp>(op, simplified->getAffineMap(),
59  simplified->getOperands());
60  return success();
61  }
62 
63  const FlatAffineValueConstraints &constraints;
64 };

67 DiagnosedSilenceableFailure
68 SimplifyBoundedAffineOpsOp::apply(TransformResults &results,
69  TransformState &state) {
70  // Get constraints for bounded values.
71  SmallVector<int64_t> lbs;
72  SmallVector<int64_t> ubs;
73  SmallVector<Value> boundedValues;
74  DenseSet<Operation *> boundedOps;
75  for (const auto &it : llvm::zip_equal(getBoundedValues(), getLowerBounds(),
76  getUpperBounds())) {
77  Value handle = std::get<0>(it);
78  ArrayRef<Operation *> boundedValueOps = state.getPayloadOps(handle);
79  for (Operation *op : boundedValueOps) {
80  if (op->getNumResults() != 1 || !op->getResult(0).getType().isIndex()) {
81  auto diag =
82  emitDefiniteFailure()
83  << "expected bounded value handle to point to one or multiple "
84  "single-result index-typed ops";
85  diag.attachNote(op->getLoc()) << "multiple/non-index result";
86  return diag;
87  }
88  boundedValues.push_back(op->getResult(0));
89  boundedOps.insert(op);
90  lbs.push_back(std::get<1>(it));
91  ubs.push_back(std::get<2>(it));
92  }
93  }
94 
95  // Build constraint set.
96  FlatAffineValueConstraints cstr;
97  for (const auto &it : llvm::zip(boundedValues, lbs, ubs)) {
98  unsigned pos;
99  if (!cstr.findVar(std::get<0>(it), &pos))
100  pos = cstr.appendSymbolVar(std::get<0>(it));
101  cstr.addBound(presburger::BoundType::LB, pos, std::get<1>(it));
102  // Note: addBound bounds are inclusive, but specified UB is exclusive.
103  cstr.addBound(presburger::BoundType::UB, pos, std::get<2>(it) - 1);
104  }
105 
106  // Transform all targets.
107  ArrayRef<Operation *> targets = state.getPayloadOps(getTarget());
108  for (Operation *target : targets) {
109  if (!isa<AffineMinOp, AffineMaxOp>(target)) {
110  auto diag = emitDefiniteFailure()
111  << "target must be affine.min or affine.max";
112  diag.attachNote(target->getLoc()) << "target op";
113  return diag;
114  }
115  if (boundedOps.contains(target)) {
116  auto diag = emitDefiniteFailure()
117  << "target op result must not be constrainted";
118  diag.attachNote(target->getLoc()) << "target/constrained op";
119  return diag;
120  }
121  }
122  SmallVector<Operation *> transformed;
123  RewritePatternSet patterns(getContext());
124  // Canonicalization patterns are needed so that affine.apply ops are composed
125  // with the remaining affine.min/max ops.
126  AffineMaxOp::getCanonicalizationPatterns(patterns, getContext());
127  AffineMinOp::getCanonicalizationPatterns(patterns, getContext());
128  patterns.insert<SimplifyAffineMinMaxOp<AffineMinOp>,
129  SimplifyAffineMinMaxOp<AffineMaxOp>>(getContext(), cstr);
130  FrozenRewritePatternSet frozenPatterns(std::move(patterns));
131  GreedyRewriteConfig config;
132  config.strictMode = GreedyRewriteStrictness::ExistingAndNewOps;
133  // Apply the simplification pattern to a fixpoint.
134  if (failed(applyOpPatternsAndFold(targets, frozenPatterns, config))) {
135  auto diag = emitDefiniteFailure()
136  << "affine.min/max simplification did not converge";
137  return diag;
138  }
139  return DiagnosedSilenceableFailure::success();
140 }
