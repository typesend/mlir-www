38  assert(createdWritten.second && "bbarg access info exists already");

42 /// Return the unique ReturnOp that terminates `funcOp`.
43 /// Return nullptr if there is no such unique ReturnOp.
44 static func::ReturnOp getAssumedUniqueReturnOp(FuncOp funcOp) {
45  func::ReturnOp returnOp;
46  for (Block &b : funcOp.getBody()) {
47  if (auto candidateOp = dyn_cast<func::ReturnOp>(b.getTerminator())) {
48  if (returnOp)
49  return nullptr;
50  returnOp = candidateOp;
51  }
52  }
53  return returnOp;
54 }

56 /// Return the index-th bufferized function argument type. This assumes that the
57 /// specified argument is a tensor. If the tensor is ranked, a layout map may be
58 /// specified by the user (as per `options.functionArgTypeConverterFn`).
59 static BaseMemRefType
60 getBufferizedFunctionArgType(FuncOp funcOp, int64_t index,
61  const BufferizationOptions &options) {
62  auto tensorType =
63  funcOp.getFunctionType().getInput(index).dyn_cast<TensorType>();
64  assert(tensorType && "expected TensorType");
65 
66  BaseMemRefType memrefType = options.functionArgTypeConverterFn(
67  tensorType, *options.defaultMemorySpace, funcOp, options);
68 
69  auto layoutAttr = funcOp.getArgAttrOfType<AffineMapAttr>(
70  index, BufferizationDialect::kBufferLayoutAttrName);
71  if (!layoutAttr)
72  return memrefType;
73 
74  auto rankedMemrefType = memrefType.dyn_cast<MemRefType>();
75  assert(rankedMemrefType && "buffer layout not supported on unranked tensors");
76  return MemRefType::get(
77  rankedMemrefType.getShape(), rankedMemrefType.getElementType(),
78  layoutAttr.getValue(), rankedMemrefType.getMemorySpace());
79 }

100 /// Return the state (phase) of analysis of the FuncOp.
101 static FuncOpAnalysisState getFuncOpAnalysisState(const AnalysisState &state,
102  FuncOp funcOp) {
103  if (!isa<OneShotAnalysisState>(state))
104  return FuncOpAnalysisState::NotAnalyzed;
105  auto *funcState = static_cast<const OneShotAnalysisState &>(state)
106  .getExtension<FuncAnalysisState>();
107  if (!funcState)
108  return FuncOpAnalysisState::NotAnalyzed;
109  const auto &analyzedFuncOps = funcState->analyzedFuncOps;
110  auto it = analyzedFuncOps.find(funcOp);
111  if (it == analyzedFuncOps.end())
112  return FuncOpAnalysisState::NotAnalyzed;
113  return it->second;

116 /// Return the index of the bbArg in the given FuncOp that is equivalent to the
117 /// specified return value (if any).
118 static std::optional<int64_t>
119 getEquivalentFuncArgIdx(FuncOp funcOp, const FuncAnalysisState &state,
120  int64_t returnValIdx) {
121  auto funcOpIt = state.equivalentFuncArgs.find(funcOp);
122  if (funcOpIt == state.equivalentFuncArgs.end())
123  // No equivalence info stores for funcOp.
124  return std::nullopt;
125 
126  auto retValIt = funcOpIt->getSecond().find(returnValIdx);
127  if (retValIt == funcOpIt->getSecond().end())
128  // Return value has no equivalent bbArg.
129  return std::nullopt;
130 
131  return retValIt->getSecond();
132 }

167  AliasingOpResultList getAliasingOpResults(Operation *op, OpOperand &opOperand,
168  const AnalysisState &state) const {
169  func::CallOp callOp = cast<func::CallOp>(op);
170  FuncOp funcOp = getCalledFunction(callOp);
171  assert(funcOp && "expected CallOp to a FuncOp");
172  if (getFuncOpAnalysisState(state, funcOp) != FuncOpAnalysisState::Analyzed)
173  // FuncOp not analyzed yet. Any OpResult may be aliasing.
174  return detail::unknownGetAliasingOpResults(opOperand);
175 
176  // Get aliasing results from state.
177  const FuncAnalysisState &funcState = getFuncAnalysisState(state);
178  auto aliasingReturnVals =
179  funcState.aliasingReturnVals.lookup(funcOp).lookup(
180  opOperand.getOperandNumber());
181 
182  // Check if the aliasing OpResult is equivalent to the OpOperand.
183  std::optional<int64_t> equivalent = {};
184  if (aliasingReturnVals.size() == 1) {
185  equivalent = getEquivalentFuncArgIdx(funcOp, funcState,
186  aliasingReturnVals.front());
187  assert((!equivalent.has_value() ||
188  *equivalent == opOperand.getOperandNumber()) &&
189  "inconsistent analysis state");
190  }
191  AliasingOpResultList result;
192  for (int64_t resultIdx : aliasingReturnVals)
193  result.addAlias({callOp->getOpResult(resultIdx),
194  equivalent.has_value() ? BufferRelation::Equivalent
195  : BufferRelation::Unknown,
196  /*isDefinite=*/equivalent.has_value()});
197  return result;
198  }

200  /// All function arguments are writable. It is the responsibility of the
201  /// CallOp to insert buffer copies where necessary.
202  LogicalResult bufferize(Operation *op, RewriterBase &rewriter,
203  const BufferizationOptions &options) const {
204  func::CallOp callOp = cast<func::CallOp>(op);
205  unsigned numResults = callOp.getNumResults();
206  unsigned numOperands = callOp->getNumOperands();
207  FuncOp funcOp = getCalledFunction(callOp);
208  assert(funcOp && "expected CallOp to a FuncOp");
209  FunctionType funcType = funcOp.getFunctionType();
210 
211  // Result types of the bufferized CallOp.
212  SmallVector<Type> resultTypes;
213  // Replacement values for the existing CallOp. These are usually the results
214  // of the bufferized CallOp, unless a tensor result folds onto an operand.
215  SmallVector<Value> replacementValues(numResults, Value());
216  // For non-tensor results: A mapping from return val indices of the old
217  // CallOp to return val indices of the bufferized CallOp.
218  SmallVector<std::optional<unsigned>> retValMapping(numResults,
219  std::nullopt);
220  // Operands of the bufferized CallOp.
221  SmallVector<Value> newOperands(numOperands, Value());
222 
223  // 1. Compute the result types of the new CallOp.
224  for (const auto &it : llvm::enumerate(callOp.getResultTypes())) {
225  unsigned returnValIdx = it.index();
226  Type returnType = it.value();
227  if (!returnType.isa<TensorType>()) {
228  // Non-tensor values are returned.
229  retValMapping[returnValIdx] = resultTypes.size();
230  resultTypes.push_back(returnType);
231  continue;
232  }
233 
234  // Returning a memref.
235  retValMapping[returnValIdx] = resultTypes.size();
236  resultTypes.push_back(funcType.getResult(resultTypes.size()));
237  }
238 
239  // 2. Rewrite tensor operands as memrefs based on `bufferizedFuncType`.
240  for (OpOperand &opOperand : callOp->getOpOperands()) {
241  unsigned idx = opOperand.getOperandNumber();
242  Value tensorOperand = opOperand.get();
243 
244  // Non-tensor operands are just copied.
245  if (!tensorOperand.getType().isa<TensorType>()) {
246  newOperands[idx] = tensorOperand;
247  continue;
248  }
249 
250  // Retrieve buffers for tensor operands.
251  Value buffer = newOperands[idx];
252  if (!buffer) {
253  FailureOr<Value> maybeBuffer =
254  getBuffer(rewriter, opOperand.get(), options);
255  if (failed(maybeBuffer))
256  return failure();
257  buffer = *maybeBuffer;
258  }
259 
260  // Caller / callee type mismatch is handled with a CastOp.
261  auto memRefType = funcType.getInput(idx);
262  // Since we don't yet have a clear layout story, to_memref may
263  // conservatively turn tensors into more dynamic memref than necessary.
264  // If the memref type of the callee fails, introduce an extra memref.cast
265  // that will either canonicalize away or fail compilation until we can do
266  // something better.
267  if (buffer.getType() != memRefType) {
268  assert(
269  memref::CastOp::areCastCompatible(buffer.getType(), memRefType) &&
270  "CallOp::bufferize: cast incompatible");
271  Value castBuffer = rewriter.create<memref::CastOp>(callOp.getLoc(),
272  memRefType, buffer);
273  buffer = castBuffer;
274  }
275  newOperands[idx] = buffer;
276  }
277 
278  // 3. Create the new CallOp.
279  Operation *newCallOp = rewriter.create<func::CallOp>(
280  callOp.getLoc(), funcOp.getSymName(), resultTypes, newOperands);
281  newCallOp->setAttrs(callOp->getAttrs());
282  // Get replacement values.
283  for (unsigned i = 0; i < replacementValues.size(); ++i) {
284  if (replacementValues[i])
285  continue;
286  replacementValues[i] = newCallOp->getResult(*retValMapping[i]);
287  }
288 
289  // 4. Replace the old op with the new op.
290  replaceOpWithBufferizedValues(rewriter, callOp, replacementValues);
291 
292  return success();
293  }

327 struct FuncOpInterface
328  : public BufferizableOpInterface::ExternalModel<FuncOpInterface, FuncOp> {
329  /// Rewrite function bbArgs and return values into buffer form. This function
330  /// bufferizes the function signature and the ReturnOp. When the entire
331  /// function body has been bufferized, function return types can be switched
332  /// to more concise memref types as part of `foldMemRefCasts`.
333  ///
334  /// All function bbArgs are writable unless they are explicitly marked as
335  /// read-only. Callers must insert copies when needed.
336  LogicalResult bufferize(Operation *op, RewriterBase &rewriter,
337  const BufferizationOptions &options) const {
338  auto funcOp = cast<FuncOp>(op);
339  FunctionType funcType = funcOp.getFunctionType();
340 
341  // Construct the bufferized function type.
342  SmallVector<Type> argTypes;
343  for (const auto &it : llvm::enumerate(funcType.getInputs())) {
344  Type argType = it.value();
345  if (auto tensorType = argType.dyn_cast<TensorType>()) {
346  argTypes.push_back(
347  getBufferizedFunctionArgType(funcOp, it.index(), options));
348  continue;
349  }
350  argTypes.push_back(argType);
351  }
352 
353  // Bodiless functions are assumed opaque and we cannot know the
354  // bufferization contract they want to enforce. As a consequence, only
355  // support functions that don't return any tensors atm.
356  if (funcOp.getBody().empty()) {
357  SmallVector<Type> retTypes;
358  for (Type resultType : funcType.getResults()) {
359  if (resultType.isa<TensorType>())
360  return funcOp->emitError() << "cannot bufferize bodiless function "
361  << "that returns a tensor";
362  retTypes.push_back(resultType);
363  }
364  funcOp.setType(FunctionType::get(op->getContext(), argTypes, retTypes));
365  return success();
366  }
367 
368  // TODO: Support functions with multiple returns.
369  func::ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
370  assert(returnOp && "expected func with single return op");
371  Location loc = returnOp.getLoc();
372 
373  // 1. Rewrite the bbArgs. Turn every tensor bbArg into a memref bbArg.
374  Block &frontBlock = funcOp.getBody().front();
375  for (BlockArgument &bbArg : frontBlock.getArguments()) {
376  auto tensorType = bbArg.getType().dyn_cast<TensorType>();
377  // Non-tensor types stay the same.
378  if (!tensorType)
379  continue;
380 
381  // Collect all uses of the bbArg.
382  SmallVector<OpOperand *> bbArgUses;
383  for (OpOperand &use : bbArg.getUses())
384  bbArgUses.push_back(&use);
385 
386  // Change the bbArg type to memref.
387  Type memrefType =
388  getBufferizedFunctionArgType(funcOp, bbArg.getArgNumber(), options);
389  bbArg.setType(memrefType);
390 
391  // Replace all uses of the original tensor bbArg.
392  rewriter.setInsertionPointToStart(&frontBlock);
393  if (!bbArgUses.empty()) {
394  // Insert to_tensor because the remaining function body has not been
395  // bufferized yet.
396  Value toTensorOp =
397  rewriter.create<bufferization::ToTensorOp>(funcOp.getLoc(), bbArg);
398  for (OpOperand *use : bbArgUses)
399  use->set(toTensorOp);
400  }
401  }
402 
403  // 2. For each result, keep track of which inplace argument it reuses.
404  SmallVector<Value> returnValues;
405  for (OpOperand &returnOperand : returnOp->getOpOperands()) {
406  Value returnVal = returnOperand.get();
407  auto tensorType = returnVal.getType().dyn_cast<TensorType>();
408  rewriter.setInsertionPoint(returnOp);
409 
410  // If not a tensor type just forward it.
411  if (!tensorType) {
412  returnValues.push_back(returnVal);
413  continue;
414  }
415 
416  // Note: If `inferFunctionResultLayout = true`, cast are later folded
417  // away.
418  BaseMemRefType resultType = options.functionArgTypeConverterFn(
419  tensorType, *options.defaultMemorySpace, funcOp, options);
420  Value toMemrefOp = rewriter.create<bufferization::ToMemrefOp>(
421  loc, resultType, returnVal);
422  returnValues.push_back(toMemrefOp);
423  }
424 
425  // 3. Rewrite the terminator without the in-place bufferizable values.
426  returnOp.getOperandsMutable().assign(returnValues);
427 
428  // 4. Rewrite the FuncOp type to buffer form.
429  funcOp.setType(FunctionType::get(op->getContext(), argTypes,
430  ValueRange(returnValues).getTypes()));
431 
432  return success();
433  }
434 
435  /// Return `true` if the given function argument is writable.
436  bool isWritable(Operation *op, Value value,
437  const AnalysisState &state) const {
438  auto funcOp = cast<FuncOp>(op);
439  BlockArgument bbArg = value.dyn_cast<BlockArgument>();
440  assert(bbArg && "expected BlockArgument");
441 
442  // "bufferization.writable" overrides other writability decisions. This is
443  // currently used for testing only.
444  if (BoolAttr writable = funcOp.getArgAttrOfType<BoolAttr>(
445  bbArg.getArgNumber(), BufferizationDialect::kWritableAttrName))
446  return writable.getValue();
447 
448  // All function arguments are writable by default.
449  return true;
450  }
