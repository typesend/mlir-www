55 static bool noAliasingUseInLoop(vector::TransferReadOp transferRead,
56  LoopLikeOpInterface loop) {
57  Value source = transferRead.getSource();
58  while (auto subView = source.getDefiningOp<memref::SubViewOp>())
59  source = subView.getSource();
60  llvm::SmallVector<Operation *, 32> users(source.getUsers().begin(),
61  source.getUsers().end());
62  llvm::SmallDenseSet<Operation *, 32> processed;
63  while (!users.empty()) {
64  Operation *user = users.pop_back_val();
65  // If the user has already been processed skip.
66  if (!processed.insert(user).second)
67  continue;
68  if (auto subView = dyn_cast<memref::SubViewOp>(user)) {
69  users.append(subView->getUsers().begin(), subView->getUsers().end());
70  continue;
71  }
72  if (isMemoryEffectFree(user) || isa<vector::TransferReadOp>(user))
73  continue;
74  return false;
75  }
76  return true;
77 }

79 void mlir::linalg::hoistRedundantVectorTransfers(func::FuncOp func) {
80  bool changed = true;
81  while (changed) {
82  changed = false;
83  // First move loop invariant ops outside of their loop. This needs to be
84  // done before as we cannot move ops without interrupting the function walk.
85  func.walk(
86  [&](LoopLikeOpInterface loopLike) { moveLoopInvariantCode(loopLike); });
87 
88  func.walk([&](vector::TransferReadOp transferRead) {
89  if (!transferRead.getShapedType().isa<MemRefType>())
90  return WalkResult::advance();
91 
92  LLVM_DEBUG(DBGS() << "Candidate for hoisting: "
93  << *transferRead.getOperation() << "\n");
94  auto loop = dyn_cast<LoopLikeOpInterface>(transferRead->getParentOp());
95  LLVM_DEBUG(DBGS() << "Parent op: " << *transferRead->getParentOp()
96  << "\n");
97  if (!isa_and_nonnull<scf::ForOp, affine::AffineForOp>(loop))
98  return WalkResult::advance();
99 
100  LLVM_DEBUG(DBGS() << "Candidate read: " << *transferRead.getOperation()
101  << "\n");
102 
103  SetVector<Operation *> forwardSlice;
104  getForwardSlice(transferRead.getOperation(), &forwardSlice);
105 
106  // Look for the last TransferWriteOp in the forwardSlice of
107  // `transferRead` that operates on the same memref.
108  vector::TransferWriteOp transferWrite;
109  for (auto *sliceOp : llvm::reverse(forwardSlice)) {
110  auto candidateWrite = dyn_cast<vector::TransferWriteOp>(sliceOp);
111  if (!candidateWrite ||
112  candidateWrite.getSource() != transferRead.getSource())
113  continue;
114  transferWrite = candidateWrite;
115  }
116 
117  // All operands of the TransferRead must be defined outside of the loop.
118  for (auto operand : transferRead.getOperands())
119  if (!loop.isDefinedOutsideOfLoop(operand))
120  return WalkResult::advance();
121 
122  // Only hoist transfer_read / transfer_write pairs and singleton
123  // transfer_reads for now.
124  if (!transferWrite) {
125  // Make sure there are no other accesses to the memref before
126  // hoisting transfer_read.
127  if (noAliasingUseInLoop(transferRead, loop))
128  loop.moveOutOfLoop(transferRead);
129  return WalkResult::advance();
130  }
131 
132  LLVM_DEBUG(DBGS() << "Candidate: " << *transferWrite.getOperation()
133  << "\n");
134 
135  // Approximate aliasing by checking that:
136  // 1. indices are the same,
137  // 2. no other operations in the loop access the same memref except
138  // for transfer_read/transfer_write accessing statically disjoint
139  // slices.
140  if (transferRead.getIndices() != transferWrite.getIndices() &&
141  transferRead.getVectorType() == transferWrite.getVectorType())
142  return WalkResult::advance();
143 
144  // TODO: may want to memoize this information for performance but it
145  // likely gets invalidated often.
146  DominanceInfo dom(loop);
147  if (!dom.properlyDominates(transferRead.getOperation(), transferWrite))
148  return WalkResult::advance();
149  for (auto &use : transferRead.getSource().getUses()) {
150  if (!loop->isAncestor(use.getOwner()))
151  continue;
152  if (use.getOwner() == transferRead.getOperation() ||
153  use.getOwner() == transferWrite.getOperation())
154  continue;
155  if (auto transferWriteUse =
156  dyn_cast<vector::TransferWriteOp>(use.getOwner())) {
157  if (!vector::isDisjointTransferSet(
158  cast<VectorTransferOpInterface>(transferWrite.getOperation()),
159  cast<VectorTransferOpInterface>(
160  transferWriteUse.getOperation())))
161  return WalkResult::advance();
162  } else if (auto transferReadUse =
163  dyn_cast<vector::TransferReadOp>(use.getOwner())) {
164  if (!vector::isDisjointTransferSet(
165  cast<VectorTransferOpInterface>(transferWrite.getOperation()),
166  cast<VectorTransferOpInterface>(
167  transferReadUse.getOperation())))
168  return WalkResult::advance();
169  } else {
170  // Unknown use, we cannot prove that it doesn't alias with the
171  // transferRead/transferWrite operations.
172  return WalkResult::advance();
173  }
174  }
175 
176  // Hoist read before.
177  loop.moveOutOfLoop(transferRead);
178 
179  // Hoist write after.
180  transferWrite->moveAfter(loop);
181 
182  // Rewrite `loop` with new yields by cloning and erase the original loop.
183  OpBuilder b(transferRead);
184  NewYieldValueFn yieldFn = [&](OpBuilder &b, Location loc,
185  ArrayRef<BlockArgument> newBBArgs) {
186  return SmallVector<Value>{transferWrite.getVector()};
187  };
188 
189  // Transfer write has been hoisted, need to update the written vector by
190  // the value yielded by the newForOp.
191  return TypeSwitch<Operation *, WalkResult>(loop)
192  .Case<scf::ForOp>([&](scf::ForOp scfForOp) {
193  auto newForOp = replaceLoopWithNewYields(
194  b, scfForOp, transferRead.getVector(), yieldFn);
195  transferWrite.getVectorMutable().assign(
196  newForOp.getResults().back());
197  changed = true;
198  loop.erase();
199  // Need to interrupt and restart because erasing the loop messes up
200  // the walk.
201  return WalkResult::interrupt();
202  })
203  .Case<affine::AffineForOp>([&](affine::AffineForOp affineForOp) {
204  auto newForOp = replaceForOpWithNewYields(
205  b, affineForOp, transferRead.getVector(),
206  SmallVector<Value>{transferWrite.getVector()},
207  transferWrite.getVector());
208  // Replace all uses of the `transferRead` with the corresponding
209  // basic block argument.
210  transferRead.getVector().replaceUsesWithIf(
211  newForOp.getLoopBody().getArguments().back(),
212  [&](OpOperand &use) {
213  Operation *user = use.getOwner();
214  return newForOp->isProperAncestor(user);
215  });
216  transferWrite.getVectorMutable().assign(
217  newForOp.getResults().back());
218  changed = true;
219  loop.erase();
220  // Need to interrupt and restart because erasing the loop messes up
221  // the walk.
222  return WalkResult::interrupt();
223  })
224  .Default([](Operation *) { return WalkResult::interrupt(); });
225  });
226  }
