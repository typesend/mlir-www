25 /// Check whether the given alloca is an input to a lifetime intrinsic,
26 /// optionally passing through one or more casts on the way. This is not
27 /// transitive through block arguments.
28 static bool hasLifetimeMarkers(LLVM::AllocaOp allocaOp) {
29  SmallVector<Operation *> stack(allocaOp->getUsers().begin(),
30  allocaOp->getUsers().end());
31  while (!stack.empty()) {
32  Operation *op = stack.pop_back_val();
33  if (isa<LLVM::LifetimeStartOp, LLVM::LifetimeEndOp>(op))
34  return true;
35  if (isa<LLVM::BitcastOp>(op))
36  stack.append(op->getUsers().begin(), op->getUsers().end());
37  }
38  return false;
39 }

41 /// Handles alloca operations in the inlined blocks:
42 /// - Moves all alloca operations with a constant size in the former entry block
43 /// of the callee into the entry block of the caller, so they become part of
44 /// the function prologue/epilogue during code generation.
45 /// - Inserts lifetime intrinsics that limit the scope of inlined static allocas
46 /// to the inlined blocks.
47 /// - Inserts StackSave and StackRestore operations if dynamic allocas were
48 /// inlined.
49 static void
50 handleInlinedAllocas(Operation *call,
51  iterator_range<Region::iterator> inlinedBlocks) {
52  Block *calleeEntryBlock = &(*inlinedBlocks.begin());
53  Block *callerEntryBlock = &(*calleeEntryBlock->getParent()->begin());
54  if (calleeEntryBlock == callerEntryBlock)
55  // Nothing to do.
56  return;
57  SmallVector<std::tuple<LLVM::AllocaOp, IntegerAttr, bool>> allocasToMove;
58  bool shouldInsertLifetimes = false;
59  bool hasDynamicAlloca = false;
60  // Conservatively only move static alloca operations that are part of the
61  // entry block and do not inspect nested regions, since they may execute
62  // conditionally or have other unknown semantics.
63  for (auto allocaOp : calleeEntryBlock->getOps<LLVM::AllocaOp>()) {
64  IntegerAttr arraySize;
65  if (!matchPattern(allocaOp.getArraySize(), m_Constant(&arraySize))) {
66  hasDynamicAlloca = true;
67  continue;
68  }
69  bool shouldInsertLifetime =
70  arraySize.getValue() != 0 && !hasLifetimeMarkers(allocaOp);
71  shouldInsertLifetimes |= shouldInsertLifetime;
72  allocasToMove.emplace_back(allocaOp, arraySize, shouldInsertLifetime);
73  }
74  // Check the remaining inlined blocks for dynamic allocas as well.
75  for (Block &block : llvm::drop_begin(inlinedBlocks)) {
76  if (hasDynamicAlloca)
77  break;
78  hasDynamicAlloca =
79  llvm::any_of(block.getOps<LLVM::AllocaOp>(), [](auto allocaOp) {
80  return !matchPattern(allocaOp.getArraySize(), m_Constant());
81  });
82  }
83  if (allocasToMove.empty() && !hasDynamicAlloca)
84  return;
85  OpBuilder builder(calleeEntryBlock, calleeEntryBlock->begin());
86  Value stackPtr;
87  if (hasDynamicAlloca) {
88  // This may result in multiple stacksave/stackrestore intrinsics in the same
89  // scope if some are already present in the body of the caller. This is not
90  // invalid IR, but LLVM cleans these up in InstCombineCalls.cpp, along with
91  // other cases where the stacksave/stackrestore is redundant.
92  stackPtr = builder.create<LLVM::StackSaveOp>(
93  call->getLoc(), LLVM::LLVMPointerType::get(call->getContext()));
94  }
95  builder.setInsertionPoint(callerEntryBlock, callerEntryBlock->begin());
96  for (auto &[allocaOp, arraySize, shouldInsertLifetime] : allocasToMove) {
97  auto newConstant = builder.create<LLVM::ConstantOp>(
98  allocaOp->getLoc(), allocaOp.getArraySize().getType(), arraySize);
99  // Insert a lifetime start intrinsic where the alloca was before moving it.
100  if (shouldInsertLifetime) {
101  OpBuilder::InsertionGuard insertionGuard(builder);
102  builder.setInsertionPoint(allocaOp);
103  builder.create<LLVM::LifetimeStartOp>(
104  allocaOp.getLoc(), arraySize.getValue().getLimitedValue(),
105  allocaOp.getResult());
106  }
107  allocaOp->moveAfter(newConstant);
108  allocaOp.getArraySizeMutable().assign(newConstant.getResult());
109  }
110  if (!shouldInsertLifetimes && !hasDynamicAlloca)
111  return;
112  // Insert a lifetime end intrinsic before each return in the callee function.
113  for (Block &block : inlinedBlocks) {
114  if (!block.getTerminator()->hasTrait<OpTrait::ReturnLike>())
115  continue;
116  builder.setInsertionPoint(block.getTerminator());
117  if (hasDynamicAlloca)
118  builder.create<LLVM::StackRestoreOp>(call->getLoc(), stackPtr);
119  for (auto &[allocaOp, arraySize, shouldInsertLifetime] : allocasToMove) {
120  if (shouldInsertLifetime)
121  builder.create<LLVM::LifetimeEndOp>(
122  allocaOp.getLoc(), arraySize.getValue().getLimitedValue(),
123  allocaOp.getResult());
124  }
125  }
126 }

128 /// If `requestedAlignment` is higher than the alignment specified on `alloca`,
129 /// realigns `alloca` if this does not exceed the natural stack alignment.
130 /// Returns the post-alignment of `alloca`, whether it was realigned or not.
131 static unsigned tryToEnforceAllocaAlignment(LLVM::AllocaOp alloca,
132  unsigned requestedAlignment,
133  DataLayout const &dataLayout) {
134  unsigned allocaAlignment = alloca.getAlignment().value_or(1);
135  if (requestedAlignment <= allocaAlignment)
136  // No realignment necessary.
137  return allocaAlignment;
138  unsigned naturalStackAlignmentBits = dataLayout.getStackAlignment();
139  // If the natural stack alignment is not specified, the data layout returns
140  // zero. Optimistically allow realignment in this case.
141  if (naturalStackAlignmentBits == 0 ||
142  // If the requested alignment exceeds the natural stack alignment, this
143  // will trigger a dynamic stack realignment, so we prefer to copy...
144  8 * requestedAlignment <= naturalStackAlignmentBits ||
145  // ...unless the alloca already triggers dynamic stack realignment. Then
146  // we might as well further increase the alignment to avoid a copy.
147  8 * allocaAlignment > naturalStackAlignmentBits) {
148  alloca.setAlignment(requestedAlignment);
149  allocaAlignment = requestedAlignment;
150  }
151  return allocaAlignment;
152 }

154 /// Tries to find and return the alignment of the pointer `value` by looking for
155 /// an alignment attribute on the defining allocation op or function argument.
156 /// If the found alignment is lower than `requestedAlignment`, tries to realign
157 /// the pointer, then returns the resulting post-alignment, regardless of
158 /// whether it was realigned or not. If no existing alignment attribute is
159 /// found, returns 1 (i.e., assume that no alignment is guaranteed).
160 static unsigned tryToEnforceAlignment(Value value, unsigned requestedAlignment,
161  DataLayout const &dataLayout) {
162  if (Operation *definingOp = value.getDefiningOp()) {
163  if (auto alloca = dyn_cast<LLVM::AllocaOp>(definingOp))
164  return tryToEnforceAllocaAlignment(alloca, requestedAlignment,
165  dataLayout);
166  if (auto addressOf = dyn_cast<LLVM::AddressOfOp>(definingOp))
167  if (auto global = SymbolTable::lookupNearestSymbolFrom<LLVM::GlobalOp>(
168  definingOp, addressOf.getGlobalNameAttr()))
169  return global.getAlignment().value_or(1);
170  // We don't currently handle this operation; assume no alignment.
171  return 1;
172  }
173  // Since there is no defining op, this is a block argument. Probably this
174  // comes directly from a function argument, so check that this is the case.
175  Operation *parentOp = value.getParentBlock()->getParentOp();
176  if (auto func = dyn_cast<LLVM::LLVMFuncOp>(parentOp)) {
177  // Use the alignment attribute set for this argument in the parent function
178  // if it has been set.
179  auto blockArg = value.cast<BlockArgument>();
180  if (Attribute alignAttr = func.getArgAttr(
181  blockArg.getArgNumber(), LLVM::LLVMDialect::getAlignAttrName()))
182  return cast<IntegerAttr>(alignAttr).getValue().getLimitedValue();
183  }
184  // We didn't find anything useful; assume no alignment.
185  return 1;
186 }

188 /// Introduces a new alloca and copies the memory pointed to by `argument` to
189 /// the address of the new alloca, then returns the value of the new alloca.
190 static Value handleByValArgumentInit(OpBuilder &builder, Location loc,
191  Value argument, Type elementType,
192  unsigned elementTypeSize,
193  unsigned targetAlignment) {
194  // Allocate the new value on the stack.
195  Value allocaOp;
196  {
197  // Since this is a static alloca, we can put it directly in the entry block,
198  // so they can be absorbed into the prologue/epilogue at code generation.
199  OpBuilder::InsertionGuard insertionGuard(builder);
200  Block *entryBlock = &(*argument.getParentRegion()->begin());
201  builder.setInsertionPointToStart(entryBlock);
202  Value one = builder.create<LLVM::ConstantOp>(loc, builder.getI64Type(),
203  builder.getI64IntegerAttr(1));
204  allocaOp = builder.create<LLVM::AllocaOp>(
205  loc, argument.getType(), elementType, one, targetAlignment);
206  }
207  // Copy the pointee to the newly allocated value.
208  Value copySize = builder.create<LLVM::ConstantOp>(
209  loc, builder.getI64Type(), builder.getI64IntegerAttr(elementTypeSize));
210  Value isVolatile = builder.create<LLVM::ConstantOp>(
211  loc, builder.getI1Type(), builder.getBoolAttr(false));
212  builder.create<LLVM::MemcpyOp>(loc, allocaOp, argument, copySize, isVolatile);
213  return allocaOp;
214 }

216 /// Handles a function argument marked with the byval attribute by introducing a
217 /// memcpy or realigning the defining operation, if required either due to the
218 /// pointee being writeable in the callee, and/or due to an alignment mismatch.
219 /// `requestedAlignment` specifies the alignment set in the "align" argument
220 /// attribute (or 1 if no align attribute was set).
221 static Value handleByValArgument(OpBuilder &builder, Operation *callable,
222  Value argument, Type elementType,
223  unsigned requestedAlignment) {
224  auto func = cast<LLVM::LLVMFuncOp>(callable);
225  LLVM::MemoryEffectsAttr memoryEffects = func.getMemoryAttr();
226  // If there is no memory effects attribute, assume that the function is
227  // not read-only.
228  bool isReadOnly = memoryEffects &&
229  memoryEffects.getArgMem() != LLVM::ModRefInfo::ModRef &&
230  memoryEffects.getArgMem() != LLVM::ModRefInfo::Mod;
231  // Check if there's an alignment mismatch requiring us to copy.
232  DataLayout dataLayout = DataLayout::closest(callable);
233  unsigned minimumAlignment = dataLayout.getTypeABIAlignment(elementType);
234  if (isReadOnly) {
235  if (requestedAlignment <= minimumAlignment)
236  return argument;
237  unsigned currentAlignment =
238  tryToEnforceAlignment(argument, requestedAlignment, dataLayout);
239  if (currentAlignment >= requestedAlignment)
240  return argument;
241  }
242  unsigned targetAlignment = std::max(requestedAlignment, minimumAlignment);
243  return handleByValArgumentInit(builder, func.getLoc(), argument, elementType,
244  dataLayout.getTypeSize(elementType),
245  targetAlignment);
246 }

274  bool isLegalToInline(Operation *call, Operation *callable,
275  bool wouldBeCloned) const final {
276  if (!wouldBeCloned)
277  return false;
278  auto callOp = dyn_cast<LLVM::CallOp>(call);
279  if (!callOp) {
280  LLVM_DEBUG(llvm::dbgs()
281  << "Cannot inline: call is not an LLVM::CallOp\n");
282  return false;
283  }
284  auto funcOp = dyn_cast<LLVM::LLVMFuncOp>(callable);
285  if (!funcOp) {
286  LLVM_DEBUG(llvm::dbgs()
287  << "Cannot inline: callable is not an LLVM::LLVMFuncOp\n");
288  return false;
289  }
290  if (auto attrs = funcOp.getArgAttrs()) {
291  for (DictionaryAttr attrDict : attrs->getAsRange<DictionaryAttr>()) {
292  for (NamedAttribute attr : attrDict) {
293  if (!isArgOrResAttrSupported(attr)) {
294  LLVM_DEBUG(llvm::dbgs() << "Cannot inline " << funcOp.getSymName()
295  << ": unhandled argument attribute "
296  << attr.getName() << "\n");
297  return false;
298  }
299  }
300  }
301  }
302  if (auto attrs = funcOp.getResAttrs()) {
303  for (DictionaryAttr attrDict : attrs->getAsRange<DictionaryAttr>()) {
304  for (NamedAttribute attr : attrDict) {
305  if (!isArgOrResAttrSupported(attr)) {
306  LLVM_DEBUG(llvm::dbgs() << "Cannot inline " << funcOp.getSymName()
307  << ": unhandled return attribute "
308  << attr.getName() << "\n");
309  return false;
310  }
311  }
312  }
313  }
314  // TODO: Handle exceptions.
315  if (funcOp.getPersonality()) {
316  LLVM_DEBUG(llvm::dbgs() << "Cannot inline " << funcOp.getSymName()
317  << ": unhandled function personality\n");
318  return false;
319  }
320  if (funcOp.getPassthrough()) {
321  // TODO: Used attributes should not be passthrough.
322  if (llvm::any_of(*funcOp.getPassthrough(), [&](Attribute attr) {
323  auto stringAttr = dyn_cast<StringAttr>(attr);
324  if (!stringAttr)
325  return false;
326  if (disallowedFunctionAttrs.contains(stringAttr)) {
327  LLVM_DEBUG(llvm::dbgs()
328  << "Cannot inline " << funcOp.getSymName()
329  << ": found disallowed function attribute "
330  << stringAttr << "\n");
331  return true;
332  }
333  return false;
334  }))
335  return false;
336  }
337  return true;
338  }

344  /// Conservative allowlist of operations supported so far.
345  bool isLegalToInline(Operation *op, Region *, bool, IRMapping &) const final {
346  if (isPure(op))
347  return true;
348  // Some attributes on memory operations require handling during
349  // inlining. Since this is not yet implemented, refuse to inline memory
350  // operations that have any of these attributes.
351  if (auto iface = dyn_cast<LLVM::AliasAnalysisOpInterface>(op)) {
352  if (iface.getAliasScopesOrNull() || iface.getNoAliasScopesOrNull()) {
353  LLVM_DEBUG(llvm::dbgs()
354  << "Cannot inline: unhandled alias analysis metadata\n");
355  return false;
356  }
357  }
358  if (auto iface = dyn_cast<LLVM::AccessGroupOpInterface>(op)) {
359  if (iface.getAccessGroupsOrNull()) {
360  LLVM_DEBUG(llvm::dbgs()
361  << "Cannot inline: unhandled access group metadata\n");
362  return false;
363  }
364  }
365  // clang-format off
366  if (isa<LLVM::AllocaOp,
367  LLVM::CallOp,
368  LLVM::DbgDeclareOp,
369  LLVM::DbgValueOp,
370  LLVM::LifetimeEndOp,
371  LLVM::LifetimeStartOp,
372  LLVM::LoadOp,
373  LLVM::MemcpyOp,
374  LLVM::MemmoveOp,
375  LLVM::MemsetOp,
376  LLVM::StackRestoreOp,
377  LLVM::StackSaveOp,
378  LLVM::StoreOp,
379  LLVM::UnreachableOp>(op))
380  return true;
381  // clang-format on
382  LLVM_DEBUG(llvm::dbgs()
383  << "Cannot inline: unhandled side effecting operation \""
384  << op->getName() << "\"\n");
385  return false;
386  }

388  /// Handle the given inlined return by replacing it with a branch. This
389  /// overload is called when the inlined region has more than one block.
390  void handleTerminator(Operation *op, Block *newDest) const final {
391  // Only return needs to be handled here.
392  auto returnOp = dyn_cast<LLVM::ReturnOp>(op);
393  if (!returnOp)
394  return;
395 
396  // Replace the return with a branch to the dest.
397  OpBuilder builder(op);
398  builder.create<LLVM::BrOp>(op->getLoc(), returnOp.getOperands(), newDest);
399  op->erase();
400  }

402  /// Handle the given inlined return by replacing the uses of the call with the
403  /// operands of the return. This overload is called when the inlined region
404  /// only contains one block.
405  void handleTerminator(Operation *op,
406  ArrayRef<Value> valuesToRepl) const final {
407  // Return will be the only terminator present.
408  auto returnOp = cast<LLVM::ReturnOp>(op);
409 
410  // Replace the values directly with the return operands.
411  assert(returnOp.getNumOperands() == valuesToRepl.size());
412  for (const auto &[dst, src] :
413  llvm::zip(valuesToRepl, returnOp.getOperands()))
414  dst.replaceAllUsesWith(src);
415  }

417  Value handleArgument(OpBuilder &builder, Operation *call, Operation *callable,
418  Value argument,
419  DictionaryAttr argumentAttrs) const final {
420  if (std::optional<NamedAttribute> attr =
421  argumentAttrs.getNamed(LLVM::LLVMDialect::getByValAttrName())) {
