26 // Returns `success` when any of the elements in `ofrs` was produced by
27 // arith::ConstantIndexOp. In that case the constant attribute replaces the
28 // Value. Returns `failure` when no folding happened.
29 LogicalResult mlir::foldDynamicIndexList(Builder &b,
30  SmallVectorImpl<OpFoldResult> &ofrs) {
31  bool valuesChanged = false;
32  for (OpFoldResult &ofr : ofrs) {
33  if (ofr.is<Attribute>())
34  continue;
35  // Newly static, move from Value to constant.
36  if (auto cstOp =
37  ofr.dyn_cast<Value>().getDefiningOp<arith::ConstantIndexOp>()) {
38  ofr = b.getIndexAttr(cstOp.value());
39  valuesChanged = true;
40  }
41  }
42  return success(valuesChanged);
43 }

66 Value mlir::getValueOrCreateCastToIndexLike(OpBuilder &b, Location loc,
67  Type targetType, Value value) {
68  if (targetType == value.getType())
69  return value;
70 
71  bool targetIsIndex = targetType.isIndex();
72  bool valueIsIndex = value.getType().isIndex();
73  if (targetIsIndex ^ valueIsIndex)
74  return b.create<arith::IndexCastOp>(loc, targetType, value);
75 
76  auto targetIntegerType = targetType.dyn_cast<IntegerType>();
77  auto valueIntegerType = value.getType().dyn_cast<IntegerType>();
78  assert(targetIntegerType && valueIntegerType &&
79  "unexpected cast between types other than integers and index");
80  assert(targetIntegerType.getSignedness() == valueIntegerType.getSignedness());
81 
82  if (targetIntegerType.getWidth() > valueIntegerType.getWidth())
83  return b.create<arith::ExtSIOp>(loc, targetIntegerType, value);
84  return b.create<arith::TruncIOp>(loc, targetIntegerType, value);
85 }

87 Value mlir::convertScalarToDtype(OpBuilder &b, Location loc, Value operand,
88  Type toType, bool isUnsignedCast) {
89  if (operand.getType() == toType)
90  return operand;
91  if (auto toIntType = toType.dyn_cast<IntegerType>()) {
92  // If operand is floating point, cast directly to the int type.
93  if (operand.getType().isa<FloatType>()) {
94  if (isUnsignedCast)
95  return b.create<arith::FPToUIOp>(loc, toType, operand);
96  return b.create<arith::FPToSIOp>(loc, toType, operand);
97  }
98  // Cast index operands directly to the int type.
99  if (operand.getType().isIndex())
100  return b.create<arith::IndexCastOp>(loc, toType, operand);
101  if (auto fromIntType = operand.getType().dyn_cast<IntegerType>()) {
102  // Either extend or truncate.
103  if (toIntType.getWidth() > fromIntType.getWidth()) {
104  if (isUnsignedCast)
105  return b.create<arith::ExtUIOp>(loc, toType, operand);
106  return b.create<arith::ExtSIOp>(loc, toType, operand);
107  }
108  if (toIntType.getWidth() < fromIntType.getWidth())
109  return b.create<arith::TruncIOp>(loc, toType, operand);
110  }
111  } else if (auto toFloatType = toType.dyn_cast<FloatType>()) {
112  // If operand is integer, cast directly to the float type.
113  // Note that it is unclear how to cast from BF16<->FP16.
114  if (operand.getType().isa<IntegerType>()) {
115  if (isUnsignedCast)
116  return b.create<arith::UIToFPOp>(loc, toFloatType, operand);
117  return b.create<arith::SIToFPOp>(loc, toFloatType, operand);
118  }
119  if (auto fromFloatType = operand.getType().dyn_cast<FloatType>()) {
120  if (toFloatType.getWidth() > fromFloatType.getWidth())
121  return b.create<arith::ExtFOp>(loc, toFloatType, operand);
122  if (toFloatType.getWidth() < fromFloatType.getWidth())
123  return b.create<arith::TruncFOp>(loc, toFloatType, operand);
124  }
125  }
126  emitWarning(loc) << "could not cast operand of type " << operand.getType()
127  << " to " << toType;
128  return operand;
129 }
