30 /// This class represents an abstract lattice. A lattice contains information
31 /// about an SSA value and is what's propagated across the IR by sparse
32 /// data-flow analysis.
33 class AbstractSparseLattice : public AnalysisState {
34 public:
35  /// Lattices can only be created for values.
36  AbstractSparseLattice(Value value) : AnalysisState(value) {}
37 
38  /// Return the program point this lattice is located at.
39  Value getPoint() const { return AnalysisState::getPoint().get<Value>(); }
40 
41  /// Join the information contained in 'rhs' into this lattice. Returns
42  /// if the value of the lattice changed.
43  virtual ChangeResult join(const AbstractSparseLattice &rhs) {
44  return ChangeResult::NoChange;
45  }
46 
47  /// Meet (intersect) the information in this lattice with 'rhs'. Returns
48  /// if the value of the lattice changed.
49  virtual ChangeResult meet(const AbstractSparseLattice &rhs) {
50  return ChangeResult::NoChange;
51  }
52 
53  /// When the lattice gets updated, propagate an update to users of the value
54  /// using its use-def chain to subscribed analyses.
55  void onUpdate(DataFlowSolver *solver) const override;
56 
57  /// Subscribe an analysis to updates of the lattice. When the lattice changes,
58  /// subscribed analyses are re-invoked on all users of the value. This is
59  /// more efficient than relying on the dependency map.
60  void useDefSubscribe(DataFlowAnalysis *analysis) {
61  useDefSubscribers.insert(analysis);
62  }
63 
64 private:
65  /// A set of analyses that should be updated when this lattice changes.
66  SetVector<DataFlowAnalysis *, SmallVector<DataFlowAnalysis *, 4>,
67  SmallPtrSet<DataFlowAnalysis *, 4>>
68  useDefSubscribers;
69 };

75 /// This class represents a lattice holding a specific value of type `ValueT`.
76 /// Lattice values (`ValueT`) are required to adhere to the following:
77 ///
78 /// * static ValueT join(const ValueT &lhs, const ValueT &rhs);
79 /// - This method conservatively joins the information held by `lhs`
80 /// and `rhs` into a new value. This method is required to be monotonic.
81 /// * bool operator==(const ValueT &rhs) const;
82 ///
83 template <typename ValueT>
84 class Lattice : public AbstractSparseLattice {
85 public:
86  using AbstractSparseLattice::AbstractSparseLattice;
87 
88  /// Return the program point this lattice is located at.
89  Value getPoint() const { return point.get<Value>(); }
90 
91  /// Return the value held by this lattice. This requires that the value is
92  /// initialized.
93  ValueT &getValue() { return value; }
94  const ValueT &getValue() const {
95  return const_cast<Lattice<ValueT> *>(this)->getValue();
96  }
97 
98  using LatticeT = Lattice<ValueT>;
99 
100  /// Join the information contained in the 'rhs' lattice into this
101  /// lattice. Returns if the state of the current lattice changed.
102  ChangeResult join(const AbstractSparseLattice &rhs) override {
103  return join(static_cast<const LatticeT &>(rhs).getValue());
104  }
105 
106  /// Meet (intersect) the information contained in the 'rhs' lattice with
107  /// this lattice. Returns if the state of the current lattice changed.
108  ChangeResult meet(const AbstractSparseLattice &rhs) override {
109  return meet(static_cast<const LatticeT &>(rhs).getValue());
110  }
111 
112  /// Join the information contained in the 'rhs' value into this
113  /// lattice. Returns if the state of the current lattice changed.
114  ChangeResult join(const ValueT &rhs) {
115  // Otherwise, join rhs with the current optimistic value.
116  ValueT newValue = ValueT::join(value, rhs);
117  assert(ValueT::join(newValue, value) == newValue &&
118  "expected `join` to be monotonic");
119  assert(ValueT::join(newValue, rhs) == newValue &&
120  "expected `join` to be monotonic");
121 
122  // Update the current optimistic value if something changed.
123  if (newValue == value)
124  return ChangeResult::NoChange;
125 
126  value = newValue;
127  return ChangeResult::Change;
128  }
129 
130  /// Trait to check if `T` provides a `meet` method. Needed since for forward
131  /// analysis, lattices will only have a `join`, no `meet`, but we want to use
132  /// the same `Lattice` class for both directions.
133  template <typename T, typename... Args>
134  using has_meet = decltype(std::declval<T>().meet());
135  template <typename T>
136  using lattice_has_meet = llvm::is_detected<has_meet, T>;
137 
138  /// Meet (intersect) the information contained in the 'rhs' value with this
139  /// lattice. Returns if the state of the current lattice changed. If the
140  /// lattice elements don't have a `meet` method, this is a no-op (see below.)
141  template <typename VT, std::enable_if_t<lattice_has_meet<VT>::value>>
142  ChangeResult meet(const VT &rhs) {
143  ValueT newValue = ValueT::meet(value, rhs);
144  assert(ValueT::meet(newValue, value) == newValue &&
145  "expected `meet` to be monotonic");
146  assert(ValueT::meet(newValue, rhs) == newValue &&
147  "expected `meet` to be monotonic");
148 
149  // Update the current optimistic value if something changed.
150  if (newValue == value)
151  return ChangeResult::NoChange;
152 
153  value = newValue;
154  return ChangeResult::Change;
155  }
156 
157  template <typename VT>
158  ChangeResult meet(const VT &rhs) {
159  return ChangeResult::NoChange;
160  }
161 
162  /// Print the lattice element.
163  void print(raw_ostream &os) const override { value.print(os); }
164 
165 private:
166  /// The currently computed value that is optimistically assumed to be true.
167  ValueT value;
168 };

174 /// Base class for sparse (forward) data-flow analyses. A sparse analysis
175 /// implements a transfer function on operations from the lattices of the
176 /// operands to the lattices of the results. This analysis will propagate
177 /// lattices across control-flow edges and the callgraph using liveness
178 /// information.
179 class AbstractSparseDataFlowAnalysis : public DataFlowAnalysis {
180 public:
181  /// Initialize the analysis by visiting every owner of an SSA value: all
182  /// operations and blocks.
183  LogicalResult initialize(Operation *top) override;
184 
185  /// Visit a program point. If this is a block and all control-flow
186  /// predecessors or callsites are known, then the arguments lattices are
187  /// propagated from them. If this is a call operation or an operation with
188  /// region control-flow, then its result lattices are set accordingly.
189  /// Otherwise, the operation transfer function is invoked.
190  LogicalResult visit(ProgramPoint point) override;
191 
192 protected:
193  explicit AbstractSparseDataFlowAnalysis(DataFlowSolver &solver);
194 
195  /// The operation transfer function. Given the operand lattices, this
196  /// function is expected to set the result lattices.
197  virtual void
198  visitOperationImpl(Operation *op,
199  ArrayRef<const AbstractSparseLattice *> operandLattices,
200  ArrayRef<AbstractSparseLattice *> resultLattices) = 0;
201 
202  /// Given an operation with region control-flow, the lattices of the operands,
203  /// and a region successor, compute the lattice values for block arguments
204  /// that are not accounted for by the branching control flow (ex. the bounds
205  /// of loops).
206  virtual void visitNonControlFlowArgumentsImpl(
207  Operation *op, const RegionSuccessor &successor,
208  ArrayRef<AbstractSparseLattice *> argLattices, unsigned firstIndex) = 0;
209 
210  /// Get the lattice element of a value.
211  virtual AbstractSparseLattice *getLatticeElement(Value value) = 0;
212 
213  /// Get a read-only lattice element for a value and add it as a dependency to
214  /// a program point.
215  const AbstractSparseLattice *getLatticeElementFor(ProgramPoint point,
216  Value value);
217 
218  /// Set the given lattice element(s) at control flow entry point(s).
219  virtual void setToEntryState(AbstractSparseLattice *lattice) = 0;
220  void setAllToEntryStates(ArrayRef<AbstractSparseLattice *> lattices);
221 
222  /// Join the lattice element and propagate and update if it changed.
223  void join(AbstractSparseLattice *lhs, const AbstractSparseLattice &rhs);
224 
225 private:
226  /// Recursively initialize the analysis on nested operations and blocks.
227  LogicalResult initializeRecursively(Operation *op);
228 
229  /// Visit an operation. If this is a call operation or an operation with
230  /// region control-flow, then its result lattices are set accordingly.
231  /// Otherwise, the operation transfer function is invoked.
232  void visitOperation(Operation *op);
233 
234  /// Visit a block to compute the lattice values of its arguments. If this is
235  /// an entry block, then the argument values are determined from the block's
236  /// "predecessors" as set by `PredecessorState`. The predecessors can be
237  /// region terminators or callable callsites. Otherwise, the values are
238  /// determined from block predecessors.
239  void visitBlock(Block *block);
240 
241  /// Visit a program point `point` with predecessors within a region branch
242  /// operation `branch`, which can either be the entry block of one of the
243  /// regions or the parent operation itself, and set either the argument or
244  /// parent result lattices.
245  void visitRegionSuccessors(ProgramPoint point, RegionBranchOpInterface branch,
246  std::optional<unsigned> successorIndex,
247  ArrayRef<AbstractSparseLattice *> lattices);
248 };

254 /// A sparse (forward) data-flow analysis for propagating SSA value lattices
255 /// across the IR by implementing transfer functions for operations.
256 ///
257 /// `StateT` is expected to be a subclass of `AbstractSparseLattice`.
258 template <typename StateT>
259 class SparseDataFlowAnalysis : public AbstractSparseDataFlowAnalysis {
260  static_assert(
261  std::is_base_of<AbstractSparseLattice, StateT>::value,
262  "analysis state class expected to subclass AbstractSparseLattice");
263 
264 public:
265  explicit SparseDataFlowAnalysis(DataFlowSolver &solver)
266  : AbstractSparseDataFlowAnalysis(solver) {}
267 
268  /// Visit an operation with the lattices of its operands. This function is
269  /// expected to set the lattices of the operation's results.
270  virtual void visitOperation(Operation *op, ArrayRef<const StateT *> operands,
271  ArrayRef<StateT *> results) = 0;
272 
273  /// Given an operation with possible region control-flow, the lattices of the
274  /// operands, and a region successor, compute the lattice values for block
275  /// arguments that are not accounted for by the branching control flow (ex.
276  /// the bounds of loops). By default, this method marks all such lattice
277  /// elements as having reached a pessimistic fixpoint. `firstIndex` is the
278  /// index of the first element of `argLattices` that is set by control-flow.
279  virtual void visitNonControlFlowArguments(Operation *op,
280  const RegionSuccessor &successor,
281  ArrayRef<StateT *> argLattices,
282  unsigned firstIndex) {
283  setAllToEntryStates(argLattices.take_front(firstIndex));
284  setAllToEntryStates(argLattices.drop_front(
285  firstIndex + successor.getSuccessorInputs().size()));
286  }
287 
288 protected:
289  /// Get the lattice element for a value.
290  StateT *getLatticeElement(Value value) override {
291  return getOrCreate<StateT>(value);
292  }
293 
294  /// Get the lattice element for a value and create a dependency on the
295  /// provided program point.
296  const StateT *getLatticeElementFor(ProgramPoint point, Value value) {
297  return static_cast<const StateT *>(
298  AbstractSparseDataFlowAnalysis::getLatticeElementFor(point, value));
299  }
300 
301  /// Set the given lattice element(s) at control flow entry point(s).
302  virtual void setToEntryState(StateT *lattice) = 0;
303  void setAllToEntryStates(ArrayRef<StateT *> lattices) {
304  AbstractSparseDataFlowAnalysis::setAllToEntryStates(
305  {reinterpret_cast<AbstractSparseLattice *const *>(lattices.begin()),
306  lattices.size()});
307  }
308 
309 private:
310  /// Type-erased wrappers that convert the abstract lattice operands to derived
311  /// lattices and invoke the virtual hooks operating on the derived lattices.
312  void visitOperationImpl(
313  Operation *op, ArrayRef<const AbstractSparseLattice *> operandLattices,
314  ArrayRef<AbstractSparseLattice *> resultLattices) override {
315  visitOperation(
316  op,
317  {reinterpret_cast<const StateT *const *>(operandLattices.begin()),
318  operandLattices.size()},
319  {reinterpret_cast<StateT *const *>(resultLattices.begin()),
320  resultLattices.size()});
321  }
322  void visitNonControlFlowArgumentsImpl(
323  Operation *op, const RegionSuccessor &successor,
324  ArrayRef<AbstractSparseLattice *> argLattices,
325  unsigned firstIndex) override {
326  visitNonControlFlowArguments(
327  op, successor,
328  {reinterpret_cast<StateT *const *>(argLattices.begin()),
329  argLattices.size()},
330  firstIndex);
331  }
332  void setToEntryState(AbstractSparseLattice *lattice) override {
333  return setToEntryState(reinterpret_cast<StateT *>(lattice));
334  }
335 };
