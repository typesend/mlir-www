52 /// Attempts to apply the pattern specified as template argument to the given
53 /// operation. The pattern is expected to have a `returningMatchAndRewrite`
54 /// function that returns the "main" result or failure. Returns failure if the
55 /// pattern failed to apply. Extra arguments are forwarded to the pattern
56 /// constructor.
57 template <typename PatternTy, typename... Args>
58 static FailureOr<LinalgOp> tryApply(Operation *operation, Args &&...args) {
59  // Check if the given operation has the type expected by the pattern.
60  using OpTy = typename llvm::function_traits<
61  decltype(&PatternTy::returningMatchAndRewrite)>::template arg_t<0>;
62  auto op = dyn_cast<OpTy>(operation);
63  if (!op)
64  return failure();
65 
66  // Apply the pattern directly to the op.
67  PatternTy pattern(operation->getContext(), std::forward<Args>(args)...);
68  // We want to discourage direct use of PatternRewriter in APIs but In this
69  // very specific case, an IRRewriter is not enough.
70  struct TrivialPatternRewriter : public PatternRewriter {
71  public:
72  explicit TrivialPatternRewriter(MLIRContext *context)
73  : PatternRewriter(context) {}
74  };
75  TrivialPatternRewriter rewriter(operation->getContext());
76  rewriter.setInsertionPoint(operation);
77  auto result = pattern.returningMatchAndRewrite(op, rewriter);
78  if (failed(result))
79  return failure();
80  return cast<LinalgOp>(result->getOperation());
81 }

209 /// Apply a tiling transformation to all payload ops and store both the
210 /// tiled operation as well as the created tile loops.
211 static LogicalResult applyTilingToAll(
212  RewriterBase &rewriter, Operation *transformOp,
213  ArrayRef<Operation *> payloadOps, unsigned numLoops,
214  transform::TransformResults &transformResults,
215  function_ref<FailureOr<scf::SCFTileAndFuseResult>(TilingInterface)>
216  applyFn) {
217  SmallVector<Operation *> tiledLinalgOps;
218  SmallVector<SmallVector<Operation *>> loopOps(numLoops);
219  for (unsigned int i = 0; i < numLoops; ++i)
220  loopOps[i].reserve(payloadOps.size());
221 
222  for (Operation *target : payloadOps) {
223  auto tilingInterfaceOp = dyn_cast<TilingInterface>(target);
224  if (!tilingInterfaceOp)
225  return transformOp->emitError("only TilingInterface ops are supported");
226 
227  rewriter.setInsertionPoint(target);
228  FailureOr<scf::SCFTileAndFuseResult> tiledResults =
229  applyFn(tilingInterfaceOp);
230  if (failed(tiledResults))
231  return failure();
232 
233  // Perform the replacement of tiled and fused values.
234  SmallVector<Operation *> opsToReplace{target};
235  llvm::append_range(opsToReplace, tiledResults->fusedProducers);
236  for (Operation *toReplace : opsToReplace) {
237  SmallVector<Value> replacements;
238  replacements.reserve(toReplace->getNumResults());
239  for (OpResult res : toReplace->getResults()) {
240  auto it = tiledResults->replacements.find(res);
241  if (it == tiledResults->replacements.end())
242  replacements.push_back(res);
243  else
244  replacements.push_back(it->getSecond());
245  }
246  rewriter.replaceOp(toReplace, replacements);
247  }
248 
249  // Report back the relevant handles to the transform op.
250  tiledLinalgOps.push_back(tiledResults->tiledAndFusedOps.front());
251  assert(tiledResults->loops.size() == numLoops &&
252  "Mismatched number of loops, tile and fuse transform should have "
253  "failed");
254  for (unsigned int i = 0; i < numLoops; ++i)
255  loopOps[i].push_back(tiledResults->loops[i]);
256  }
257 
258  transformResults.set(transformOp->getOpResult(0), tiledLinalgOps);
259  for (unsigned int i = 0; i < numLoops; ++i)
260  transformResults.set(transformOp->getOpResult(i + 1), loopOps[i]);
261 
262  return success();
263 }

265 /// Parse a tiling-like operation that returns the tiled op as well as the
266 /// created tile loops. The function counts the non-zero tile sizes to compute
267 /// the number of results.
268 static ParseResult parseTileLikeOp(OpAsmParser &parser, OperationState &result,
269  StringRef sizesAttrName) {
270  OpAsmParser::UnresolvedOperand targetOperand;
271  SMLoc opLoc = parser.getCurrentLocation();
272  if (parser.parseOperand(targetOperand) ||
273  parser.parseOptionalAttrDict(result.attributes))
274  return failure();
275  Attribute sizesAttr = result.attributes.get(sizesAttrName);
276  if (!sizesAttr)
277  return parser.emitError(opLoc)
278  << "expected '" << sizesAttrName << "' attribute";
279  auto sizesArrayAttr = sizesAttr.dyn_cast<ArrayAttr>();
280  if (!sizesArrayAttr)
281  return parser.emitError(opLoc)
282  << "'" << sizesAttrName << "' attribute must be an array";
283  Type pdlOpType = parser.getBuilder().getType<pdl::OperationType>();
284  size_t numExpectedLoops =
285  sizesArrayAttr.size() -
286  llvm::count(extractFromI64ArrayAttr(sizesArrayAttr), 0);
287  result.addTypes(SmallVector<Type>(numExpectedLoops + 1, pdlOpType));
288  if (parser.resolveOperand(targetOperand, pdlOpType, result.operands))
289  return failure();
290  return success();
291 }

293 DiagnosedSilenceableFailure
294 transform::FuseOp::apply(mlir::transform::TransformResults &transformResults,
295  mlir::transform::TransformState &state) {
296  SmallVector<int64_t> tileSizes = extractFromI64ArrayAttr(getTileSizes());
297  SmallVector<int64_t> tileInterchange =
298  extractFromI64ArrayAttr(getTileInterchange());
299 
300  scf::SCFTilingOptions tilingOptions;
301  tilingOptions.interchangeVector = tileInterchange;
302  tilingOptions = tilingOptions.setTileSizes(tileSizes);
303  scf::SCFTileAndFuseOptions tileAndFuseOptions;
304  tileAndFuseOptions.tilingOptions = tilingOptions;
305  TrackingListener listener(state, *this);
306  IRRewriter rewriter(getContext(), &listener);
307  LogicalResult result = applyTilingToAll(
308  rewriter, getOperation(), state.getPayloadOps(getTarget()),
309  tileSizes.size() - llvm::count(tileSizes, 0), transformResults,
310  [&](TilingInterface tilingInterfaceOp)
311  -> FailureOr<scf::SCFTileAndFuseResult> {
312  return tileConsumerAndFuseProducerGreedilyUsingSCFForOp(
313  rewriter, tilingInterfaceOp, tileAndFuseOptions);
314  });
315  return failed(result) ? DiagnosedSilenceableFailure::definiteFailure()
316  : DiagnosedSilenceableFailure::success();
317 }

356 /// Find the first "extract" user of `producerOp` and tile it right before its
357 /// use. The tiled op is fused under the `containingOp`.
358 /// Return this fused op on success or nullptr if anything fails.
359 static SmallVector<Operation *>
360 tileAndFuseFirstExtractUse(RewriterBase &rewriter, Diagnostic &diag,
361  Operation *producerOp, Operation *containingOp) {
362  LLVM_DEBUG(DBGS() << "Try to fuse a direct extract use\n");
363  auto tileableProducer = dyn_cast<TilingInterface>(producerOp);
364  if (!tileableProducer) {
365  diag.attachNote(producerOp->getLoc())
366  << "producer is not a TileableInterface: " << *producerOp;
367  return {};
368  }
369 
370  // Search the producer slices accessed within the containing operation.
371  // TODO: Generalize to more extract/insert/parallel_insert triples, maybe
372  // evolve into an interface.
373  auto it = llvm::find_if(tileableProducer->getUsers(), [&](Operation *user) {
374  auto sliceOp = dyn_cast<tensor::ExtractSliceOp>(user);
375  return sliceOp && containingOp->isProperAncestor(sliceOp);
376  });
377 
378  // Find a fusion opportunity.
379  if (it == tileableProducer->getUsers().end()) {
380  diag.attachNote(tileableProducer->getLoc())
381  << "could not find fusion opportunity for: " << *tileableProducer;
382  return {};
383  }
384  auto sliceOpToTile = cast<tensor::ExtractSliceOp>(*it);
385 
386  // Try to fuse the producer in-place.
387  OpBuilder::InsertionGuard guard(rewriter);
388  rewriter.setInsertionPoint(sliceOpToTile);
389 
390  // Tile the producer.
391  int64_t resultNumber =
392  sliceOpToTile.getSource().cast<OpResult>().getResultNumber();
393  LLVM_DEBUG(DBGS() << "resultNumber: " << resultNumber << "\n");
394 
395  FailureOr<TilingResult> tileAndFuseResult =
396  tileableProducer.generateResultTileValue(rewriter, resultNumber,
397  sliceOpToTile.getMixedOffsets(),
398  sliceOpToTile.getMixedSizes());
399  if (failed(tileAndFuseResult)) {
400  diag.attachNote(tileableProducer->getLoc())
401  << "failed to tile producer op: " << *tileableProducer;
402  return {};
403  }
404 
405 #ifndef NDEBUG
406  for (auto tiledOp : tileAndFuseResult->tiledOps) {
407  LLVM_DEBUG(DBGS() << "tiledProducer: " << *tiledOp << "\n");
408  }
409 #endif
410 
411  // Replace the extract op.
412  auto maybeRankReduced = tensor::ExtractSliceOp::rankReduceIfNeeded(
413  rewriter, sliceOpToTile->getLoc(), tileAndFuseResult->tiledValues[0],
414  sliceOpToTile->getResult(0)
415  .getType()
416  .cast<RankedTensorType>()
417  .getShape());
418  assert(succeeded(maybeRankReduced) && "unexpected shape");
419  rewriter.replaceOp(sliceOpToTile, *maybeRankReduced);
420  return tileAndFuseResult->tiledOps;
421 }

423 /// First, find the first "scf::ForallOp" user of `producerOp` and ensure
424 /// it is exactly the `containingOp`, otherwise bail.
425 /// Then, find the first "extract" user of the tied block argument and tile it
426 /// right before its "extract" use. The tiled op is fused under the
427 /// `containingOp`.
428 /// Return this fused op on success or nullptr if anything fails.
429 static SmallVector<Operation *>
430 tileAndFuseFirstExtractUseThroughContainingOpBlockArgument(
431  RewriterBase &rewriter, Diagnostic &diag, Operation *producerOp,
432  Operation *containingOp) {
433  LLVM_DEBUG(DBGS() << "Try to fuse an extract use through block argument\n");
434 
435  auto tileableProducer = dyn_cast<TilingInterface>(producerOp);
436  if (!tileableProducer) {
437  diag.attachNote(producerOp->getLoc())
438  << "producer is not a TileableInterface: " << *producerOp;
439  return {};
440  }
441 
442  // Search the first use by a "scf::ForallOp" user.
443  scf::ForallOp forallOp;
444  auto itProducerUses =
445  llvm::find_if(tileableProducer->getUses(), [&](OpOperand &use) {
446  forallOp = dyn_cast<scf::ForallOp>(use.getOwner());
447  return forallOp;
448  });
449  // If it's not from the containing op, return.
450  if (!forallOp || forallOp != containingOp) {
451  diag.attachNote(tileableProducer->getLoc())
452  << "could not find a use by the containing op: " << *tileableProducer;
453  return {};
454  }
455 
456  // Search the producer slices accessed within the containing
457  // operation.
458  // TODO: Generalize to more extract/insert/parallel_insert triples.
459  // Maybe evolve into an interface.
460  OpOperand *pUse = &(*itProducerUses);
461  BlockArgument bbArg = forallOp.getTiedBlockArgument(pUse);
462 
463  // Search the producer slices accessed within the containing operation.
464  // TODO: Generalize to more extract/insert/parallel_insert triples, maybe
465  // evolve into an interface.
466  auto itBBArgUsers = llvm::find_if(bbArg.getUsers(), [&](Operation *user) {
467  auto sliceOp = dyn_cast<tensor::ExtractSliceOp>(user);
468  return sliceOp && containingOp->isProperAncestor(sliceOp);
469  });
470 
471  // Find a fusion opportunity.
472  if (itBBArgUsers == bbArg.getUsers().end()) {
473  diag.attachNote(containingOp->getLoc())
474  << "could not find fusion opportunity for bbArg: " << bbArg;
475  return {};
476  }
477  auto sliceOpToTile = cast<tensor::ExtractSliceOp>(*itBBArgUsers);
478 
479  // Try to fuse the producer in-place.
480  OpBuilder::InsertionGuard guard(rewriter);
481  rewriter.setInsertionPoint(sliceOpToTile);
482 
483  // Replace the use in the tileableProducer before tiling: clone, replace and
484  // then tile.
485  int64_t resultNumber = pUse->get().cast<OpResult>().getResultNumber();
486  LLVM_DEBUG(DBGS() << "resultNumber: " << resultNumber << "\n");
487 
488  // Gather destination tensors.
489  SmallVector<Value> destinationTensors;
490  if (failed(tensor::getOrCreateDestinations(
491  rewriter, tileableProducer->getLoc(), tileableProducer,
492  destinationTensors))) {
493  diag.attachNote(tileableProducer->getLoc())
494  << "failed to get destination tensors for: " << *tileableProducer;
495  return {};
496  }
497 
498  IRMapping bvm;
499  bvm.map(destinationTensors[resultNumber], bbArg);
500  auto tileableProducerClone =
501  cast<TilingInterface>(rewriter.clone(*tileableProducer, bvm));
502  auto scopeGuard =
503  llvm::make_scope_exit([&]() { rewriter.eraseOp(tileableProducerClone); });
504 
505  // Tile the producer.
506  FailureOr<TilingResult> tileAndFuseResult =
507  tileableProducerClone.generateResultTileValue(
508  rewriter, resultNumber, sliceOpToTile.getMixedOffsets(),
509  sliceOpToTile.getMixedSizes());
510  if (failed(tileAndFuseResult)) {
511  diag.attachNote(tileableProducer->getLoc())
512  << "failed to tile producer op: " << *tileableProducer;
513  return {};
514  }
515 
516  // Replace the extract op.
517  auto maybeRankReduced = tensor::ExtractSliceOp::rankReduceIfNeeded(
518  rewriter, sliceOpToTile->getLoc(), tileAndFuseResult->tiledValues[0],
519  sliceOpToTile->getResult(0)
520  .getType()
521  .cast<RankedTensorType>()
522  .getShape());
523  assert(succeeded(maybeRankReduced) && "unexpected shape");
524  rewriter.replaceOp(sliceOpToTile, *maybeRankReduced);
525 
526  // Replace the use in containingOp.
527  rewriter.updateRootInPlace(containingOp, [&]() {
528  containingOp->setOperand(pUse->getOperandNumber(),
529  destinationTensors.front());
530  });
531 
532  return tileAndFuseResult->tiledOps;
533 }

535 static Operation *cloneAndFuseFirstUse(RewriterBase &rewriter, Diagnostic &diag,
536  Operation *producerOp,
537  Operation *containingOp) {
538  LLVM_DEBUG(DBGS() << "Try to fuse an use by cloning\n");
539 
540  // Gather all uses inside the containing op.
541  SmallVector<OpOperand *> uses;
542  for (OpResult result : producerOp->getOpResults()) {
543  for (OpOperand &use : result.getUses()) {
544  if (containingOp->isProperAncestor(use.getOwner())) {
545  uses.push_back(&use);
546  continue;
547  }
548  // Cannot clone and fuse if the use is by the containing op itself: fail
549  // immediately.
550  if (containingOp == use.getOwner()) {
551  diag.attachNote(producerOp->getLoc())
552  << "producer op use by containing op cannot be fused by cloning";
553  return nullptr;
554  }
555  }
556  }
557 
558  // Check for a non-empty list of fusion opportunities.
559  if (uses.empty()) {
560  diag.attachNote(producerOp->getLoc()) << "no fusion opportunity by cloning";
561  return nullptr;
562  }
563 
564  // Clone and fuse inside the containing op.
565  Operation *fusedOp = nullptr;
566  OpOperand *use = uses.front();
567  // Parallel insert slice is not a valid clone destination.

583 DiagnosedSilenceableFailure
584 transform::FuseIntoContainingOp::apply(transform::TransformResults &results,
585  transform::TransformState &state) {
586  SmallVector<Operation *> fusedOps;
587  ArrayRef<Operation *> producerOps = state.getPayloadOps(getProducerOp());
588  // If nothing to fuse, propagate success.
589  if (producerOps.empty()) {
590  results.set(getFusedOp().cast<OpResult>(),
591  SmallVector<mlir::Operation *>{});
592  return DiagnosedSilenceableFailure::success();
593  }
594  ArrayRef<Operation *> containingOps = state.getPayloadOps(getContainingOp());
595  if (containingOps.size() != 1) {
596  return emitDefiniteFailure()
597  << "requires exactly one containing_op handle (got "
598  << containingOps.size() << ")";
599  }
600  Operation *containingOp = containingOps.front();
601 
602  // Helper function to find the next producer that should be fused. Take any
603  // producer that has a use inside the containing op.
604  SmallVector<Operation *> remainingProducers(producerOps.begin(),
605  producerOps.end());
606  auto getNextProducer = [&]() -> FailureOr<Operation *> {
607  for (const auto &it : enumerate(remainingProducers)) {
608  Operation *producerOp = it.value();
609  // The containing op may be a user of producerOp: use isAncestor.
610  int64_t numUsesInContainingOp =
611  llvm::count_if(producerOp->getUsers(), [&](Operation *op) {
612  return containingOp->isAncestor(op);
613  });
614  // TODO: When resolving the TODO below (no duplicate ops), take an op
615  // that has no use among the remaining producers. This is a topological
616  // sorting.
617  if (numUsesInContainingOp > 0) {
618  if (numUsesInContainingOp == 1)
619  remainingProducers.erase(remainingProducers.begin() + it.index());
620  return producerOp;
621  }
622  }
623  return failure();
624  };
625 
626  TrackingListener listener(state, *this);
627  IRRewriter rewriter(getContext(), &listener);
628  while (!remainingProducers.empty()) {
629  auto nextProducer = getNextProducer();
630  if (failed(nextProducer)) {
631  return mlir::emitSilenceableFailure(containingOp->getLoc())
632  << "could not find next producer to fuse into container";
633  }
634 
635  Operation *producerOp = *nextProducer;
636 
637  // Default diagnostic, to be complemented with more failure information.
638  Diagnostic diag(producerOp->getLoc(), DiagnosticSeverity::Remark);
639  diag << "could not fuse " << *producerOp << " into " << *containingOp;
640 
641  // TODO: If there are multiple uses of the producer in the containing op,
642  // we currently tile/clone the op multiple times (once per use). In some
643  // cases, we can tile/clone once and reuse the value for each use.
644  // Futhermore, producers should then be traversed according to a
645  // topological sorting.
646  SmallVector<Operation *> tiledOps =
647  tileAndFuseFirstExtractUse(rewriter, diag, producerOp, containingOp);
648  if (!tiledOps.empty()) {
649  LLVM_DEBUG(DBGS() << "\nFused a direct extract use\n" << *containingOp);
650  fusedOps.append(tiledOps);
651  continue;
652  }
653 
654  SmallVector<Operation *> tiledContainingOpOperand =
655  tileAndFuseFirstExtractUseThroughContainingOpBlockArgument(
656  rewriter, diag, producerOp, containingOp);
657  if (!tiledContainingOpOperand.empty()) {
658  LLVM_DEBUG(DBGS() << "\nFused an extract use through block argument\n"
659  << *containingOp);
660  fusedOps.append(tiledContainingOpOperand);
661  continue;
662  }
663 
664  Operation *cloned =
665  cloneAndFuseFirstUse(rewriter, diag, producerOp, containingOp);
666  if (cloned) {
667  LLVM_DEBUG(DBGS() << "\nFused an use by cloning\n" << *containingOp);
668  fusedOps.push_back(cloned);
669  continue;
670  }
671  return DiagnosedSilenceableFailure::silenceableFailure(std::move(diag));
672  }
673 
674  results.set(getFusedOp().cast<OpResult>(), fusedOps);
675  return DiagnosedSilenceableFailure::success();
676 }

813 transform::MatchOp::apply(transform::TransformResults &results,
814  transform::TransformState &state) {
815  llvm::StringSet<> strs;
816  if (getOps().has_value())
817  strs.insert(getOps()->getAsValueRange<StringAttr>().begin(),
818  getOps()->getAsValueRange<StringAttr>().end());
819 
820  ArrayRef<Operation *> payloadOps = state.getPayloadOps(getTarget());
821  if (payloadOps.size() != 1) {
822  return emitDefiniteFailure("requires exactly one target handle");
823  }
824 
825  SmallVector<Operation *> res;
826  auto matchFun = [&](Operation *op) {
827  if (getOps().has_value() && !strs.contains(op->getName().getStringRef()))
828  return;
829 
830  // Interfaces cannot be matched by name, just by ID.
831  // So we specifically encode the interfaces we care about for this op.
832  if (getInterface().has_value()) {
833  auto iface = getInterface().value();
834  if (iface == transform::MatchInterfaceEnum::LinalgOp &&
835  !isa<LinalgOp>(op))
836  return;
837  if (iface == transform::MatchInterfaceEnum::TilingInterface &&
838  isa<TilingInterface>(op))
839  return;
840  }
841 
842  // Check if all specified attributes match.
843  if (getOpAttrs().has_value()) {
844  DictionaryAttr opAttrs = getOpAttrs().value();
845  for (NamedAttribute attr : opAttrs) {
846  if (attr.getName() == getInterfaceAttrName() ||
847  attr.getName() == getOpsAttrName())
848  continue;
849  if (!op->hasAttr(attr.getName()))
850  return;
851  if (op->getAttr(attr.getName()) != attr.getValue())
852  return;
853  }
854  }
855 
856  if (getFilterResultType().has_value()) {
857  Type t = getFilterResultType().value();
858  if (op->getNumResults() != 1 || op->getResultTypes().front() != t)
859  return;
860  }
861 
862  // All constraints are satisfied.
863  res.push_back(op);
864  return;
865  };
866 
867  payloadOps.front()->walk(matchFun);
868  results.set(getResult().cast<OpResult>(), res);
869  return DiagnosedSilenceableFailure::success();
870 }

1078 /// Pack a LinalgOp by greedily inferring matmul dimensions (m, n, k) where m
1079 /// and n are proper parallel dimensions and k is a proper reduction
1080 /// dimension. Packing occurs by rewriting the op as a linalg.generic and
1081 /// calling linalg::pack by `mnkPackedSizes`. The order of the packed
1082 /// dimensions is customizable: the `mnkOrder` is a permutation of {0, 1, 2}
1083 /// to reorder {m, n, k} into one of the 8 possible forms. The outer
1084 /// dimensions of the operands are not permuted at this time, this is left for
1085 /// future work.
1086 static FailureOr<PackResult>
1087 packMatmulGreedily(RewriterBase &rewriter, LinalgOp linalgOp,
1088  ArrayRef<OpFoldResult> mnkPackedSizes,
1089  ArrayRef<int64_t> mnkPaddedSizesNextMultipleOf,
1090  ArrayRef<int64_t> mnkOrder) {
1091  assert(mnkPackedSizes.size() == 3 && "unexpected num of packing sizes");
1092  assert((mnkPaddedSizesNextMultipleOf.empty() ||
1093  mnkPaddedSizesNextMultipleOf.size() == 3) &&
1094  "num of packing sizes next multiple should be empty or of size 3");
1095  assert(mnkOrder.size() == 3 && "unexpected mnkOrder size");
1096  assert(isPermutationVector(mnkOrder) && "expected a permutation");
1097 
1098  int64_t numLoops = linalgOp.getNumLoops();
1099  if (numLoops <= 2) {
1100  return rewriter.notifyMatchFailure(
1101  linalgOp, "need 3+ loops to find a matmul to pack");
1102  }
1103 
1104  // Locally adjust the desired iterator position of mnk and packing sizes.
1105  int64_t numPackedDims = mnkPackedSizes.size();
1106  SmallVector<int64_t> mmnnkkPos(numPackedDims);
1107  for (int64_t i = 0, e = numPackedDims; i < e; ++i)
1108  mmnnkkPos[i] = numLoops - numPackedDims + mnkOrder[i];
1109  SmallVector<OpFoldResult> packedSizes(numPackedDims);
1110  for (int64_t i = 0, e = numPackedDims; i < e; ++i)
1111  packedSizes[mnkOrder[i]] = mnkPackedSizes[i];
1112  SmallVector<int64_t> paddedSizesNextMultipleOf(numPackedDims);
1113  for (int64_t i = 0, e = numPackedDims; i < e; ++i) {
1114  paddedSizesNextMultipleOf[mnkOrder[i]] =
1115  mnkPaddedSizesNextMultipleOf.empty() ? 0
1116  : mnkPaddedSizesNextMultipleOf[i];
1117  }
1118 
1119  // 1. Infer dims that are important for matmul.
1120  FailureOr<EmbeddedMatmulDimsCandidates> res = inferMatmulDims(linalgOp);
1121  if (failed(res)) {
1122  return rewriter.notifyMatchFailure(linalgOp,
1123  "couldn't infer matmul iterators");
1124  }
1125 
1126  // 2. Normalize linalgOp to an kmn-matmul-like with [red, par, par] most
1127  // minor iterators. If we wanted a different normalization order, this is
1128  // where it would have to plug a heuristic.
1129  int64_t mPos = *(res->mPos.begin()), nPos = *(res->nPos.begin()),
1130  kPos = *(res->kPos.begin());
1131  LLVM_DEBUG(DBGSNL(); DBGSNL(); DBGSNL();
1132  DBGS() << "Start packing generic op greedily with (m@" << mPos
1133  << ", n@" << nPos << ", k@" << kPos << "): " << linalgOp
1134  << "\n";);
1135 
1136  // 2.a. Rewrite as a generic.
1137  auto genericOp = dyn_cast<GenericOp>(linalgOp.getOperation());
1138  if (!genericOp) {
1139  FailureOr<GenericOp> generalizeResult =
1140  generalizeNamedOp(rewriter, linalgOp);
1141  assert(succeeded(generalizeResult) && "unexpected failure generalizing op");
1142  genericOp = *generalizeResult;
1143  }
1144 
1145  // 2.b. Interchange to move the dimensions (k, m, n) as most-minor
1146  // iterators. Note that this only normalized the iteration order and does
1147  // not change the indexings of any operand.
1148  SmallVector<int64_t> permutation =
1149  computePermutationVector(numLoops, {mPos, nPos, kPos}, mmnnkkPos);
1150  LLVM_DEBUG(llvm::interleaveComma(permutation, DBGS() << "perm: "); DBGSNL(););
1151  // Sign .. unsigned pollution.
1152  SmallVector<unsigned> unsignedPerm(permutation.begin(), permutation.end());
1153  FailureOr<GenericOp> interchangeResult =
1154  interchangeGenericOp(rewriter, genericOp, unsignedPerm);
1155  assert(succeeded(interchangeResult) && "unexpected failure interchanging op");
1156  genericOp = *interchangeResult;
1157  LLVM_DEBUG(DBGS() << "Generalized Op to pack: " << genericOp << "\n";);
1158 
1159  // At this point, the op iterators are normalized to {leading, k, m, n}.
1160  // The layouts induced by packing will always be:
1161  // - LHS{leading_lhs, kk, mm}
1162  // - RHS{leading_rhs, kk, nn}
1163  // - RES{leading_res, mm, nn}
1164  // If we wanted to change the packed order, we would reorder (k, m, n) to
1165  // something else above.
1166  //
1167  // Additional permutations of the outer dims of the operands (i.e.
1168  // leading_lhs, leading_rhs and leading_res) could follow by computing the
1169  // desired outerPerm for each operand.
1170  // This is left for future work.
1171 
1172  // TODO: this creates too much IR, go use reifyResultShapes.
1173  SmallVector<Range, 4> loopRanges =
1174  cast<LinalgOp>(genericOp.getOperation())
1175  .createLoopRanges(rewriter, genericOp.getLoc());
1176 
1177  // Add leading zeros to match numLoops, we only pack the last 3 dimensions
1178  // post interchange.
1179  LLVM_DEBUG(llvm::interleaveComma(paddedSizesNextMultipleOf,
1180  DBGS() << "paddedSizesNextMultipleOf: ");
1181  DBGSNL(););
1182  LLVM_DEBUG(llvm::interleaveComma(loopRanges, DBGS() << "loopRanges: ",
1183  [](Range r) { llvm::dbgs() << r.size; });
1184  DBGSNL(););
1185  SmallVector<OpFoldResult> adjustedPackedSizes(numLoops - packedSizes.size(),
1186  rewriter.getIndexAttr(0));
1187  for (int64_t i = 0, e = numPackedDims; i < e; ++i) {
1188  if (paddedSizesNextMultipleOf[i] == 0) {
1189  adjustedPackedSizes.push_back(packedSizes[i]);
1190  continue;
1191  }
1192  AffineExpr d0, s0;
1193  bindDims(rewriter.getContext(), d0);
1194  bindSymbols(rewriter.getContext(), s0);
1195  adjustedPackedSizes.push_back(affine::makeComposedFoldedAffineApply(
1196  rewriter, genericOp->getLoc(), d0.ceilDiv(s0) * s0,
1197  {loopRanges[adjustedPackedSizes.size()].size,
1198  rewriter.getIndexAttr(paddedSizesNextMultipleOf[i])}));
1199  }
1200  LLVM_DEBUG(llvm::interleaveComma(adjustedPackedSizes,
1201  DBGS() << "adjustedPackedSizes: ");
1202  DBGSNL(););
1203 
1204  // TODO: If we wanted to give the genericOp a name after packing, after
1205  // calling `pack` would be a good time. One would still need to check that
1206  // `containsMostMinorMatmul(packingRes->packedLinalgOp)` is true, since we
1207  // also allow degenerate matmul cases (i.e. matvec, dot).
1208  return linalg::pack(rewriter, genericOp, adjustedPackedSizes);
1209 }

1288 /// Return true if `permutation` is a valid permutation of the
1289 /// `outer_dims_perm` (case OuterOrInnerPerm::Outer) or `inner_dims_pos`
1290 /// (OuterOrInnerPerm::Inner) of the `tensor.pack` or `tensor.unpack` `op.
1291 /// This is the case when the `permutation` rank matches the rank expected by
1292 /// `op` and `permutation` is itself a permutation vector.
1293 /// Return true if either `op` or `permutation` are empty to allow a simpler
1294 /// polymorphic implementation.
1295 template <typename RelayoutOpTy>
1296 bool isValidPackingPermutation(
1297  RelayoutOpTy op, ArrayRef<int64_t> permutation,
1298  OuterOrInnerPerm outerOrInnerPerm = OuterOrInnerPerm::Outer) {
1299  static_assert(
1300  llvm::is_one_of<RelayoutOpTy, tensor::PackOp, tensor::UnPackOp>::value,
1301  "applies to only pack or unpack operations");
1302  if (!op || permutation.empty())
1303  return true;
1304  size_t innerRank = op.getInnerDimsPos().size();
1305  if (outerOrInnerPerm == OuterOrInnerPerm::Inner)
1306  return permutation.size() == innerRank && isPermutationVector(permutation);
1307  // op.getOuterDimsPerm() may be empty, in which case it is identity.
1308  // Don't rely on it.
1309  if (std::is_same<RelayoutOpTy, tensor::PackOp>::value) {
1310  return permutation.size() == op.getSourceRank() &&
1311  isPermutationVector(permutation);
1312  }
1313  return permutation.size() == op.getDestRank() &&
1314  isPermutationVector(permutation);
1315 }

1317 DiagnosedSilenceableFailure
1318 transform::PackTransposeOp::apply(transform::TransformResults &transformResults,
1319  transform::TransformState &state) {
1320  ArrayRef<Operation *> packOrUnpackOps =
1321  state.getPayloadOps(getTargetPackOrUnPackOp());
1322  ArrayRef<Operation *> linalgOps = state.getPayloadOps(getTargetLinalgOp());
1323  // Step 1. If nothing to pack, propagate success.
1324  if (packOrUnpackOps.empty()) {
1325  transformResults.set(getPackedOp().cast<OpResult>(), {});
1326  transformResults.set(getPackOp().cast<OpResult>(), {});
1327  transformResults.set(getUnPackOp().cast<OpResult>(), {});
1328  return DiagnosedSilenceableFailure::success();
1329  }
1330 
1331  // Step 2. Bunch of runtime sanity check and error messages.
1332  // Step 2.1. Fail on multi-op handles.
1333  if (packOrUnpackOps.size() != 1 || linalgOps.size() != 1) {
1334  return emitSilenceableError() << "requires target to map to exactly 1 "
1335  "packing op and 1 packed op ("
1336  << "got " << packOrUnpackOps.size() << " and "
1337  << linalgOps.size() << ")";
1338  }
1339 
1340  // Step 2.2. Fail on wrong type.
1341  auto packOp = dyn_cast<tensor::PackOp>(packOrUnpackOps.front());
1342  auto unPackOp = dyn_cast<tensor::UnPackOp>(packOrUnpackOps.front());
1343  if ((!packOp && !unPackOp)) {
1344  return emitSilenceableError() << "requires target to map to a "
1345  "tensor.pack or tensor.unpack";
1346  }
1347  LinalgOp linalgOpTarget = dyn_cast<LinalgOp>(linalgOps.front());
1348  if (!linalgOpTarget)
1349  return emitSilenceableError() << "requires a LinalgOp target";
1350 
1351  // Step 2.3. Fail if we can't get the producer / consumer Linalg op.
1352  LinalgOp linalgOp;
1353  if (packOp && packOp.getResult().hasOneUse())
1354  linalgOp = dyn_cast<LinalgOp>(*(packOp.getResult().getUsers().begin()));
1355  else if (unPackOp)
1356  linalgOp = unPackOp.getSource().getDefiningOp<LinalgOp>();
1357  if (linalgOp != linalgOpTarget) {
1358  auto errorMsg =
1359  packOp ? StringLiteral{"not a single use by the LinalgOp target"}
1360  : StringLiteral{"not produced by the LinalgOp target"};
1361  return emitSilenceableError() << errorMsg;
1362  }
1363 
1364  // Step 2.4. If we have an UnPackOp, we need to fetch the symmetrical
1365  // PackOp.
1366  if (unPackOp) {
1367  assert(!packOp && "packOp must be null on entry when unPackOp is not null");
1368  OpOperand *packUse = linalgOp.getDpsInitOperand(
1369  unPackOp.getSource().cast<OpResult>().getResultNumber());
1370  packOp = dyn_cast_or_null<tensor::PackOp>(packUse->get().getDefiningOp());
1371  if (!packOp || !packOp.getResult().hasOneUse())
1372  return emitSilenceableError() << "could not find matching pack op";
1373  }
1374 
1375  // Step 2.5. Fail if any permutation does not validate.
1376  for (auto permType : {OuterOrInnerPerm::Outer, OuterOrInnerPerm::Inner}) {
1377  ArrayRef<int64_t> perm =
1378  (permType == OuterOrInnerPerm::Outer) ? getOuterPerm() : getInnerPerm();
1379  auto errorMsg = (permType == OuterOrInnerPerm::Outer)
1380  ? StringLiteral{"invalid outer_perm"}
1381  : StringLiteral{"invalid inner_perm"};
1382  if (!isValidPackingPermutation(packOp, perm, permType) ||
1383  !isValidPackingPermutation(unPackOp, perm, permType)) {
1384  Operation *packOrUnpackOp =
1385  unPackOp ? unPackOp.getOperation() : packOp.getOperation();
1386  return emitSilenceableError() << errorMsg << ": " << *packOrUnpackOp;
1387  }
1388  }
1389 
1390  // From here on, packOp and linalgOp are always present, unPackOp may or may
1391  // not be present.
1392  assert(packOp && linalgOp && "unexpected null op");
1393 
1394  // Step 3. Actually transpose the ops.
1395  TrackingListener listener(state, *this);
1396  IRRewriter rewriter(getContext(), &listener);
1397  FailureOr<PackTransposeResult> res = packTranspose(
1398  rewriter, packOp, linalgOp, unPackOp, getOuterPerm(), getInnerPerm());
1399  // Preconditions have been checked, it is an error to fail here.
1400  assert(succeeded(res) && "unexpected packTranspose failure");
1401 
1402  // Step 4. Return results.
1403  transformResults.set(getPackOp().cast<OpResult>(), {res->transposedPackOp});
1404  transformResults.set(getPackedOp().cast<OpResult>(),
1405  {res->transposedLinalgOp});
1406  if (unPackOp) {
1407  transformResults.set(getUnPackOp().cast<OpResult>(),
1408  {res->transposedUnPackOp});
1409  } else {
1410  transformResults.set(getUnPackOp().cast<OpResult>(), {});
1411  }
1412 
1413  return DiagnosedSilenceableFailure::success();
1414 }

1481  rewriter.replaceOp(target, *result);
1482  results.push_back(paddedOp);
1483  return DiagnosedSilenceableFailure::success();

1830  auto diag = DiagnosedSilenceableFailure::success();
1831  if (getDynamicSplitPoint().getType().isa<TransformHandleTypeInterface>()) {
1832  splitPoints = llvm::to_vector(llvm::map_range(
1833  state.getPayloadOps(getDynamicSplitPoint()), [&](Operation *op) {
1834  if (op->getNumResults() != 1 ||
1835  !op->getResult(0).getType().isIndex()) {
1836  diag = emitSilenceableError()

1842  }));
1843  } else {

1846  [](Attribute attr) { return OpFoldResult(attr); }));
1847  }
1848  if (diag.isSilenceableFailure())
1849  return diag;
1850 
1851  if (splitPoints.size() != payload.size()) {
1852  return emitDefiniteFailure()
1853  << "expected the dynamic split point handle to point to as "
1854  "many operations ("
1855  << splitPoints.size() << ") as the target handle ("
1856  << payload.size() << ")";
1857  }
1858  } else {
1859  splitPoints.resize(payload.size(),
1860  rewriter.getIndexAttr(getStaticSplitPoint()));
1861  }
1862 
1863  // Split each target operation.
1864  SmallVector<Operation *> first, second;
1865  Operation *noSecondPart = nullptr;
1866  for (const auto &pair : llvm::zip(payload, splitPoints)) {
1867  Operation *target = std::get<0>(pair);
1868  auto linalgOp = dyn_cast<LinalgOp>(target);
1869  if (!linalgOp) {
1870  auto diag = emitSilenceableError() << "only applies to structured ops";
1871  diag.attachNote(target->getLoc()) << "target op";
1872  return diag;
1873  }
1874 
1875  if (getDimension() >= linalgOp.getNumLoops()) {
1876  auto diag = emitSilenceableError() << "dimension " << getDimension()
1877  << " does not exist in target op";
1878  diag.attachNote(target->getLoc()) << "target op";
1879  return diag;
1880  }
1881 
1882  rewriter.setInsertionPoint(linalgOp);
1883  std::tie(first.emplace_back(), second.emplace_back()) = linalg::splitOp(
1884  rewriter, cast<TilingInterface>(linalgOp.getOperation()),
1885  getDimension(), std::get<1>(pair));
1886 
1887  // Propagate errors.
1888  if (!first.back() && !second.back()) {
1889  auto diag = emitDefiniteFailure() << "internal failure in splitting";
1890  diag.attachNote(target->getLoc()) << "target op";
1891  return diag;
1892  }
1893 
1894  // Do not add null second parts.
1895  if (!second.back()) {
1896  noSecondPart = target;
1897  second.pop_back();
1898  }
1899  }
1900 
1901  if (second.size() != first.size() && !second.empty()) {
1902  auto diag = emitSilenceableError()

1908  << "first target with no second part";
1909  return diag;
1910  }
1911 
1912  results.set(getFirst().cast<OpResult>(), first);
1913  results.set(getSecond().cast<OpResult>(), second);
1914  return DiagnosedSilenceableFailure::success();
1915 }
