31 /// Makes target array's elements sorted according to the `order` array.
32 static void sortArrayBasedOnOrder(std::vector<LoopId> &target,
33  ArrayRef<LoopId> order) {
34  std::sort(target.begin(), target.end(), [&order](LoopId l, LoopId r) {
35  assert(l != r);
36  int idxL = -1, idxR = -1;
37  for (int i = 0, e = order.size(); i < e; i++) {
38  if (order[i] == l)
39  idxL = i;
40  if (order[i] == r)
41  idxR = i;
42  }
43  assert(idxL >= 0 && idxR >= 0);
44  return idxL < idxR;
45  });
46 }

72 void CodegenEnv::startEmit() {
73  assert(insChain == nullptr && "must only start emitting once");
74  if (sparseOut) {
75  insChain = sparseOut->get();
76  latticeMerger.setHasSparseOut(true);
77  }
78 
79  // Sort the related loop array such that they are in the same order as they
80  // appears on the topoOrder.
81  // TODO: since we only handle affine addition for slice based codegen, and
82  // addition is assoicative, the order how we evaluate the expression does
83  // not matter. However, to support multiplication, the order of the loop
84  // index should match the evaluation order to the affine expression AST.
85 
86  // Initialize loop emitter.
87  SmallVector<Value> tensors; // input tensors passed to loop emitter
88  for (OpOperand &t : linalgOp->getOpOperands()) {
89  tensors.push_back(t.get());
90  const TensorId tid = makeTensorId(t.getOperandNumber());
91  const Level lvlRank = linalgOp.getMatchingIndexingMap(&t).getNumResults();
92  const auto enc = getSparseTensorEncoding(t.get().getType());
93  (void)enc;
94  assert(!enc || lvlRank == enc.getLvlRank());
95  for (Level lvl = 0; lvl < lvlRank; lvl++)
96  sortArrayBasedOnOrder(latticeMerger.getDependentLoops(tid, lvl), topSort);
97  }
98 
99  loopEmitter.initialize(
100  tensors,
101  StringAttr::get(linalgOp.getContext(),
102  linalg::GenericOp::getOperationName()),
103  /*hasOutput=*/true,
104  /*isSparseOut=*/sparseOut != nullptr, topSort,
105  // TODO: compute the map and pass it to loop emitter directly instead of
106  // passing in a callback.
107  [this](TensorId t, Level lvl) -> std::vector<std::pair<TensorId, Level>> {
108  // Translates from a list of loop index to a list of [tid, dim] pair.
109  std::vector<LoopId> rLoops = this->merger().getDependentLoops(t, lvl);
110  std::vector<std::pair<TensorId, Level>> ret;
111  ret.reserve(rLoops.size());
112  for (LoopId l : rLoops)
113  ret.emplace_back(this->merger().getLoopDefiningLvl(l));
114  return ret;
115  });
116 }

118 std::optional<Operation *> CodegenEnv::genLoopBoundary(
119  function_ref<std::optional<Operation *>(MutableArrayRef<Value> parameters)>
120  callback) {
121  SmallVector<Value> params;
122  if (isReduc()) {
123  params.push_back(redVal);
124  if (redValidLexInsert)
125  params.push_back(redValidLexInsert);
126  } else {
127  assert(!redValidLexInsert);
128  }
129  if (isExpand())
130  params.push_back(expCount);
131  if (insChain != nullptr)
132  params.push_back(insChain);
133  auto r = callback(params); // may update parameters
134  unsigned i = 0;
135  if (isReduc()) {
136  // FIXME: This requires `updateExprValue` to perform updates without
137  // checking for a previous value; but it's not clear whether that's
138  // by design or might be a potential source for bugs.
139  updateReduc(params[i++]);
140  if (redValidLexInsert)
141  setValidLexInsert(params[i++]);
142  }
143  if (isExpand())
144  updateExpandCount(params[i++]);
145  if (insChain != nullptr)
146  updateInsertionChain(params[i]);
147  return r;
148 }

154 bool CodegenEnv::isAdmissibleTensorExp(ExprId exp) {
155  // We reject any expression that makes a reduction from `-outTensor`, as those
156  // expressions create a dependency between the current iteration (i) and the
157  // previous iteration (i-1). It would require iterating over the whole
158  // coordinate space, which prevent exploiting sparsity for faster code.
159  for (utils::IteratorType it : linalgOp.getIteratorTypesArray()) {
160  if (it == utils::IteratorType::reduction) {
161  if (latticeMerger.hasNegateOnOut(exp))
162  return false;
163  break;
164  }
165  }
166 
167  OpOperand *lhs = linalgOp.getDpsInitOperand(0);
168  const TensorId tensor = makeTensorId(lhs->getOperandNumber());
169  // An non-annotated output tensor is assumed dense, and becomes a random
170  // access n-dim memref. Admissible since insertions cannot occur.
171  if (getSparseTensorType(lhs->get()).isAllDense())
172  return true;
173 
174  // A tensor expression with a sparse output tensor that changes its values
175  // but not its nonzero structure, an operation called "simply dynamic" in
176  // [Bik96,Ch9], is also admissible without special env.
177  if (latticeMerger.isSingleCondition(tensor, exp))
178  return true;
179 
180  // Accept "truly dynamic" if the output tensor materializes uninitialized
181  // into the computation and insertions occur in lexicographic index order.
182  sparseOut = lhs;
183  return isMaterializing(lhs->get());
184 }

186 bool CodegenEnv::isAdmissibleTopoOrder() {
187  if (!hasSparseOutput())
188  return true;
189 
190  OpOperand *lhs = linalgOp.getDpsInitOperand(0);
191  // Accept "truly dynamic" if the output tensor materializes uninitialized
192  // into the computation and insertions occur in lexicographic index order.
193  LoopOrd nest = 0;
194  const auto iteratorTypes = linalgOp.getIteratorTypesArray();
195  assert(topSortSize() == latticeMerger.getNumLoops());
196  for (const LoopId i : topSort) {
197  if (!latticeMerger.isFilterLoop(i)) {
198  // We only count non-filter loops as filter loops should be considered
199  // a special type of parallel loops.
200  if (linalg::isReductionIterator(iteratorTypes[i]))
201  break; // terminate at first reduction
202  nest++;
203  }
204  }
205  // Determine admissible dynamic insertion situations:
206  // (1) fully injective, since there are no reductions,
207  // (2) admissible 1-d expansion in innermost dimension.
208  if (static_cast<int64_t>(nest) >= linalgOp.getRank(lhs) - 1) {
209  outerParNest = nest;
210  return true;
211  }
212  return false;
213 }

231 Value CodegenEnv::getLoopVar(LoopId i) const {
232  // TODO: this class should store the inverse of `topSort` so that
233  // it can do this conversion directly, instead of searching through
234  // `topSort` every time. (Or else, `LoopEmitter` should handle this.)
235  for (LoopOrd n = 0, numLoops = topSortSize(); n < numLoops; n++)
236  if (topSort[n] == i)
237  return loopEmitter.getLoopIV(n);
238  llvm_unreachable("invalid loop identifier");
239 }
