26 Operation *
27 tensor::TrackingListener::findReplacementOp(Operation *op,
28  ValueRange newValues) const {
29  SmallVector<Value> values(newValues.begin(), newValues.end());
30  do {
31  if (Operation *replacement =
32  transform::TrackingListener::findReplacementOp(op, values))
33  return replacement;
34 
35  Operation *defOp = getCommonDefiningOp(values);
36  if (!defOp)
37  return nullptr;
38 
39  // Skip cast-like operations.
40  // TODO: CastOpInterface could be used if CollapseShapeOp and ExpandShapeOp
41  // implement that interface
42  values.clear();
43  llvm::TypeSwitch<Operation *>(defOp)
44  .Case<CastOp>([&](CastOp op) { values.push_back(op.getSource()); })
45  .Case<CollapseShapeOp>(
46  [&](CollapseShapeOp op) { values.push_back(op.getSrc()); })
47  .Case<ExpandShapeOp>(
48  [&](ExpandShapeOp op) { values.push_back(op.getSrc()); })
49  .Case<ReshapeOp>(
50  [&](ReshapeOp op) { values.push_back(op.getSource()); })
51  .Default([](Operation *op) {});
52  } while (!values.empty());
53 
54  return nullptr;
55 }

61 DiagnosedSilenceableFailure transform::MakeLoopIndependentOp::applyToOne(
62  Operation *target, transform::ApplyToEachResultList &results,
63  transform::TransformState &state) {
64  // Gather IVs.
65  SmallVector<Value> ivs;
66  Operation *nextOp = target;
67  for (uint64_t i = 0, e = getNumLoops(); i < e; ++i) {
68  nextOp = nextOp->getParentOfType<scf::ForOp>();
69  if (!nextOp) {
70  DiagnosedSilenceableFailure diag = emitSilenceableError()
71  << "could not find " << i
72  << "-th enclosing loop";
73  diag.attachNote(target->getLoc()) << "target op";
74  return diag;
75  }
76  ivs.push_back(cast<scf::ForOp>(nextOp).getInductionVar());
77  }
78 
79  // Rewrite IR.
80  IRRewriter rewriter(target->getContext());
81  FailureOr<Value> replacement = failure();
82  if (auto padOp = dyn_cast<tensor::PadOp>(target)) {
83  replacement = tensor::buildIndependentOp(rewriter, padOp, ivs);
84  } else if (auto emptyOp = dyn_cast<tensor::EmptyOp>(target)) {
85  replacement = tensor::buildIndependentOp(rewriter, emptyOp, ivs);
86  } else {
87  DiagnosedSilenceableFailure diag = emitSilenceableError()
88  << "unsupported target op";
89  diag.attachNote(target->getLoc()) << "target op";
90  return diag;
91  }
92  if (failed(replacement)) {
93  DiagnosedSilenceableFailure diag =
94  emitSilenceableError() << "could not make target op loop-independent";
95  diag.attachNote(target->getLoc()) << "target op";
96  return diag;
97  }
98  rewriter.replaceOp(target, *replacement);
99  results.push_back(replacement->getDefiningOp());
100  return DiagnosedSilenceableFailure::success();
101 }
