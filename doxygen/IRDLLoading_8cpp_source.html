25 /// Define and load an operation represented by a `irdl.operation`
26 /// operation.
27 static WalkResult loadOperation(OperationOp op, ExtensibleDialect *dialect) {
28  // IRDL does not support defining custom parsers or printers.
29  auto parser = [](OpAsmParser &parser, OperationState &result) {
30  return failure();
31  };
32  auto printer = [](Operation *op, OpAsmPrinter &printer, StringRef) {
33  printer.printGenericOp(op);
34  };
35 
36  auto verifier = [](Operation *op) { return success(); };
37 
38  // IRDL does not support defining regions.
39  auto regionVerifier = [](Operation *op) { return success(); };
40 
41  auto opDef = DynamicOpDefinition::get(
42  op.getName(), dialect, std::move(verifier), std::move(regionVerifier),
43  std::move(parser), std::move(printer));
44  dialect->registerDynamicOp(std::move(opDef));
45 
46  return WalkResult::advance();
47 }

49 /// Load all dialects in the given module, without loading any operation, type
50 /// or attribute definitions.
51 static DenseMap<DialectOp, ExtensibleDialect *> loadEmptyDialects(ModuleOp op) {
52  DenseMap<DialectOp, ExtensibleDialect *> dialects;
53  op.walk([&](DialectOp dialectOp) {
54  MLIRContext *ctx = dialectOp.getContext();
55  StringRef dialectName = dialectOp.getName();
56 
57  DynamicDialect *dialect = ctx->getOrLoadDynamicDialect(
58  dialectName, [](DynamicDialect *dialect) {});
59 
60  dialects.insert({dialectOp, dialect});
61  });
62  return dialects;
63 }

65 /// Preallocate type definitions objects with empty verifiers.
66 /// This in particular allocates a TypeID for each type definition.
67 static DenseMap<TypeOp, std::unique_ptr<DynamicTypeDefinition>>
68 preallocateTypeDefs(ModuleOp op,
69  DenseMap<DialectOp, ExtensibleDialect *> dialects) {
70  DenseMap<TypeOp, std::unique_ptr<DynamicTypeDefinition>> typeDefs;
71  op.walk([&](TypeOp typeOp) {
72  ExtensibleDialect *dialect = dialects[typeOp.getParentOp()];
73  auto typeDef = DynamicTypeDefinition::get(
74  typeOp.getName(), dialect,
75  [](function_ref<InFlightDiagnostic()>, ArrayRef<Attribute>) {
76  return success();
77  });
78  typeDefs.try_emplace(typeOp, std::move(typeDef));
79  });
80  return typeDefs;
81 }

101 LogicalResult mlir::irdl::loadDialects(ModuleOp op) {
102  // Preallocate all dialects, and type and attribute definitions.
103  // In particular, this allocates TypeIDs so type and attributes can have
104  // verifiers that refer to each other.
105  DenseMap<DialectOp, ExtensibleDialect *> dialects = loadEmptyDialects(op);
106  DenseMap<TypeOp, std::unique_ptr<DynamicTypeDefinition>> types =
107  preallocateTypeDefs(op, dialects);
108  DenseMap<AttributeOp, std::unique_ptr<DynamicAttrDefinition>> attrs =
109  preallocateAttrDefs(op, dialects);
110 
111  // Define and load all operations.
112  WalkResult res = op.walk([&](OperationOp opOp) {
113  return loadOperation(opOp, dialects[opOp.getParentOp()]);
114  });
115  if (res.wasInterrupted())
116  return failure();
117 
118  // Load all types in their dialects.
119  for (auto &pair : types) {
120  ExtensibleDialect *dialect = dialects[pair.first.getParentOp()];
121  dialect->registerDynamicType(std::move(pair.second));
122  }
123 
124  // Load all attributes in their dialects.
125  for (auto &pair : attrs) {
126  ExtensibleDialect *dialect = dialects[pair.first.getParentOp()];
127  dialect->registerDynamicAttr(std::move(pair.second));
128  }
129 
130  return success();
131 }
