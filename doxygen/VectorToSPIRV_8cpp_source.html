41 /// Returns the number of bits for the given scalar/vector type.
42 static int getNumBits(Type type) {
43  // TODO: This does not take into account any memory layout or widening
44  // constraints. E.g., a vector<3xi57> may report to occupy 3x57=171 bit, even
45  // though in practice it will likely be stored as in a 4xi64 vector register.
46  if (auto vectorType = type.dyn_cast<VectorType>())
47  return vectorType.getNumElements() * vectorType.getElementTypeBitWidth();
48  return type.getIntOrFloatBitWidth();
49 }

53 struct VectorBitcastConvert final
54  : public OpConversionPattern<vector::BitCastOp> {
55  using OpConversionPattern::OpConversionPattern;
56 
57  LogicalResult
58  matchAndRewrite(vector::BitCastOp bitcastOp, OpAdaptor adaptor,
59  ConversionPatternRewriter &rewriter) const override {
60  Type dstType = getTypeConverter()->convertType(bitcastOp.getType());
61  if (!dstType)
62  return failure();
63 
64  if (dstType == adaptor.getSource().getType()) {
65  rewriter.replaceOp(bitcastOp, adaptor.getSource());
66  return success();
67  }
68 
69  // Check that the source and destination type have the same bitwidth.
70  // Depending on the target environment, we may need to emulate certain
71  // types, which can cause issue with bitcast.
72  Type srcType = adaptor.getSource().getType();
73  if (getNumBits(dstType) != getNumBits(srcType)) {
74  return rewriter.notifyMatchFailure(
75  bitcastOp,
76  llvm::formatv("different source ({0}) and target ({1}) bitwidth",
77  srcType, dstType));
78  }
79 
80  rewriter.replaceOpWithNewOp<spirv::BitcastOp>(bitcastOp, dstType,
81  adaptor.getSource());
82  return success();
83  }
84 };

281  Value dstVector = adaptor.getOperands().back();

292  rewriter.getI32ArrayAttr(offset));
293  return success();
294  }
295

298  uint64_t insertSize =

316 struct VectorReductionPattern final
317  : public OpConversionPattern<vector::ReductionOp> {
318  using OpConversionPattern::OpConversionPattern;
319 
320  LogicalResult
321  matchAndRewrite(vector::ReductionOp reduceOp, OpAdaptor adaptor,
322  ConversionPatternRewriter &rewriter) const override {
323  Type resultType = typeConverter->convertType(reduceOp.getType());
324  if (!resultType)
325  return failure();
326 
327  auto srcVectorType = adaptor.getVector().getType().dyn_cast<VectorType>();
328  if (!srcVectorType || srcVectorType.getRank() != 1)
329  return rewriter.notifyMatchFailure(reduceOp, "not 1-D vector source");
330 
331  // Extract all elements.
332  int numElements = srcVectorType.getDimSize(0);
333  SmallVector<Value, 4> values;
334  values.reserve(numElements + (adaptor.getAcc() != nullptr));
335  Location loc = reduceOp.getLoc();
336  for (int i = 0; i < numElements; ++i) {
337  values.push_back(rewriter.create<spirv::CompositeExtractOp>(
338  loc, srcVectorType.getElementType(), adaptor.getVector(),
339  rewriter.getI32ArrayAttr({i})));
340  }
341  if (Value acc = adaptor.getAcc())
342  values.push_back(acc);
343 
344  // Reduce them.
345  Value result = values.front();
346  for (Value next : llvm::ArrayRef(values).drop_front()) {
347  switch (reduceOp.getKind()) {
348 
349 #define INT_AND_FLOAT_CASE(kind, iop, fop) \
350  case vector::CombiningKind::kind: \
351  if (resultType.isa<IntegerType>()) { \
352  result = rewriter.create<spirv::iop>(loc, resultType, result, next); \
353  } else { \
354  assert(resultType.isa<FloatType>()); \
355  result = rewriter.create<spirv::fop>(loc, resultType, result, next); \
356  } \
357  break
358 
359 #define INT_OR_FLOAT_CASE(kind, fop) \
360  case vector::CombiningKind::kind: \
361  result = rewriter.create<fop>(loc, resultType, result, next); \
362  break
363 
364  INT_AND_FLOAT_CASE(ADD, IAddOp, FAddOp);
365  INT_AND_FLOAT_CASE(MUL, IMulOp, FMulOp);
366 
367  INT_OR_FLOAT_CASE(MAXF, SPIRVFMaxOp);
368  INT_OR_FLOAT_CASE(MINF, SPIRVFMinOp);
369  INT_OR_FLOAT_CASE(MINUI, SPIRVUMinOp);
370  INT_OR_FLOAT_CASE(MINSI, SPIRVSMinOp);
371  INT_OR_FLOAT_CASE(MAXUI, SPIRVUMaxOp);
372  INT_OR_FLOAT_CASE(MAXSI, SPIRVSMaxOp);
373 
374  case vector::CombiningKind::AND:
375  case vector::CombiningKind::OR:
376  case vector::CombiningKind::XOR:
377  return rewriter.notifyMatchFailure(reduceOp, "unimplemented");
378  }
379  }
380 
381  rewriter.replaceOp(reduceOp, result);
382  return success();
383  }

413  LogicalResult
414  matchAndRewrite(vector::ShuffleOp shuffleOp, OpAdaptor adaptor,

417  if (!spirv::CompositeType::isValid(oldResultType))
418  return failure();
419  Type newResultType = getTypeConverter()->convertType(oldResultType);
420

425  return attr.cast<IntegerAttr>().getValue().getZExtValue();
426  }));
427  rewriter.replaceOpWithNewOp<spirv::VectorShuffleOp>(
428  shuffleOp, newResultType, adaptor.getV1(), adaptor.getV2(),
429  rewriter.getI32ArrayAttr(components));
430  return success();
431  }
432 
433  SmallVector<Value, 2> oldOperands = {adaptor.getV1(), adaptor.getV2()};
434  SmallVector<Value, 4> newOperands;
435  newOperands.reserve(oldResultType.getNumElements());
436  for (const APInt &i : shuffleOp.getMask().getAsValueRange<IntegerAttr>()) {
437  newOperands.push_back(oldOperands[i.getZExtValue()]);
438  }
439  rewriter.replaceOpWithNewOp<spirv::CompositeConstructOp>(
440  shuffleOp, newResultType, newOperands);

442  return success();

446 struct VectorReductionToDotProd final : OpRewritePattern<vector::ReductionOp> {
447  using OpRewritePattern::OpRewritePattern;
448 
449  LogicalResult matchAndRewrite(vector::ReductionOp op,
450  PatternRewriter &rewriter) const override {
451  if (op.getKind() != vector::CombiningKind::ADD)
452  return rewriter.notifyMatchFailure(op, "combining kind is not 'add'");
453 
454  auto resultType = dyn_cast<IntegerType>(op.getType());
455  if (!resultType)
456  return rewriter.notifyMatchFailure(op, "result is not an integer");
457 
458  int64_t resultBitwidth = resultType.getIntOrFloatBitWidth();
459  if (!llvm::is_contained({32, 64}, resultBitwidth))
460  return rewriter.notifyMatchFailure(op, "unsupported integer bitwidth");
461 
462  VectorType inVecTy = op.getSourceVectorType();
463  if (!llvm::is_contained({4, 3}, inVecTy.getNumElements()) ||
464  inVecTy.getShape().size() != 1 || inVecTy.isScalable())
465  return rewriter.notifyMatchFailure(op, "unsupported vector shape");
466 
467  auto mul = op.getVector().getDefiningOp<arith::MulIOp>();
468  if (!mul)
469  return rewriter.notifyMatchFailure(
470  op, "reduction operand is not 'arith.muli'");
471 
472  if (succeeded(handleCase<arith::ExtSIOp, arith::ExtSIOp, spirv::SDotOp,
473  spirv::SDotAccSatOp, false>(op, mul, rewriter)))
474  return success();
475 
476  if (succeeded(handleCase<arith::ExtUIOp, arith::ExtUIOp, spirv::UDotOp,
477  spirv::UDotAccSatOp, false>(op, mul, rewriter)))
478  return success();
479 
480  if (succeeded(handleCase<arith::ExtSIOp, arith::ExtUIOp, spirv::SUDotOp,
481  spirv::SUDotAccSatOp, false>(op, mul, rewriter)))
482  return success();
483 
484  if (succeeded(handleCase<arith::ExtUIOp, arith::ExtSIOp, spirv::SUDotOp,
485  spirv::SUDotAccSatOp, true>(op, mul, rewriter)))
486  return success();
487 
488  return failure();
489  }
490 
491 private:
492  template <typename LhsExtensionOp, typename RhsExtensionOp, typename DotOp,
493  typename DotAccOp, bool SwapOperands>
494  static LogicalResult handleCase(vector::ReductionOp op, arith::MulIOp mul,
495  PatternRewriter &rewriter) {
496  auto lhs = mul.getLhs().getDefiningOp<LhsExtensionOp>();
497  if (!lhs)
498  return failure();
499  Value lhsIn = lhs.getIn();
500  auto lhsInType = cast<VectorType>(lhsIn.getType());
501  if (!lhsInType.getElementType().isInteger(8))
502  return failure();
503 
504  auto rhs = mul.getRhs().getDefiningOp<RhsExtensionOp>();
505  if (!rhs)
506  return failure();
507  Value rhsIn = rhs.getIn();
508  auto rhsInType = cast<VectorType>(rhsIn.getType());
509  if (!rhsInType.getElementType().isInteger(8))
510  return failure();
511 
512  if (op.getSourceVectorType().getNumElements() == 3) {
513  IntegerType i8Type = rewriter.getI8Type();
514  auto v4i8Type = VectorType::get({4}, i8Type);
515  Location loc = op.getLoc();
516  Value zero = spirv::ConstantOp::getZero(i8Type, loc, rewriter);
517  lhsIn = rewriter.create<spirv::CompositeConstructOp>(
518  loc, v4i8Type, ValueRange{lhsIn, zero});
519  rhsIn = rewriter.create<spirv::CompositeConstructOp>(
520  loc, v4i8Type, ValueRange{rhsIn, zero});
521  }
522 
523  // There's no variant of dot prod ops for unsigned LHS and signed RHS, so
524  // we have to swap operands instead in that case.
525  if (SwapOperands)
526  std::swap(lhsIn, rhsIn);
527 
528  if (Value acc = op.getAcc()) {
529  rewriter.replaceOpWithNewOp<DotAccOp>(op, op.getType(), lhsIn, rhsIn, acc,
530  nullptr);
531  } else {
532  rewriter.replaceOpWithNewOp<DotOp>(op, op.getType(), lhsIn, rhsIn,
533  nullptr);
534  }
535 
536  return success();
537  }
