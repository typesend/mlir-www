59 /// Find 2 parallel (m and n) and 1 reduction (k) dimension candidates that form
60 /// a matmul subcomputation within `linalgOp`. These dimensions are such that:
61 /// 1. The m dimension is involved in an outer-product along LHS
62 /// (i.e. it is a permutation on RES and LHS and does not appear in RHS).
63 /// 2. The n dimension is involved in an outer-product along RHS
64 /// (i.e. it is a permutation on RES and RHS and does not appear in LHS).
65 /// 3. The k dimension appears as a permutation on LHS and RHS.
66 /// 4. m, n and k appear only once in any given indexing.
67 /// This allows detecting that some matmul is embedded within `linalgOp` with
68 /// some orthogonal heuristic.

283 /// Scheme used to distribute loops to processors.
284 enum class DistributionMethod {
285  /// Cyclic distribution where no assumption is made about the dynamic
286  /// relationship between number of processors and number of iterations of
287  /// the
288  /// distributed loop. Distributes the following loop
289  ///
290  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
291  ///
292  /// to
293  ///
294  /// scf.parallel(%iv)= (%lb + %procId * %step) to (%ub) step (%step *
295  /// %nprocs)
296  Cyclic = 0,
297 
298  /// Cyclic distribution where the number of processors can be assumed to be
299  /// more than or equal to the number of iterations of the distributed loop.
300  /// In
301  /// such cases, a simple in-bounds check is enough (instead of materializing
302  /// a
303  /// loop). Distributes the following loop
304  ///
305  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
306  ///
307  /// to
308  ///
309  /// %iv = %lb + %procId * %step
310  /// %cond = arith.cmpi "slt", %iv, %ub
311  /// scf.if %cond {
312  /// ...
313  /// }
314  CyclicNumProcsGeNumIters = 1,
315 
316  /// Cyclic distribution where the number of processors can be assumed to be
317  /// equal to the number of iterations of the distributed loop. In such
318  /// cases,
319  /// no bounds check is needed. Distributes the following loop
320  ///
321  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
322  ///
323  /// to
324  ///
325  /// %iv = %lb + %procId * %step
326  CyclicNumProcsEqNumIters = 2,
327 
328  /// No Distribution.
329  None = 3
330 };
