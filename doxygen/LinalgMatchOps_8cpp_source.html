27 DiagnosedSilenceableFailure transform::MatchStructuredOp::matchOperation(
28  Operation *current, transform::TransformResults &results,
29  transform::TransformState &state) {
30  // First, check if the payload operation is a structured Linalg operation.
31  if (!isa<linalg::LinalgOp>(current)) {
32  if (getFailurePropagationMode().value_or(
33  FailurePropagationMode::Propagate) ==
34  FailurePropagationMode::Propagate) {
35  return emitSilenceableError() << "expected a Linalg op";
36  }
37  // If errors are suppressed, succeed and set all results to empty lists.
38  LLVM_DEBUG(DBGS() << "optional nested matcher expected a Linalg op");
39  results.setRemainingToEmpty(cast<TransformOpInterface>(getOperation()));
40  return DiagnosedSilenceableFailure::success();
41  }
42 
43  // Bind `current` to the block argument.
44  auto scope = state.make_region_scope(getBodyRegion());
45  if (failed(state.mapBlockArgument(getBody()->getArgument(0),
46  MappedValue(current)))) {
47  return DiagnosedSilenceableFailure::definiteFailure();
48  }
49 
50  for (Operation &nested : getBody()->without_terminator()) {
51  DiagnosedSilenceableFailure diag =
52  state.applyTransform(cast<TransformOpInterface>(nested));
53  if (diag.isDefiniteFailure())
54  return diag;
55  if (diag.succeeded())
56  continue;
57 
58  // If propagating errors, do this immediately.
59  assert(diag.isSilenceableFailure());
60  if (getFailurePropagationMode().value_or(
61  FailurePropagationMode::Propagate) ==
62  FailurePropagationMode::Propagate) {
63  return diag;
64  }
65 
66  // If suppressing errors, print the message into the debug stream before
67  // silencing it. Then set all results value that are already known.
68  // Results come from the terminator operands, which may be defined in the
69  // (single) block of this operation or above it. When they are defined
70  // above, they are known to be mapped at this point per SSA dominance.
71  // When they are defined in this block, we additionally check if we have
72  // already applied the operation that defines them. If not, the
73  // corresponding results will be set to empty lists.
74  LLVM_DEBUG(DBGS() << "optional nested matcher failed: " << diag.getMessage()
75  << "\n");
76  (void)diag.silence();
77  SmallVector<OpOperand *> undefinedOperands;
78  for (OpOperand &terminatorOperand :
79  getBody()->getTerminator()->getOpOperands()) {
80  Operation *definingOp = terminatorOperand.get().getDefiningOp();
81  if (!definingOp)
82  continue;
83  if (definingOp->getBlock() != getBody())
84  continue;
85  if (definingOp->isBeforeInBlock(&nested))
86  continue;
87 
88  undefinedOperands.push_back(&terminatorOperand);
89  }
90 
91  SmallVector<SmallVector<transform::MappedValue>> mappings;
92  auto filtered = llvm::make_filter_range(
93  getBody()->getTerminator()->getOpOperands(), [&](OpOperand &opOperand) {
94  return !llvm::is_contained(undefinedOperands, &opOperand);
95  });
96  SmallVector<Value> definedOperands = llvm::to_vector(llvm::map_range(
97  filtered, [](OpOperand &opOperand) { return opOperand.get(); }));
98  detail::prepareValueMappings(mappings, definedOperands, state);
99  for (auto &&[operand, mapping] : llvm::zip_equal(filtered, mappings)) {
100  results.setMappedValues(getResults()[operand.getOperandNumber()],
101  mapping);
102  }
103  results.setRemainingToEmpty(cast<TransformOpInterface>(getOperation()));
104  return DiagnosedSilenceableFailure::success();
105  }
106 
107  // Set the results.
108  detail::forwardTerminatorOperands(getBody(), state, results);
109  return DiagnosedSilenceableFailure::success();
110 }

184  return emitSilenceableError() << "not a passthrough";
185  }
186  return DiagnosedSilenceableFailure::success();
187  }
188  return emitDefiniteFailure() << "unknown body condition";
189 }

203 /// Checks if all values from `list` are also contained in `reference`. Returns
204 /// a silenceable error with the given message at the given location when it is
205 /// not the case. The error message must contain the "{0}" placeholder that
206 /// will be substituted with the value from `list` that is not contained in
207 /// `reference`.
208 static DiagnosedSilenceableFailure containsAll(ArrayRef<unsigned> reference,
209  ArrayRef<int64_t> list,
210  Location loc,
211  const char *message) {
212  for (int64_t value : list) {
213  if (llvm::any_of(reference, [&](unsigned ref) {
214  return static_cast<int64_t>(ref) == value;
215  })) {
216  continue;
217  }
218  return emitSilenceableFailure(loc) << llvm::formatv(message, value);
219  }
220  return DiagnosedSilenceableFailure::success();
221 }

223 /// Populates `result` with the positional identifiers relative to `maxNumber`.
224 /// If `isAll` is set, the result will contain all numbers from `0` to
225 /// `maxNumber - 1` inclusive regardless of `rawList`. Otherwise, negative
226 /// values from `rawList` are are interpreted as counting backwards from
227 /// `maxNumber`, i.e., `-1` is interpreted a `maxNumber - 1`, while positive
228 /// numbers remain as is. If `isInverted` is set, populates `result` with those
229 /// values from the `0` to `maxNumber - 1` inclusive range that don't appear in
230 /// `rawList`. If `rawList` contains values that are greater than or equal to
231 /// `maxNumber` or less than `-maxNumber`, produces a silenceable error at the
232 /// given location. `maxNumber` must be positive. If `rawList` contains
233 /// duplicate numbers or numbers that become duplicate after negative value
234 /// remapping, emits a silenceable error.
235 static DiagnosedSilenceableFailure
236 expandTargetSpecification(Location loc, bool isAll, bool isInverted,
237  ArrayRef<int64_t> rawList, int64_t maxNumber,
238  SmallVectorImpl<int64_t> &result) {
239  assert(maxNumber > 0 && "expected size to be positive");
240  assert(!(isAll && isInverted) && "cannot invert all");
241  if (isAll) {
242  result = llvm::to_vector(llvm::seq<int64_t>(0, maxNumber));
243  return DiagnosedSilenceableFailure::success();
244  }
245 
246  SmallVector<int64_t> expanded;
247  llvm::SmallDenseSet<int64_t> visited;
248  expanded.reserve(rawList.size());
249  SmallVectorImpl<int64_t> &target = isInverted ? expanded : result;
250  for (int64_t raw : rawList) {
251  int64_t updated = raw < 0 ? maxNumber + raw : raw;
252  if (updated >= maxNumber) {
253  return emitSilenceableFailure(loc)
254  << "position overflow " << updated << " (updated from " << raw
255  << ") for maximum " << maxNumber;
256  }
257  if (updated < 0) {
258  return emitSilenceableFailure(loc) << "position underflow " << updated
259  << " (updated from " << raw << ")";
260  }
261  if (!visited.insert(updated).second) {
262  return emitSilenceableFailure(loc) << "repeated position " << updated
263  << " (updated from " << raw << ")";
264  }
265  target.push_back(updated);
266  }
267 
268  if (!isInverted)
269  return DiagnosedSilenceableFailure::success();
270 
271  result.reserve(result.size() + (maxNumber - expanded.size()));
272  for (int64_t candidate : llvm::seq<int64_t>(0, maxNumber)) {
273  if (llvm::is_contained(expanded, candidate))
274  continue;
275  result.push_back(candidate);
276  }
277 
278  return DiagnosedSilenceableFailure::success();
279 }

281 /// Checks if the positional specification defined is valid and reports errors
282 /// otherwise.
283 LogicalResult verifyStructuredTransformDimsOp(Operation *op,
284  ArrayRef<int64_t> raw,
285  bool inverted, bool all) {
286  if (all) {
287  if (inverted) {
288  return op->emitOpError()
289  << "cannot request both 'all' and 'inverted' values in the list";
290  }
291  if (!raw.empty()) {
292  return op->emitOpError()
293  << "cannot both request 'all' and specific values in the list";
294  }
295  }
296  if (!all && raw.empty()) {
297  return op->emitOpError() << "must request specific values in the list if "
298  "'all' is not specified";
299  }
300  SmallVector<int64_t> rawVector = llvm::to_vector(raw);
301  auto *it = std::unique(rawVector.begin(), rawVector.end());
302  if (it != rawVector.end())
303  return op->emitOpError() << "expected the listed values to be unique";
304 
305  return success();
306 }

312 DiagnosedSilenceableFailure transform::MatchStructuredDimOp::matchOperation(
313  Operation *current, transform::TransformResults &results,
314  transform::TransformState &state) {
315  auto linalgOp = cast<linalg::LinalgOp>(current);
316  SmallVector<int64_t> dimensions;
317  DiagnosedSilenceableFailure diag = getDimensionsFor(linalgOp, dimensions);
318  if (!diag.succeeded())
319  return diag;
320 
321  // If asked to check for the kind of dimension, perform the check.
322  if (getParallel() || getReduction()) {
323  SmallVector<unsigned> reference;
324  if (getParallel())
325  linalgOp.getParallelDims(reference);
326  else if (getReduction())
327  linalgOp.getReductionDims(reference);
328 
329  DiagnosedSilenceableFailure diag =
330  containsAll(reference, dimensions, getLoc(),
331  getParallel() ? "expects dimension #{0} to be parallel"
332  : "expects dimension #{0} to be reduction");
333  if (!diag.succeeded())
334  return diag;
335  }
336 
337  // If not capturing, we are done here.
338  if (!getResult())
339  return diag;
340 
341  SmallVector<int64_t, 4> ranges = linalgOp.getStaticLoopRanges();
342  Builder builder(current);
343  SmallVector<Attribute> captured = llvm::to_vector(
344  llvm::map_range(dimensions, [&](int64_t dim) -> Attribute {
345  return builder.getI64IntegerAttr(ranges[dim]);
346  }));
347  results.setParams(cast<OpResult>(getResult()), captured);
348  return DiagnosedSilenceableFailure::success();
349 }

402 DiagnosedSilenceableFailure transform::MatchStructuredInputOp::matchOperation(
403  Operation *current, transform::TransformResults &results,
404  transform::TransformState &state) {
405  auto linalgOp = cast<linalg::LinalgOp>(current);
406  SmallVector<int64_t> positions;
407  DiagnosedSilenceableFailure diag = getPositionsFor(linalgOp, positions);
408  if (!diag.succeeded())
409  return diag;
410 
411  SmallVector<MappedValue> operandMapping;
412  operandMapping.reserve(positions.size());
413  for (int64_t position : positions) {
414  AffineMap indexingMap =
415  linalgOp.getMatchingIndexingMap(linalgOp.getDpsInputOperand(position));
416  if (getPermutation() && !indexingMap.isPermutation()) {
417  return emitSilenceableError() << "the indexing map for input #"
418  << position << " is not a permutation";
419  }
420  if (getProjectedPermutation() && !indexingMap.isProjectedPermutation()) {
421  return emitSilenceableError()
422  << "the indexing map for input #" << position
423  << " is not a projected permutation";
424  }
425 
426  // If capture not requested, skip it.
427  if (!getResult())
428  continue;
429 
430  Value operand = linalgOp.getDpsInputOperand(position)->get();
431  if (isa<TransformValueHandleTypeInterface>(getResult().getType())) {
432  operandMapping.emplace_back(operand);
433  continue;
434  }
435 
436  Operation *operandProducer = operand.getDefiningOp();
437  if (!operandProducer) {
438  return emitSilenceableError()
439  << "input #" << position << " is not produced by an operation";
440  }
441  operandMapping.emplace_back(operandProducer);
442  }
443  if (getResult())
444  results.setMappedValues(cast<OpResult>(getResult()), operandMapping);
445  return DiagnosedSilenceableFailure::success();
446 }

600 DiagnosedSilenceableFailure transform::MatchStructuredResultOp::matchOperation(
601  Operation *op, transform::TransformResults &results,
602  transform::TransformState &state) {
603  auto linalgOp = cast<linalg::LinalgOp>(op);
604  int64_t position;
605  DiagnosedSilenceableFailure diag = getPositionFor(linalgOp, position);
606  if (!diag.succeeded())
607  return diag;
608 
609  Value result = linalgOp.getTiedOpResult(linalgOp.getDpsInitOperand(position));
610  if (getResult().getType().isa<TransformValueHandleTypeInterface>()) {
611  results.setValues(cast<OpResult>(getResult()), result);
612  return DiagnosedSilenceableFailure::success();
613  }
614 
615  if (result.getUsers().empty()) {
616  return emitSilenceableError()
617  << "no users of the result #" << getPosition();
618  }
619  Operation *firstUser = *result.getUsers().begin();
620  if (getAny()) {
621  results.set(cast<OpResult>(getResult()), firstUser);

681 constexpr const static llvm::StringLiteral kDimExceptKeyword = "except";

683 /// Keyword syntax for full inclusion in positional specification.
684 constexpr const static llvm::StringLiteral kDimAllKeyword = "all";

686 /// Parses a positional specification for structured transform operations. The
687 /// following forms are accepted:
688 ///
689 /// - `all`: sets `isAll` and returns;
690 /// - comma-separated-integer-list: populates `rawDimList` with the values;
691 /// - `except` `(` comma-separated-integer-list `)`: populates `rawDimList`
692 /// with the values and sets `isInverted`.
693 static ParseResult parseStructuredTransformDims(OpAsmParser &parser,
694  DenseI64ArrayAttr &rawDimList,
695  UnitAttr &isInverted,
696  UnitAttr &isAll) {
697  Builder &builder = parser.getBuilder();
698  if (parser.parseOptionalKeyword(kDimAllKeyword).succeeded()) {
699  rawDimList = builder.getDenseI64ArrayAttr({});
700  isInverted = nullptr;
701  isAll = builder.getUnitAttr();
702  return success();
703  }
704 
705  isAll = nullptr;
706  isInverted = nullptr;
707  if (parser.parseOptionalKeyword(kDimExceptKeyword).succeeded()) {
708  isInverted = builder.getUnitAttr();
709  }
710 
711  if (isInverted) {
712  if (parser.parseLParen().failed())
713  return failure();
714  }
715 
716  SmallVector<int64_t> values;
717  ParseResult listResult = parser.parseCommaSeparatedList(
718  [&]() { return parser.parseInteger(values.emplace_back()); });
719  if (listResult.failed())
720  return failure();
721 
722  rawDimList = builder.getDenseI64ArrayAttr(values);
723 
724  if (isInverted) {
725  if (parser.parseRParen().failed())
726  return failure();
727  }
728  return success();
729 }

731 /// Prints a positional specification for structured transform operations.
732 static void printStructuredTransformDims(OpAsmPrinter &printer, Operation *op,
733  DenseI64ArrayAttr rawDimList,
734  UnitAttr isInverted, UnitAttr isAll) {
735  if (isAll) {
736  printer << kDimAllKeyword;
737  return;
738  }
739  if (isInverted) {
740  printer << kDimExceptKeyword << "(";
741  }
742  llvm::interleaveComma(rawDimList.asArrayRef(), printer.getStream(),
743  [&](int64_t value) { printer << value; });
744  if (isInverted) {
745  printer << ")";
746  }
747 }
748 /// Parses a single non-function type or a function type with at least one
749 /// argument. This allows for the following syntax:
750 ///
751 /// - type: just the argument type;
752 /// - `(` type `)` `->` type: one argument and one result type;
753 /// - `(` type `)` `->` `(` comma-separated-type-list `)`: one argument and
754 /// multiple result types.
755 ///
756 /// Unlike FunctionType, this allows and requires one to omit the parens around
757 /// the argument type in absence of result types, and does not accept the
758 /// trailing `-> ()` construct, which makes the syntax nicer for operations.
759 static ParseResult parseSemiFunctionType(OpAsmParser &parser,
760  Type &argumentType, Type &resultType) {
761  argumentType = resultType = nullptr;
762  bool hasLParen = parser.parseOptionalLParen().succeeded();
763  if (parser.parseType(argumentType).failed())
764  return failure();
765  if (!hasLParen)
766  return success();
767 
768  return failure(parser.parseRParen().failed() ||
769  parser.parseArrow().failed() ||
770  parser.parseType(resultType).failed());
771 }
772 static ParseResult parseSemiFunctionType(OpAsmParser &parser,
773  Type &argumentType,
774  SmallVectorImpl<Type> &resultTypes) {
775  argumentType = nullptr;
776  bool hasLParen = parser.parseOptionalLParen().succeeded();
777  if (parser.parseType(argumentType).failed())
778  return failure();
779  if (!hasLParen)
780  return success();
781 
782  if (parser.parseRParen().failed() || parser.parseArrow().failed())
783  return failure();
784 
785  if (parser.parseOptionalLParen().failed()) {
786  Type type;
787  if (parser.parseType(type).failed())
788  return failure();
789  resultTypes.push_back(type);
790  return success();
791  }
792  if (parser.parseTypeList(resultTypes).failed() ||
793  parser.parseRParen().failed()) {
794  resultTypes.clear();
795  return failure();
796  }
797  return success();
798 }

800 /// Prints argument and result types in a syntax similar to that of FunctionType
801 /// but allowing and requiring one to omit the parens around the argument type
802 /// in absence of result types, and without the trailing `-> ()`.
803 static void printSemiFunctionType(OpAsmPrinter &printer, Operation *op,
804  Type argumentType, TypeRange resultType) {
805  if (!resultType.empty())
806  printer << "(";
807  printer << argumentType;
808  if (resultType.empty())
809  return;

811 
812  if (resultType.size() > 1)
813  printer << "(";
814  llvm::interleaveComma(resultType, printer.getStream());
815  if (resultType.size() > 1)
816  printer << ")";
817 }

819  Type argumentType, Type resultType) {
820  return printSemiFunctionType(printer, op, argumentType,
821  resultType ? TypeRange(resultType)
822  : TypeRange());
