22 template <typename ExprType>
23 SmallVector<ExprType> computeSuffixProductImpl(ArrayRef<ExprType> sizes,
24  ExprType unit) {
25  if (sizes.empty())
26  return {};
27  SmallVector<ExprType> strides(sizes.size(), unit);
28  for (int64_t r = strides.size() - 2; r >= 0; --r)
29  strides[r] = strides[r + 1] * sizes[r + 1];
30  return strides;
31 }

33 template <typename ExprType>
34 SmallVector<ExprType> computeElementwiseMulImpl(ArrayRef<ExprType> v1,
35  ArrayRef<ExprType> v2) {
36  // Early exit if both are empty, let zip_equal fail if only 1 is empty.
37  if (v1.empty() && v2.empty())
38  return {};
39  SmallVector<ExprType> result;
40  for (auto it : llvm::zip_equal(v1, v2))
41  result.push_back(std::get<0>(it) * std::get<1>(it));
42  return result;
43 }

45 template <typename ExprType>
46 ExprType linearizeImpl(ArrayRef<ExprType> offsets, ArrayRef<ExprType> basis,
47  ExprType zero) {
48  assert(offsets.size() == basis.size());
49  ExprType linearIndex = zero;
50  for (unsigned idx = 0, e = basis.size(); idx < e; ++idx)
51  linearIndex = linearIndex + offsets[idx] * basis[idx];
52  return linearIndex;
53 }

55 template <typename ExprType, typename DivOpTy>
56 SmallVector<ExprType> delinearizeImpl(ExprType linearIndex,
57  ArrayRef<ExprType> strides,
58  DivOpTy divOp) {
59  int64_t rank = strides.size();
60  SmallVector<ExprType> offsets(rank);
61  for (int64_t r = 0; r < rank; ++r) {
62  offsets[r] = divOp(linearIndex, strides[r]);
63  linearIndex = linearIndex % strides[r];
64  }
65  return offsets;
66 }

108 std::optional<SmallVector<int64_t>>
109 mlir::computeShapeRatio(ArrayRef<int64_t> shape, ArrayRef<int64_t> subShape) {
110  if (shape.size() < subShape.size())
111  return std::nullopt;
112  assert(llvm::all_of(shape, [](int64_t s) { return s > 0; }) &&
113  "shape must be nonnegative");
114  assert(llvm::all_of(subShape, [](int64_t s) { return s > 0; }) &&
115  "subShape must be nonnegative");
116 
117  // Starting from the end, compute the integer divisors.
118  std::vector<int64_t> result;
119  result.reserve(shape.size());
120  for (auto [size, subSize] :
121  llvm::zip(llvm::reverse(shape), llvm::reverse(subShape))) {
122  // If integral division does not occur, return and let the caller decide.
123  if (size % subSize != 0)
124  return std::nullopt;
125  result.push_back(size / subSize);
126  }
127  // At this point we computed the ratio (in reverse) for the common size.
128  // Fill with the remaining entries from the shape (still in reverse).
129  int commonSize = subShape.size();
130  std::copy(shape.rbegin() + commonSize, shape.rend(),
131  std::back_inserter(result));
132  // Reverse again to get it back in the proper order and return.
133  return SmallVector<int64_t>{result.rbegin(), result.rend()};
134 }

216 SmallVector<int64_t>
217 mlir::computePermutationVector(int64_t permSize, ArrayRef<int64_t> positions,
218  ArrayRef<int64_t> desiredPositions) {
219  SmallVector<int64_t> res(permSize, -1);
220  DenseSet<int64_t> seen;
221  for (auto [pos, desiredPos] : llvm::zip_equal(positions, desiredPositions)) {
222  res[desiredPos] = pos;
223  seen.insert(pos);
224  }
225  int64_t nextPos = 0;
226  for (int64_t &entry : res) {
227  if (entry != -1)
228  continue;
229  while (seen.contains(nextPos))
230  ++nextPos;
231  entry = nextPos;
232  ++nextPos;
233  }
234  return res;
235 }
