25 /// Performs constant folding `calculate` with element-wise behavior on the two
26 /// attributes in `operands` and returns the result if possible.
27 /// Uses `resultType` for the type of the returned attribute.
28 template <class AttrElementT,
29  class ElementValueT = typename AttrElementT::ValueType,
30  class CalculationT = function_ref<
31  std::optional<ElementValueT>(ElementValueT, ElementValueT)>>
32 Attribute constFoldBinaryOpConditional(ArrayRef<Attribute> operands,
33  Type resultType,
34  const CalculationT &calculate) {
35  assert(operands.size() == 2 && "binary op takes two operands");
36  if (!resultType || !operands[0] || !operands[1])
37  return {};
38 
39  if (operands[0].isa<AttrElementT>() && operands[1].isa<AttrElementT>()) {
40  auto lhs = operands[0].cast<AttrElementT>();
41  auto rhs = operands[1].cast<AttrElementT>();
42  if (lhs.getType() != rhs.getType())
43  return {};
44 
45  auto calRes = calculate(lhs.getValue(), rhs.getValue());
46 
47  if (!calRes)
48  return {};
49 
50  return AttrElementT::get(resultType, *calRes);
51  }
52 
53  if (operands[0].isa<SplatElementsAttr>() &&
54  operands[1].isa<SplatElementsAttr>()) {
55  // Both operands are splats so we can avoid expanding the values out and
56  // just fold based on the splat value.
57  auto lhs = operands[0].cast<SplatElementsAttr>();
58  auto rhs = operands[1].cast<SplatElementsAttr>();
59  if (lhs.getType() != rhs.getType())
60  return {};
61 
62  auto elementResult = calculate(lhs.getSplatValue<ElementValueT>(),
63  rhs.getSplatValue<ElementValueT>());
64  if (!elementResult)
65  return {};
66 
67  return DenseElementsAttr::get(cast<ShapedType>(resultType), *elementResult);
68  }
69 
70  if (operands[0].isa<ElementsAttr>() && operands[1].isa<ElementsAttr>()) {
71  // Operands are ElementsAttr-derived; perform an element-wise fold by
72  // expanding the values.
73  auto lhs = operands[0].cast<ElementsAttr>();
74  auto rhs = operands[1].cast<ElementsAttr>();
75  if (lhs.getType() != rhs.getType())
76  return {};
77 
78  auto lhsIt = lhs.value_begin<ElementValueT>();
79  auto rhsIt = rhs.value_begin<ElementValueT>();
80  SmallVector<ElementValueT, 4> elementResults;
81  elementResults.reserve(lhs.getNumElements());
82  for (size_t i = 0, e = lhs.getNumElements(); i < e; ++i, ++lhsIt, ++rhsIt) {
83  auto elementResult = calculate(*lhsIt, *rhsIt);
84  if (!elementResult)
85  return {};
86  elementResults.push_back(*elementResult);
87  }
88 
89  return DenseElementsAttr::get(cast<ShapedType>(resultType), elementResults);
90  }
91  return {};
92 }

94 /// Performs constant folding `calculate` with element-wise behavior on the two
95 /// attributes in `operands` and returns the result if possible.
96 /// Uses the operand element type for the element type of the returned
97 /// attribute.
98 template <class AttrElementT,
99  class ElementValueT = typename AttrElementT::ValueType,
100  class CalculationT = function_ref<
101  std::optional<ElementValueT>(ElementValueT, ElementValueT)>>
102 Attribute constFoldBinaryOpConditional(ArrayRef<Attribute> operands,
103  const CalculationT &calculate) {
104  assert(operands.size() == 2 && "binary op takes two operands");
105  auto getResultType = [](Attribute attr) -> Type {
106  if (auto typed = attr.dyn_cast_or_null<TypedAttr>())
107  return typed.getType();
108  return {};
109  };
110 
111  Type lhsType = getResultType(operands[0]);
112  Type rhsType = getResultType(operands[1]);
113  if (!lhsType || !rhsType)
114  return {};
115  if (lhsType != rhsType)
116  return {};
117 
118  return constFoldBinaryOpConditional<AttrElementT, ElementValueT,
119  CalculationT>(operands, lhsType,
120  calculate);
121 }

149 /// Performs constant folding `calculate` with element-wise behavior on the one
150 /// attributes in `operands` and returns the result if possible.
151 template <class AttrElementT,
152  class ElementValueT = typename AttrElementT::ValueType,
153  class CalculationT =
154  function_ref<std::optional<ElementValueT>(ElementValueT)>>
155 Attribute constFoldUnaryOpConditional(ArrayRef<Attribute> operands,
156  const CalculationT &&calculate) {
157  assert(operands.size() == 1 && "unary op takes one operands");
158  if (!operands[0])
159  return {};
160 
161  if (operands[0].isa<AttrElementT>()) {
162  auto op = operands[0].cast<AttrElementT>();
163 
164  auto res = calculate(op.getValue());
165  if (!res)
166  return {};
167  return AttrElementT::get(op.getType(), *res);
168  }
169  if (operands[0].isa<SplatElementsAttr>()) {
170  // Both operands are splats so we can avoid expanding the values out and
171  // just fold based on the splat value.
172  auto op = operands[0].cast<SplatElementsAttr>();
173 
174  auto elementResult = calculate(op.getSplatValue<ElementValueT>());
175  if (!elementResult)
176  return {};
177  return DenseElementsAttr::get(op.getType(), *elementResult);
178  } else if (operands[0].isa<ElementsAttr>()) {
179  // Operands are ElementsAttr-derived; perform an element-wise fold by
180  // expanding the values.
181  auto op = operands[0].cast<ElementsAttr>();
182 
183  auto opIt = op.value_begin<ElementValueT>();
184  SmallVector<ElementValueT> elementResults;
185  elementResults.reserve(op.getNumElements());
186  for (size_t i = 0, e = op.getNumElements(); i < e; ++i, ++opIt) {
187  auto elementResult = calculate(*opIt);
188  if (!elementResult)
189  return {};
190  elementResults.push_back(*elementResult);
191  }
192  return DenseElementsAttr::get(op.getShapedType(), elementResults);
193  }
194  return {};
195 }

208 template <
209  class AttrElementT, class TargetAttrElementT,
210  class ElementValueT = typename AttrElementT::ValueType,
211  class TargetElementValueT = typename TargetAttrElementT::ValueType,
212  class CalculationT = function_ref<TargetElementValueT(ElementValueT, bool)>>
213 Attribute constFoldCastOp(ArrayRef<Attribute> operands, Type resType,
214  const CalculationT &calculate) {
215  assert(operands.size() == 1 && "Cast op takes one operand");
216  if (!operands[0])
217  return {};
218 
219  if (operands[0].isa<AttrElementT>()) {
220  auto op = operands[0].cast<AttrElementT>();
221  bool castStatus = true;
222  auto res = calculate(op.getValue(), castStatus);
223  if (!castStatus)
224  return {};
225  return TargetAttrElementT::get(resType, res);
226  }
227  if (operands[0].isa<SplatElementsAttr>()) {
228  // The operand is a splat so we can avoid expanding the values out and
229  // just fold based on the splat value.
230  auto op = operands[0].cast<SplatElementsAttr>();
231  bool castStatus = true;
232  auto elementResult =
233  calculate(op.getSplatValue<ElementValueT>(), castStatus);
234  if (!castStatus)
235  return {};
236  return DenseElementsAttr::get(cast<ShapedType>(resType), elementResult);
237  }
238  if (operands[0].isa<ElementsAttr>()) {
239  // Operand is ElementsAttr-derived; perform an element-wise fold by
240  // expanding the value.
241  auto op = operands[0].cast<ElementsAttr>();
242  bool castStatus = true;
243  auto opIt = op.value_begin<ElementValueT>();
244  SmallVector<TargetElementValueT> elementResults;
245  elementResults.reserve(op.getNumElements());
246  for (size_t i = 0, e = op.getNumElements(); i < e; ++i, ++opIt) {
247  auto elt = calculate(*opIt, castStatus);
248  if (!castStatus)
249  return {};
250  elementResults.push_back(elt);
251  }
252 
253  return DenseElementsAttr::get(cast<ShapedType>(resType), elementResults);
254  }
255  return {};
256 }
