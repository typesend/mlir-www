38 /// A listener that updates a TransformState based on IR modifications. This
39 /// listener can be used during a greedy pattern rewrite to keep the transform
40 /// state up-to-date.
41 class TrackingListener : public RewriterBase::Listener,
42  public TransformState::Extension {
43 public:
44  /// Create a new TrackingListener for usage in the specified transform op.
45  explicit TrackingListener(TransformState &state, TransformOpInterface op)
46  : TransformState::Extension(state), transformOp(op) {}
47 
48 protected:
49  /// Return a replacement payload op for the given op, which is going to be
50  /// replaced with the given values. By default, if all values are defined by
51  /// the same newly-created op, which also has the same type as the given op,
52  /// that defining op is used as a replacement.
53  virtual Operation *findReplacementOp(Operation *op,
54  ValueRange newValues) const;
55 
56  /// Notify the listener that the pattern failed to match the given operation,
57  /// and provide a callback to populate a diagnostic with the reason why the
58  /// failure occurred.
59  LogicalResult
60  notifyMatchFailure(Location loc,
61  function_ref<void(Diagnostic &)> reasonCallback) override;
62 
63  /// This function is called when a tracked payload op is dropped because no
64  /// replacement op was found. Derived classes can implement this function for
65  /// custom error handling.
66  virtual void notifyPayloadReplacementNotFound(Operation *op,
67  ValueRange values) {}
68 
69  /// Return "true" if the given op is a new op.
70  bool isNewOp(Operation *op) const;
71 
72  /// Return the single op that defines all given values (if any).
73  static Operation *getCommonDefiningOp(ValueRange values);
74 
75 private:
76  void notifyOperationInserted(Operation *op) override;
77 
78  void notifyOperationRemoved(Operation *op) override;
79 
80  void notifyOperationReplaced(Operation *op, ValueRange newValues) override;
81 
82  /// Ops that were newly created during the transform.
83  DenseMap<OperationName, DenseSet<Operation *>> newOps;
84 
85  /// The transform op in which this TrackingListener is used.
86  TransformOpInterface transformOp;
87 };
