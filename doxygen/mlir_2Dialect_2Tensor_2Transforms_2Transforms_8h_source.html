68 /// Build a new tensor::PadOp with low/high padding that is independent of all
69 /// given independencies. If the op is already independent of all
70 /// independencies, the same PadOp result is returned.
71 ///
72 /// Failure indicates the no suitable upper bound for low/high padding could be
73 /// found.
74 ///
75 /// Example:
76 /// scf.for %iv = %lb to %ub step %step {
77 /// %high = affine.apply affine_map<(d0)[s0] -> (s0 - d0)> (%i)[%ub]
78 /// %p = tensor.pad %t low[5] high[%high] ...
79 /// ...
80 /// }
81 ///
82 /// The function builds IR such as:
83 /// %high_new = affine.apply affine_map<()[s0, s1] -> (-s0 + s1)> ()[%lb, %ub]
84 /// %p_hoistable = tensor.pad %t low[5] high[%high_new]
85 /// %dim = tensor.dim %t, %c0
86 /// %size = affine.apply affine_map<(d0)[s0, s1] -> (-d0 + s0 + s1 + 5)>
87 /// (%iv)[%ub, %dim]
88 /// %slice = tensor.extract_slice %p_hoistable [0] [%size] [1]
89 ///
90 /// The slice is returned.
91 FailureOr<Value> buildIndependentOp(OpBuilder &b, tensor::PadOp padOp,
92  ValueRange independencies);
93 
94 /// Build a new tensor::EmptyOp who's dynamic sizes are independent of all
95 /// given independencies. If the op is already independent of all
96 /// independencies, the same EmptyOp result is returned.
97 ///
98 /// Failure indicates the no suitable upper bound for the dynamic sizes could be
99 /// found.
100 FailureOr<Value> buildIndependentOp(OpBuilder &b, tensor::EmptyOp emptyOp,
101  ValueRange independencies);
