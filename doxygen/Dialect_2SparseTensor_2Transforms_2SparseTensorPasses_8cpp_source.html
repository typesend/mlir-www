105 struct SparseTensorConversionPass
106  : public impl::SparseTensorConversionPassBase<SparseTensorConversionPass> {
107 
108  SparseTensorConversionPass() = default;
109  SparseTensorConversionPass(const SparseTensorConversionPass &pass) = default;
110  SparseTensorConversionPass(const SparseTensorConversionOptions &options) {
111  sparseToSparse = static_cast<int32_t>(options.sparseToSparseStrategy);
112  }
113 
114  void runOnOperation() override {
115  auto *ctx = &getContext();
116  RewritePatternSet patterns(ctx);
117  SparseTensorTypeToPtrConverter converter;
118  ConversionTarget target(*ctx);
119  // Everything in the sparse dialect must go!
120  target.addIllegalDialect<SparseTensorDialect>();
121  // All dynamic rules below accept new function, call, return, and various
122  // tensor and bufferization operations as legal output of the rewriting
123  // provided that all sparse tensor types have been fully rewritten.
124  target.addDynamicallyLegalOp<func::FuncOp>([&](func::FuncOp op) {
125  return converter.isSignatureLegal(op.getFunctionType());
126  });
127  target.addDynamicallyLegalOp<func::CallOp>([&](func::CallOp op) {
128  return converter.isSignatureLegal(op.getCalleeType());
129  });
130  target.addDynamicallyLegalOp<func::ReturnOp>([&](func::ReturnOp op) {
131  return converter.isLegal(op.getOperandTypes());
132  });
133  target.addDynamicallyLegalOp<tensor::DimOp>([&](tensor::DimOp op) {
134  return converter.isLegal(op.getOperandTypes());
135  });
136  target.addDynamicallyLegalOp<tensor::CastOp>([&](tensor::CastOp op) {
137  return converter.isLegal(op.getSource().getType()) &&
138  converter.isLegal(op.getDest().getType());
139  });
140  target.addDynamicallyLegalOp<tensor::ExpandShapeOp>(
141  [&](tensor::ExpandShapeOp op) {
142  return converter.isLegal(op.getSrc().getType()) &&
143  converter.isLegal(op.getResult().getType());
144  });
145  target.addDynamicallyLegalOp<tensor::CollapseShapeOp>(
146  [&](tensor::CollapseShapeOp op) {
147  return converter.isLegal(op.getSrc().getType()) &&
148  converter.isLegal(op.getResult().getType());
149  });
150  target.addDynamicallyLegalOp<bufferization::AllocTensorOp>(
151  [&](bufferization::AllocTensorOp op) {
152  return converter.isLegal(op.getType());
153  });
154  target.addDynamicallyLegalOp<bufferization::DeallocTensorOp>(
155  [&](bufferization::DeallocTensorOp op) {
156  return converter.isLegal(op.getTensor().getType());
157  });
158  // The following operations and dialects may be introduced by the
159  // rewriting rules, and are therefore marked as legal.
160  target.addLegalOp<complex::ConstantOp, complex::NotEqualOp, linalg::FillOp,
161  linalg::YieldOp, tensor::ExtractOp>();
162  target.addLegalDialect<
163  arith::ArithDialect, bufferization::BufferizationDialect,
164  LLVM::LLVMDialect, memref::MemRefDialect, scf::SCFDialect>();
165  // Translate strategy flags to strategy options.
166  SparseTensorConversionOptions options(
167  sparseToSparseConversionStrategy(sparseToSparse));
168  // Populate with rules and apply rewriting rules.
169  populateFunctionOpInterfaceTypeConversionPattern<func::FuncOp>(patterns,
170  converter);
171  populateCallOpTypeConversionPattern(patterns, converter);
172  scf::populateSCFStructuralTypeConversionsAndLegality(converter, patterns,
173  target);
174  populateSparseTensorConversionPatterns(converter, patterns, options);
175  if (failed(applyPartialConversion(getOperation(), target,
176  std::move(patterns))))
177  signalPassFailure();
178  }
179 };
