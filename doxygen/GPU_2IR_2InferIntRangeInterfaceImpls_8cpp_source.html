41 static Value valueByDim(KernelDim3 dims, Dimension dim) {
42  switch (dim) {
43  case Dimension::x:

55 template <typename Op>
56 static std::optional<uint64_t> getKnownLaunchDim(Op op, LaunchDims type) {
57  Dimension dim = op.getDimension();
58  if (auto launch = op->template getParentOfType<LaunchOp>()) {
59  KernelDim3 bounds;
60  switch (type) {
61  case LaunchDims::Block:
62  bounds = launch.getBlockSizeOperandValues();
63  break;
64  case LaunchDims::Grid:
65  bounds = launch.getGridSizeOperandValues();
66  break;
67  }
68  Value maybeBound = valueByDim(bounds, dim);
69  APInt value;
70  if (matchPattern(maybeBound, m_ConstantInt(&value)))
71  return value.getZExtValue();
72  }
73 
74  if (auto func = op->template getParentOfType<GPUFuncOp>()) {
75  switch (type) {
76  case LaunchDims::Block:
77  return llvm::transformOptional(func.getKnownBlockSize(dim), zext);
78  case LaunchDims::Grid:
79  return llvm::transformOptional(func.getKnownGridSize(dim), zext);
80  }
81  }
82  return std::nullopt;
83 }

146 void LaunchOp::inferResultRanges(ArrayRef<ConstantIntRanges> argRanges,
147  SetIntRangeFn setResultRange) {
148  auto setRange = [&](const ConstantIntRanges &argRange, Value dimResult,
149  Value idxResult) {
150  if (argRange.umin().getBitWidth() != IndexType::kInternalStorageBitWidth)
151  return;
152  ConstantIntRanges dimRange =
153  argRange.intersection(getIndexRange(1, kMaxDim));
154  setResultRange(dimResult, dimRange);
155  ConstantIntRanges idxRange =
156  getIndexRange(0, dimRange.umax().getZExtValue() - 1);
157  setResultRange(idxResult, idxRange);
158  };
159 
160  argRanges = argRanges.drop_front(getAsyncDependencies().size());
161  KernelDim3 gridDims = getGridSize();
162  KernelDim3 blockIds = getBlockIds();
163  setRange(argRanges[0], gridDims.x, blockIds.x);
164  setRange(argRanges[1], gridDims.y, blockIds.y);
165  setRange(argRanges[2], gridDims.z, blockIds.z);
166  KernelDim3 blockDims = getBlockSize();
167  KernelDim3 threadIds = getThreadIds();
168  setRange(argRanges[3], blockDims.x, threadIds.x);
169  setRange(argRanges[4], blockDims.y, threadIds.y);
170  setRange(argRanges[5], blockDims.z, threadIds.z);
171 }
