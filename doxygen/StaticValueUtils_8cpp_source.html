18 bool isZeroIndex(OpFoldResult v) {
19  if (!v)
20  return false;
21  if (auto attr = v.dyn_cast<Attribute>()) {
22  IntegerAttr intAttr = attr.dyn_cast<IntegerAttr>();
23  return intAttr && intAttr.getValue().isZero();
24  }
25  if (auto cst = v.get<Value>().getDefiningOp<arith::ConstantIndexOp>())
26  return cst.value() == 0;
27  return false;
28 }

45 /// Helper function to dispatch an OpFoldResult into `staticVec` if:
46 /// a) it is an IntegerAttr
47 /// In other cases, the OpFoldResult is dispached to the `dynamicVec`.
48 /// In such dynamic cases, a copy of the `sentinel` value is also pushed to
49 /// `staticVec`. This is useful to extract mixed static and dynamic entries that
50 /// come from an AttrSizedOperandSegments trait.
51 void dispatchIndexOpFoldResult(OpFoldResult ofr,
52  SmallVectorImpl<Value> &dynamicVec,
53  SmallVectorImpl<int64_t> &staticVec) {
54  auto v = ofr.dyn_cast<Value>();
55  if (!v) {
56  APInt apInt = ofr.get<Attribute>().cast<IntegerAttr>().getValue();
57  staticVec.push_back(apInt.getSExtValue());
58  return;
59  }
60  dynamicVec.push_back(v);
61  staticVec.push_back(ShapedType::kDynamic);
62 }

116 /// If ofr is a constant integer or an IntegerAttr, return the integer.
117 std::optional<int64_t> getConstantIntValue(OpFoldResult ofr) {
118  // Case 1: Check for Constant integer.
119  if (auto val = ofr.dyn_cast<Value>()) {
120  APSInt intVal;
121  if (matchPattern(val, m_ConstantInt(&intVal)))
122  return intVal.getSExtValue();
123  return std::nullopt;
124  }
125  // Case 2: Check for IntegerAttr.
126  Attribute attr = ofr.dyn_cast<Attribute>();
127  if (auto intAttr = attr.dyn_cast_or_null<IntegerAttr>())
128  return intAttr.getValue().getSExtValue();
129  return std::nullopt;
130 }

138 /// Return true if ofr1 and ofr2 are the same integer constant attribute values
139 /// or the same SSA value.
140 /// Ignore integer bitwidth and type mismatch that come from the fact there is
141 /// no IndexAttr and that IndexType has no bitwidth.
142 bool isEqualConstantIntOrValue(OpFoldResult ofr1, OpFoldResult ofr2) {
143  auto cst1 = getConstantIntValue(ofr1), cst2 = getConstantIntValue(ofr2);
144  if (cst1 && cst2 && *cst1 == *cst2)
145  return true;
146  auto v1 = ofr1.dyn_cast<Value>(), v2 = ofr2.dyn_cast<Value>();
147  return v1 && v1 == v2;
148 }

150 bool isEqualConstantIntOrValueArray(ArrayRef<OpFoldResult> ofrs1,
151  ArrayRef<OpFoldResult> ofrs2) {
152  if (ofrs1.size() != ofrs2.size())
153  return false;
154  for (auto [ofr1, ofr2] : llvm::zip_equal(ofrs1, ofrs2))
155  if (!isEqualConstantIntOrValue(ofr1, ofr2))
156  return false;
157  return true;

160 /// Return a vector of OpFoldResults with the same size a staticValues, but all
161 /// elements for which ShapedType::isDynamic is true, will be replaced by
162 /// dynamicValues.
163 SmallVector<OpFoldResult> getMixedValues(ArrayRef<int64_t> staticValues,
164  ValueRange dynamicValues, Builder &b) {
165  SmallVector<OpFoldResult> res;
166  res.reserve(staticValues.size());
167  unsigned numDynamic = 0;
168  unsigned count = static_cast<unsigned>(staticValues.size());
169  for (unsigned idx = 0; idx < count; ++idx) {
170  int64_t value = staticValues[idx];
171  res.push_back(ShapedType::isDynamic(value)
172  ? OpFoldResult{dynamicValues[numDynamic++]}
173  : OpFoldResult{b.getI64IntegerAttr(staticValues[idx])});
174  }
175  return res;
176 }

178 /// Decompose a vector of mixed static or dynamic values into the corresponding
179 /// pair of arrays. This is the inverse function of `getMixedValues`.
180 std::pair<ArrayAttr, SmallVector<Value>>
181 decomposeMixedValues(Builder &b,
182  const SmallVectorImpl<OpFoldResult> &mixedValues) {
183  SmallVector<int64_t> staticValues;
184  SmallVector<Value> dynamicValues;
185  for (const auto &it : mixedValues) {
186  if (it.is<Attribute>()) {
187  staticValues.push_back(it.get<Attribute>().cast<IntegerAttr>().getInt());
188  } else {
189  staticValues.push_back(ShapedType::kDynamic);
190  dynamicValues.push_back(it.get<Value>());
191  }
192  }
193  return {b.getI64ArrayAttr(staticValues), dynamicValues};
194 }

196 /// Helper to sort `values` according to matching `keys`.
197 template <typename K, typename V>
198 static SmallVector<V>
199 getValuesSortedByKeyImpl(ArrayRef<K> keys, ArrayRef<V> values,
200  llvm::function_ref<bool(K, K)> compare) {
201  if (keys.empty())
202  return SmallVector<V>{values};
203  assert(keys.size() == values.size() && "unexpected mismatching sizes");
204  auto indices = llvm::to_vector(llvm::seq<int64_t>(0, values.size()));
205  std::sort(indices.begin(), indices.end(),
206  [&](int64_t i, int64_t j) { return compare(keys[i], keys[j]); });
207  SmallVector<V> res;
208  res.reserve(values.size());
209  for (int64_t i = 0, e = indices.size(); i < e; ++i)
210  res.push_back(values[indices[i]]);
211  return res;
212 }

232 /// Return the number of iterations for a loop with a lower bound `lb`, upper
233 /// bound `ub` and step `step`.
234 std::optional<int64_t> constantTripCount(OpFoldResult lb, OpFoldResult ub,
235  OpFoldResult step) {
236  if (lb == ub)
237  return 0;
238 
239  std::optional<int64_t> lbConstant = getConstantIntValue(lb);
240  if (!lbConstant)
241  return std::nullopt;
242  std::optional<int64_t> ubConstant = getConstantIntValue(ub);
243  if (!ubConstant)
244  return std::nullopt;
245  std::optional<int64_t> stepConstant = getConstantIntValue(step);
246  if (!stepConstant)
247  return std::nullopt;
248 
249  return mlir::ceilDiv(*ubConstant - *lbConstant, *stepConstant);
250 }
