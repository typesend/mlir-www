26 // SparseTensorDescriptor and helpers that manage the sparse tensor memory
27 // layout scheme during "direct code generation" (i.e. when sparsification
28 // generates the buffers as part of actual IR, in constrast with the library
29 // approach where data structures are hidden behind opaque pointers).
30 //
31 // The sparse tensor storage scheme for a rank-dimensional tensor is organized
32 // as a single compound type with the following fields. Note that every memref
33 // with ? size actually behaves as a "vector", i.e. the stored size is the
34 // capacity and the used size resides in the storage_specifier struct.
35 //
36 // struct {
37 // ; per-level l:
38 // ; if dense:
39 // <nothing>
40 // ; if compresed:
41 // memref<? x pos> positions-l ; positions for sparse level l
42 // memref<? x crd> coordinates-l ; coordinates for sparse level l
43 // ; if singleton:
44 // memref<? x crd> coordinates-l ; coordinates for singleton level l
45 //
46 // memref<? x eltType> values ; values
47 //
48 // struct sparse_tensor.storage_specifier {
49 // array<rank x int> lvlSizes ; sizes/cardinalities for each level
50 // array<n x int> memSizes; ; sizes/lengths for each data memref
51 // }
52 // };
53 //
54 // In addition, for a "trailing COO region", defined as a compressed level
55 // followed by one or more singleton levels, the default SOA storage that
56 // is inherent to the TACO format is optimized into an AOS storage where
57 // all coordinates of a stored element appear consecutively. In such cases,
58 // a special operation (sparse_tensor.coordinates_buffer) must be used to
59 // access the AOS coordinates array. In the code below, the method `getCOOStart`
60 // is used to find the start of the "trailing COO region".
61 //
62 // If the sparse tensor is a slice (produced by `tensor.extract_slice`
63 // operation), instead of allocating a new sparse tensor for it, it reuses the
64 // same sets of MemRefs but attaching a additional set of slicing-metadata for
65 // per-dimension slice offset and stride.
66 //
67 // Examples.
68 //
69 // #CSR storage of 2-dim matrix yields
70 // memref<?xindex> ; positions-1
71 // memref<?xindex> ; coordinates-1
72 // memref<?xf64> ; values
73 // struct<(array<2 x i64>, array<3 x i64>)>) ; lvl0, lvl1, 3xsizes
74 //
75 // #COO storage of 2-dim matrix yields
76 // memref<?xindex>, ; positions-0, essentially [0,sz]
77 // memref<?xindex> ; AOS coordinates storage
78 // memref<?xf64> ; values
79 // struct<(array<2 x i64>, array<3 x i64>)>) ; lvl0, lvl1, 3xsizes
80 //
81 // Slice on #COO storage of 2-dim matrix yields
82 // ;; Inherited from the original sparse tensors
83 // memref<?xindex>, ; positions-0, essentially [0,sz]
84 // memref<?xindex> ; AOS coordinates storage
85 // memref<?xf64> ; values
86 // struct<(array<2 x i64>, array<3 x i64>, ; lvl0, lvl1, 3xsizes
87 // ;; Extra slicing-metadata
88 // array<2 x i64>, array<2 x i64>)>) ; dim offset, dim stride.
89 //

181  static void foreachFieldInSparseTensor(
182  const SparseTensorEncodingAttr enc,
183  llvm::function_ref<bool(FieldIndex, SparseTensorFieldKind, Level,
184  DimLevelType)>
185  callback) {
186  return sparse_tensor::foreachFieldInSparseTensor(enc, callback);
187  }
188 
189  std::pair<FieldIndex, unsigned>
190  getFieldIndexAndStride(SparseTensorFieldKind kind,
191  std::optional<Level> lvl) const {
192  FieldIndex fieldIdx = -1u;
193  unsigned stride = 1;
194  if (kind == SparseTensorFieldKind::CrdMemRef) {
195  assert(lvl.has_value());
196  const Level cooStart = getCOOStart(enc);
197  const Level lvlRank = enc.getLvlRank();
198  if (lvl.value() >= cooStart && lvl.value() < lvlRank) {
199  lvl = cooStart;
200  stride = lvlRank - cooStart;
201  }
202  }
203  foreachFieldInSparseTensor(
204  enc,
205  [lvl, kind, &fieldIdx](FieldIndex fIdx, SparseTensorFieldKind fKind,
206  Level fLvl, DimLevelType dlt) -> bool {
207  if ((lvl && fLvl == lvl.value() && kind == fKind) ||
208  (kind == fKind && fKind == SparseTensorFieldKind::ValMemRef)) {
209  fieldIdx = fIdx;
210  // Returns false to break the iteration.
211  return false;
212  }
213  return true;
214  });
215  assert(fieldIdx != -1u);
216  return std::pair<FieldIndex, unsigned>(fieldIdx, stride);
217  }
218 
219 private:
220  SparseTensorEncodingAttr enc;
221 };

244 /// A helper class around an array of values that corresponds to a sparse
245 /// tensor. This class provides a set of meaningful APIs to query and update

247 /// on how a sparse tensor is laid out but instead rely on this class to access
248 /// the right value for the right field.
249 template <typename ValueArrayRef>
250 class SparseTensorDescriptorImpl {
251 protected:
252  SparseTensorDescriptorImpl(SparseTensorType stt, ValueArrayRef fields)
253  : rType(stt), fields(fields) {
254  assert(stt.hasEncoding() &&
255  getNumFieldsFromEncoding(stt.getEncoding()) == getNumFields());
256  // We should make sure the class is trivially copyable (and should be small
257  // enough) such that we can pass it by value.
258  static_assert(std::is_trivially_copyable_v<
259  SparseTensorDescriptorImpl<ValueArrayRef>>);
260  }

264  std::optional<Level> lvl) const {
