68 bool LLVM::LoadOp::canUsesBeRemoved(
69  const MemorySlot &slot, const SmallPtrSetImpl<OpOperand *> &blockingUses,
70  SmallVectorImpl<OpOperand *> &newBlockingUses) {
71  if (blockingUses.size() != 1)
72  return false;
73  Value blockingUse = (*blockingUses.begin())->get();
74  // If the blocking use is the slot ptr itself, there will be enough
75  // context to reconstruct the result of the load at removal time, so it can
76  // be removed (provided it loads the exact stored value and is not
77  // volatile).
78  return blockingUse == slot.ptr && getAddr() == slot.ptr &&
79  getResult().getType() == slot.elemType && !getVolatile_();
80 }

91 bool LLVM::StoreOp::canUsesBeRemoved(
92  const MemorySlot &slot, const SmallPtrSetImpl<OpOperand *> &blockingUses,
93  SmallVectorImpl<OpOperand *> &newBlockingUses) {
94  if (blockingUses.size() != 1)
95  return false;
96  Value blockingUse = (*blockingUses.begin())->get();
97  // If the blocking use is the slot ptr itself, dropping the store is
98  // fine, provided we are currently promoting its target value. Don't allow a
99  // store OF the slot pointer, only INTO the slot pointer.
100  return blockingUse == slot.ptr && getAddr() == slot.ptr &&
101  getValue() != slot.ptr && getValue().getType() == slot.elemType &&
102  !getVolatile_();
103 }

105 DeletionKind LLVM::StoreOp::removeBlockingUses(
106  const MemorySlot &slot, const SmallPtrSetImpl<OpOperand *> &blockingUses,
107  OpBuilder &builder, Value reachingDefinition) {
108  // `canUsesBeRemoved` checked this blocking use must be the stored slot
109  // pointer.
110  for (Operation *user : slot.ptr.getUsers())
111  if (auto declareOp = llvm::dyn_cast<LLVM::DbgDeclareOp>(user))
112  builder.create<LLVM::DbgValueOp>(declareOp->getLoc(), getValue(),
113  declareOp.getVarInfo());
114  return DeletionKind::Delete;
115 }

121 /// Conditions the deletion of the operation to the removal of all its uses.
122 static bool forwardToUsers(Operation *op,
123  SmallVectorImpl<OpOperand *> &newBlockingUses) {
124  for (Value result : op->getResults())
125  for (OpOperand &use : result.getUses())
126  newBlockingUses.push_back(&use);
127  return true;
128 }
