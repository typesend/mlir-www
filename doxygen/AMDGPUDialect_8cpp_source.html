49 template <typename T>
50 static LogicalResult verifyRawBufferOp(T &op) {
51  MemRefType bufferType = op.getMemref().getType().template cast<MemRefType>();
52  Attribute memorySpace = bufferType.getMemorySpace();
53  bool isGlobal = false;
54  if (!memorySpace)
55  isGlobal = true;
56  else if (auto intMemorySpace = memorySpace.dyn_cast<IntegerAttr>())
57  isGlobal = intMemorySpace.getInt() == 0 || intMemorySpace.getInt() == 1;
58  else if (auto gpuMemorySpace = memorySpace.dyn_cast<gpu::AddressSpaceAttr>())
59  isGlobal = gpuMemorySpace.getValue() == gpu::AddressSpace::Global;
60 
61  if (!isGlobal)
62  return op.emitOpError(
63  "Buffer ops must operate on a memref in global memory");
64  if (!bufferType.hasRank())
65  return op.emitOpError(
66  "Cannot meaningfully buffer_store to an unranked memref");
67  if (static_cast<int64_t>(op.getIndices().size()) != bufferType.getRank())
68  return op.emitOpError("Expected " + Twine(bufferType.getRank()) +
69  " indices to memref");
70  return success();
71 }

97 static std::optional<uint32_t> getConstantUint32(Value v) {
98  APInt cst;

100  return std::nullopt;

106 template <typename OpType>
107 static bool staticallyOutOfBounds(OpType op) {
108  if (!op.getBoundsCheck())
109  return false;
110  MemRefType bufferType = op.getMemref().getType();
111  if (!bufferType.hasStaticShape())
112  return false;
113  int64_t offset;
114  SmallVector<int64_t> strides;
115  if (failed(getStridesAndOffset(bufferType, strides, offset)))
116  return false;
117  int64_t result = offset + op.getIndexOffset().value_or(0);
118  if (op.getSgprOffset()) {
119  std::optional<uint32_t> sgprOffset = getConstantUint32(op.getSgprOffset());
120  if (!sgprOffset)
121  return false;
122  result += *sgprOffset;
123  }
124  if (strides.size() != op.getIndices().size())
125  return false;
126  int64_t indexVal = 0;
127  for (auto pair : llvm::zip(strides, op.getIndices())) {
128  int64_t stride = std::get<0>(pair);
129  Value idx = std::get<1>(pair);
130  std::optional<uint32_t> idxVal = getConstantUint32(idx);
131  if (!idxVal)
132  return false;
133  indexVal += stride * *idxVal;
134  }
135  result += indexVal;
136  if (result > std::numeric_limits<uint32_t>::max())
137  // Overflow means don't drop
138  return false;
139  return result >= bufferType.getNumElements();
140 }

143 template <typename OpType>
144 struct RemoveStaticallyOobBufferLoads final : public OpRewritePattern<OpType> {
145  using OpRewritePattern<OpType>::OpRewritePattern;
146 
147  LogicalResult matchAndRewrite(OpType op, PatternRewriter &rw) const override {
148  if (!staticallyOutOfBounds(op))
149  return failure();
150  Type loadType = op.getResult().getType();
151  rw.replaceOpWithNewOp<arith::ConstantOp>(op, loadType,
152  rw.getZeroAttr(loadType));
153  return success();
154  }
155 };

210 LogicalResult MFMAOp::verify() {
211  constexpr uint32_t waveSize = 64;
212  Builder b(getContext());
213 
214  Type sourceType = getSourceA().getType();
215  Type destType = getDestC().getType();
216 
217  Type sourceElem = sourceType, destElem = destType;
218  uint32_t sourceLen = 1, destLen = 1;
219  if (auto sourceVector = sourceType.dyn_cast<VectorType>()) {
220  sourceLen = sourceVector.getNumElements();
221  sourceElem = sourceVector.getElementType();
222  }
223  if (auto destVector = destType.dyn_cast<VectorType>()) {
224  destLen = destVector.getNumElements();
225  destElem = destVector.getElementType();
226  }
227 
228  Type sourceBType = getSourceB().getType();
229  if (sourceElem.isFloat8E5M2FNUZ() || sourceElem.isFloat8E4M3FNUZ()) {
230  int64_t sourceBLen = 1;
231  Type sourceBElem = sourceBType;
232  if (auto sourceBVector = sourceBType.dyn_cast<VectorType>()) {
233  sourceBLen = sourceBVector.getNumElements();
234  sourceBElem = sourceBVector.getElementType();
235  }
236  if (!sourceBElem.isFloat8E5M2FNUZ() && !sourceBElem.isFloat8E4M3FNUZ())
237  return emitOpError("expected both source operands to have f8 elements");
238  if (sourceLen != sourceBLen)
239  return emitOpError(
240  "expected both f8 source vectors to have the same length");
241  } else {
242  if (sourceType != sourceBType)
243  return emitOpError(
244  "expected both non-f8 source operand types to match exactly");
245  }
246  // Normalize the wider integer types the compiler expects to i8
247  if (sourceElem.isInteger(32)) {
248  sourceLen *= 4;
249  sourceElem = b.getI8Type();
250  }
251  if (sourceElem.isInteger(64)) {
252  sourceLen *= 8;
253  sourceElem = b.getI8Type();
254  }
255 
256  int64_t numSourceElems = (getM() * getK() * getBlocks()) / waveSize;
257  if (sourceLen != numSourceElems)
258  return emitOpError("expected " + Twine(numSourceElems) +
259  " source values for this operation but got " +
260  Twine(sourceLen));
261 
262  int64_t numDestElems = (getM() * getN() * getBlocks()) / waveSize;
263  if (destLen != numDestElems)
264  return emitOpError("expected " + Twine(numDestElems) +
265  " result values for this operation but got " +
266  Twine(destLen));
267 
268  if (destElem.isF64() && getBlgp() != MFMAPermB::none)
269  return emitOpError(
270  "double-precision ops do not support permuting lanes of B");
271  if (destElem.isF64() && getCbsz() != 0)
272  return emitOpError(
273  "double-precision ops do not support permuting lanes of A");
274  if (getAbid() >= (1u << getCbsz()))
275  return emitOpError(
276  "block ID for permuting A (abid) must be below 2 ** cbsz");
277 
278  if ((getNegateA() || getNegateB() || getNegateC()) && !destElem.isF64())
279  return emitOpError(
280  "negation flags only available for double-precision operations");
281 
282  return success();
283 }
