78 /// Return the multi-dimensional integral ratio of `subShape` to the trailing
79 /// dimensions of `shape`. This represents how many times `subShape` fits
80 /// within `shape`. If integral division is not possible, return std::nullopt.
81 /// The trailing `subShape.size()` entries of both shapes are assumed (and
82 /// enforced) to only contain non-negative values.
83 ///
84 /// Examples:
85 /// - shapeRatio({3, 5, 8}, {2, 5, 2}) returns {3, 2, 1}.
86 /// - shapeRatio({3, 8}, {2, 5, 2}) returns std::nullopt (subshape has
87 /// higher
88 /// rank).
89 /// - shapeRatio({42, 2, 10, 32}, {2, 5, 2}) returns {42, 1, 2, 16} which is
90 /// derived as {42(leading shape dim), 2/2, 10/5, 32/2}.

181 /// Apply the permutation defined by `permutation` to `inVec`.
182 /// Element `i` in `inVec` is mapped to location `j = permutation[i]`.
183 /// E.g.: for an input vector `inVec = ['a', 'b', 'c']` and a permutation
184 /// vector `permutation = [2, 0, 1]`, this function leaves `inVec = ['c', 'a',
185 /// 'b']`.
186 template <typename T, unsigned N>
187 void applyPermutationToVector(SmallVector<T, N> &inVec,
188  ArrayRef<int64_t> permutation) {
189  SmallVector<T, N> auxVec(inVec.size());
190  for (const auto &en : enumerate(permutation))
191  auxVec[en.index()] = inVec[en.value()];
192  inVec = auxVec;
193 }
