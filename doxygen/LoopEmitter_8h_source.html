23 /// The position of a loop in the loop-stack, or the position of a
24 /// `LoopId` in a topologically-sorted list of `LoopId`s.
25 ///
26 /// Although this type may have the same cardinality as `LoopId`, it must
27 /// not be confused with that type. The `LoopId` type is used by the `Merger`
28 /// as a unique identifier for loop-variables, regardless of the ordering
29 /// of those loops. Whereas the `LoopOrd` type is used by the `LoopEmitter`
30 /// (and `CodegenEnv`) to refer to the actual order in which loops are
31 /// generated.
32 ///
33 /// TODO: further explicate the correspondences between these various
34 /// types. In particular, since the `$dim` argument to `linalg::IndexOp`
35 /// is a De Bruijn index, it seems like that should correspond to `LoopOrd`,
36 /// and yet the `Merger` has that correspond with `LoopId` instead.
37 /// In addition `LoopEmitter::genAffine` has `AffineDimExpr::position`
38 /// correspond to `LoopId`, however it is unclear what the providence
39 /// of those `AffineDimExpr` is.
40 //
41 // TODO: use a struct/class rather than a typedef, so that we can actually
42 // typecheck this to avoid mixups in the code.
43 using LoopOrd = unsigned;

48 // SparseTensorLoopEmiter class, manages sparse tensors and helps to
49 // generate loop structure to (co)-iterate sparse tensors.
50 //
51 // An example usage:
52 // To generate the following loops over T1<?x?> and T2<?x?>
53 //
54 // for i in TENSOR_1_0 {
55 // for j : TENSOR_2_0 {
56 // for k : TENSOR_1_1 {}
57 // for k : TENSOR_2_1 {}
58 // }
59 // }
60 //
61 // One can use
62 //
63 // LoopEmiter loopEmiter({T1, T1});
64 // loopEmiter.initializeLoopEmit();
65 // loopEmiter.enterLoopOverTensorAtLvl(T1, 0);
66 // loopEmiter.enterLoopOverTensorAtLvl(T2, 0);
67 // loopEmiter.enterLoopOverTensorAtLvl(T1, 1);
68 // loopEmiter.exitCurrentLoop();
69 // loopEmiter.enterLoopOverTensorAtLvl(T2, 1);
70 // loopEmiter.exitCurrentLoop(); // exit k
71 // loopEmiter.exitCurrentLoop(); // exit j
72 // loopEmiter.exitCurrentLoop(); // exit i

81  // Map from [tid, dim] to a list of dependent [tid, dim] for affine expression

83  // E.g., for affine index (d0 + d1), it depends on two [tid, dim] that defines

174  /// Fills the out-parameter with the loop induction variables for all
175  /// loops in the current loop-stack. The variables are given in the
176  /// same order as the loop-stack, hence `ivs` should be indexed into
177  /// by `LoopOrd` (not `LoopId`).
178  void getLoopIVs(SmallVectorImpl<Value> &ivs) const {
179  ivs.clear();
180  ivs.reserve(getCurrentDepth());
181  for (auto &l : loopStack)
182  ivs.push_back(l.iv);

208  /// Converts a range of TensorLevel to a range of std::pair<TensorId, Level>
209  template <class ContainerTy>

212  static_assert(std::is_same_v<llvm::remove_cvref_t<EltTy>, TensorLevel>,

400  /// Get the collapse reassociation for `tensors[tid][dstLvl]`.
401  /// For unreshaped operands, the reassociation is simply an identity
402  /// transformation.
403  ///
404  /// NOTE: the result uses `Level` rather than the `int64_t` of
405  /// `ReassociationIndices`, since the former gives clarity to what
406  /// the values actually mean.
407  ///
408  /// TODO: why not do this computation when we first store the reassoc,
409  /// instead of doing it every time we look it up?
410  SmallVector<Level, 2> getCollapseReassociation(TensorId tid, Level dstLvl) {
411  assert(tid < getNumTensors() && "Invalid TensorId");
412  assert(collapseReassoc.size() == getNumTensors());
413  if (const auto reassoc = collapseReassoc[tid]) {
414  // TODO: store the dstLvlRank in the LoopEmitter so that we can
415  // check `dstLvl < dstLvlRank` at the top; and only here need to
416  // assert that `reassoc.size() == dstLvlRank`.
417  assert(dstLvl < reassoc.size() && "Level is out-of-bounds");
418  const auto srcLvls = reassoc[dstLvl].cast<ArrayAttr>();
419  return llvm::to_vector<2>(
420  llvm::map_range(srcLvls, [&](Attribute srcLvl) -> Level {
421  // TODO: replace this with the converter for `LevelAttr`.
422  return srcLvl.cast<IntegerAttr>().getValue().getZExtValue();
423  }));
424  }
425  return {dstLvl};
426  }
