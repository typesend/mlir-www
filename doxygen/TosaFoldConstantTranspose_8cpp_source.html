24 template <typename BaseType>
25 DenseElementsAttr transposeType(ElementsAttr attr, ShapedType inputType,
26  ShapedType outputType,
27  llvm::ArrayRef<int64_t> permValues) {
28  if (inputType.getNumElements() == 0)
29  return DenseElementsAttr::get(outputType, llvm::ArrayRef<BaseType>{});
30 
31  auto attrValues = attr.getValues<BaseType>();
32  auto inputShape = inputType.getShape();
33 
34  // The inverted permutation map and strides of the output are used to compute
35  // the contribution of a given dimension to the destination linear index in
36  // an order-independent way.
37  auto outputStrides = computeStrides(outputType.getShape());
38  auto invertedPermValues = invertPermutationVector(permValues);
39 
40  auto initialValue = *std::begin(attrValues);
41  SmallVector<BaseType> outputValues(inputType.getNumElements(), initialValue);
42 
43  for (const auto &it : llvm::enumerate(attrValues)) {
44  auto srcLinearIndex = it.index();
45 
46  uint64_t dstLinearIndex = 0;
47  for (int64_t dim = inputShape.size() - 1; dim >= 0; --dim) {
48  // Compute the index into the current dimension of the source vector.
49  auto sourceIndexForDim = srcLinearIndex % inputShape[dim];
50  srcLinearIndex /= inputShape[dim];
51 
52  // Add the contribution of the current dimension to the output using the
53  // permutation map.
54  dstLinearIndex +=
55  outputStrides[invertedPermValues[dim]] * sourceIndexForDim;
56  }
57 
58  outputValues[dstLinearIndex] = it.value();
59  }
60 
61  return DenseElementsAttr::get(outputType,
62  llvm::ArrayRef<BaseType>(outputValues));
63 }

65 // A type specialized transposition of an ElementsAttr.
66 // This implementation tries to operate on the underlying data in its raw
67 // representation when possible to avoid allocating a large number of Attribute
68 // objects.
69 DenseElementsAttr transpose(ElementsAttr attr, ShapedType inputType,
70  ShapedType outputType,
71  llvm::ArrayRef<int64_t> permValues) {
72  auto baseType = inputType.getElementType();
73 
74  // Handle possible integer types
75  if (auto intType = baseType.dyn_cast<IntegerType>()) {
76  switch (intType.getWidth()) {
77  case 1:
78  return transposeType<bool>(attr, inputType, outputType, permValues);
79  case 8:
80  return transposeType<int8_t>(attr, inputType, outputType, permValues);
81  case 16:
82  return transposeType<int16_t>(attr, inputType, outputType, permValues);
83  case 32:
84  return transposeType<int32_t>(attr, inputType, outputType, permValues);
85  case 64:
86  return transposeType<int64_t>(attr, inputType, outputType, permValues);
87  default:
88  return transposeType<APInt>(attr, inputType, outputType, permValues);
89  }
90  }
91 
92  // Handle possible float types
93  if (baseType.isF32()) {
94  return transposeType<float>(attr, inputType, outputType, permValues);
95  }
96 
97  return transposeType<APFloat>(attr, inputType, outputType, permValues);
98 }

100 struct TosaFoldConstantTranspose : public OpRewritePattern<tosa::TransposeOp> {
101  using OpRewritePattern::OpRewritePattern;
102 
103  LogicalResult matchAndRewrite(tosa::TransposeOp op,
104  PatternRewriter &rewriter) const override {
105  auto outputType = op.getType().cast<ShapedType>();
106  // TOSA supports quantized types.
107  if (!outputType.getElementType().isIntOrIndexOrFloat())
108  return failure();
109 
110  ElementsAttr inputValues;
111  if (!matchPattern(op.getInput1(), m_Constant(&inputValues)))
112  return failure();
113  // Make sure the input is a constant that has a single user.
114  if (!llvm::hasSingleElement(op.getInput1().getDefiningOp()->getUsers()))
115  return failure();
116 
117  DenseIntElementsAttr permAttr;
118  if (!matchPattern(op.getPerms(), m_Constant(&permAttr)))
119  return failure();
120  auto permValues = llvm::to_vector<6>(llvm::map_range(
121  // TOSA allows both 32- and 64-bit integer tensors here.
122  permAttr.getValues<APInt>(),
123  [](const APInt &val) { return val.getSExtValue(); }));
124 
125  auto inputType = op.getInput1().getType().cast<ShapedType>();
126 
127  auto resultAttr = transpose(inputValues, inputType, outputType, permValues);
128  rewriter.replaceOpWithNewOp<tosa::ConstOp>(op, outputType, resultAttr);
129  return success();
130  }
131 };
