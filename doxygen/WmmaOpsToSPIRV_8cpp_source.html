27 /// Creates a SPIR-V op to replace the given GPU subgroup mma elementwise op
28 /// when the elementwise op directly supports with cooperative matrix type.
29 /// Returns false if cannot.
30 ///
31 /// See SPV_NV_cooperative_matrix for supported elementwise ops.
32 static bool createElementwiseOp(ConversionPatternRewriter &builder,
33  gpu::SubgroupMmaElementwiseOp op,
34  spirv::CooperativeMatrixNVType coopType,
35  ValueRange operands) {
36  switch (op.getOpType()) {
37  case gpu::MMAElementwiseOp::ADDF:
38  builder.replaceOpWithNewOp<spirv::FAddOp>(op, coopType, operands);
39  return true;
40  case gpu::MMAElementwiseOp::ADDI:
41  builder.replaceOpWithNewOp<spirv::IAddOp>(op, coopType, operands);
42  return true;
43  case gpu::MMAElementwiseOp::SUBF:
44  builder.replaceOpWithNewOp<spirv::FSubOp>(op, coopType, operands);
45  return true;
46  case gpu::MMAElementwiseOp::SUBI:
47  builder.replaceOpWithNewOp<spirv::ISubOp>(op, coopType, operands);
48  return true;
49  case gpu::MMAElementwiseOp::DIVF:
50  builder.replaceOpWithNewOp<spirv::FDivOp>(op, coopType, operands);
51  return true;
52  case gpu::MMAElementwiseOp::DIVS:
53  builder.replaceOpWithNewOp<spirv::SDivOp>(op, coopType, operands);
54  return true;
55  case gpu::MMAElementwiseOp::DIVU:
56  builder.replaceOpWithNewOp<spirv::UDivOp>(op, coopType, operands);
57  return true;
58  case gpu::MMAElementwiseOp::NEGATEF:
59  builder.replaceOpWithNewOp<spirv::FNegateOp>(op, coopType, operands);
60  return true;
61  case gpu::MMAElementwiseOp::NEGATES:
62  builder.replaceOpWithNewOp<spirv::SNegateOp>(op, coopType, operands);
63  return true;
64  default:
65  break;
66  }
67  return false;
68 }

72 /// This class implements the conversion of GPU MMA loadOp to
73 /// CooperativeMatrixLoad op in the SPIRV dialect.
74 struct WmmaLoadOpToSPIRVLowering
75  : public OpConversionPattern<gpu::SubgroupMmaLoadMatrixOp> {
76  using OpConversionPattern::OpConversionPattern;
77 
78  LogicalResult
79  matchAndRewrite(gpu::SubgroupMmaLoadMatrixOp subgroupMmaLoadMatrixOp,
80  OpAdaptor adaptor,
81  ConversionPatternRewriter &rewriter) const override {
82  Location loc = subgroupMmaLoadMatrixOp->getLoc();
83  gpu::MMAMatrixType retType =
84  subgroupMmaLoadMatrixOp.getRes().getType().cast<gpu::MMAMatrixType>();
85  auto memrefType =
86  subgroupMmaLoadMatrixOp.getSrcMemref().getType().cast<MemRefType>();
87  Value bufferPtr = spirv::getElementPtr(
88  *getTypeConverter<SPIRVTypeConverter>(), memrefType,
89  adaptor.getSrcMemref(), adaptor.getIndices(), loc, rewriter);
90  auto coopType = convertMMAToSPIRVType(retType);
91  int64_t stride = subgroupMmaLoadMatrixOp.getLeadDimension().getSExtValue();
92  auto i32Type = rewriter.getI32Type();
93  auto strideValue = rewriter.create<spirv::ConstantOp>(
94  loc, i32Type, IntegerAttr::get(i32Type, stride));
95  bool isColMajor = static_cast<bool>(subgroupMmaLoadMatrixOp.getTranspose());
96  auto columnMajor = rewriter.create<spirv::ConstantOp>(
97  loc, rewriter.getI1Type(), rewriter.getBoolAttr(isColMajor));
98  rewriter.replaceOpWithNewOp<spirv::NVCooperativeMatrixLoadOp>(
99  subgroupMmaLoadMatrixOp, coopType, bufferPtr, strideValue, columnMajor,
100  spirv::MemoryAccessAttr());
101  return success();
102  }
103 };

153 /// Convert GPU MMA ConstantMatrixOp to constant SPIR-V cooperative matrix ops.
154 struct WmmaConstantOpToSPIRVLowering
155  : public OpConversionPattern<gpu::SubgroupMmaConstantMatrixOp> {
156  using OpConversionPattern::OpConversionPattern;
157 
158  LogicalResult
159  matchAndRewrite(gpu::SubgroupMmaConstantMatrixOp subgroupMmaConstantMatrixOp,
160  OpAdaptor adaptor,
161  ConversionPatternRewriter &rewriter) const override {
162  Value cst = adaptor.getOperands()[0];
163  auto coopType = convertMMAToSPIRVType(

171 /// Converts elementwise ops to SPIR-V cooperative matrix elementwise ops for
172 /// the default case.
173 struct WmmaElementwiseOpToSPIRVDefaultLowering
174  : public OpConversionPattern<gpu::SubgroupMmaElementwiseOp> {
175  using OpConversionPattern::OpConversionPattern;
176 
177  LogicalResult
178  matchAndRewrite(gpu::SubgroupMmaElementwiseOp elementwiseOp,
179  OpAdaptor adaptor,
180  ConversionPatternRewriter &rewriter) const override {
181  // All operands should be of cooperative matrix types.
182  for (Value operand : adaptor.getOperands()) {
183  if (!operand.getType().isa<spirv::CooperativeMatrixNVType>())
184  return failure();
185  }
186  auto coopType = convertMMAToSPIRVType(
187  elementwiseOp.getType().cast<gpu::MMAMatrixType>());
188  return success(createElementwiseOp(rewriter, elementwiseOp, coopType,
189  adaptor.getOperands()));
190  }
191 };

193 /// Converts elementwise ops to SPIR-V cooperative matrix elementwise ops for
194 /// matrix times scalar case.
195 struct WmmaElementwiseOpToSPIRVScalarMulLowering
196  : public OpConversionPattern<gpu::SubgroupMmaElementwiseOp> {
197  using OpConversionPattern::OpConversionPattern;
198 
199  LogicalResult
200  matchAndRewrite(gpu::SubgroupMmaElementwiseOp elementwiseOp,
201  OpAdaptor adaptor,
202  ConversionPatternRewriter &rewriter) const override {
203  if (adaptor.getOperands().size() != 2)
204  return failure();
205  // All operands should be of cooperative matrix types.
206  for (Value operand : adaptor.getOperands()) {
207  if (!operand.getType().isa<spirv::CooperativeMatrixNVType>())
208  return failure();
209  }
210 
211  if (elementwiseOp.getOpType() != gpu::MMAElementwiseOp::MULF)
212  return failure();
213 
214  // Use the original operands to check whether one of the operands is a splat
215  // scalar value.
216  Value lhs = elementwiseOp.getOperands().front();
217  Value rhs = elementwiseOp.getOperands().back();
218  Value splat = nullptr;
219  Value matrix = nullptr;
220  if (lhs.getDefiningOp<gpu::SubgroupMmaConstantMatrixOp>()) {
221  splat = adaptor.getOperands().front();
222  matrix = adaptor.getOperands().back();
223  } else if (rhs.getDefiningOp<gpu::SubgroupMmaConstantMatrixOp>()) {
224  matrix = adaptor.getOperands().front();
225  splat = adaptor.getOperands().back();
226  }
227  if (!splat || !matrix)
228  return failure();
229 
230  // Constant MMA matrix ops are converted to spirv.CompositeConstruct ops.
231  Value scalar = nullptr;
232  auto cc = splat.getDefiningOp<spirv::CompositeConstructOp>();
233  if (!cc)
234  return failure();
235  assert(cc.getConstituents().size() == 1);
236  scalar = cc.getConstituents().front();
237 
238  auto coopType = convertMMAToSPIRVType(
239  elementwiseOp.getType().cast<gpu::MMAMatrixType>());
240  rewriter.replaceOpWithNewOp<spirv::MatrixTimesScalarOp>(
241  elementwiseOp, coopType, ValueRange{matrix, scalar});
242  return success();
243  }
244 };

248 /// Return the LLVMStructureType corresponding to the MMAMatrixType `type`.
249 mlir::spirv::CooperativeMatrixNVType
250 mlir::convertMMAToSPIRVType(gpu::MMAMatrixType type) {
251  ArrayRef<int64_t> retTypeShape = type.getShape();
252  Type elementType = type.getElementType();
253  return spirv::CooperativeMatrixNVType::get(
254  elementType, spirv::Scope::Subgroup, retTypeShape[0], retTypeShape[1]);

257 void mlir::populateGpuWMMAToSPIRVConversionPatterns(
258  SPIRVTypeConverter &converter, RewritePatternSet &patterns) {
259  MLIRContext *context = patterns.getContext();
260  patterns.add<WmmaLoadOpToSPIRVLowering, WmmaMmaOpToSPIRVLowering,
261  WmmaStoreOpToSPIRVLowering, WmmaConstantOpToSPIRVLowering,
262  WmmaElementwiseOpToSPIRVDefaultLowering>(converter, context);
263  // Give the following patterns higher benefit to prevail over the default one.
264  patterns.add<WmmaElementwiseOpToSPIRVScalarMulLowering>(converter, context,
265  /*benefit=*/2);
266 }
