25 /// The matcher that matches a certain kind of Attribute and binds the value
26 /// inside the Attribute.
27 template <
28  typename AttrClass,
29  // Require AttrClass to be a derived class from Attribute and get its
30  // value type
31  typename ValueType = typename std::enable_if_t<
32  std::is_base_of<Attribute, AttrClass>::value, AttrClass>::ValueType,
33  // Require the ValueType is not void
34  typename = std::enable_if_t<!std::is_void<ValueType>::value>>
35 struct attr_value_binder {
36  ValueType *bind_value;
37 
38  /// Creates a matcher instance that binds the value to bv if match succeeds.
39  attr_value_binder(ValueType *bv) : bind_value(bv) {}
40 
41  bool match(const Attribute &attr) {
42  if (auto intAttr = attr.dyn_cast<AttrClass>()) {
43  *bind_value = intAttr.getValue();
44  return true;
45  }
46  return false;
47  }
48 };

71 /// The matcher that matches operations that have the `ConstantLike` trait, and
72 /// binds the folded attribute value.
73 template <typename AttrT>
74 struct constant_op_binder {
75  AttrT *bind_value;
76 
77  /// Creates a matcher instance that binds the constant attribute value to
78  /// bind_value if match succeeds.
79  constant_op_binder(AttrT *bind_value) : bind_value(bind_value) {}
80  /// Creates a matcher instance that doesn't bind if match succeeds.
81  constant_op_binder() : bind_value(nullptr) {}
82 
83  bool match(Operation *op) {
84  if (!op->hasTrait<OpTrait::ConstantLike>())
85  return false;
86 
87  // Fold the constant to an attribute.
88  SmallVector<OpFoldResult, 1> foldedOp;
89  LogicalResult result = op->fold(/*operands=*/std::nullopt, foldedOp);
90  (void)result;
91  assert(succeeded(result) && "expected ConstantLike op to be foldable");
92 
93  if (auto attr = foldedOp.front().get<Attribute>().dyn_cast<AttrT>()) {
94  if (bind_value)
95  *bind_value = attr;
96  return true;
97  }
98  return false;
99  }
100 };

105 struct AttrOpBinder {
106  /// Creates a matcher instance that binds the attribute value to

111  AttrOpBinder(StringRef attrName) : attrName(attrName), bindValue(nullptr) {}

125 /// The matcher that matches a constant scalar / vector splat / tensor splat
126 /// float operation and binds the constant float value.
127 struct constant_float_op_binder {
128  FloatAttr::ValueType *bind_value;
129 
130  /// Creates a matcher instance that binds the value to bv if match succeeds.
131  constant_float_op_binder(FloatAttr::ValueType *bv) : bind_value(bv) {}
132 
133  bool match(Operation *op) {
134  Attribute attr;
135  if (!constant_op_binder<Attribute>(&attr).match(op))
136  return false;
137  auto type = op->getResult(0).getType();
138 
139  if (type.isa<FloatType>())
140  return attr_value_binder<FloatAttr>(bind_value).match(attr);
141  if (type.isa<VectorType, RankedTensorType>()) {
142  if (auto splatAttr = attr.dyn_cast<SplatElementsAttr>()) {
143  return attr_value_binder<FloatAttr>(bind_value)
144  .match(splatAttr.getSplatValue<Attribute>());
145  }
146  }
147  return false;
148  }
149 };

267 /// RecursivePatternMatcher that composes.
268 template <typename OpType, typename... OperandMatchers>
269 struct RecursivePatternMatcher {
270  RecursivePatternMatcher(OperandMatchers... matchers)
271  : operandMatchers(matchers...) {}
272  bool match(Operation *op) {
273  if (!isa<OpType>(op) || op->getNumOperands() != sizeof...(OperandMatchers))
274  return false;
275  bool res = true;
276  enumerate(operandMatchers, [&](size_t index, auto &matcher) {
277  res &= matchOperandOrValueAtIndex(op, index, matcher);
278  });
279  return res;
280  }
281  std::tuple<OperandMatchers...> operandMatchers;
282 };
