47  tileSizeComputationFunction = [=](OpBuilder &, Operation *) { return ts; };

106 /// Reconstruct the fused producer from within the tiled-and-fused code. Based
107 /// on the slice of the producer computed in place it is possible that within
108 /// the loop nest same slice of the producer is computed multiple times. It is
109 /// in general not possible to recompute the value of the fused producer from
110 /// the tiled loop code in such cases. For the cases where no slice of the
111 /// producer is computed in a redundant fashion it is possible to reconstruct
112 /// the value of the original producer from within the tiled loop. It is upto
113 /// the caller to ensure that the producer is not computed redundantly within
114 /// the tiled loop nest. For example, consider
115 ///
116 /// ```mlir
117 /// %0 = linalg.matmul ins(...) outs(...) -> tensor<?x?xf32>
118 /// %1 = linalg.matmul ins(%0, ..) outs(...) -> tensor<?x?x?f32>
119 /// ```
120 ///
121 /// If `%1` is tiled in a 2D fashion and `%0` is fused with it, the resulting IR
122 /// is,
123 ///
124 /// ```mlir
125 /// %t1_0 = scf.for .... iter_args(%arg0 = ...) {
126 /// %t1_1 = scf.for ... iter_args(%arg1 = %arg0) {
127 /// ...
128 /// %t1_2 = linalg.matmul ins(...) outs(...) -> tensor<?x?xf32>
129 /// %t1_3 = linalg.matmul ins(%t1_2, ...)
130 /// %t1_4 = tensor.insert_slice %t1_3 into %arg1 ...
131 /// scf.yield %t1_4
132 /// }
133 /// scf.yield %t1_1
134 /// }
135 /// ```
136 ///
137 /// Here `%t1_2` is the same for all iterations of the inner `scf.for`. Instead
138 /// if `%1` were tiled only along the rows, the resultant code would be
139 ///
140 /// ```mlir
141 /// %t2_0 = scf.for .... iter_args(%arg0 = ...) {
142 /// ...
143 /// %t2_1 = linalg.matmul ins(...) outs(...) -> tensor<?x?xf32>
144 /// %t2_2 = linalg.matmul ins(%t2_1, ...)
145 /// %t2_3 = tensor.insert_slice %t2_2 into %arg0 ...
146 /// scf.yield %t2_3
147 /// }
148 /// ```
149 ///
150 /// Here there is no intersection in the different slices of `%t2_1` computed
151 /// across iterations of the `scf.for`. In such cases, the value of the original
152 /// `%0` can be reconstructed from within the loop body. This is useful in cases
153 /// where `%0` had other uses as well. If not reconstructed from within the loop
154 /// body, uses of `%0` could not be replaced, making it still live and the
155 /// fusion immaterial.
156 void yieldReplacementForFusedProducer(
157  RewriterBase &rewriter, tensor::ExtractSliceOp sliceOp,
158  scf::SCFFuseProducerOfSliceResult fusedProducerInfo,
159  MutableArrayRef<scf::ForOp> loops);

175 /// Method to tile and fuse a sequence of operations, by tiling the consumer
176 /// and fusing its producers. Note that this assumes that it is valid to
177 /// tile+fuse the producer into the innermost tiled loop. Its up to the caller
178 /// to ensure that the tile sizes provided make this fusion valid.
179 ///
180 /// For example, for the following sequence
181 ///
182 /// ```mlir
183 /// %0 =
184 /// %1 = linalg.fill ... outs(%0 : ... )
185 /// %2 = linalg.matmul ... outs(%1 : ...) ...
186 /// ```
187 ///
188 /// it is legal to fuse the fill with the matmul only if the matmul is tiled
189 /// along the parallel dimensions and not the reduction dimension, i.e. the tile
190 /// size for the reduction dimension should be 0. The resulting fused
191 /// transformation is
192 ///
193 /// ```mlir
194 /// %1 = scf.for ... iter_args(%arg0 = %0)
195 /// %2 = tensor.extract_slice %arg0
196 /// %3 = linalg.fill .. outs(%2 : ... )
197 /// %4 = linalg.matmul .. outs(%3 : ...)
198 /// }
199 /// ```

222 /// Method to tile a reduction and generate a parallel op within a serial loop.
223 /// Each of the partial reductions are calculated in parallel. Then after the
224 /// loop all the partial reduction are merged into a final reduction.
225 /// For example for the following sequence
226 ///
227 /// ```mlir
228 /// %0 = linalg.generic %in ["parallel", "reduction"]
229 /// : tensor<7x9xf32> -> tensor<7xf32>
230 /// ```
231 ///
232 /// into:
233 ///
234 /// ```mlir
235 /// %0 = linalg.fill ... : tensor<7x4xf32>
236 /// %1 = scf.for ... iter_args(%arg0 = %0)
237 /// %2 = tensor.extract_slice %arg0 : tensor<7x4xf32> -> tensor<7x?xf32>
238 /// %3 = tensor.extract_slice %in : tensor<7x9xf32> -> tensor<7x?xf32>
239 /// %4 = linalg.generic %2, %3 ["parallel", "parallel"]
240 /// : tensor<7x?xf32> -> tensor<7x?xf32>
241 /// %5 = tensor.insert_slice %3, %0[0, 0] : tensor<7x4xf32>
242 /// }
243 /// %6 = linalg.generic %1 ["parallel", "reduction"]
244 /// : tensor<7x4xf32> -> tensor<7xf32>
245 /// ```
