67  // XXX: I think that the original types and the converted types are the same
68  // iff there was no non-identity type conversion. If that is true, the
69  // patterns could actually test whether there is anything useful to do

112 /// Builds an `UnrealizedConversionCastOp` from the given inputs to the given
113 /// result types. Returns the result values of the cast.
114 static ValueRange buildUnrealizedCast(OpBuilder &builder, TypeRange resultTypes,
115  ValueRange inputs, CastKind kind) {
116  // Create cast.
117  Location loc = builder.getUnknownLoc();
118  if (!inputs.empty())
119  loc = inputs.front().getLoc();
120  auto castOp =
121  builder.create<UnrealizedConversionCastOp>(loc, resultTypes, inputs);
122 
123  // Store cast kind as attribute.
124  auto kindAttr = StringAttr::get(builder.getContext(), getCastKindName(kind));
125  castOp->setAttr(castKindAttrName, kindAttr);
126 
127  return castOp->getResults();
128 }

130 /// Builds one `UnrealizedConversionCastOp` for each of the given original
131 /// values using the respective target types given in the provided conversion
132 /// mapping and returns the results of these casts. If the conversion mapping of
133 /// a value maps a type to itself (i.e., is an identity conversion), then no
134 /// cast is inserted and the original value is returned instead.
135 /// Note that these unrealized casts are different from target materializations
136 /// in that they are *always* inserted, even if they immediately fold away, such
137 /// that patterns always see valid intermediate IR, whereas materializations are
138 /// only used in the places where the unrealized casts *don't* fold away.
139 static SmallVector<Value>
140 buildUnrealizedForwardCasts(ValueRange originalValues,
141  OneToNTypeMapping &conversion,
142  RewriterBase &rewriter, CastKind kind) {
143 
144  // Convert each operand one by one.
145  SmallVector<Value> convertedValues;
146  convertedValues.reserve(conversion.getConvertedTypes().size());
147  for (auto [idx, originalValue] : llvm::enumerate(originalValues)) {
148  TypeRange convertedTypes = conversion.getConvertedTypes(idx);
149 
150  // Identity conversion: keep operand as is.
151  if (isIdentityConversion(originalValue.getType(), convertedTypes)) {
152  convertedValues.push_back(originalValue);
153  continue;
154  }
155 
156  // Non-identity conversion: materialize target types.
157  ValueRange castResult =
158  buildUnrealizedCast(rewriter, convertedTypes, originalValue, kind);
159  convertedValues.append(castResult.begin(), castResult.end());
160  }
161 
162  return convertedValues;
163 }

165 /// Builds one `UnrealizedConversionCastOp` for each sequence of the given
166 /// original values to one value of the type they originated from, i.e., a
167 /// "reverse" conversion from N converted values back to one value of the
168 /// original type, using the given (forward) type conversion. If a given value
169 /// was mapped to a value of the same type (i.e., the conversion in the mapping
170 /// is an identity conversion), then the "converted" value is returned without
171 /// cast.
172 /// Note that these unrealized casts are different from source materializations
173 /// in that they are *always* inserted, even if they immediately fold away, such
174 /// that patterns always see valid intermediate IR, whereas materializations are
175 /// only used in the places where the unrealized casts *don't* fold away.
176 static SmallVector<Value>
177 buildUnrealizedBackwardsCasts(ValueRange convertedValues,
178  const OneToNTypeMapping &typeConversion,
179  RewriterBase &rewriter) {
180  assert(typeConversion.getConvertedTypes() == convertedValues.getTypes());
181 
182  // Create unrealized cast op for each converted result of the op.
183  SmallVector<Value> recastValues;
184  TypeRange originalTypes = typeConversion.getOriginalTypes();
185  recastValues.reserve(originalTypes.size());
186  auto convertedValueIt = convertedValues.begin();
187  for (auto [idx, originalType] : llvm::enumerate(originalTypes)) {
188  TypeRange convertedTypes = typeConversion.getConvertedTypes(idx);
189  size_t numConvertedValues = convertedTypes.size();
190  if (isIdentityConversion(originalType, convertedTypes)) {
191  // Identity conversion: take result as is.
192  recastValues.push_back(*convertedValueIt);
193  } else {
194  // Non-identity conversion: cast back to source type.
195  ValueRange recastValue = buildUnrealizedCast(
196  rewriter, originalType,
197  ValueRange{convertedValueIt, convertedValueIt + numConvertedValues},
198  CastKind::Source);
199  assert(recastValue.size() == 1);
200  recastValues.push_back(recastValue.front());
201  }
202  convertedValueIt += numConvertedValues;
203  }
204 
205  return recastValues;
206 }

208 void OneToNPatternRewriter::replaceOp(Operation *op, ValueRange newValues,
209  const OneToNTypeMapping &resultMapping) {
210  // Create a cast back to the original types and replace the results of the
211  // original op with those.
212  assert(newValues.size() == resultMapping.getConvertedTypes().size());
213  assert(op->getResultTypes() == resultMapping.getOriginalTypes());
214  SmallVector<Value> castResults =
215  buildUnrealizedBackwardsCasts(newValues, resultMapping, *this);
216  replaceOp(op, castResults);
217 }

219 Block *OneToNPatternRewriter::applySignatureConversion(
220  Block *block, OneToNTypeMapping &argumentConversion) {
221  // Split the block at the beginning to get a new block to use for the
222  // updated signature.
223  SmallVector<Location> locs;
224  argumentConversion.convertLocations(block->getArguments(), locs);
225  Block *newBlock =
226  createBlock(block, argumentConversion.getConvertedTypes(), locs);
227  replaceAllUsesWith(block, newBlock);
228 
229  // Create necessary casts in new block.
230  SmallVector<Value> castResults;
231  for (auto [i, arg] : llvm::enumerate(block->getArguments())) {
232  TypeRange convertedTypes = argumentConversion.getConvertedTypes(i);
233  ValueRange newArgs =
234  argumentConversion.getConvertedValues(newBlock->getArguments(), i);
235  if (isIdentityConversion(arg.getType(), convertedTypes)) {
236  // Identity conversion: take argument as is.
237  assert(newArgs.size() == 1);
238  castResults.push_back(newArgs.front());
239  } else {
240  // Non-identity conversion: cast the converted arguments to the original
241  // type.
242  PatternRewriter::InsertionGuard g(*this);
243  setInsertionPointToStart(newBlock);
244  ValueRange castResult = buildUnrealizedCast(*this, arg.getType(), newArgs,
245  CastKind::Argument);
246  assert(castResult.size() == 1);
247  castResults.push_back(castResult.front());
248  }
249  }
250 
251  // Merge old block into new block such that we only have the latter with the
252  // new signature.
253  mergeBlocks(block, newBlock, castResults);
254 
255  return newBlock;
256 }

258 LogicalResult
259 OneToNConversionPattern::matchAndRewrite(Operation *op,
260  PatternRewriter &rewriter) const {
261  auto *typeConverter = getTypeConverter<OneToNTypeConverter>();
262 
263  // Construct conversion mapping for results.
264  Operation::result_type_range originalResultTypes = op->getResultTypes();
265  OneToNTypeMapping resultMapping(originalResultTypes);
266  if (failed(typeConverter->computeTypeMapping(originalResultTypes,
267  resultMapping)))
268  return failure();
269 
270  // Construct conversion mapping for operands.
271  Operation::operand_type_range originalOperandTypes = op->getOperandTypes();
272  OneToNTypeMapping operandMapping(originalOperandTypes);
273  if (failed(typeConverter->computeTypeMapping(originalOperandTypes,
274  operandMapping)))
275  return failure();
276 
277  // Cast operands to target types.
278  SmallVector<Value> convertedOperands = buildUnrealizedForwardCasts(
279  op->getOperands(), operandMapping, rewriter, CastKind::Target);
280 
281  // Create a `OneToNPatternRewriter` for the pattern, which provides additional
282  // functionality.
283  // TODO(ingomueller): I guess it would be better to use only one rewriter
284  // throughout the whole pass, but that would require to
285  // drive the pattern application ourselves, which is a lot
286  // of additional boilerplate code. This seems to work fine,
287  // so I leave it like this for the time being.
288  OneToNPatternRewriter oneToNPatternRewriter(rewriter.getContext());
289  oneToNPatternRewriter.restoreInsertionPoint(rewriter.saveInsertionPoint());
290  oneToNPatternRewriter.setListener(rewriter.getListener());
291 
292  // Apply actual pattern.
293  if (failed(matchAndRewrite(op, oneToNPatternRewriter, operandMapping,
294  resultMapping, convertedOperands)))
295  return failure();
296 
297  return success();
298 }

302 // This function applies the provided patterns using
303 // `applyPatternsAndFoldGreedily` and then replaces all newly inserted
304 // `UnrealizedConversionCastOps` that haven't folded away. ("Backward" casts
305 // from target to source types inserted by a `OneToNConversionPattern` normally
306 // fold away with the "forward" casts from source to target types inserted by
307 // the next pattern.) To understand which casts are "newly inserted", all casts
308 // inserted by this pass are annotated with a string attribute that also
309 // documents which kind of the cast (source, argument, or target).
310 LogicalResult
311 applyPartialOneToNConversion(Operation *op, OneToNTypeConverter &typeConverter,
312  const FrozenRewritePatternSet &patterns) {
313 #ifndef NDEBUG
314  // Remember existing unrealized casts. This data structure is only used in
315  // asserts; building it only for that purpose may be an overkill.
316  SmallSet<UnrealizedConversionCastOp, 4> existingCasts;
317  op->walk([&](UnrealizedConversionCastOp castOp) {
318  assert(!castOp->hasAttr(castKindAttrName));
319  existingCasts.insert(castOp);
320  });
321 #endif // NDEBUG
322 
323  // Apply provided conversion patterns.
324  if (failed(applyPatternsAndFoldGreedily(op, patterns))) {
325  emitError(op->getLoc()) << "failed to apply conversion patterns";
326  return failure();
327  }
328 
329  // Find all unrealized casts inserted by the pass that haven't folded away.
330  SmallVector<UnrealizedConversionCastOp> worklist;
331  op->walk([&](UnrealizedConversionCastOp castOp) {
332  if (castOp->hasAttr(castKindAttrName)) {
333  assert(!existingCasts.contains(castOp));
334  worklist.push_back(castOp);
335  }
336  });
337 
338  // Replace new casts with user materializations.
339  IRRewriter rewriter(op->getContext());
340  for (UnrealizedConversionCastOp castOp : worklist) {
341  TypeRange resultTypes = castOp->getResultTypes();
342  ValueRange operands = castOp->getOperands();
343  StringRef castKind =
344  castOp->getAttrOfType<StringAttr>(castKindAttrName).getValue();
345  rewriter.setInsertionPoint(castOp);
346 
347 #ifndef NDEBUG
348  // Determine whether operands or results are already legal to test some
349  // assumptions for the different kind of materializations. These properties
350  // are only used it asserts and it may be overkill to compute them.
351  bool areOperandTypesLegal = llvm::all_of(
352  operands.getTypes(), [&](Type t) { return typeConverter.isLegal(t); });
353  bool areResultsTypesLegal = llvm::all_of(
354  resultTypes, [&](Type t) { return typeConverter.isLegal(t); });
355 #endif // NDEBUG
356 
357  // Add materialization and remember materialized results.
358  SmallVector<Value> materializedResults;
359  if (castKind == getCastKindName(CastKind::Target)) {
360  // Target materialization.
361  assert(!areOperandTypesLegal && areResultsTypesLegal &&
362  operands.size() == 1 && "found unexpected target cast");
363  std::optional<SmallVector<Value>> maybeResults =
364  typeConverter.materializeTargetConversion(
365  rewriter, castOp->getLoc(), resultTypes, operands.front());
366  if (!maybeResults) {
367  emitError(castOp->getLoc())
368  << "failed to create target materialization";
369  return failure();
370  }
371  materializedResults = maybeResults.value();
372  } else {
373  // Source and argument materializations.
374  assert(areOperandTypesLegal && !areResultsTypesLegal &&
375  resultTypes.size() == 1 && "found unexpected cast");
376  std::optional<Value> maybeResult;
377  if (castKind == getCastKindName(CastKind::Source)) {
378  // Source materialization.
379  maybeResult = typeConverter.materializeSourceConversion(
380  rewriter, castOp->getLoc(), resultTypes.front(),
381  castOp.getOperands());
382  } else {
383  // Argument materialization.
384  assert(castKind == getCastKindName(CastKind::Argument) &&
385  "unexpected value of cast kind attribute");
386  assert(llvm::all_of(operands,
387  [&](Value v) { return v.isa<BlockArgument>(); }));
388  maybeResult = typeConverter.materializeArgumentConversion(
389  rewriter, castOp->getLoc(), resultTypes.front(),
390  castOp.getOperands());
391  }
392  if (!maybeResult.has_value() || !maybeResult.value()) {
393  emitError(castOp->getLoc())
394  << "failed to create " << castKind << " materialization";
395  return failure();
396  }
397  materializedResults = {maybeResult.value()};
398  }
399 
400  // Replace the cast with the result of the materialization.
401  rewriter.replaceOp(castOp, materializedResults);
402  }
403 
404  return success();
405 }
