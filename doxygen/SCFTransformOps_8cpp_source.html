28 DiagnosedSilenceableFailure
29 transform::GetParentForOp::apply(transform::TransformResults &results,
30  transform::TransformState &state) {
31  SetVector<Operation *> parents;
32  for (Operation *target : state.getPayloadOps(getTarget())) {
33  Operation *loop, *current = target;
34  for (unsigned i = 0, e = getNumLoops(); i < e; ++i) {
35  loop = getAffine()
36  ? current->getParentOfType<AffineForOp>().getOperation()
37  : current->getParentOfType<scf::ForOp>().getOperation();
38  if (!loop) {
39  DiagnosedSilenceableFailure diag =
40  emitSilenceableError()
41  << "could not find an '"
42  << (getAffine() ? AffineForOp::getOperationName()
43  : scf::ForOp::getOperationName())
44  << "' parent";
45  diag.attachNote(target->getLoc()) << "target op";
46  return diag;
47  }
48  current = loop;
49  }
50  parents.insert(loop);
51  }
52  results.set(getResult().cast<OpResult>(), parents.getArrayRef());
53  return DiagnosedSilenceableFailure::success();
54 }

60 /// Wraps the given operation `op` into an `scf.execute_region` operation. Uses
61 /// the provided rewriter for all operations to remain compatible with the
62 /// rewriting infra, as opposed to just splicing the op in place.
63 static scf::ExecuteRegionOp wrapInExecuteRegion(RewriterBase &b,
64  Operation *op) {
65  if (op->getNumRegions() != 1)
66  return nullptr;
67  OpBuilder::InsertionGuard g(b);
68  b.setInsertionPoint(op);
69  scf::ExecuteRegionOp executeRegionOp =
70  b.create<scf::ExecuteRegionOp>(op->getLoc(), op->getResultTypes());
71  {
72  OpBuilder::InsertionGuard g(b);
73  b.setInsertionPointToStart(&executeRegionOp.getRegion().emplaceBlock());
74  Operation *clonedOp = b.cloneWithoutRegions(*op);
75  Region &clonedRegion = clonedOp->getRegions().front();
76  assert(clonedRegion.empty() && "expected empty region");
77  b.inlineRegionBefore(op->getRegions().front(), clonedRegion,
78  clonedRegion.end());
79  b.create<scf::YieldOp>(op->getLoc(), clonedOp->getResults());
80  }
81  b.replaceOp(op, executeRegionOp.getResults());
82  return executeRegionOp;
83 }

85 DiagnosedSilenceableFailure
86 transform::LoopOutlineOp::apply(transform::TransformResults &results,
87  transform::TransformState &state) {
88  SmallVector<Operation *> transformed;
89  DenseMap<Operation *, SymbolTable> symbolTables;
90  for (Operation *target : state.getPayloadOps(getTarget())) {
91  Location location = target->getLoc();
92  Operation *symbolTableOp = SymbolTable::getNearestSymbolTable(target);
93  IRRewriter rewriter(getContext());
94  scf::ExecuteRegionOp exec = wrapInExecuteRegion(rewriter, target);
95  if (!exec) {
96  DiagnosedSilenceableFailure diag = emitSilenceableError()
97  << "failed to outline";
98  diag.attachNote(target->getLoc()) << "target op";
99  return diag;
100  }
101  func::CallOp call;
102  FailureOr<func::FuncOp> outlined = outlineSingleBlockRegion(
103  rewriter, location, exec.getRegion(), getFuncName(), &call);
104 
105  if (failed(outlined))
106  return emitDefaultDefiniteFailure(target);
107 
108  if (symbolTableOp) {
109  SymbolTable &symbolTable =
110  symbolTables.try_emplace(symbolTableOp, symbolTableOp)
111  .first->getSecond();
112  symbolTable.insert(*outlined);
113  call.setCalleeAttr(FlatSymbolRefAttr::get(*outlined));
114  }
115  transformed.push_back(*outlined);
116  }
117  results.set(getTransformed().cast<OpResult>(), transformed);
118  return DiagnosedSilenceableFailure::success();
119 }

146 /// Callback for PipeliningOption. Populates `schedule` with the mapping from an
147 /// operation to its logical time position given the iteration interval and the
148 /// read latency. The latter is only relevant for vector transfers.
149 static void
150 loopScheduling(scf::ForOp forOp,
151  std::vector<std::pair<Operation *, unsigned>> &schedule,
152  unsigned iterationInterval, unsigned readLatency) {
153  auto getLatency = [&](Operation *op) -> unsigned {
154  if (isa<vector::TransferReadOp>(op))
155  return readLatency;
156  return 1;
157  };
158 
159  DenseMap<Operation *, unsigned> opCycles;
160  std::map<unsigned, std::vector<Operation *>> wrappedSchedule;
161  for (Operation &op : forOp.getBody()->getOperations()) {
162  if (isa<scf::YieldOp>(op))
163  continue;
164  unsigned earlyCycle = 0;
165  for (Value operand : op.getOperands()) {
166  Operation *def = operand.getDefiningOp();
167  if (!def)
168  continue;
169  earlyCycle = std::max(earlyCycle, opCycles[def] + getLatency(def));
170  }
171  opCycles[&op] = earlyCycle;
172  wrappedSchedule[earlyCycle % iterationInterval].push_back(&op);
173  }
174  for (const auto &it : wrappedSchedule) {
175  for (Operation *op : it.second) {
176  unsigned cycle = opCycles[op];
177  schedule.emplace_back(op, cycle / iterationInterval);
178  }
179  }
180 }

182 DiagnosedSilenceableFailure
183 transform::LoopPipelineOp::applyToOne(scf::ForOp target,
184  transform::ApplyToEachResultList &results,
185  transform::TransformState &state) {
186  scf::PipeliningOption options;
187  options.getScheduleFn =
188  [this](scf::ForOp forOp,
189  std::vector<std::pair<Operation *, unsigned>> &schedule) mutable {
190  loopScheduling(forOp, schedule, getIterationInterval(),
191  getReadLatency());
192  };
193  scf::ForLoopPipeliningPattern pattern(options, target->getContext());
194  IRRewriter rewriter(getContext());
195  rewriter.setInsertionPoint(target);
196  FailureOr<scf::ForOp> patternResult =
197  scf::pipelineForLoop(rewriter, target, options);
198  if (succeeded(patternResult)) {
199  results.push_back(*patternResult);
200  return DiagnosedSilenceableFailure::success();
201  }
202  return emitDefaultSilenceableFailure(target);
203 }

253 /// Replaces the given op with the contents of the given single-block region,
254 /// using the operands of the block terminator to replace operation results.
255 static void replaceOpWithRegion(RewriterBase &rewriter, Operation *op,
256  Region &region) {
257  assert(llvm::hasSingleElement(region) && "expected single-region block");
258  Block *block = &region.front();
259  Operation *terminator = block->getTerminator();
260  ValueRange results = terminator->getOperands();
261  rewriter.inlineBlockBefore(block, op, /*blockArgs=*/{});
262  rewriter.replaceOp(op, results);
263  rewriter.eraseOp(terminator);
264 }

266 DiagnosedSilenceableFailure transform::TakeAssumedBranchOp::applyToOne(
267  scf::IfOp ifOp, transform::ApplyToEachResultList &results,
268  transform::TransformState &state) {
269  TrackingListener listener(state, *this);
270  IRRewriter rewriter(ifOp->getContext(), &listener);
271  rewriter.setInsertionPoint(ifOp);
272 
273  Region &region =
274  getTakeElseBranch() ? ifOp.getElseRegion() : ifOp.getThenRegion();
275  if (!llvm::hasSingleElement(region)) {
276  return emitDefiniteFailure()
277  << "requires an scf.if op with a single-block "
278  << ((getTakeElseBranch()) ? "`else`" : "`then`") << " region";
279  }
280  replaceOpWithRegion(rewriter, ifOp, region);
281  return DiagnosedSilenceableFailure::success();
282 }
