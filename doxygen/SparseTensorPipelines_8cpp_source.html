31 /// Return configuration options for One-Shot Bufferize.
32 static bufferization::OneShotBufferizationOptions
33 getBufferizationOptions(bool analysisOnly) {
34  using namespace bufferization;
35  OneShotBufferizationOptions options;
36  options.bufferizeFunctionBoundaries = true;
37  // TODO(springerm): To spot memory leaks more easily, returning dense allocs
38  // should be disallowed.
39  options.allowReturnAllocs = true;
40  options.setFunctionBoundaryTypeConversion(LayoutMapOption::IdentityLayoutMap);
41  options.unknownTypeConverterFn = [](Value value, Attribute memorySpace,
42  const BufferizationOptions &options) {
43  return getMemRefTypeWithStaticIdentityLayout(
44  value.getType().cast<TensorType>(), memorySpace);
45  };
46  if (analysisOnly) {
47  options.testAnalysisOnly = true;
48  options.printConflicts = true;
49  }
50  return options;
51 }

57 void mlir::sparse_tensor::buildSparseCompiler(
58  OpPassManager &pm, const SparseCompilerOptions &options) {
59  pm.addNestedPass<func::FuncOp>(createLinalgGeneralizationPass());
60  pm.addPass(createSparsificationAndBufferizationPass(
61  getBufferizationOptions(options.testBufferizationAnalysisOnly),
62  options.sparsificationOptions(), options.sparseTensorConversionOptions(),
63  options.createSparseDeallocs, options.enableRuntimeLibrary,
64  options.enableBufferInitialization, options.vectorLength,
65  /*enableVLAVectorization=*/options.armSVE,
66  /*enableSIMDIndex32=*/options.force32BitVectorIndices));
67  if (options.testBufferizationAnalysisOnly)
68  return;
69  pm.addNestedPass<func::FuncOp>(createCanonicalizerPass());
70  pm.addNestedPass<func::FuncOp>(
71  mlir::bufferization::createFinalizingBufferizePass());
72 
73  // GPU code generation.
74  const bool gpuCodegen = options.gpuTriple.hasValue();
75  if (gpuCodegen) {
76  pm.addPass(createSparseGPUCodegenPass());
77  pm.addNestedPass<gpu::GPUModuleOp>(createStripDebugInfoPass());
78  pm.addNestedPass<gpu::GPUModuleOp>(createConvertSCFToCFPass());
79  pm.addNestedPass<gpu::GPUModuleOp>(createLowerGpuOpsToNVVMOpsPass());
80  }
81 
82  // TODO(springerm): Add sparse support to the BufferDeallocation pass and add
83  // it to this pipeline.
84  pm.addNestedPass<func::FuncOp>(createConvertLinalgToLoopsPass());
85  pm.addNestedPass<func::FuncOp>(createConvertVectorToSCFPass());
86  pm.addNestedPass<func::FuncOp>(createConvertSCFToCFPass());
87  pm.addPass(memref::createExpandStridedMetadataPass());
88  pm.addPass(createLowerAffinePass());
89  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
90  pm.addPass(createFinalizeMemRefToLLVMConversionPass());
91  pm.addNestedPass<func::FuncOp>(createConvertComplexToStandardPass());
92  pm.addNestedPass<func::FuncOp>(arith::createArithExpandOpsPass());
93  pm.addNestedPass<func::FuncOp>(createConvertMathToLLVMPass());
94  pm.addPass(createConvertMathToLibmPass());
95  pm.addPass(createConvertComplexToLibmPass());
96  // Repeat convert-vector-to-llvm.
97  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
98  pm.addPass(createConvertComplexToLLVMPass());
99  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
100  pm.addPass(createConvertFuncToLLVMPass());
101 
102  // Finalize GPU code generation.
103  if (gpuCodegen) {
104 #if MLIR_GPU_TO_CUBIN_PASS_ENABLE
105  pm.addNestedPass<gpu::GPUModuleOp>(createGpuSerializeToCubinPass(
106  options.gpuTriple, options.gpuChip, options.gpuFeatures));
107 #endif
108  pm.addPass(createGpuToLLVMConversionPass());
109  }
110 
111  pm.addPass(createReconcileUnrealizedCastsPass());
112 }
