23 /// Hoist vector.transfer_read/vector.transfer_write on buffers pairs out of
24 /// immediately enclosing scf::ForOp iteratively, if the following conditions
25 /// are true:
26 /// 1. The two ops access the same memref with the same indices.
27 /// 2. All operands are invariant under the enclosing scf::ForOp.
28 /// 3. No uses of the memref either dominate the transfer_read or are
29 /// dominated by the transfer_write (i.e. no aliasing between the write and
30 /// the read across the loop)
31 /// To improve hoisting opportunities, call the `moveLoopInvariantCode` helper
32 /// function on the candidate loop above which to hoist. Hoisting the transfers
33 /// results in scf::ForOp yielding the value that originally transited through
34 /// memory.
35 ///
36 /// WARNING: This hoisting does not model parallelism and is generally incorrect
37 /// when used on distributed loops with memref semantics!
38 void hoistRedundantVectorTransfers(func::FuncOp func);

40 /// Greedily hoist redundant subset extract/insert operations on tensors outside
41 /// of `forOp`. The logic follows:
42 /// 1. Look for a write walking back from the `forOp` yield.
43 /// 2. Check the uses of the matching block argument and look for a matching
44 /// read (i.e. extract_slice of transfer_read) with matching indices.
45 /// 3. In the case of a transfer_write, we can bypass other non-conflicting
46 /// operations and find more hoisting opportunities.
47 /// 4. Hoist the read/write pair and update the tensor SSA links.
48 ///
49 /// Return the unmodified `forOp` if no hoisting occured.
50 /// Return a new scf::ForOp if hoisting on tensors occured.
51 ///
52 /// After this transformation the returned scf::ForOp may have unused arguments
53 /// that can be removed by application of canonicalization patterns.
54 ///
55 /// Example:
56 /// ========
57 /// IR Resembling:
58 ///
59 /// ```
60 /// %0 = scf.for %i = %l to %u step %s iter_args(%a0 = %t0)->(tensor<10xf32>) {
61 /// %1 = scf.for %j = %l to %u step %s iter_args(%a6 = %a0)->(tensor<10xf32>) {
62 /// %e = tensor.extract_slice %a6[%i][%sz][1]: tensor<10xf32> to tensor<?xf32>
63 /// %r = vector.transfer_read %e[%c0], %cst: tensor<?xf32>, vector<4xf32>
64 /// %u = "some_use"(%r) : (vector<4xf32>) -> vector<4xf32>
65 /// %w = vector.transfer_write %u, %e[%c0] : vector<4xf32>, tensor<?xf32>
66 /// %st = tensor.insert_slice %w into %a6[%i][%sz][1]
67 /// : tensor<?xf32> into tensor<10xf32>
68 /// scf.yield %st: tensor<10xf32>
69 /// }
70 /// scf.yield %1: tensor<10xf32>
71 /// }
72 /// ```
73 ///
74 /// Progressively hoists to:
75 ///
76 /// ```
77 /// %0 = scf.for %i = %l to %u step %s iter_args(%a0 = %t0) -> (tensor<10xf32>){
78 /// %e = tensor.extract_slice %a0[%i][%sz][1]: tensor<10xf32> to tensor<?xf32>
79 /// %1:2 = scf.for %j = %l to %u step %s iter_args(%a6 = a0, %a7 = %e)
80 /// -> (tensor<10xf32>, tensor<?xf32>) {
81 /// %r = vector.transfer_read %a7[%c0], %cst: tensor<?xf32>, vector<4xf32>
82 /// %u = "some_use"(%r) : (vector<4xf32>) -> vector<4xf32>
83 /// %w = vector.transfer_write %u, %a7[%c0] : vector<4xf32>, tensor<?xf32>
84 /// scf.yield %a6, %w: tensor<10xf32>, tensor<?xf32>
85 /// }
86 /// %st = tensor.insert_slice %1#1 into %1#0[%i][%sz][1]
87 /// : tensor<?xf32> into tensor<10xf32>
88 /// scf.yield %1: tensor<10xf32>
89 /// }
90 /// ```
91 ///
92 /// and
93 ///
94 /// ```
95 /// %0 = scf.for %i = %l to %u step %s iter_args(%a0 = %t0) -> (tensor<10xf32>){
96 /// %e = tensor.extract_slice %a0[%i][%sz][1]: tensor<10xf32> to tensor<?xf32>
97 /// %r = vector.transfer_read %a7[%c0], %cst: tensor<?xf32>, vector<4xf32>
98 /// %1:3 = scf.for %j = %l to %u step %s iter_args(%a6 = a0, %a7 = %e, %a7 = r)
99 /// -> (tensor<10xf32>, tensor<?xf32>, vector<4xf32>) {
100 /// %u = "some_use"(%r) : (vector<4xf32>) -> vector<4xf32>
101 /// scf.yield %a6, %a7, %u: tensor<10xf32>, tensor<?xf32>, vector<4xf32>
102 /// }
103 /// %w = vector.transfer_write %1#2, %1#1[%c0] : vector<4xf32>, tensor<?xf32>
104 /// %st = tensor.insert_slice %w into %1#0[%i][%sz][1]
105 /// : tensor<?xf32> into tensor<10xf32>
106 /// scf.yield %1: tensor<10xf32>
107 /// }
108 /// ```
109 ///
110 /// It can then canonicalize to:
111 ///
112 /// ```
113 /// %0 = scf.for %i = %l to %u step %s iter_args(%a0 = %t0) -> (tensor<10xf32>){
114 /// %e = tensor.extract_slice %a0[%i][%sz][1]: tensor<10xf32> to tensor<?xf32>
115 /// %r = vector.transfer_read %a7[%c0], %cst: tensor<?xf32>, vector<4xf32>
116 /// %1 = scf.for %j = %l to %u step %s iter_args(%a7 = r)
117 /// -> (tensor<10xf32>, tensor<?xf32>, vector<4xf32>) {
118 /// %u = "some_use"(%r) : (vector<4xf32>) -> vector<4xf32>
119 /// scf.yield %u: vector<4xf32>
120 /// }
121 /// %w = vector.transfer_write %1, %e[%c0] : vector<4xf32>, tensor<?xf32>
122 /// %st = tensor.insert_slice %w into %a0[%i][%sz][1]
123 /// : tensor<?xf32> into tensor<10xf32>
124 /// scf.yield %1: tensor<10xf32>
125 /// }
126 /// ```
127 ///

129 // - Other loops than scf.ForOp that operate on tensors (both sequential and
130 // parallel loops).
131 // - Other subset extract/insert pairs than tensor.extract/insert_slice and
132 // vector.transfer_read/write.
