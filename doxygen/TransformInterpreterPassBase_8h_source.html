55 /// Base class for transform dialect interpreter passes that can consume and
56 /// dump transform dialect scripts in separate files. The pass is controlled by
57 /// three string options:
58 ///
59 /// - transformFileName: if non-empty, the name of the file containing the
60 /// transform script. If empty, `debugTransformRootTag` is considered or the
61 /// pass root operation must contain a single top-level transform op that
62 /// will be interpreted.
63 /// - transformLibraryFileName: if non-empty, the name of the file containing
64 /// definitions of external symbols referenced in the transform script.
65 /// These definitions will be used to replace declarations.
66 /// - debugPayloadRootTag: if non-empty, the value of the attribute named
67 /// `kTransformDialectTagAttrName` indicating the single op that is
68 /// considered the payload root of the transform interpreter; otherwise, the
69 /// root operation of the pass is used.
70 /// - debugTransformRootTag: if non-empty, the value of the attribute named
71 /// `kTransformDialectTagAttrName` indicating the single top-level transform
72 /// op contained in the payload root to be used as the entry point by the
73 /// transform interpreter; mutually exclusive with `transformFileName`.
74 ///
75 /// The pass runs the transform dialect interpreter as directed by the options.
76 /// It also provides the mechanism to dump reproducers into stderr
77 /// (-debug-only=transform-dialect-dump-repro) or into a temporary file
78 /// (-debug-only=transform-dialect-save-repro) that can be used with this
79 /// pass in a standalone mode.
80 ///
81 /// Concrete passes must derive from this class instead of their generated base
82 /// class (or PassWrapper), and supply themselves and the generated base class
83 /// as template arguments. They are *not* expected to to implement `initialize`
84 /// or `runOnOperation`. They *are* expected to call the copy constructor of
85 /// this class in their copy constructors, short of which the file-based
86 /// transform dialect script injection facility will become nonoperational.
87 ///
88 /// Concrete passes may implement the `runBeforeInterpreter` and
89 /// `runAfterInterpreter` to customize the behavior of the pass.
90 template <typename Concrete, template <typename> typename GeneratedBase>
91 class TransformInterpreterPassBase : public GeneratedBase<Concrete> {
92 public:
93  explicit TransformInterpreterPassBase(
94  const TransformOptions &options = TransformOptions())
95  : options(options) {}

129  /// Hook for passes to run additional logic in the pass before the
130  /// interpreter. If failure is returned, the pass fails and the interpreter is
131  /// not run.
132  LogicalResult runBeforeInterpreter(Operation *) { return success(); }
133 
134  /// Hook for passes to run additional logic in the pass after the interpreter.
135  /// Only runs if everything succeeded before. If failure is returned, the pass
136  /// fails.
137  LogicalResult runAfterInterpreter(Operation *) { return success(); }
138 
139  void runOnOperation() override {
140  auto *pass = static_cast<Concrete *>(this);
141  Operation *op = pass->getOperation();
142  StringRef binaryName = Concrete::getBinaryName();
143  if (failed(pass->runBeforeInterpreter(op)) ||
144  failed(detail::interpreterBaseRunOnOperationImpl(
145  op, pass->getArgument(), sharedTransformModule,
146  transformLibraryModule,
147  /*extraMappings=*/{}, options, pass->transformFileName,
148  pass->transformLibraryFileName, pass->debugPayloadRootTag,
149  pass->debugTransformRootTag, binaryName)) ||
150  failed(pass->runAfterInterpreter(op))) {
151  return pass->signalPassFailure();
152  }
153  }
154 
155 protected:
156  /// Transform interpreter options.
157  TransformOptions options;
158 
159  /// Returns a read-only reference to shared transform module.
160  const std::shared_ptr<OwningOpRef<ModuleOp>> &
161  getSharedTransformModule() const {
162  return sharedTransformModule;
163  }
164 
165  /// Returns a read-only reference to the transform library module.
166  const std::shared_ptr<OwningOpRef<ModuleOp>> &
167  getTransformLibraryModule() const {
168  return transformLibraryModule;
169  }
170 
171 private:
172  /// The separate transform module to be used for transformations, shared
173  /// across multiple instances of the pass if it is applied in parallel to
174  /// avoid potentially expensive cloning. MUST NOT be modified after the pass
175  /// has been initialized.
176  std::shared_ptr<OwningOpRef<ModuleOp>> sharedTransformModule = nullptr;
177 
178  /// The transform module containing symbol definitions that become available
179  /// in the transform scripts. Similar to dynamic linking for binaries. This is
180  /// shared across multiple instances of the pass and therefore MUST NOT be
181  /// modified after the pass has been initialized.
182  std::shared_ptr<OwningOpRef<ModuleOp>> transformLibraryModule = nullptr;
183 };
