48 /// Parse a region, and add a single block if the region is empty.
49 /// If no region is parsed, create a new region with a single empty block.
50 static ParseResult parseSingleBlockRegion(OpAsmParser &p, Region &region) {
51  auto regionParseRes = p.parseOptionalRegion(region);
52  if (regionParseRes.has_value() && failed(regionParseRes.value()))
53  return failure();
54 
55  // If the region is empty, add a single empty block.
56  if (region.empty())
57  region.push_back(new Block());
58 
59  return success();
60 }
