23 PadOp mlir::tensor::createPadHighOp(RankedTensorType type, Value source,
24  Value pad, bool nofold, Location loc,
25  OpBuilder &b) {
26  auto zero = b.createOrFold<arith::ConstantIndexOp>(loc, 0);
27  SmallVector<OpFoldResult> low(type.getRank(), zero);
28  SmallVector<OpFoldResult> high(type.getRank(), zero);
29  for (const auto &en : enumerate(type.getShape())) {
30  // Pad only the static dimensions of the result tensor type.
31  if (ShapedType::isDynamic(en.value()))
32  continue;
33  // Compute the padding width.
34  AffineExpr d0;
35  bindDims(b.getContext(), d0);
36  auto dimOp = b.createOrFold<tensor::DimOp>(loc, source, en.index());
37  high[en.index()] =
38  affine::makeComposedAffineApply(b, loc, en.value() - d0, {dimOp})
39  .getResult();
40  }
41  return b.create<PadOp>(loc, type, source, low, high, pad, nofold);
42 }

57 FailureOr<OpFoldResult> mlir::tensor::createDimValue(OpBuilder &b, Location loc,
58  Value rankedTensor,
59  int64_t dim) {
60  auto tensorTy = rankedTensor.getType().dyn_cast<RankedTensorType>();
61  if (!tensorTy)
62  return failure();
63  auto shape = tensorTy.getShape();
64  if (dim >= static_cast<int64_t>(shape.size()))
65  return failure();
66  if (ShapedType::isDynamic(shape[dim]))
67  return OpFoldResult(b.createOrFold<tensor::DimOp>(loc, rankedTensor, dim));
68  return OpFoldResult(b.getIndexAttr(shape[dim]));
69 }

86 FailureOr<RankedTensorType>
87 mlir::tensor::computeTransposedType(RankedTensorType rankedTensorType,
88  ArrayRef<int64_t> transposeVector) {
89  if (transposeVector.empty())
90  return rankedTensorType;
91 
92  if (!isPermutationVector(transposeVector) ||
93  transposeVector.size() != static_cast<size_t>(rankedTensorType.getRank()))
94  return failure();
95 
96  SmallVector<int64_t> transposedShape(rankedTensorType.getShape().begin(),
97  rankedTensorType.getShape().end());
98  applyPermutationToVector(transposedShape, transposeVector);
99 
100  using RTTBuilder = RankedTensorType::Builder;
101  RankedTensorType transposedTensorType =
102  RTTBuilder(rankedTensorType).setShape(transposedShape);
103  return transposedTensorType;
104 }
