31 static llvm::FastMathFlags getFastmathFlags(FastmathFlagsInterface &op) {
32  using llvmFMF = llvm::FastMathFlags;
33  using FuncT = void (llvmFMF::*)(bool);
34  const std::pair<FastmathFlags, FuncT> handlers[] = {
35  // clang-format off
36  {FastmathFlags::nnan, &llvmFMF::setNoNaNs},
37  {FastmathFlags::ninf, &llvmFMF::setNoInfs},
38  {FastmathFlags::nsz, &llvmFMF::setNoSignedZeros},
39  {FastmathFlags::arcp, &llvmFMF::setAllowReciprocal},
40  {FastmathFlags::contract, &llvmFMF::setAllowContract},
41  {FastmathFlags::afn, &llvmFMF::setApproxFunc},
42  {FastmathFlags::reassoc, &llvmFMF::setAllowReassoc},
43  // clang-format on
44  };
45  llvm::FastMathFlags ret;
46  ::mlir::LLVM::FastmathFlags fmfMlir = op.getFastmathAttr().getValue();
47  for (auto it : handlers)
48  if (bitEnumContainsAll(fmfMlir, it.first))
49  (ret.*(it.second))(true);
50  return ret;
51 }

60 /// Get the declaration of an overloaded llvm intrinsic. First we get the
61 /// overloaded argument types and/or result type from the CallIntrinsicOp, and
62 /// then use those to get the correct declaration of the overloaded intrinsic.
63 static FailureOr<llvm::Function *>
64 getOverloadedDeclaration(CallIntrinsicOp &op, llvm::Intrinsic::ID id,
65  llvm::Module *module,
66  LLVM::ModuleTranslation &moduleTranslation) {
67  SmallVector<llvm::Type *, 8> allArgTys;
68  for (Type type : op->getOperandTypes())
69  allArgTys.push_back(moduleTranslation.convertType(type));
70 
71  llvm::Type *resTy;
72  if (op.getNumResults() == 0)
73  resTy = llvm::Type::getVoidTy(module->getContext());
74  else
75  resTy = moduleTranslation.convertType(op.getResult(0).getType());
76 
77  // ATM we do not support variadic intrinsics.
78  llvm::FunctionType *ft = llvm::FunctionType::get(resTy, allArgTys, false);
79 
80  SmallVector<llvm::Intrinsic::IITDescriptor, 8> table;
81  getIntrinsicInfoTableEntries(id, table);
82  ArrayRef<llvm::Intrinsic::IITDescriptor> tableRef = table;
83 
84  SmallVector<llvm::Type *, 8> overloadedArgTys;
85  if (llvm::Intrinsic::matchIntrinsicSignature(ft, tableRef,
86  overloadedArgTys) !=
87  llvm::Intrinsic::MatchIntrinsicTypesResult::MatchIntrinsicTypes_Match) {
88  return op.emitOpError("intrinsic type is not a match");
89  }
90 
91  ArrayRef<llvm::Type *> overloadedArgTysRef = overloadedArgTys;
92  return llvm::Intrinsic::getDeclaration(module, id, overloadedArgTysRef);
93 }

95 /// Builder for LLVM_CallIntrinsicOp
96 static LogicalResult
97 convertCallLLVMIntrinsicOp(CallIntrinsicOp &op, llvm::IRBuilderBase &builder,
98  LLVM::ModuleTranslation &moduleTranslation) {
99  llvm::Module *module = builder.GetInsertBlock()->getModule();
100  llvm::Intrinsic::ID id =
101  llvm::Function::lookupIntrinsicID(op.getIntrinAttr());
102  if (!id)
103  return op.emitOpError()
104  << "couldn't find intrinsic: " << op.getIntrinAttr();
105 
106  llvm::Function *fn = nullptr;
107  if (llvm::Intrinsic::isOverloaded(id)) {
108  auto fnOrFailure =
109  getOverloadedDeclaration(op, id, module, moduleTranslation);
110  if (failed(fnOrFailure))
111  return failure();
112  fn = *fnOrFailure;
113  } else {
114  fn = llvm::Intrinsic::getDeclaration(module, id, {});
115  }
116 
117  auto *inst =
118  builder.CreateCall(fn, moduleTranslation.lookupValues(op.getOperands()));
119  if (op.getNumResults() == 1)
120  moduleTranslation.mapValue(op->getResults().front()) = inst;
121  return success();
122 }

124 /// Constructs branch weights metadata if the provided `weights` hold a value,
125 /// otherwise returns nullptr.
126 static llvm::MDNode *
127 convertBranchWeights(std::optional<ElementsAttr> weights,
128  LLVM::ModuleTranslation &moduleTranslation) {
129  if (!weights)
130  return nullptr;
131  SmallVector<uint32_t> weightValues;
132  weightValues.reserve(weights->size());
133  for (APInt weight : weights->cast<DenseIntElementsAttr>())
134  weightValues.push_back(weight.getLimitedValue());
135  return llvm::MDBuilder(moduleTranslation.getLLVMContext())
136  .createBranchWeights(weightValues);
137 }

140 convertOperationImpl(Operation &opInst, llvm::IRBuilderBase &builder,
141  LLVM::ModuleTranslation &moduleTranslation) {

150  // Helper function to reconstruct the function type for an indirect call given
151  // the result and argument types. The function cannot reconstruct the type of
152  // variadic functions since the call operation does not carry enough
153  // information to distinguish normal and variadic arguments. Supporting
154  // indirect variadic calls requires an additional type attribute on the call
155  // operation that stores the LLVM function type of the callee.
156  // TODO: Support indirect calls to variadic function pointers.
157  auto getCalleeFunctionType = [&](TypeRange resultTypes, ValueRange args) {
158  Type resultType = resultTypes.empty()
159  ? LLVMVoidType::get(opInst.getContext())
160  : resultTypes.front();
161  return llvm::cast<llvm::FunctionType>(moduleTranslation.convertType(
162  LLVMFunctionType::get(opInst.getContext(), resultType,
163  llvm::to_vector(args.getTypes()), false)));
164  };
165 
166  // Emit function calls. If the "callee" attribute is present, this is a
167  // direct function call and we also need to look up the remapped function
168  // itself. Otherwise, this is an indirect call and the callee is the first
169  // operand, look it up as a normal value.
170  if (auto callOp = dyn_cast<LLVM::CallOp>(opInst)) {
171  auto operands = moduleTranslation.lookupValues(callOp.getOperands());
172  ArrayRef<llvm::Value *> operandsRef(operands);
173  llvm::CallInst *call;
174  if (auto attr = callOp.getCalleeAttr()) {
175  call = builder.CreateCall(
176  moduleTranslation.lookupFunction(attr.getValue()), operandsRef);
177  } else {
178  call = builder.CreateCall(getCalleeFunctionType(callOp.getResultTypes(),
179  callOp.getArgOperands()),
180  operandsRef.front(), operandsRef.drop_front());
181  }
182  llvm::MDNode *branchWeights =
183  convertBranchWeights(callOp.getBranchWeights(), moduleTranslation);
184  if (branchWeights)
185  call->setMetadata(llvm::LLVMContext::MD_prof, branchWeights);
186  // If the called function has a result, remap the corresponding value. Note
187  // that LLVM IR dialect CallOp has either 0 or 1 result.
188  if (opInst.getNumResults() != 0) {
189  moduleTranslation.mapValue(opInst.getResult(0), call);
190  return success();
191  }
192  // Check that LLVM call returns void for 0-result functions.
193  return success(call->getType()->isVoidTy());
194  }
195 
196  if (auto inlineAsmOp = dyn_cast<LLVM::InlineAsmOp>(opInst)) {
197  // TODO: refactor function type creation which usually occurs in std-LLVM
198  // conversion.
199  SmallVector<Type, 8> operandTypes;
200  llvm::append_range(operandTypes, inlineAsmOp.getOperands().getTypes());
201 
202  Type resultType;
203  if (inlineAsmOp.getNumResults() == 0) {
204  resultType = LLVM::LLVMVoidType::get(&moduleTranslation.getContext());
205  } else {
206  assert(inlineAsmOp.getNumResults() == 1);
207  resultType = inlineAsmOp.getResultTypes()[0];
208  }
209  auto ft = LLVM::LLVMFunctionType::get(resultType, operandTypes);
210  llvm::InlineAsm *inlineAsmInst =
211  inlineAsmOp.getAsmDialect()
212  ? llvm::InlineAsm::get(
213  static_cast<llvm::FunctionType *>(
214  moduleTranslation.convertType(ft)),
215  inlineAsmOp.getAsmString(), inlineAsmOp.getConstraints(),
216  inlineAsmOp.getHasSideEffects(),
217  inlineAsmOp.getIsAlignStack(),
218  convertAsmDialectToLLVM(*inlineAsmOp.getAsmDialect()))
219  : llvm::InlineAsm::get(static_cast<llvm::FunctionType *>(
220  moduleTranslation.convertType(ft)),
221  inlineAsmOp.getAsmString(),
222  inlineAsmOp.getConstraints(),
223  inlineAsmOp.getHasSideEffects(),
224  inlineAsmOp.getIsAlignStack());
225  llvm::CallInst *inst = builder.CreateCall(
226  inlineAsmInst,
227  moduleTranslation.lookupValues(inlineAsmOp.getOperands()));
228  if (auto maybeOperandAttrs = inlineAsmOp.getOperandAttrs()) {
229  llvm::AttributeList attrList;
230  for (const auto &it : llvm::enumerate(*maybeOperandAttrs)) {
231  Attribute attr = it.value();
232  if (!attr)
233  continue;
234  DictionaryAttr dAttr = attr.cast<DictionaryAttr>();
235  TypeAttr tAttr =
236  dAttr.get(InlineAsmOp::getElementTypeAttrName()).cast<TypeAttr>();
237  llvm::AttrBuilder b(moduleTranslation.getLLVMContext());
238  llvm::Type *ty = moduleTranslation.convertType(tAttr.getValue());
239  b.addTypeAttr(llvm::Attribute::ElementType, ty);
240  // shift to account for the returned value (this is always 1 aggregate
241  // value in LLVM).
242  int shift = (opInst.getNumResults() > 0) ? 1 : 0;
243  attrList = attrList.addAttributesAtIndex(
244  moduleTranslation.getLLVMContext(), it.index() + shift, b);
245  }
246  inst->setAttributes(attrList);
247  }
248 
249  if (opInst.getNumResults() != 0)
250  moduleTranslation.mapValue(opInst.getResult(0), inst);
251  return success();
252  }
253 
254  if (auto invOp = dyn_cast<LLVM::InvokeOp>(opInst)) {
255  auto operands = moduleTranslation.lookupValues(invOp.getCalleeOperands());
256  ArrayRef<llvm::Value *> operandsRef(operands);
257  llvm::Instruction *result;
258  if (auto attr = opInst.getAttrOfType<FlatSymbolRefAttr>("callee")) {
259  result = builder.CreateInvoke(
260  moduleTranslation.lookupFunction(attr.getValue()),
261  moduleTranslation.lookupBlock(invOp.getSuccessor(0)),
262  moduleTranslation.lookupBlock(invOp.getSuccessor(1)), operandsRef);
263  } else {
264  result = builder.CreateInvoke(
265  getCalleeFunctionType(invOp.getResultTypes(), invOp.getArgOperands()),
266  operandsRef.front(),
267  moduleTranslation.lookupBlock(invOp.getSuccessor(0)),
268  moduleTranslation.lookupBlock(invOp.getSuccessor(1)),
269  operandsRef.drop_front());
270  }
271  llvm::MDNode *branchWeights =
272  convertBranchWeights(invOp.getBranchWeights(), moduleTranslation);
273  if (branchWeights)
274  result->setMetadata(llvm::LLVMContext::MD_prof, branchWeights);
275  moduleTranslation.mapBranch(invOp, result);
276  // InvokeOp can only have 0 or 1 result
277  if (invOp->getNumResults() != 0) {
278  moduleTranslation.mapValue(opInst.getResult(0), result);
279  return success();
280  }
281  return success(result->getType()->isVoidTy());
282  }
283 
284  if (auto lpOp = dyn_cast<LLVM::LandingpadOp>(opInst)) {
285  llvm::Type *ty = moduleTranslation.convertType(lpOp.getType());
286  llvm::LandingPadInst *lpi =
287  builder.CreateLandingPad(ty, lpOp.getNumOperands());
288  lpi->setCleanup(lpOp.getCleanup());
289 
290  // Add clauses
291  for (llvm::Value *operand :
292  moduleTranslation.lookupValues(lpOp.getOperands())) {
293  // All operands should be constant - checked by verifier
294  if (auto *constOperand = dyn_cast<llvm::Constant>(operand))
295  lpi->addClause(constOperand);
296  }
297  moduleTranslation.mapValue(lpOp.getResult(), lpi);
298  return success();
299  }
300 
301  // Emit branches. We need to look up the remapped blocks and ignore the
302  // block arguments that were transformed into PHI nodes.
303  if (auto brOp = dyn_cast<LLVM::BrOp>(opInst)) {
304  llvm::BranchInst *branch =
305  builder.CreateBr(moduleTranslation.lookupBlock(brOp.getSuccessor()));
306  moduleTranslation.mapBranch(&opInst, branch);
307  moduleTranslation.setLoopMetadata(&opInst, branch);
308  return success();
309  }
310  if (auto condbrOp = dyn_cast<LLVM::CondBrOp>(opInst)) {
311  llvm::MDNode *branchWeights =
312  convertBranchWeights(condbrOp.getBranchWeights(), moduleTranslation);
313  llvm::BranchInst *branch = builder.CreateCondBr(
314  moduleTranslation.lookupValue(condbrOp.getOperand(0)),
315  moduleTranslation.lookupBlock(condbrOp.getSuccessor(0)),
316  moduleTranslation.lookupBlock(condbrOp.getSuccessor(1)), branchWeights);
317  moduleTranslation.mapBranch(&opInst, branch);
318  moduleTranslation.setLoopMetadata(&opInst, branch);
319  return success();
320  }
321  if (auto switchOp = dyn_cast<LLVM::SwitchOp>(opInst)) {
322  llvm::MDNode *branchWeights =
323  convertBranchWeights(switchOp.getBranchWeights(), moduleTranslation);
324  llvm::SwitchInst *switchInst = builder.CreateSwitch(
325  moduleTranslation.lookupValue(switchOp.getValue()),
326  moduleTranslation.lookupBlock(switchOp.getDefaultDestination()),
327  switchOp.getCaseDestinations().size(), branchWeights);
328 
329  auto *ty = llvm::cast<llvm::IntegerType>(
330  moduleTranslation.convertType(switchOp.getValue().getType()));
331  for (auto i :
332  llvm::zip(switchOp.getCaseValues()->cast<DenseIntElementsAttr>(),
333  switchOp.getCaseDestinations()))
334  switchInst->addCase(
335  llvm::ConstantInt::get(ty, std::get<0>(i).getLimitedValue()),
336  moduleTranslation.lookupBlock(std::get<1>(i)));
337 
338  moduleTranslation.mapBranch(&opInst, switchInst);
339  return success();
340  }
341 
342  // Emit addressof. We need to look up the global value referenced by the
343  // operation and store it in the MLIR-to-LLVM value mapping. This does not
344  // emit any LLVM instruction.
345  if (auto addressOfOp = dyn_cast<LLVM::AddressOfOp>(opInst)) {
346  LLVM::GlobalOp global =
347  addressOfOp.getGlobal(moduleTranslation.symbolTable());
348  LLVM::LLVMFuncOp function =
349  addressOfOp.getFunction(moduleTranslation.symbolTable());
350 
351  // The verifier should not have allowed this.
352  assert((global || function) &&
353  "referencing an undefined global or function");
354 
355  moduleTranslation.mapValue(
356  addressOfOp.getResult(),
357  global ? moduleTranslation.lookupGlobal(global)
358  : moduleTranslation.lookupFunction(function.getName()));
359  return success();
360  }
361 
362  return failure();
363 }
