110 /// Given a list of MappedValues, cast them to the value kind implied by the
111 /// interface of the handle type, and dispatch to one of the callbacks.
112 static DiagnosedSilenceableFailure dispatchMappedValues(
113  Value handle, ArrayRef<transform::MappedValue> values,
114  function_ref<LogicalResult(ArrayRef<Operation *>)> operationsFn,
115  function_ref<LogicalResult(ArrayRef<transform::Param>)> paramsFn,
116  function_ref<LogicalResult(ValueRange)> valuesFn) {
117  if (handle.getType().isa<transform::TransformHandleTypeInterface>()) {
118  SmallVector<Operation *> operations;
119  operations.reserve(values.size());
120  for (transform::MappedValue value : values) {
121  if (auto *op = value.dyn_cast<Operation *>()) {
122  operations.push_back(op);
123  continue;
124  }
125  return emitSilenceableFailure(handle.getLoc())
126  << "wrong kind of value provided for top-level operation handle";
127  }
128  if (failed(operationsFn(operations)))
129  return DiagnosedSilenceableFailure::definiteFailure();
130  return DiagnosedSilenceableFailure::success();
131  }
132 
133  if (handle.getType().isa<transform::TransformValueHandleTypeInterface>()) {
134  SmallVector<Value> payloadValues;
135  payloadValues.reserve(values.size());
136  for (transform::MappedValue value : values) {
137  if (auto v = value.dyn_cast<Value>()) {
138  payloadValues.push_back(v);
139  continue;
140  }
141  return emitSilenceableFailure(handle.getLoc())
142  << "wrong kind of value provided for the top-level value handle";
143  }
144  if (failed(valuesFn(payloadValues)))
145  return DiagnosedSilenceableFailure::definiteFailure();
146  return DiagnosedSilenceableFailure::success();
147  }
148 
149  assert(handle.getType().isa<transform::TransformParamTypeInterface>() &&
150  "unsupported kind of block argument");
151  SmallVector<transform::Param> parameters;
152  parameters.reserve(values.size());
153  for (transform::MappedValue value : values) {
154  if (auto attr = value.dyn_cast<Attribute>()) {
155  parameters.push_back(attr);
156  continue;
157  }
158  return emitSilenceableFailure(handle.getLoc())
159  << "wrong kind of value provided for top-level parameter";
160  }
161  if (failed(paramsFn(parameters)))
162  return DiagnosedSilenceableFailure::definiteFailure();
163  return DiagnosedSilenceableFailure::success();
164 }

184 transform::TransformState::setPayloadOps(Value value,
185  ArrayRef<Operation *> targets) {
186  assert(value != kTopLevelValue &&
187  "attempting to reset the transformation root");
188  assert(value.getType().isa<TransformHandleTypeInterface>() &&
189  "wrong handle type");
190 
191  for (Operation *target : targets) {
192  if (target)
193  continue;
194  return emitError(value.getLoc())
195  << "attempting to assign a null payload op to this transform value";
196  }
197 
198  auto iface = value.getType().cast<TransformHandleTypeInterface>();
199  DiagnosedSilenceableFailure result =
200  iface.checkPayload(value.getLoc(), targets);
201  if (failed(result.checkAndReport()))
202  return failure();
203 
204  // Setting new payload for the value without cleaning it first is a misuse of
205  // the API, assert here.
206  SmallVector<Operation *> storedTargets(targets.begin(), targets.end());
207  Mappings &mappings = getMapping(value);
208  bool inserted =
209  mappings.direct.insert({value, std::move(storedTargets)}).second;
210  assert(inserted && "value is already associated with another list");
211  (void)inserted;
212 
213  for (Operation *op : targets)
214  mappings.reverse[op].push_back(value);
215 
216  return success();

328 LogicalResult
329 transform::TransformState::replacePayloadOp(Operation *op,
330  Operation *replacement) {
331  // Drop the mapping between the op and all handles that point to it. Don't
332  // care if there are on such handles.
333  SmallVector<Value> opHandles;
334  (void)getHandlesForPayloadOp(op, opHandles);
335  for (Value handle : opHandles) {
336  Mappings &mappings = getMapping(handle);
337  dropMappingEntry(mappings.reverse, op, handle);
338  }
339 
340 #ifndef NDEBUG
341  for (Value opResult : op->getResults()) {
342  SmallVector<Value> valueHandles;
343  (void)getHandlesForPayloadValue(opResult, valueHandles);
344  assert(valueHandles.empty() && "expected no mapping to old results");
345  }
346 #endif // NDEBUG
347 
348 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
349  if (options.getExpensiveChecksEnabled()) {
350  auto it = cachedNames.find(op);
351  assert(it != cachedNames.end() && "entry not found");
352  assert(it->second == op->getName() && "operation name mismatch");
353  cachedNames.erase(it);
354  }
355 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
356 
357  // TODO: consider invalidating the handles to nested objects here.
358 
359  // If replacing with null, that is erasing the mapping, drop the mapping
360  // between the handles and the IR objects and return.
361  if (!replacement) {
362  for (Value handle : opHandles) {
363  Mappings &mappings = getMapping(handle);
364  dropMappingEntry(mappings.direct, handle, op);
365  }
366  return success();
367  }
368 
369 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
370  if (options.getExpensiveChecksEnabled()) {
371  auto insertion = cachedNames.insert({replacement, replacement->getName()});
372  if (!insertion.second) {
373  assert(insertion.first->second == replacement->getName() &&
374  "operation is already cached with a different name");
375  }
376  }
377 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
378 
379  // Otherwise, replace the pointed-to object of all handles while preserving
380  // their relative order. First, replace the mapped operation if present.
381  for (Value handle : opHandles) {
382  Mappings &mappings = getMapping(handle);
383  auto it = mappings.direct.find(handle);
384  if (it == mappings.direct.end())
385  continue;
386 
387  SmallVector<Operation *, 2> &association = it->getSecond();
388  // Note that an operation may be associated with the handle more than once.
389  for (Operation *&mapped : association) {
390  if (mapped == op)
391  mapped = replacement;
392  }
393  mappings.reverse[replacement].push_back(handle);
394  }
395 
396  return success();
397 }

400 transform::TransformState::replacePayloadValue(Value value, Value replacement) {
401  SmallVector<Value> valueHandles;
402  (void)getHandlesForPayloadValue(value, valueHandles);
403 
404  for (Value handle : valueHandles) {

415  continue;

417  SmallVector<Value> &association = it->getSecond();

419  if (mapped == value)

421  }
422  mappings.reverseValues[replacement].push_back(handle);
423  }
424  }
425 
426  return success();

429 void transform::TransformState::recordOpHandleInvalidationOne(
430  OpOperand &consumingHandle, ArrayRef<Operation *> potentialAncestors,
431  Operation *payloadOp, Value otherHandle, Value throughValue) {
432  // If the op is associated with invalidated handle, skip the check as it
433  // may be reading invalid IR.
434  if (invalidatedHandles.count(otherHandle))
435  return;
436 
437  FULL_LDBG("--recordOpHandleInvalidationOne\n");
438  DEBUG_WITH_TYPE(
439  DEBUG_TYPE_FULL,
440  llvm::interleaveComma(potentialAncestors, DBGS() << "--ancestors: ",
441  [](Operation *op) { llvm::dbgs() << *op; });
442  llvm::dbgs() << "\n");
443  for (Operation *ancestor : potentialAncestors) {
444  // clang-format off
445  DEBUG_WITH_TYPE(DEBUG_TYPE_FULL,
446  { (DBGS() << "----handle one ancestor: " << *ancestor << "\n"); });
447  DEBUG_WITH_TYPE(DEBUG_TYPE_FULL,
448  { (DBGS() << "----of payload with name: "
449  << payloadOp->getName().getIdentifier() << "\n"); });
450  DEBUG_WITH_TYPE(DEBUG_TYPE_FULL,
451  { (DBGS() << "----of payload: " << *payloadOp << "\n"); });
452  // clang-format on
453  if (!ancestor->isAncestor(payloadOp))
454  continue;
455 
456  // Make sure the error-reporting lambda doesn't capture anything
457  // by-reference because it will go out of scope. Additionally, extract
458  // location from Payload IR ops because the ops themselves may be
459  // deleted before the lambda gets called.
460  Location ancestorLoc = ancestor->getLoc();
461  Location opLoc = payloadOp->getLoc();
462  Operation *owner = consumingHandle.getOwner();
463  unsigned operandNo = consumingHandle.getOperandNumber();
464  std::optional<Location> throughValueLoc =
465  throughValue ? std::make_optional(throughValue.getLoc()) : std::nullopt;
466  invalidatedHandles[otherHandle] = [ancestorLoc, opLoc, owner, operandNo,
467  otherHandle,
468  throughValueLoc](Location currentLoc) {
469  InFlightDiagnostic diag = emitError(currentLoc)
470  << "op uses a handle invalidated by a "
471  "previously executed transform op";
472  diag.attachNote(otherHandle.getLoc()) << "handle to invalidated ops";
473  diag.attachNote(owner->getLoc())
474  << "invalidated by this transform op that consumes its operand #"
475  << operandNo
476  << " and invalidates all handles to payload IR entities associated "
477  "with this operand and entities nested in them";
478  diag.attachNote(ancestorLoc) << "ancestor payload op";
479  diag.attachNote(opLoc) << "nested payload op";
480  if (throughValueLoc) {
481  diag.attachNote(*throughValueLoc)
482  << "consumed handle points to this payload value";
483  }
484  };
485  }

488 void transform::TransformState::recordValueHandleInvalidationByOpHandleOne(
489  OpOperand &consumingHandle, ArrayRef<Operation *> potentialAncestors,
490  Value payloadValue, Value valueHandle) {
491  // If the op is associated with invalidated handle, skip the check as it
492  // may be reading invalid IR.
493  if (invalidatedHandles.count(valueHandle))
494  return;
495 
496  for (Operation *ancestor : potentialAncestors) {
497  Operation *definingOp;
498  std::optional<unsigned> resultNo;
499  unsigned argumentNo, blockNo, regionNo;
500  if (auto opResult = payloadValue.dyn_cast<OpResult>()) {
501  definingOp = opResult.getOwner();
502  resultNo = opResult.getResultNumber();
503  } else {
504  auto arg = payloadValue.cast<BlockArgument>();
505  definingOp = arg.getParentBlock()->getParentOp();
506  argumentNo = arg.getArgNumber();
507  blockNo = std::distance(arg.getOwner()->getParent()->begin(),
508  arg.getOwner()->getIterator());
509  regionNo = arg.getOwner()->getParent()->getRegionNumber();
510  }
511  assert(definingOp && "expected the value to be defined by an op as result "
512  "or block argument");
513  if (!ancestor->isAncestor(definingOp))
514  continue;
515 
516  Operation *owner = consumingHandle.getOwner();
517  unsigned operandNo = consumingHandle.getOperandNumber();
518  Location ancestorLoc = ancestor->getLoc();
519  Location opLoc = definingOp->getLoc();
520  Location valueLoc = payloadValue.getLoc();
521  invalidatedHandles[valueHandle] =
522  [valueHandle, owner, operandNo, resultNo, argumentNo, blockNo, regionNo,
523  ancestorLoc, opLoc, valueLoc](Location currentLoc) {
524  InFlightDiagnostic diag = emitError(currentLoc)
525  << "op uses a handle invalidated by a "
526  "previously executed transform op";
527  diag.attachNote(valueHandle.getLoc()) << "invalidated handle";
528  diag.attachNote(owner->getLoc())
529  << "invalidated by this transform op that consumes its operand #"
530  << operandNo
531  << " and invalidates all handles to payload IR entities "
532  "associated with this operand and entities nested in them";
533  diag.attachNote(ancestorLoc)
534  << "ancestor op associated with the consumed handle";
535  if (resultNo) {
536  diag.attachNote(opLoc)
537  << "op defining the value as result #" << *resultNo;
538  } else {
539  diag.attachNote(opLoc)
540  << "op defining the value as block argument #" << argumentNo
541  << " of block #" << blockNo << " in region #" << regionNo;
542  }
543  diag.attachNote(valueLoc) << "payload value";
544  };
545  }

548 void transform::TransformState::recordOpHandleInvalidation(
549  OpOperand &handle, ArrayRef<Operation *> potentialAncestors,
550  Value throughValue) {
551  // Iterate over the mapping and invalidate aliasing handles. This is quite
552  // expensive and only necessary for error reporting in case of transform
553  // dialect misuse with dangling handles. Iteration over the handles is based
554  // on the assumption that the number of handles is significantly less than the
555  // number of IR objects (operations and values). Alternatively, we could walk
556  // the IR nested in each payload op associated with the given handle and look
557  // for handles associated with each operation and value.
558  for (const Mappings &mapping : llvm::make_second_range(mappings)) {
559  // Go over all op handle mappings and mark as invalidated any handle
560  // pointing to any of the payload ops associated with the given handle or
561  // any op nested in them.
562  for (const auto &[payloadOp, otherHandles] : mapping.reverse) {
563  for (Value otherHandle : otherHandles)
564  recordOpHandleInvalidationOne(handle, potentialAncestors, payloadOp,
565  otherHandle, throughValue);
566  }
567  // Go over all value handle mappings and mark as invalidated any handle
568  // pointing to any result of the payload op associated with the given handle
569  // or any op nested in them. Similarly invalidate handles to argument of
570  // blocks belonging to any region of any payload op associated with the
571  // given handle or any op nested in them.
572  for (const auto &[payloadValue, valueHandles] : mapping.reverseValues) {
573  for (Value valueHandle : valueHandles)
574  recordValueHandleInvalidationByOpHandleOne(handle, potentialAncestors,
575  payloadValue, valueHandle);
576  }
577  }
578 }

627  (DBGS() << "----iterate on handle: " << target.get() << "\n");
628  });
629  // If the operand uses an invalidated handle, report it.
630  auto it = invalidatedHandles.find(target.get());
631  if (!transform.allowsRepeatedHandleOperands() &&
632  it != invalidatedHandles.end()) {
633  FULL_LDBG("--End checkAndRecordHandleInvalidation -> FAILURE\n");
634  return it->getSecond()(transform->getLoc()), failure();
635  }
636 
637  // Invalidate handles pointing to the operations nested in the operation
638  // associated with the handle consumed by this operation.
639  auto consumesTarget = [&](const MemoryEffects::EffectInstance &effect) {
640  return isa<MemoryEffects::Free>(effect.getEffect()) &&
641  effect.getValue() == target.get();
642  };
643  if (llvm::any_of(effects, consumesTarget)) {
644  FULL_LDBG("----found consume effect -> SKIP\n");
645  if (target.get().getType().isa<TransformHandleTypeInterface>()) {
646  FULL_LDBG("----recordOpHandleInvalidation\n");
647  ArrayRef<Operation *> payloadOps = getPayloadOps(target.get());
648  recordOpHandleInvalidation(target, payloadOps);
649  } else if (target.get()
650  .getType()
651  .isa<TransformValueHandleTypeInterface>()) {
652  FULL_LDBG("----recordValueHandleInvalidation\n");
653  recordValueHandleInvalidation(target);
654  } else {
655  FULL_LDBG("----not a TransformHandle -> SKIP AND DROP ON THE FLOOR\n");
656  }
657  } else {
658  FULL_LDBG("----no consume effect -> SKIP\n");
659  }
660  }
661 
662  FULL_LDBG("--End checkAndRecordHandleInvalidation -> SUCCESS\n");
663  return success();
664 }

667 DiagnosedSilenceableFailure
668 checkRepeatedConsumptionInOperand(ArrayRef<T> payload,
669  transform::TransformOpInterface transform,
670  unsigned operandNumber) {
671  DenseSet<T> seen;
672  for (T p : payload) {
673  if (!seen.insert(p).second) {
674  DiagnosedSilenceableFailure diag =
675  transform.emitSilenceableError()
676  << "a handle passed as operand #" << operandNumber
677  << " and consumed by this operation points to a payload "
678  "entity more than once";
679  if constexpr (std::is_pointer_v<T>)
680  diag.attachNote(p->getLoc()) << "repeated target op";
681  else
682  diag.attachNote(p.getLoc()) << "repeated target value";
683  return diag;
684  }
685  }
686  return DiagnosedSilenceableFailure::success();
687 }

689 DiagnosedSilenceableFailure
690 transform::TransformState::applyTransform(TransformOpInterface transform) {
691  LLVM_DEBUG({
692  DBGS() << "applying: ";
693  transform->print(llvm::dbgs(), OpPrintingFlags().skipRegions());
694  llvm::dbgs() << "\n";
695  });
696  DEBUG_WITH_TYPE(DEBUG_TYPE_FULL,
697  DBGS() << "Top-level payload before application:\n"
698  << *getTopLevel() << "\n");
699  auto printOnFailureRAII = llvm::make_scope_exit([this] {
700  (void)this;
701  LLVM_DEBUG(DBGS() << "Failing Top-level payload:\n"; getTopLevel()->print(
702  llvm::dbgs(), mlir::OpPrintingFlags().printGenericOpForm()););
703  });
704  if (options.getExpensiveChecksEnabled()) {
705  FULL_LDBG("ExpensiveChecksEnabled\n");
706  if (failed(checkAndRecordHandleInvalidation(transform)))
707  return DiagnosedSilenceableFailure::definiteFailure();
708 
709  for (OpOperand &operand : transform->getOpOperands()) {
710  DEBUG_WITH_TYPE(DEBUG_TYPE_FULL, {
711  (DBGS() << "iterate on handle: " << operand.get() << "\n");
712  });
713  if (!isHandleConsumed(operand.get(), transform)) {
714  FULL_LDBG("--handle not consumed -> SKIP\n");
715  continue;
716  }
717  FULL_LDBG("--handle is consumed\n");
718 
719  Type operandType = operand.get().getType();
720  if (operandType.isa<TransformHandleTypeInterface>()) {
721  FULL_LDBG("--checkRepeatedConsumptionInOperand for Operation*\n");
722  DiagnosedSilenceableFailure check =
723  checkRepeatedConsumptionInOperand<Operation *>(
724  getPayloadOps(operand.get()), transform,
725  operand.getOperandNumber());
726  if (!check.succeeded()) {
727  FULL_LDBG("----FAILED\n");
728  return check;
729  }
730  } else if (operandType.isa<TransformValueHandleTypeInterface>()) {
731  FULL_LDBG("--checkRepeatedConsumptionInOperand For Value\n");
732  DiagnosedSilenceableFailure check =
733  checkRepeatedConsumptionInOperand<Value>(
734  getPayloadValues(operand.get()), transform,
735  operand.getOperandNumber());
736  if (!check.succeeded()) {
737  FULL_LDBG("----FAILED\n");
738  return check;
739  }
740  } else {
741  FULL_LDBG("--not a TransformHandle -> SKIP AND DROP ON THE FLOOR\n");
742  }
743  }
744 
745 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
746  // Cache Operation* -> OperationName mappings. These will be checked after
747  // the transform has been applied to detect incorrect memory side effects
748  // and missing op tracking.
749  for (Mappings &mapping : llvm::make_second_range(mappings)) {
750  for (Operation *op : llvm::make_first_range(mapping.reverse)) {
751  auto insertion = cachedNames.insert({op, op->getName()});
752  if (!insertion.second) {
753  if (insertion.first->second != op->getName()) {
754  // Operation is already in the cache, but with a different name.
755  DiagnosedDefiniteFailure diag =
756  emitDefiniteFailure(transform->getLoc())
757  << "expensive checks failure: operation mismatch, expected "
758  << insertion.first->second;
759  diag.attachNote(op->getLoc()) << "payload op: " << op->getName();
760  return diag;
761  }
762  }
763  }
764  }
765 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
766  }
767 
768  // Find which operands are consumed.
769  DenseSet<unsigned> consumedOperands;
770  auto memEffectInterface =
771  cast<MemoryEffectOpInterface>(transform.getOperation());
772  SmallVector<MemoryEffects::EffectInstance, 2> effects;
773  for (OpOperand &target : transform->getOpOperands()) {
774  effects.clear();
775  memEffectInterface.getEffectsOnValue(target.get(), effects);
776  if (llvm::any_of(effects, [](const MemoryEffects::EffectInstance &effect) {
777  return isa<transform::TransformMappingResource>(
778  effect.getResource()) &&
779  isa<MemoryEffects::Free>(effect.getEffect());
780  })) {
781  consumedOperands.insert(target.getOperandNumber());
782  }
783  }
784 
785  // Remember the results of the payload ops associated with the consumed
786  // op handles or the ops defining the value handles so we can drop the
787  // association with them later. This must happen here because the
788  // transformation may destroy or mutate them so we cannot traverse the payload
789  // IR after that.
790  SmallVector<Value> origOpFlatResults;
791  SmallVector<Operation *> origAssociatedOps;
792 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
793  DenseSet<Operation *> consumedPayloadOps;
794 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
795  for (unsigned index : consumedOperands) {
796  Value operand = transform->getOperand(index);
797  if (operand.getType().isa<TransformHandleTypeInterface>()) {
798  for (Operation *payloadOp : getPayloadOps(operand)) {
799  llvm::append_range(origOpFlatResults, payloadOp->getResults());
800 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
801  if (options.getExpensiveChecksEnabled()) {
802  // Store all consumed payload ops (and their nested ops) in a set for
803  // extra error checking.
804  payloadOp->walk(
805  [&](Operation *op) { consumedPayloadOps.insert(op); });
806  }
807 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
808  }
809  continue;
810  }
811  if (operand.getType().isa<TransformValueHandleTypeInterface>()) {
812  for (Value payloadValue : getPayloadValues(operand)) {
813  if (payloadValue.isa<OpResult>()) {
814  origAssociatedOps.push_back(payloadValue.getDefiningOp());
815  continue;
816  }
817  llvm::append_range(
818  origAssociatedOps,
819  llvm::map_range(*payloadValue.cast<BlockArgument>().getOwner(),
820  [](Operation &op) { return &op; }));
821  }
822  continue;
823  }
824  DiagnosedDefiniteFailure diag =
825  emitDefiniteFailure(transform->getLoc())
826  << "unexpectedly consumed a value that is not a handle as operand #"
827  << index;
828  diag.attachNote(operand.getLoc())
829  << "value defined here with type " << operand.getType();
830  return diag;
831  }
832 
833  // Compute the result but do not short-circuit the silenceable failure case as
834  // we still want the handles to propagate properly so the "suppress" mode can
835  // proceed on a best effort basis.
836  transform::TransformResults results(transform->getNumResults());
837  DiagnosedSilenceableFailure result(transform.apply(results, *this));
838  if (result.isDefiniteFailure())
839  return result;
840 
841  // If a silenceable failure was produced, some results may be unset, set them
842  // to empty lists.
843  if (result.isSilenceableFailure())
844  results.setRemainingToEmpty(transform);
845 
846  // Remove the mapping for the operand if it is consumed by the operation. This
847  // allows us to catch use-after-free with assertions later on.
848  for (unsigned index : consumedOperands) {
849  Value operand = transform->getOperand(index);
850  if (operand.getType().isa<TransformHandleTypeInterface>()) {
851  forgetMapping(operand, origOpFlatResults);
852  } else if (operand.getType().isa<TransformValueHandleTypeInterface>()) {
853  forgetValueMapping(operand, origAssociatedOps);
854  }
855  }
856 
857 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
858  if (options.getExpensiveChecksEnabled()) {
859  // Remove erased ops from the transform state.
860  for (Operation *op : consumedPayloadOps) {
861  // This payload op was consumed but it may still be mapped to one or
862  // multiple handles. Forget all handles that are mapped to the op, so that
863  // there are no dangling pointers in the transform dialect state. This is
864  // necessary so that the `cachedNames`-based checks work correctly.
865  //
866  // Note: Dangling pointers to erased payload ops are allowed if the
867  // corresponding handles are not used anymore. There is another
868  // "expensive-check" that looks for future uses of dangling payload op
869  // pointers (through arbitrary handles). Removing handles to erased ops
870  // does not interfere with the other expensive checks: handle invalidation
871  // happens earlier and keeps track of invalidated handles with
872  // pre-generated error messages, so we do not need the association to
873  // still be there when the invalidated handle is accessed.
874  SmallVector<Value> handles;
875  (void)getHandlesForPayloadOp(op, handles);
876  for (Value handle : handles)
877  forgetMapping(handle, /*origOpFlatResults=*/ValueRange());
878  cachedNames.erase(op);
879  }
880 
881  // Check cached operation names.
882  for (Mappings &mapping : llvm::make_second_range(mappings)) {
883  for (Operation *op : llvm::make_first_range(mapping.reverse)) {
884  // Make sure that the name of the op has not changed. If it has changed,
885  // the op was removed and a new op was allocated at the same memory
886  // location. This means that we are missing op tracking somewhere.
887  auto cacheIt = cachedNames.find(op);
888  if (cacheIt == cachedNames.end()) {
889  DiagnosedDefiniteFailure diag =
890  emitDefiniteFailure(transform->getLoc())
891  << "expensive checks failure: operation not found in cache";
892  diag.attachNote(op->getLoc()) << "payload op";
893  return diag;
894  }
895  // If the `getName` call (or the above `attachNote`) is crashing, we
896  // have a dangling pointer. This usually means that an op was erased but
897  // the transform dialect was not made aware of that; e.g., missing
898  // "consumesHandle" or rewriter usage.
899  if (cacheIt->second != op->getName()) {
900  DiagnosedDefiniteFailure diag =
901  emitDefiniteFailure(transform->getLoc())
902  << "expensive checks failure: operation mismatch, expected "
903  << cacheIt->second;
904  diag.attachNote(op->getLoc()) << "payload op: " << op->getName();
905  return diag;
906  }
907  }
908  }
909  }
910 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
911 
912  if (failed(updateStateFromResults(results, transform->getResults())))
913  return DiagnosedSilenceableFailure::definiteFailure();
914 
915  printOnFailureRAII.release();
916  DEBUG_WITH_TYPE(DEBUG_PRINT_AFTER_ALL, {
917  DBGS() << "Top-level payload:\n";
918  getTopLevel()->print(llvm::dbgs());
919  });
920  return result;
921 }

946  }
947  }
948  }
949  return success();

1109 LogicalResult
1110 transform::detail::checkApplyToOne(Operation *transformOp,
1111  Location payloadOpLoc,
1112  const ApplyToEachResultList &partialResult) {
1113  Location transformOpLoc = transformOp->getLoc();
1114  StringRef transformOpName = transformOp->getName().getStringRef();
1115  unsigned expectedNumResults = transformOp->getNumResults();
1116 
1117  // Reuse the emission of the diagnostic note.
1118  auto emitDiag = [&]() {
1119  auto diag = mlir::emitError(transformOpLoc);
1120  diag.attachNote(payloadOpLoc) << "when applied to this op";
1121  return diag;
1122  };
1123 
1124  if (partialResult.size() != expectedNumResults) {
1125  auto diag = emitDiag() << "application of " << transformOpName
1126  << " expected to produce " << expectedNumResults
1127  << " results (actually produced "
1128  << partialResult.size() << ").";
1129  diag.attachNote(transformOpLoc)
1130  << "if you need variadic results, consider a generic `apply` "
1131  << "instead of the specialized `applyToOne`.";
1132  return failure();
1133  }
1134 
1135  // Check that the right kind of value was produced.
1136  for (const auto &[ptr, res] :
1137  llvm::zip(partialResult, transformOp->getResults())) {
1138  if (ptr.isNull())
1139  continue;
1140  if (res.getType().template isa<TransformHandleTypeInterface>() &&
1141  !ptr.is<Operation *>()) {
1142  return emitDiag() << "application of " << transformOpName
1143  << " expected to produce an Operation * for result #"
1144  << res.getResultNumber();
1145  }
1146  if (res.getType().template isa<TransformParamTypeInterface>() &&
1147  !ptr.is<Attribute>()) {
1148  return emitDiag() << "application of " << transformOpName
1149  << " expected to produce an Attribute for result #"
1150  << res.getResultNumber();
1151  }
1152  if (res.getType().template isa<TransformValueHandleTypeInterface>() &&
1153  !ptr.is<Value>()) {
1154  return emitDiag() << "application of " << transformOpName
1155  << " expected to produce a Value for result #"
1156  << res.getResultNumber();
1157  }
1158  }
1159  return success();
1160 }

1217 void transform::detail::forwardTerminatorOperands(
1218  Block *block, transform::TransformState &state,
1219  transform::TransformResults &results) {
1220  for (auto &&[terminatorOperand, result] :
1221  llvm::zip(block->getTerminator()->getOperands(),
1222  block->getParentOp()->getOpResults())) {
1223  if (result.getType().isa<transform::TransformHandleTypeInterface>()) {
1224  results.set(result, state.getPayloadOps(terminatorOperand));
1225  } else if (result.getType()
1226  .isa<transform::TransformValueHandleTypeInterface>()) {
