49 /// Pad the `opOperand` in the `paddingDimensions` using the padding value and
50 /// the nofold flag found in `paddingValues` and `packPaddings`, respectively.
51 /// Exit early and return the `opOperand` value if the shape dimensions that
52 /// match `paddingDimensions` have a static size and the nofold flag is not set.
53 /// Otherwise, try to pad the shape dimensions that match the iterator
54 /// dimensions `paddingDimensions` and return the tensor::PadOp result if
55 /// padding succeeds or failure otherwise.
56 static FailureOr<Value> padOperandToSmallestStaticBoundingBox(
57  RewriterBase &rewriter, linalg::LinalgOp opToPad, OpOperand *opOperand,
58  ArrayRef<int64_t> paddingDimensions, ArrayRef<Attribute> paddingValues,
59  ArrayRef<bool> packPaddings) {
60  AffineMap indexingMap = opToPad.getMatchingIndexingMap(opOperand);
61  ArrayRef<int64_t> shape = opToPad.getShape(opOperand);
62 
63  // Collect the shape dimension that are a function of the `paddingDimensions`.
64  llvm::SmallDenseSet<int64_t> shapeDimsToPad;
65  for (int64_t dim : paddingDimensions)
66  for (const auto &en : enumerate(indexingMap.getResults()))
67  if (en.value().isFunctionOfDim(dim))
68  shapeDimsToPad.insert(en.index());
69 
70  // Return the unpadded operand if padding to a static shape is not needed and
71  // if the nofold flag is not set.
72  bool nofold = opOperand->getOperandNumber() < packPaddings.size()
73  ? packPaddings[opOperand->getOperandNumber()]
74  : false;
75  bool hasStaticShape = llvm::none_of(shapeDimsToPad, [&](int64_t dim) {
76  return ShapedType::isDynamic(shape[dim]);
77  });
78  if (!nofold && hasStaticShape)
79  return opOperand->get();
80 
81  // Fail if `paddingValues` specifies no padding value.
82  if (opOperand->getOperandNumber() >= paddingValues.size()) {
83  return rewriter.notifyMatchFailure(opToPad, "--no padding value specified");
84  }
85  Attribute paddingAttr = paddingValues[opOperand->getOperandNumber()];
86  Value paddingValue = rewriter.create<arith::ConstantOp>(
87  opToPad.getLoc(), cast<TypedAttr>(paddingAttr));
88 
89  // Follow the use-def chain if `currOpOperand` is defined by a LinalgOp.
90  OpOperand *currOpOperand = opOperand;
91  while (auto linalgOp = currOpOperand->get().getDefiningOp<LinalgOp>()) {
92  OpResult result = currOpOperand->get().cast<OpResult>();
93  currOpOperand = linalgOp.getDpsInitOperand(result.getResultNumber());
94  }
95 
96  // Fail if `currOpOperand` is not defined by an ExtractSliceOp or EmptyOp.
97  auto sliceOp = currOpOperand->get().getDefiningOp<tensor::ExtractSliceOp>();
98  auto emptyOp = currOpOperand->get().getDefiningOp<tensor::EmptyOp>();
99  if (!sliceOp && !emptyOp) {
100  // TODO: may want to add support for going through loop iter args.
101  // This is not strictly necessary as we can pad before hoisting but it would
102  // make the system more resilient to minor transformation reordering.
103  LLVM_DEBUG(DBGS() << "--not defined by an extractSlice or emptyOp\n");
104  return rewriter.notifyMatchFailure(
105  opToPad, "not defined by an extractSlice or emptyOp");
106  }
107 
108  llvm::SmallBitVector droppedDims;
109  SmallVector<OpFoldResult> mixedSizes;
110  if (sliceOp) {
111  // Compute the dropped dimensions if `sliceOp` is rank-reducing.
112  droppedDims = sliceOp.getDroppedDims();
113  mixedSizes = sliceOp.getMixedSizes();
114  }
115  if (emptyOp) {
116  mixedSizes = emptyOp.getMixedSizes();
117  droppedDims.resize(mixedSizes.size());
118  }
119  LLVM_DEBUG(llvm::interleaveComma(mixedSizes, DBGS() << "--mixedSizes: ");
120  llvm::dbgs() << "\n");
121 
122  // Upper bound the sizes to obtain a static bounding box.
123  SmallVector<int64_t> paddedShape(shape.begin(), shape.end());
124  int64_t shapeIdx = 0;
125  for (const auto &en : enumerate(mixedSizes)) {
126  LLVM_DEBUG(DBGS() << "----mixedSizes: " << en.value() << "\n");
127  // Skip dropped dimensions.
128  if (droppedDims.test(en.index())) {
129  LLVM_DEBUG(DBGS() << "------dim is dropped, SKIP\n");
130  continue;
131  }
132  // Skip dimensions that do not require padding.
133  if (!shapeDimsToPad.contains(shapeIdx)) {
134  shapeIdx++;
135  LLVM_DEBUG(DBGS() << "------dim does not require padding, SKIP\n");
136  continue;
137  }
138  // If the size is an attribute add it directly to `paddedShape`.
139  if (en.value().is<Attribute>()) {
140  paddedShape[shapeIdx++] =
141  en.value().get<Attribute>().dyn_cast<IntegerAttr>().getInt();
142  LLVM_DEBUG(
143  DBGS() << "------dim is an attr, add it to padded shape, SKIP\n");
144  continue;
145  }
146  // Otherwise, try to compute a constant upper bound for the size value.
147  FailureOr<int64_t> upperBound =
148  ValueBoundsConstraintSet::computeConstantBound(
149  presburger::BoundType::UB, en.value().get<Value>(),
150  /*dim=*/std::nullopt, /*stopCondition=*/nullptr, /*closedUB=*/true);
151  if (failed(upperBound)) {
152  LLVM_DEBUG(DBGS() << "--count not compute a bounding box for padding");
153  return rewriter.notifyMatchFailure(
154  opToPad, "count not compute a bounding box for padding");
155  }
156  paddedShape[shapeIdx++] = *upperBound;
157  }
158  assert(shapeIdx == static_cast<int64_t>(shape.size()) &&
159  "expect the dynamic and static ranks to match");
160 
161  // Pad the operand to the bounding box defined by `paddedShape`.
162  auto paddedTensorType = RankedTensorType::get(
163  paddedShape, getElementTypeOrSelf(opOperand->get()));
164  LLVM_DEBUG(DBGS() << "--SUCCESS, makeComposedPadHighOp with type: "
165  << paddedTensorType);
166  return makeComposedPadHighOp(rewriter, opToPad->getLoc(), paddedTensorType,
167  opOperand->get(), paddingValue, nofold);
168 }

183 FailureOr<SmallVector<Value>>
184 linalg::rewriteAsPaddedOp(RewriterBase &rewriter, LinalgOp opToPad,
185  ArrayRef<int64_t> paddingDimensions,
186  ArrayRef<Attribute> paddingValues,
187  ArrayRef<bool> packPaddings, LinalgOp &paddedOp) {
188  LLVM_DEBUG(DBGS() << "Start rewriteAsPaddedOp : " << opToPad << "\n");
189  Location loc = opToPad->getLoc();
190 
191  // TODO: there are cases where we may still want to pad to larger sizes.
192  if (!opToPad.hasTensorSemantics())
193  return rewriter.notifyMatchFailure(opToPad,
194  "expected operation on tensors");
195 
196  OpBuilder::InsertionGuard g(rewriter);
197  // Set IP after op because we also take the dims of the original output.
198  rewriter.setInsertionPointAfter(opToPad);
199 
200  // Make a copy of the shaped operands and update it.
201  SmallVector<Value> newOperands;
202  newOperands.reserve(opToPad->getNumOperands());
203  for (OpOperand &opOperand : opToPad->getOpOperands()) {
204  FailureOr<Value> paddedOperand = padOperandToSmallestStaticBoundingBox(
205  rewriter, opToPad, &opOperand, paddingDimensions, paddingValues,
206  packPaddings);
207  // Exit if `paddingDimensions` cannot be bounded statically.
208  if (failed(paddedOperand)) {
209  LLVM_DEBUG(DBGS() << "--operand cannot be bound statically : "
210  << opOperand.get() << " -> FAIL\n");
211  return rewriter.notifyMatchFailure(opToPad,
212  "operand cannot be bound statically");
213  }
214  newOperands.push_back(*paddedOperand);
215  }
216 
217  ReifiedRankedShapedTypeDims reifiedResultShapes;
218  if (failed(reifyResultShapes(rewriter, opToPad, reifiedResultShapes))) {
219  LLVM_DEBUG(DBGS() << "--failed to reify result shapes -> FAIL\n");
220  return rewriter.notifyMatchFailure(opToPad,
221  "failed to reify result shapes");
222  }
223  assert(reifiedResultShapes.size() == opToPad->getNumResults() &&
224  "expected same number of results");
225 
226  // Clone `opToPad` to operate on the statically padded shapes.
227  auto resultTensorTypes =
228  ValueRange(newOperands).take_back(opToPad.getNumDpsInits()).getTypes();
229  // clone **should** properly notify the rewriter.
230  paddedOp = clone(rewriter, opToPad, resultTensorTypes, newOperands);
231  LLVM_DEBUG(DBGS() << "--cloned padded op: " << paddedOp << "\n");
232 
233  // Recover the slice out of the new static results. This keeps the original
234  // linalg op around because it uses the dims of the original results.
235  SmallVector<Value> paddedSubtensorResults;
236  paddedSubtensorResults.reserve(opToPad->getNumResults());
237  for (const auto &en : llvm::enumerate(paddedOp->getResults())) {
238  Value paddedResult = en.value();
239  int64_t resultNumber = en.index();
240  int64_t rank = paddedResult.getType().cast<RankedTensorType>().getRank();
241  SmallVector<OpFoldResult> offsets(rank, rewriter.getIndexAttr(0));
242  SmallVector<OpFoldResult> sizes;
243  SmallVector<OpFoldResult> strides(rank, rewriter.getIndexAttr(1));
244  paddedSubtensorResults.push_back(rewriter.create<tensor::ExtractSliceOp>(
245  loc, paddedResult, offsets, reifiedResultShapes[resultNumber],
246  strides));
247  }
248  return paddedSubtensorResults;
249 }

284 FailureOr<LinalgOp>
285 mlir::linalg::padAndHoistLinalgOp(RewriterBase &rewriter, LinalgOp linalgOp,
286  LinalgPaddingOptions options) {
287  if (!linalgOp.hasTensorSemantics())
288  return rewriter.notifyMatchFailure(
289  linalgOp, "only applies to Linalg ops with tensor semantics");
290 
291  // Pad the operation.
292  LinalgOp paddedOp;
293  FailureOr<SmallVector<Value>> newResults =
294  rewriteAsPaddedOp(rewriter, linalgOp, options.paddingDimensions,
295  options.paddingValues, options.packPaddings, paddedOp);
296  if (failed(newResults))
297  return rewriter.notifyMatchFailure(linalgOp,
298  "failed to rewrite as a padded op");
299 
300  // Hoist the padding.
301  for (const auto &en : enumerate(options.hoistPaddings)) {
302  if (static_cast<int64_t>(en.index()) >= paddedOp->getNumOperands())
303  break;
304  OpOperand &opOperand = paddedOp->getOpOperand(en.index());
305  auto padOp = opOperand.get().getDefiningOp<tensor::PadOp>();
306  if (!padOp || en.value() == 0) {
307  (void)rewriter.notifyMatchFailure(linalgOp, "not a tensor.pad -- skip");
308  continue;
309  }
310 
311  // Fail hoisting if the operand shape is not fully static.
312  if (llvm::any_of(paddedOp.getShape(&opOperand), ShapedType::isDynamic)) {
313  (void)rewriter.notifyMatchFailure(linalgOp,
314  "non static padding shape -- skip");
315  continue;
316  }
317 
318  tensor::PadOp hoistedOp;
319  SmallVector<GenericOp> transposeOps;
320  SmallVector<int64_t> transposeVector =
321  en.index() < options.transposePaddings.size()
322  ? options.transposePaddings[en.index()]
323  : SmallVector<int64_t>{};
324 
325  FailureOr<Value> newResult = hoistPaddingOnTensors(
326  padOp, en.value(), transposeVector, hoistedOp, transposeOps);
327  if (failed(newResult)) {
328  (void)rewriter.notifyMatchFailure(linalgOp,
329  "failed to apply hoistPadding");
330  continue;
331  }
332  rewriter.replaceOp(padOp, *newResult);
333  }
334 
335  // Replace the original operation to pad.
336  rewriter.replaceOp(linalgOp, *newResults);
337 
338  return paddedOp;
339 }

373 /// Perform one step of packing of a LinalgOp's metadata along `dim` into the
374 /// `newDim` at `iteratorTypes.size()` by:
375 /// 1. Appending `iteratorTypes[newDim]`, equal to `iteratorTypes[dim]`.
376 /// 2. Appending a `newDim` to the domain of every indexing map.
377 /// 3. For each operand (i.e. for each map in `indexingMaps`), perform packing
378 /// by potentially adding a `newDim` result to `map`.
379 /// The preserved invariant is that `iteratorTypes.size()` is always equal to
380 /// `map.getNumDims()` for every map in `indexingMaps`.
381 ///
382 /// Update `indexingMaps` and `iteratorTypes` inplace as one step of the update.
383 /// Return a vector that records the optional packing for each operand.
384 /// Return failure if the packed indexing cannot be represented with a LinalgOp.
385 ///
386 /// Further details:
387 /// ================
388 /// The current implementation of packing (i.e. data tiling) consists of
389 /// rewriting a linearized strip-mined form into a higher-dimensional access.
390 /// e.g. consider an access `A[I][f(j, k, l)]` and packing by 4; we rewrite
391 /// `I` into `4 * i + ii`, where `0 <= ii < 4`.
392 /// The access is further rewritten as `A[i][f(j, k, l)][ii]`.
393 ///
394 /// This rewrite into higher dimensional access is not possible for general
395 /// AffineExpr in Linalg atm, it is restricted to an AffineDimExpr:
396 /// e.g. consider an access `A[I + J][f(j, k, l)]` and packing by 4; we
397 /// rewrite `I + J` into `4 * i + ii + J`, where `0 <= ii < 4`.
398 /// The rewrite of the access would be a form not representable in Linalg:
399 /// `A[i + (ii + J) / 4][f(j, k, l)][(ii + J) % 4]`.
400 /// Note however that as `J` and `ii` iterate, the accesses do not have a
401 /// particular alignment, so packing does not achieve alignment in this case
402 ///
403 /// In the future, we may want to consider a mixed-form that allows some
404 /// alignment in the presence of multiple accesses:
405 /// `A[I][f(j, k, l)]` and `B[I + J][f(j, k, l)]`
406 /// And would rewrite accesses as:
407 /// `A[i][f(j, k, l)][ii]` and `B[4 * i + ii + J][f(j, k, l)]`
408 static FailureOr<SmallVector<std::optional<int64_t>>>
409 packLinalgMetadataOnce(SmallVectorImpl<AffineMap> &indexingMaps,
410  SmallVectorImpl<utils::IteratorType> &iteratorTypes,
411  int64_t dim) {
412  int64_t newDim = iteratorTypes.size();
413  iteratorTypes.push_back(iteratorTypes[dim]);
414 
415  SmallVector<std::optional<int64_t>> packedDimPerIndexingMap(
416  indexingMaps.size(), std::nullopt);
417  SmallVector<AffineMap> newMaps;
418  for (int64_t operandIdx = 0, e = indexingMaps.size(); operandIdx < e;
419  ++operandIdx) {
420  AffineMap map = indexingMaps[operandIdx];
421 
422  // Add the `newDim` to map whatever the case.
423  assert(map.getNumDims() == newDim && "num dims invariant violation");
424  map = map.shiftDims(1, newDim);
425 
426  // Get the at-most-1 index of the result that is a function of `dim`.
427  // If we can find one, we insert `AffineDimExpr(newDim)` to the map, which
428  // logically chunks dimension `dim` into `K * dim + newDim`, where the
429  // packing factor `K` is specified separately.
430  assert(hasAtMostOneResultFunctionOfDim(map, dim) &&
431  "num results invariant violation");
432  auto maybeOperandDimensionToPack = getFirstResultIndexFunctionOf(map, dim);
433  if (!maybeOperandDimensionToPack.has_value()) {
434  newMaps.push_back(map);
435  continue;
436  }
437 
438  // We can only pack AffineDimExpr atm.
439  if (!map.getResult(maybeOperandDimensionToPack.value())
440  .isa<AffineDimExpr>())
441  return failure();
442 
443  // Add `newDim` to the results of the map.
444  map = map.insertResult(Builder(map.getContext()).getAffineDimExpr(newDim),
445  map.getNumResults());
446  newMaps.push_back(map);
447 
448  // Record the that `operandIdx` is packed.
449  packedDimPerIndexingMap[operandIdx] = maybeOperandDimensionToPack;
450  }
451  indexingMaps = newMaps;
452 
453  return packedDimPerIndexingMap;
454 }

480 FailureOr<LowerPackResult> linalg::lowerPack(RewriterBase &rewriter,
481  tensor::PackOp packOp) {
482  // 1. Filter out NYI cases.
483  auto packedTensorType =
484  packOp->getResultTypes().front().cast<RankedTensorType>();
485  if (!packedTensorType.hasStaticShape()) {
486  return rewriter.notifyMatchFailure(
487  packOp,
488  "non-static shape NYI, needs a more powerful tensor.expand_shape op");
489  }
490 
491  Location loc = packOp->getLoc();
492  OpBuilder::InsertionGuard g(rewriter);
493  rewriter.setInsertionPoint(packOp);
494 
495  // 2. Compute the permutation vector to shuffle packed shape into the shape
496  // before any outer or inner permutations have been applied. The permutation
497  // can be obtained from two permutations:
498  // a) Compute the permutation vector to move the last `numPackedDims` into
499  // the `innerPosDims` of a shape of rank `packedRank`.
500  // b) Compute the permutation vector to move outer dims if the pack op
501  // has outer_dims_perm.
502  // Apply (b) permutation on (a) permutation to get the final permutation.
503  int64_t numPackedDims = packOp.getInnerDimsPos().size();
504  int64_t packedRank = packedTensorType.getRank();
505  auto lastDims = llvm::to_vector(
506  llvm::seq<int64_t>(packedRank - numPackedDims, packedRank));
507  PackingMetadata packingMetadata = computePackingMetadata(
508  packedTensorType.getRank(), packOp.getInnerDimsPos());
509  SmallVector<int64_t> innerPositionsPerm = computePermutationVector(
510  packedRank, lastDims, packingMetadata.insertPositions);
511 
512  SmallVector<int64_t> outerPos = packingMetadata.outerPositions;
513  ArrayRef<int64_t> outerPerm = packOp.getOuterDimsPerm();
514  if (!outerPerm.empty())
515  applyPermutationToVector(outerPos, outerPerm);
516  SmallVector<int64_t> outerPositionPerm = computePermutationVector(
517  packedRank, packingMetadata.outerPositions, outerPos);
518 
519  SmallVector<int64_t> packedToStripMinedShapePerm = innerPositionsPerm;
520  applyPermutationToVector(packedToStripMinedShapePerm, outerPositionPerm);
521 
522  // 3. Compute the stripMinedShape: this is the packed shape before any outer
523  // or inner permutations have been applied.
524  SmallVector<int64_t> stripMinedShape(packedTensorType.getShape());
525  applyPermutationToVector(stripMinedShape, packedToStripMinedShapePerm);
526 
527  // 4. Pad the source of packOp to a shape we can expand into stripMinedShape.
528  RankedTensorType collapsed = tensor::CollapseShapeOp::inferCollapsedType(
529  RankedTensorType::Builder(packedTensorType).setShape(stripMinedShape),
530  packingMetadata.reassociations);
531  Value paddingValue = packOp.getPaddingValue();
532  if (!paddingValue) {
533  paddingValue = rewriter.create<arith::ConstantOp>(
534  loc, rewriter.getZeroAttr(getElementTypeOrSelf(collapsed)));
535  }
536  auto padOp =
537  tensor::createPadHighOp(collapsed, packOp.getSource(), paddingValue,
538  /*nofold=*/false, loc, rewriter);
539 
540  LLVM_DEBUG(
541  DBGSNL(); DBGSNL(); llvm::interleaveComma(packingMetadata.insertPositions,
542  DBGS() << "insertPositions: ");
543  DBGSNL(); llvm::interleaveComma(packingMetadata.outerPositions,
544  DBGS() << "outerPositions: ");
545  DBGSNL(); llvm::interleaveComma(packedTensorType.getShape(),
546  DBGS() << "packedShape: ");
547  DBGSNL();
548  llvm::interleaveComma(outerPositionPerm, DBGS() << "outerPositionPerm: ");
549  DBGSNL(); llvm::interleaveComma(innerPositionsPerm,
550  DBGS() << "innerPositionsPerm: ");
551  DBGSNL();
552  llvm::interleaveComma(packedToStripMinedShapePerm,
553  DBGS() << "packedToStripMinedShapePerm: ");
554  DBGSNL(); llvm::interleaveComma(
555  packingMetadata.reassociations, DBGS() << "reassociations: ",
556  [&](ReassociationIndices ri) {
557  llvm::interleaveComma(ri, llvm::dbgs() << "|");
558  });
559  DBGSNL();
560  llvm::interleaveComma(stripMinedShape, DBGS() << "stripMinedShape: ");
561  DBGSNL(); DBGS() << "collapsed type: " << collapsed; DBGSNL(););
562 
563  if (packOp.isLikePad()) {
564  // This pack is just a plain pad.
565  // Just insert the pad in the higher ranked tensor.
566  auto emptyOp =
567  rewriter.create<tensor::EmptyOp>(loc, packedTensorType, ValueRange{});
568  // Offsets.
569  SmallVector<OpFoldResult> zeros(packedRank, rewriter.getIndexAttr(0));
570  // Strides.
571  SmallVector<OpFoldResult> ones(packedRank, rewriter.getIndexAttr(1));
572  SmallVector<OpFoldResult> sizes =
573  getMixedDimensions(rewriter, loc, packOp.getDest());
574 
575  auto insertSliceOp = rewriter.create<tensor::InsertSliceOp>(
576  loc, /*source=*/padOp, /*dest=*/emptyOp,
577  /*offsets=*/zeros, sizes,
578  /*strides=*/ones);
579 
580  LLVM_DEBUG(DBGS() << "insert_slice op: " << insertSliceOp; DBGSNL(););
581 
582  rewriter.replaceOp(packOp, insertSliceOp->getResults());
583 
584  return LowerPackResult{padOp, /*reshapeOp=*/nullptr,
585  /*transposeOp=*/nullptr};
586  }
587  // 5. Expand from the padded result to the stripMinedShape.
588  auto reshapeOp = rewriter.create<tensor::ExpandShapeOp>(
589  loc,
590  RankedTensorType::Builder(packedTensorType).setShape(stripMinedShape),
591  padOp.getResult(), packingMetadata.reassociations);
592 
593  // 6. Transpose stripMinedShape to packedShape.
594  SmallVector<int64_t> transpPerm =
595  invertPermutationVector(packedToStripMinedShapePerm);
596  auto transposeOp = rewriter.create<linalg::TransposeOp>(
597  loc, reshapeOp.getResult(), packOp.getDest(), transpPerm);
598 
599  LLVM_DEBUG(DBGSNL(); DBGSNL(); DBGSNL();
600  DBGS() << "reshape op: " << reshapeOp; DBGSNL();
601  llvm::interleaveComma(transpPerm, DBGS() << "transpPerm: ");
602  DBGSNL(); DBGS() << "transpose op: " << transposeOp; DBGSNL(););
603 
604  // 7. Replace packOp by transposeOp.
605  rewriter.replaceOp(packOp, transposeOp->getResults());
606 
607  return LowerPackResult{padOp, reshapeOp, transposeOp};
608 }

610 FailureOr<LowerUnPackOpResult> linalg::lowerUnPack(RewriterBase &rewriter,
611  tensor::UnPackOp unPackOp) {
612  // 1. Filter out NYI cases.
613  if (!unPackOp.getOuterDimsPerm().empty())
614  return rewriter.notifyMatchFailure(unPackOp, "outer dims perm NYI");
615 
616  RankedTensorType packedTensorType = unPackOp.getSourceType();
617  if (!packedTensorType.hasStaticShape()) {
618  return rewriter.notifyMatchFailure(
619  unPackOp,
620  "non-static shape NYI, needs a more powerful tensor.expand_shape op");
621  }
622 
623  Location loc = unPackOp->getLoc();
624  OpBuilder::InsertionGuard g(rewriter);
625  rewriter.setInsertionPoint(unPackOp);
626 
627  int64_t packedRank = packedTensorType.getRank();
628 
629  OpFoldResult zero = rewriter.getIndexAttr(0), one = rewriter.getIndexAttr(1);
630  auto destTensorType = unPackOp.getDest().getType().cast<RankedTensorType>();
631  if (unPackOp.isLikeUnPad()) {
632  // This unpack is just a plain unpad.
633  // Just extract the slice from the higher ranked tensor.
634  ArrayRef<int64_t> destShape = destTensorType.getShape();
635  // The inner dimensions stay the same as the destination tensor, but the
636  // outer ones are additional 1s.
637  SmallVector<OpFoldResult> sizes(packedRank - destShape.size(), one);
638  sizes.append(getMixedDimensions(rewriter, loc, unPackOp.getDest()));
639 
640  auto extractSliceOp = rewriter.create<tensor::ExtractSliceOp>(
641  loc, destTensorType, unPackOp.getSource(),
642  SmallVector<OpFoldResult>(packedRank, zero), sizes,
643  SmallVector<OpFoldResult>(packedRank, one));
644 
645  rewriter.replaceOp(unPackOp, extractSliceOp->getResults());
646 
647  return LowerUnPackOpResult{/*emptyOp=*/nullptr, /*transposeOp=*/nullptr,
648  /*reshapeOp=*/nullptr, extractSliceOp};
649  }
650  // 2. Compute the permutation vector to move the last `numPackedDims` into
651  // the `innerPosDims` of a shape of rank `packedRank`.
652  int64_t numPackedDims = unPackOp.getInnerDimsPos().size();
653  auto lastDims = llvm::to_vector(
654  llvm::seq<int64_t>(packedRank - numPackedDims, packedRank));
655  PackingMetadata packingMetadata =
656  computePackingMetadata(packedRank, unPackOp.getInnerDimsPos());
657  SmallVector<int64_t> lastDimsToInsertPositionsPerm = computePermutationVector(
658  packedRank, lastDims, packingMetadata.insertPositions);
659 
660  // 3. Compute the stripMinedShape: this is the packed shape without outer and
661  // inner permutations.
662  SmallVector<int64_t> stripMinedShape(packedTensorType.getShape());
663  applyPermutationToVector(stripMinedShape, lastDimsToInsertPositionsPerm);
664 
665  // 4. Transpose packedShape to stripMinedShape.
666  RankedTensorType stripMinedTensorType =
667  RankedTensorType::Builder(packedTensorType).setShape(stripMinedShape);
668  RankedTensorType collapsedType = tensor::CollapseShapeOp::inferCollapsedType(
669  stripMinedTensorType, packingMetadata.reassociations);
670  auto emptyOp =
671  rewriter.create<tensor::EmptyOp>(loc, stripMinedTensorType, ValueRange{});
672  auto transposeOp = rewriter.create<linalg::TransposeOp>(
673  loc, unPackOp.getSource(), emptyOp, lastDimsToInsertPositionsPerm);
674 
675  LLVM_DEBUG(
676  DBGSNL(); DBGSNL(); llvm::interleaveComma(packingMetadata.insertPositions,
677  DBGS() << "insertPositions: ");
678  DBGSNL(); llvm::interleaveComma(packedTensorType.getShape(),
679  DBGS() << "packedShape: ");
680  DBGSNL();
681  llvm::interleaveComma(lastDimsToInsertPositionsPerm,
682  DBGS() << "lastDimsToInsertPositionsPerm: ");
683  DBGSNL(); llvm::interleaveComma(
684  packingMetadata.reassociations, DBGS() << "reassociations: ",
685  [&](ReassociationIndices ri) {
686  llvm::interleaveComma(ri, llvm::dbgs() << "|");
687  });
688  DBGSNL();
689  llvm::interleaveComma(stripMinedShape, DBGS() << "stripMinedShape: ");
690  DBGSNL(); DBGS() << "collapsed type: " << collapsedType; DBGSNL(););
691 
692  // 5. Collapse from the stripMinedShape to the padded result.
693  auto reshapeOp = rewriter.create<tensor::CollapseShapeOp>(
694  loc, collapsedType, transposeOp->getResult(0),
695  packingMetadata.reassociations);
696 
697  // 6. ExtractSlice
698  int64_t destRank = destTensorType.getRank();
699  auto extractSliceOp = rewriter.create<tensor::ExtractSliceOp>(
700  loc, destTensorType, reshapeOp->getResult(0),
701  SmallVector<OpFoldResult>(destRank, zero),
702  tensor::getMixedSizes(rewriter, loc, unPackOp->getResult(0)),
703  SmallVector<OpFoldResult>(destRank, one));
704 
705  // 7. Replace unPackOp by extractSliceOp.
706  rewriter.replaceOp(unPackOp, extractSliceOp->getResults());
707 
708  return LowerUnPackOpResult{emptyOp, transposeOp, reshapeOp, extractSliceOp};
709 }

733 /// Implement packing of a single LinalgOp by performing packing by
734 /// `packedSizes`. There must be one packedSizes entry per `linalgOp` iterator.
735 /// Return the packed Linalg op on success, failure otherwise.
736 FailureOr<PackResult> linalg::pack(RewriterBase &rewriter,
737  linalg::LinalgOp linalgOp,
738  ArrayRef<OpFoldResult> packedSizes) {
739  if (packedSizes.size() != linalgOp.getNumLoops()) {
740  return rewriter.notifyMatchFailure(linalgOp,
741  "incorrect number of pack sizes");
742  }
743 
744  Location loc = linalgOp->getLoc();
745  SmallVector<AffineMap> indexingMaps = linalgOp.getIndexingMapsArray();
746  SmallVector<utils::IteratorType> iteratorTypes =
747  linalgOp.getIteratorTypesArray();
748  LLVM_DEBUG(DBGS() << "Start packing: " << linalgOp << "\n";
749  llvm::interleaveComma(indexingMaps, DBGS() << "maps: "); DBGSNL();
750  llvm::interleaveComma(iteratorTypes, DBGS() << "iterators: ");
751  DBGSNL(););
752 
753  SmallVector<tensor::PackOp> packOps;
754  SmallVector<tensor::UnPackOp> unPackOps;
755  // Step 1. Pack each dim of the LinalgOp metadata by packedSizes[i].
756  PackedOperandsDimList listOfPackedOperandsDim;
757  for (int64_t i = 0, e = packedSizes.size(); i < e; ++i) {
758  std::optional<int64_t> maybeConstant = getConstantIntValue(packedSizes[i]);
759  // Skip tile sizes explicitly set to 0.
760  if (maybeConstant.has_value() && maybeConstant.value() == 0)
761  continue;
762 
763  PackedOperandsDim packedOperandsDims;
764  packedOperandsDims.packedSize = packedSizes[i];
765  FailureOr<SmallVector<std::optional<int64_t>>>
766  maybePackedDimForEachOperand =
767  packLinalgMetadataOnce(indexingMaps, iteratorTypes, i);
768  if (failed(maybePackedDimForEachOperand))
769  return failure();
770  packedOperandsDims.packedDimForEachOperand = *maybePackedDimForEachOperand;
771  listOfPackedOperandsDim.push_back(std::move(packedOperandsDims));
772 
773  LLVM_DEBUG(
774  DBGS() << "++++ After pack size #" << i << ": " << packedSizes[i]
775  << "\n";
776  llvm::interleaveComma(indexingMaps, DBGS() << "maps: "); DBGSNL();
777  llvm::interleaveComma(iteratorTypes, DBGS() << "iterators: "); DBGSNL();
778  llvm::interleaveComma(packedOperandsDims.packedDimForEachOperand,
779  DBGS() << "packedDimForEachOperand: ");
780  DBGSNL(););
781  }
782 
783  // Step 2. Propagate packing to all LinalgOp operands.
784  SmallVector<Value> inputsAndInits, results;
785  for (const auto &operandsList :
786  {linalgOp.getDpsInputOperands(), linalgOp.getDpsInitOperands()}) {
787  for (OpOperand *opOperandPtr : operandsList) {
788  int64_t pos = opOperandPtr->getOperandNumber();
789  Value operand = opOperandPtr->get();
790  SmallVector<int64_t> innerPos =
791  listOfPackedOperandsDim.extractPackedDimsForOperand(pos);
792  SmallVector<OpFoldResult> innerPackSizes =
793  listOfPackedOperandsDim.extractPackSizesForOperand(pos);
794  LLVM_DEBUG(
795  DBGS() << "operand: " << operand << "\n";
796  llvm::interleaveComma(innerPos, DBGS() << "innerPos: "); DBGSNL();
797  llvm::interleaveComma(innerPackSizes, DBGS() << "innerPackSizes: ");
798  DBGSNL(););
799  if (innerPackSizes.empty()) {
800  inputsAndInits.push_back(operand);
801  continue;
802  }
803  Value dest = tensor::PackOp::createDestinationTensor(
804  rewriter, loc, operand, innerPackSizes, innerPos,
805  /*outerDimsPerm=*/{});
806  // TODO: value of the padding attribute should be determined by consumers.
807  auto zeroAttr =
808  rewriter.getZeroAttr(getElementTypeOrSelf(dest.getType()));
809  Value zero = rewriter.create<arith::ConstantOp>(loc, zeroAttr);
810  packOps.push_back(rewriter.create<tensor::PackOp>(
811  loc, operand, dest, innerPos, innerPackSizes, zero));
812  inputsAndInits.push_back(packOps.back());
813  }
814  }
815 
816  // Step 3. Build the packed op, use the type of `inits` as result types.
817  ValueRange inputs =
818  ValueRange{inputsAndInits}.take_front(linalgOp.getNumDpsInputs());
819  ValueRange inits =
820  ValueRange{inputsAndInits}.take_back(linalgOp.getNumDpsInits());
821  auto packedLinalgOp = rewriter.create<linalg::GenericOp>(
822  linalgOp.getLoc(), inits.getTypes(), inputs, inits, indexingMaps,
823  iteratorTypes);
824  packedLinalgOp.getRegion().takeBody(linalgOp->getRegion(0));
825 
826  // Step 4. Propagate packing to all the op results.
827  for (OpResult result : packedLinalgOp->getResults()) {
828  int64_t resultNum = result.getResultNumber();
829  tensor::PackOp maybePackedInit =
830  inits[resultNum].getDefiningOp<tensor::PackOp>();
831  if (!maybePackedInit) {
832  results.push_back(result);
833  continue;
834  }
835  // Build the symmetrical UnPackOp to the existing PackOp.
836  unPackOps.push_back(rewriter.create<tensor::UnPackOp>(
837  packedLinalgOp->getLoc(), result, maybePackedInit.getSource(),
838  maybePackedInit.getInnerDimsPos(), maybePackedInit.getMixedTiles()));
839  results.push_back(unPackOps.back());
840  }
841 
842  // Step 5. Replace `linalgOp`.
843  rewriter.replaceOp(linalgOp, results);
844 
845  // Return packedLinalgOp.
846  return PackResult{packOps,
847  cast<linalg::LinalgOp>(packedLinalgOp.getOperation()),
848  unPackOps};
849 }

866 /// Return a new GenericOp obtained by transposing opOperand by the permutation
867 /// vector:
868 /// - the corresponding indexing map is transposed by `permutation`
869 /// - the corresponding operand value is replaced by `transposedValue`
870 /// `linalgOp` is replaced by the return op in the process.
871 /// Asserts that `transposedValue` is of the proper transposed ShapedType.
872 static LinalgOp transposeOneLinalgOperandAndReplace(
873  RewriterBase &rewriter, LinalgOp linalgOp, OpOperand &opOperand,
874  ArrayRef<int64_t> permutation, Value transposedValue) {
875  // Sanity check the operand.
876  assert(linalgOp == opOperand.getOwner() && "linalg op must own the operand");
877 
878  // Sanity check of the expected transposed tensor type.
879  auto tensorType = permuteShape(
880  opOperand.get().getType().cast<RankedTensorType>(), permutation);
881  (void)tensorType;
882  assert(tensorType == transposedValue.getType() &&
883  "expected tensor type mismatch");
884 
885  // Compute the transposed indexing map.
886  // Sigh unsigned pollution.
887  SmallVector<unsigned> tmpTransposition = llvm::to_vector(
888  llvm::map_range(permutation, [](int64_t i) -> unsigned { return i; }));
889  AffineMap permutationMap =
890  AffineMap::getPermutationMap(tmpTransposition, rewriter.getContext());
891  AffineMap transposedMap =
892  permutationMap.compose(linalgOp.getMatchingIndexingMap(&opOperand));
893 
894  // Set the transposed indexing map in the proper position.
895  SmallVector<AffineMap> indexingMaps = linalgOp.getIndexingMapsArray();
896  indexingMaps[linalgOp.getIndexingMapIndex(&opOperand)] = transposedMap;
897  // Set the transposedValue in the proper operand position.
898  SmallVector<Value> operands = linalgOp->getOperands();
899  operands[opOperand.getOperandNumber()] = transposedValue;
900 
901  ValueRange operandsRef(operands);
902  auto transposedGenericOp = rewriter.create<linalg::GenericOp>(
903  /*location=*/linalgOp->getLoc(),
904  /*resultTensorTypes=*/
905  operandsRef.drop_front(linalgOp.getNumDpsInputs()).getTypes(),
906  /*inputs=*/operandsRef.take_front(linalgOp.getNumDpsInputs()),
907  /*outputs=*/operandsRef.drop_front(linalgOp.getNumDpsInputs()),
908  /*indexingMaps=*/indexingMaps,
909  /*iteratorTypes=*/linalgOp.getIteratorTypesArray());
910  transposedGenericOp.getRegion().takeBody(linalgOp->getRegion(0));
911  rewriter.replaceOp(linalgOp, transposedGenericOp->getResults());
912 
913  return cast<linalg::LinalgOp>(transposedGenericOp.getOperation());
914 }

916 FailureOr<PackTransposeResult>
917 linalg::packTranspose(RewriterBase &rewriter, tensor::PackOp packOp,
918  linalg::LinalgOp linalgOp, tensor::UnPackOp maybeUnPackOp,
919  ArrayRef<int64_t> outerPerm,
920  ArrayRef<int64_t> innerPerm) {
921  Location loc = linalgOp.getLoc();
922 
923  // Step 1. Transpose packOp.
924  rewriter.setInsertionPoint(packOp);
925  tensor::PackOp transposedPackOp =
926  packOp.createTransposedClone(rewriter, loc, innerPerm, outerPerm);
927 
928  if (!packOp.getResult().hasOneUse())
929  return rewriter.notifyMatchFailure(linalgOp, "expect single pack use");
930 
931  OpOperand &packUse = *packOp->getUses().begin();
932  if (packUse.getOwner() != linalgOp) {
933  return rewriter.notifyMatchFailure(
934  linalgOp, "not a single use by the LinalgOp target");
935  }
936  if (maybeUnPackOp &&
937  (!linalgOp.isDpsInit(&packUse) ||
938  maybeUnPackOp.getSource() != linalgOp.getTiedOpResult(&packUse))) {
939  return rewriter.notifyMatchFailure(linalgOp,
940  "not produced by the LinalgOp target");
941  }
942 
943  // Step 2. Transpose linalgOp.
944  // transposedPackOp.getOuterDimsPerm() may be empty, in which case it is the
945  // identity. Don't rely on it.
946  int64_t numLeadingDims = packOp.getSourceRank();
947  int64_t numTrailingDims = packOp.getInnerDimsPos().size();
948  // Step 2.a. Compute the permutation on the whole operand.
949  // Leading part just reuse the outerPerm.
950  SmallVector<int64_t> permutation(outerPerm);
951  if (permutation.empty())
952  llvm::append_range(permutation, llvm::seq<int64_t>(0, numLeadingDims));
953  // Trailing part needs to reindex positions by `numLeadingDims`.
954  if (innerPerm.empty()) {
955  llvm::append_range(
956  permutation,
957  llvm::seq<int64_t>(numLeadingDims, numLeadingDims + numTrailingDims));
958  } else {
959  llvm::append_range(permutation,
960  llvm::map_range(innerPerm, [&](int64_t pos) {
961  return numLeadingDims + pos;
962  }));
963  }
964  if (!isPermutationVector(permutation))
965  return rewriter.notifyMatchFailure(linalgOp, "invalid permutation");
966 
967  // Step 2.b. Save the transposedPackUse operand number in case we need to
968  // get the tied OpResult after `linalgOp` has been replaced.
969  int64_t packUseOperandNumber = packUse.getOperandNumber();
970  // Step 2.c. Actually perform the transposition.
971  rewriter.setInsertionPoint(linalgOp);
972  linalg::LinalgOp transposedLinalgOp = transposeOneLinalgOperandAndReplace(
973  rewriter, linalgOp, packUse, permutation, transposedPackOp.getResult());
974 
975  // Step 3. Maybe transpose unPackOp.
976  tensor::UnPackOp transposedUnPackOp;
977  if (maybeUnPackOp) {
978  OpOperand &opOperand =
979  transposedLinalgOp->getOpOperand(packUseOperandNumber);
980  OpResult transposedResult = transposedLinalgOp.getTiedOpResult(&opOperand);
981  rewriter.setInsertionPoint(maybeUnPackOp);
982  transposedUnPackOp = maybeUnPackOp.createTransposedClone(
983  rewriter, loc, transposedResult, innerPerm, outerPerm);
984 
985  rewriter.replaceOp(maybeUnPackOp, transposedUnPackOp->getResults());
986  }
987 
988  // Step 4. Finally, replace packOp now that we don't need it anymore.
989  rewriter.replaceOp(packOp, transposedPackOp->getResults());
990 
991  return PackTransposeResult{transposedPackOp, transposedLinalgOp,
992  transposedUnPackOp};
993 }

1044  // Bail on non-static shapes.
1045  if (!inputShapedType.hasStaticShape())
1046  return failure();
1047  if (!resultShapedType.hasStaticShape())
1048  return failure();
1049 
1050  // Only support padding with a constant for now, i.e. either:
1051  // 1. A BBarg from a different block.
1052  // 2. A value defined outside of the current block.
1053  Block &block = padOp.getRegion().front();
1054  auto yieldOp = cast<tensor::YieldOp>(block.getTerminator());
1055  Value padValue = yieldOp.getValue();
1056  Operation *definingOp = padValue.getDefiningOp();
1057  if (definingOp && definingOp->getBlock() == &block)
1058  return failure();
1059  if (!definingOp && padValue.cast<BlockArgument>().getOwner() == &block)
1060  return failure();

1062  // Create tensor with the padded shape
1063  Location loc = padOp.getLoc();
1064  SmallVector<Value> indices(resultShapedType.getRank(),
1065  rewriter.create<arith::ConstantIndexOp>(loc, 0));
1066  Value emptyTensor = rewriter.create<tensor::EmptyOp>(
1067  loc, resultShapedType.getShape(), resultShapedType.getElementType());
1068 
1069  // Initialize tensor with the pad value
1070  Value tmpTensor = rewriter
1071  .create<linalg::FillOp>(loc, ValueRange{padValue},
1072  ValueRange{emptyTensor})
1073  .result();
1074 
1075  // Copy original contents into new tensor
1076  // Uses linalg.generic, but could be done with tensor.insert_slice
1077  SmallVector<AffineExpr, 4> outputExprs;
1078  for (unsigned i = 0; i < resultShapedType.getRank(); ++i) {
1079  outputExprs.push_back(getAffineDimExpr(i, rewriter.getContext()) +
1080  padOp.getStaticLow()[i]);
1081  }
1082 
1083  SmallVector<AffineMap, 2> transferMaps = {
1084  rewriter.getMultiDimIdentityMap(inputShapedType.getRank()),
1085  AffineMap::get(resultShapedType.getRank(),
1086  /*symbolCount=*/0, outputExprs, rewriter.getContext())};
1087 
1088  rewriter.replaceOpWithNewOp<linalg::GenericOp>(
1089  padOp, resultShapedType, padOp.getSource(), tmpTensor, transferMaps,
1090  getNParallelLoopsAttrs(resultShapedType.getRank()),
1091  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange args) {
1092  nestedBuilder.create<linalg::YieldOp>(nestedLoc, args[0]);
1093  });
1094

1096 }

1244 // Normalizes a permutation on a higher rank space to its actual size, e.g.
1245 // perm = [1, 4, 2]
1246 // becomes
1247 // norm = [0, 2, 1]
1248 static SmallVector<int64_t>
1249 getPackUnpackNormalizedPerm(int rank, ArrayRef<int64_t> perm) {
1250  constexpr int64_t kNonTiledMarker = -1;
1251  SmallVector<int64_t> vec(rank, kNonTiledMarker);
1252  for (auto [index, value] : llvm::enumerate(perm))
1253  vec[value] = index;
1254  SmallVector<int64_t> normalizedPerm = llvm::to_vector(llvm::make_filter_range(
1255  vec, [&](int64_t v) { return v != kNonTiledMarker; }));
1256  // This inverts the permutation in addition to normalizing so invert back.
1257  return invertPermutationVector(normalizedPerm);
1258 }

1260 // Gets the normalized permutation implied by innerDimsPos and outerDimsPerm
1261 // assuming rank reduction of unit outer dims.
1262 static SmallVector<int64_t>
1263 getPackUnpackRankReducedPerm(ArrayRef<int64_t> shape,
1264  ArrayRef<int64_t> innerDimsPos,
1265  ArrayRef<int64_t> outerDimsPerm) {
1266  SmallVector<int64_t> rankReducedOuterDimsPerm;
1267  SmallVector<int64_t> outerDims;
1268  SmallVector<int64_t> innerDims;
1269  int64_t dim = 0;
1270  int64_t unpackedRank = shape.size();
1271  for (auto i : llvm::seq<unsigned>(0, unpackedRank)) {
1272  if (llvm::is_contained(innerDimsPos, i)) {
1273  innerDims.push_back(dim++);
1274  continue;
1275  }
1276  if (shape[i] == 1)
1277  continue;
1278  outerDims.push_back(dim++);
1279  if (!outerDimsPerm.empty())
1280  rankReducedOuterDimsPerm.push_back(outerDimsPerm[i]);
1281  }
1282 
1283  // Get the position of the inner dims after permutation.
1284  SmallVector<int64_t> innerPerm =
1285  getPackUnpackNormalizedPerm(unpackedRank, innerDimsPos);
1286  applyPermutationToVector<int64_t>(innerDims, innerPerm);
1287 
1288  // Ditto for the outer dims.
1289  SmallVector<int64_t> perm = outerDims;
1290 
1291  rankReducedOuterDimsPerm =
1292  getPackUnpackNormalizedPerm(unpackedRank, rankReducedOuterDimsPerm);
1293  if (!rankReducedOuterDimsPerm.empty())
1294  applyPermutationToVector<int64_t>(perm, rankReducedOuterDimsPerm);
1295 
1296  // The tile always ends up as the inner most dims after packing.
1297  perm.append(innerDims);
1298 
1299  return perm;
1300 }

1303  tensor::PackOp packOp, PatternRewriter &rewriter) const {
1304  if (llvm::any_of(packOp.getMixedTiles(),
1305  [](OpFoldResult tile) { return tile.is<Value>(); })) {
1306  return rewriter.notifyMatchFailure(packOp,
1307  "require inner tile sizes being static");
1308  }
1309 
1310  // TODO: support the case that outer dimensions are not all 1s. A
1311  // tensor.expand_shape will be generated in this case.
1312  auto innerDimsPos = packOp.getInnerDimsPos();
1313  int64_t srcRank = packOp.getSourceRank();
1314  auto destShape = packOp.getDestType().getShape();
1315  if (llvm::any_of(innerDimsPos, [destShape](int64_t index) {
1316  return destShape[index] != 1;
1317  })) {
1318  return rewriter.notifyMatchFailure(
1319  packOp, "require the tiled outer dimensions of the result are all 1s");

1321

1323  // outer dims.
1324  Location loc = packOp.getLoc();

1330  Attribute oneIdxAttr = rewriter.getIndexAttr(1);

1336  if (dimAndTileMapping.count(i)) {

1339  readSizes.push_back(dimAndTileMapping[i]);

1341  }

1343  readSizes.push_back(
1344  rewriter.create<tensor::DimOp>(loc, input, i).getResult());
1345  } else {
1346  readSizes.push_back(rewriter.getIndexAttr(inputShape[i]));
1347  }
1348  if (inputShape[i] != 1)
1349  readShape.push_back(inputShape[i]);
1350  }
1351 
1352  Type elemType = packOp.getSourceType().getElementType();
1353  auto readType = RankedTensorType::get(readShape, elemType);
1354 
1355  Value tile = rewriter.create<tensor::ExtractSliceOp>(
1356  loc, readType, input, readOffsets, readSizes, readStrides);
1357

1359

1361  inputShape, innerDimsPos, packOp.getOuterDimsPerm());

1363  LLVM_DEBUG(DBGS() << "Pack permutation: " << packOp << "\n";

1371  rewriter.create<linalg::TransposeOp>(loc, tile, empty, perm);
1372

1374  int64_t destRank = packOp.getDestRank();

1376  SmallVector<OpFoldResult> writeOffsets(destRank, zeroIdxAttr);

1378  tensor::getMixedSizes(rewriter, loc, packOp.getDest());
1379 
1380  auto insert = rewriter.create<tensor::InsertSliceOp>(
1381  loc, transposedOp.getResult()[0], packOp.getDest(), writeOffsets,
1382  writeSizes, writeStrides);
1383  rewriter.replaceOp(packOp, insert.getResult());
1384

1386 }

1388 LogicalResult GeneralizeOuterUnitDimsUnPackOpPattern::matchAndRewrite(
1389  tensor::UnPackOp unpackOp, PatternRewriter &rewriter) const {
1390  int64_t srcRank = unpackOp.getSourceRank();
1391  int64_t destRank = unpackOp.getDestRank();
1392  ArrayRef<int64_t> srcShape = unpackOp.getSourceType().getShape();
1393  ArrayRef<int64_t> innerDimsPos = unpackOp.getInnerDimsPos();
1394  if (llvm::any_of(innerDimsPos, [srcShape](int64_t index) {
1395  return srcShape[index] != 1;
1396  })) {
1397  return rewriter.notifyMatchFailure(
1398  unpackOp,
1399  "require the tiled outer dimensions of the result are all 1s");
1400  }
1401 
1402  // 1. Use rank-reduced tensor.extract_slice op to extract the tile.
1403  Location loc = unpackOp.getLoc();
1404  Value source = unpackOp.getSource();
1405  DenseMap<int64_t, OpFoldResult> dimAndTileMapping =
1406  unpackOp.getDimAndTileMapping();
1407  Attribute zeroIdxAttr = rewriter.getIndexAttr(0);
1408  Attribute oneIdxAttr = rewriter.getIndexAttr(1);
1409  SmallVector<OpFoldResult> readOffsets(srcRank, zeroIdxAttr);
1410  SmallVector<OpFoldResult> readStrides(srcRank, oneIdxAttr);
1411  SmallVector<OpFoldResult> readSizes;
1412  SmallVector<int64_t> readShape;
1413  for (auto i : llvm::seq<unsigned>(0, destRank)) {
1414  if (dimAndTileMapping.count(i)) {
1415  readSizes.push_back(oneIdxAttr);
1416  continue;
1417  }
1418 
1419  if (ShapedType::isDynamic(srcShape[i])) {
1420  readSizes.push_back(
1421  rewriter.create<tensor::DimOp>(loc, source, i).getResult());
1422  } else {
1423  readSizes.push_back(rewriter.getIndexAttr(srcShape[i]));
1424  }
1425  if (srcShape[i] != 1)
1426  readShape.push_back(srcShape[i]);
1427  }
1428  auto mixedTiles = unpackOp.getMixedTiles();
1429  readSizes.append(mixedTiles.begin(), mixedTiles.end());
1430 
1431  // Explicitly create the type for extract_slice op because the inner tile
1432  // size could be 1. We want to represent the whole inner tile in this case.
1433  auto tileShape = srcShape.drop_front(destRank);
1434  // Append the inner tile shape to the permuted and rank-reduced outer shape.
1435  readShape.append(tileShape.begin(), tileShape.end());
1436  Type elemType = unpackOp.getSourceType().getElementType();
1437  auto readType = RankedTensorType::get(readShape, elemType);
1438  Value innerTile = rewriter.create<tensor::ExtractSliceOp>(
1439  loc, readType, unpackOp.getSource(), readOffsets, readSizes, readStrides);
1440 
1441  // 2. Transpose the tile to match the outer corresponding tile order.
1442  SmallVector<int64_t> perm = getPackUnpackRankReducedPerm(
1443  srcShape.take_front(destRank), innerDimsPos, unpackOp.getOuterDimsPerm());
1444  // Unpack is a transition out of packed space so we invert the permutation.
1445  perm = invertPermutationVector(perm);
1446  SmallVector<int64_t> transpShape(readShape);
1447  applyPermutationToVector<int64_t>(transpShape, perm);
1448 
1449  Value empty = rewriter.create<tensor::EmptyOp>(loc, transpShape, elemType);
1450  auto transposedOp =
1451  rewriter.create<linalg::TransposeOp>(loc, innerTile, empty, perm);
1452 
1453  // 3. Handle in-complete tiles if needed. It truncates trailing data from the
1454  // transposed tile.
1455  int numLoops = transpShape.size();
1456  SmallVector<OpFoldResult> tileStrides(numLoops, oneIdxAttr);
1457  SmallVector<OpFoldResult> tileOffsets(numLoops, zeroIdxAttr);
1458  SmallVector<OpFoldResult> tileSizes;
1459  ArrayRef<int64_t> destShape = unpackOp.getDestType().getShape();
1460  for (auto i : llvm::seq<unsigned>(0, destRank)) {
1461  if (dimAndTileMapping.count(i) || destShape[i] != 1)
1462  tileSizes.push_back(getAsOpFoldResult(
1463  rewriter.createOrFold<tensor::DimOp>(loc, unpackOp.getDest(), i)));
1464  }
1465 
1466  auto partialTile = rewriter.create<tensor::ExtractSliceOp>(
1467  loc, transposedOp.getResult()[0], tileOffsets, tileSizes, tileStrides);
1468 
1469  // 4. Insert the result to the destination tensor.
1470  SmallVector<OpFoldResult> writeSizes;
1471  SmallVector<OpFoldResult> writeStrides(destRank, oneIdxAttr);
1472  SmallVector<OpFoldResult> writeOffsets(destRank, zeroIdxAttr);
1473  for (int i = 0, idx = 0; i < destRank; ++i) {
1474  if (dimAndTileMapping.count(i) || destShape[i] != 1)
1475  writeSizes.push_back(tileSizes[idx++]);
1476  else
1477  writeSizes.push_back(oneIdxAttr);
1478  }
1479  auto insert = rewriter.create<tensor::InsertSliceOp>(
1480  loc, partialTile, unpackOp.getDest(), writeOffsets, writeSizes,
1481  writeStrides);
1482  rewriter.replaceOp(unpackOp, insert.getResult());
1483 
1484  return success();
1485 }

1498  if (convOp.hasBufferSemantics())
1499  return failure(); // To be implemented.

1513  auto [khIndex, kwIndex, ohIndex, owIndex] =
1514  TypeSwitch<Operation *, std::tuple<int64_t, int64_t, int64_t, int64_t>>(
1515  convOp)
1516  .Case([&](linalg::Conv2DNhwcHwcfOp op) {
1517  return std::make_tuple(0, 1, 1, 2);
1518  })
1519  .Case([&](linalg::Conv2DNchwFchwOp op) {
1520  return std::make_tuple(2, 3, 2, 3);
1521  })
1522  .Case([&](linalg::PoolingNhwcSumOp op) {
1523  return std::make_tuple(0, 1, 1, 2);
1524  })
1525  .Case([&](linalg::PoolingNchwSumOp op) {

1530  })

1532  return std::make_tuple(0, 1, 1, 2);
1533  })

1539  })
1540  .Case([&](linalg::PoolingNchwMaxOp op) {

1542  })

1544  llvm_unreachable("unexpected conv2d/pool2d operation.");
1545  return std::make_tuple(0, 0, 0, 0);

1549  // of size 1. Other cases can rely on tiling to reduce to such cases.
1550  int64_t khSize = kernelShape[khIndex], kwSize = kernelShape[kwIndex];
1551  int64_t ohSize = outputShape[ohIndex], owSize = outputShape[owIndex];

1553  bool removeW = (kwSize == 1 && owSize == 1);
1554  if (!removeH && !removeW)
