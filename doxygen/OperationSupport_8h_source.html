239  /// This hook implements a generalized folder for this operation. Operations
240  /// can implement this to provide simplifications rules that are applied by
241  /// the Builder::createOrFold API and the canonicalization pass.
242  ///
243  /// This is an intentionally limited interface - implementations of this
244  /// hook can only perform the following changes to the operation:
245  ///
246  /// 1. They can leave the operation alone and without changing the IR, and
247  /// return failure.
248  /// 2. They can mutate the operation in place, without changing anything
249  /// else
250  /// in the IR. In this case, return success.
251  /// 3. They can return a list of existing values that can be used instead
252  /// of
253  /// the operation. In this case, fill in the results list and return
254  /// success. The caller will remove the operation and use those results
255  /// instead.
256  ///
257  /// This allows expression of some simple in-place canonicalizations (e.g.
258  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
259  /// generalized constant folding.
260  LogicalResult foldHook(Operation *op, ArrayRef<Attribute> operands,
261  SmallVectorImpl<OpFoldResult> &results) const {
262  return getImpl()->foldHook(op, operands, results);
263  }

691 /// Unsorted string search or identifier lookups are linear scans.
692 template <typename IteratorT, typename NameT>
693 std::pair<IteratorT, bool> findAttrUnsorted(IteratorT first, IteratorT last,
694  NameT name) {
695  for (auto it = first; it != last; ++it)
696  if (it->getName() == name)
697  return {it, true};
698  return {last, false};
699 }
700 
701 /// Using llvm::lower_bound requires an extra string comparison to check whether
702 /// the returned iterator points to the found element or whether it indicates
703 /// the lower bound. Skip this redundant comparison by checking if `compare ==
704 /// 0` during the binary search.
705 template <typename IteratorT>
706 std::pair<IteratorT, bool> findAttrSorted(IteratorT first, IteratorT last,
707  StringRef name) {
708  ptrdiff_t length = std::distance(first, last);
709 
710  while (length > 0) {
711  ptrdiff_t half = length / 2;
712  IteratorT mid = first + half;
713  int compare = mid->getName().strref().compare(name);
714  if (compare < 0) {
715  first = mid + 1;
716  length = length - half - 1;
717  } else if (compare > 0) {
718  length = half;
719  } else {
720  return {mid, true};
721  }
722  }
723  return {first, false};
724 }
725 
726 /// StringAttr lookups on large attribute lists will switch to string binary
727 /// search. String binary searches become significantly faster than linear scans
728 /// with the identifier when the attribute list becomes very large.
729 template <typename IteratorT>
730 std::pair<IteratorT, bool> findAttrSorted(IteratorT first, IteratorT last,
731  StringAttr name) {
732  constexpr unsigned kSmallAttributeList = 16;
733  if (std::distance(first, last) > kSmallAttributeList)
734  return findAttrSorted(first, last, name.strref());
735  return findAttrUnsorted(first, last, name);
736 }
737 
738 /// Get an attribute from a sorted range of named attributes. Returns null if
739 /// the attribute was not found.
740 template <typename IteratorT, typename NameT>
741 Attribute getAttrFromSortedRange(IteratorT first, IteratorT last, NameT name) {
742  std::pair<IteratorT, bool> result = findAttrSorted(first, last, name);
743  return result.second ? result.first->getValue() : Attribute();
744 }
