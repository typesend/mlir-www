38 static LogicalResult checkConstantOperandPad(Operation *op) {
39  if (auto pad_op = dyn_cast<tosa::PadOp>(op)) {
40  DenseElementsAttr paddings;
41  if (!matchPattern(pad_op.getPadding(), m_Constant(&paddings)))
42  return op->emitOpError("padding of pad is not constant");
43 
44  DenseElementsAttr pad_const;
45  // Assume this op is zero-padding if pad_const is not presented.
46  if (pad_op.getPadConst() &&
47  !matchPattern(pad_op.getPadConst(), m_Constant(&pad_const)))
48  return op->emitOpError("pad_const of pad is not constant");
49  }
50  return success();
51 }
52 
53 static LogicalResult checkConstantOperandTranspose(Operation *op) {

55  DenseElementsAttr perms;
56  if (!matchPattern(transpose_op.getPerms(), m_Constant(&perms)))
57  return op->emitOpError("perms of transpose is not constant");

62 static LogicalResult checkConstantOperandFullyConnected(Operation *op) {
63  if (auto fc_op = dyn_cast<tosa::FullyConnectedOp>(op)) {
64  DenseElementsAttr weight;
65  if (!matchPattern(fc_op.getWeight(), m_Constant(&weight)))
66  return op->emitOpError("weight of fully_connected is not constant");
67 
68  DenseElementsAttr bias;
69  if (!matchPattern(fc_op.getBias(), m_Constant(&bias)))
70  return op->emitOpError("bias of fully_connected is not constant");
71  }
72  return success();
73 }

79 struct TosaValidation : public tosa::impl::TosaValidationBase<TosaValidation> {
80 public:
81  explicit TosaValidation() { populateConstantOperandChecks(); }
82  void runOnOperation() override;
83 
84  LogicalResult applyConstantOperandCheck(Operation *op) {
85  for (auto &checker : const_checkers) {
86  if (failed(checker(op)))
87  return failure();
88  }
89  return success();
90  }
91 
92 private:
93  void populateConstantOperandChecks() {
94  const_checkers.emplace_back(checkConstantOperandPad);
95  const_checkers.emplace_back(checkConstantOperandTranspose);
96  const_checkers.emplace_back(checkConstantOperandFullyConnected);
97  }
98 
99  SmallVector<std::function<LogicalResult(Operation *)>> const_checkers;
100  std::optional<TosaProfileEnum> profileType;
101 };

103 void TosaValidation::runOnOperation() {
104  profileType = symbolizeEnum<TosaProfileEnum>(profileName);
105 
106  getOperation().walk([&](Operation *op) {
107  for (Value operand : op->getOperands()) {
108  if ((profileType == TosaProfileEnum::BaseInference) &&
109  getElementTypeOrSelf(operand).isa<FloatType>()) {
110  return signalPassFailure();
111  }
112  if (getElementTypeOrSelf(operand).isF64()) {
113  return signalPassFailure();
114  }
115  }
116 
117  // Some uses of TOSA rely on the constant operands of particular operations.
118  if (StrictOperationSpecAlignment && failed(applyConstantOperandCheck(op)))
119  signalPassFailure();
120  });
121 }
122 } // namespace
