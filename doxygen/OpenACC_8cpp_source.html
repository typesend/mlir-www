223 template <typename StructureOp>
224 static ParseResult parseRegions(OpAsmParser &parser, OperationState &state,
225  unsigned nRegions = 1) {
226 
227  SmallVector<Region *, 2> regions;
228  for (unsigned i = 0; i < nRegions; ++i)
229  regions.push_back(state.addRegion());
230 
231  for (Region *region : regions) {
232  if (parser.parseRegion(*region, /*arguments=*/{}, /*argTypes=*/{}))
233  return failure();
234  }
235 
236  return success();
237 }

244 /// Pattern to remove operation without region that have constant false `ifCond`
245 /// and remove the condition from the operation if the `ifCond` is a true
246 /// constant.
247 template <typename OpTy>
248 struct RemoveConstantIfCondition : public OpRewritePattern<OpTy> {
249  using OpRewritePattern<OpTy>::OpRewritePattern;
250 
251  LogicalResult matchAndRewrite(OpTy op,
252  PatternRewriter &rewriter) const override {
253  // Early return if there is no condition.
254  Value ifCond = op.getIfCond();
255  if (!ifCond)
256  return failure();
257 
258  IntegerAttr constAttr;
259  if (!matchPattern(ifCond, m_Constant(&constAttr)))
260  return failure();
261  if (constAttr.getInt())
262  rewriter.updateRootInPlace(op, [&]() { op.getIfCondMutable().erase(0); });
263  else
264  rewriter.eraseOp(op);
265 
266  return success();
267  }
268 };

342 static ParseResult
343 parseGangClause(OpAsmParser &parser,
344  std::optional<OpAsmParser::UnresolvedOperand> &gangNum,
345  Type &gangNumType,
346  std::optional<OpAsmParser::UnresolvedOperand> &gangStatic,
347  Type &gangStaticType, UnitAttr &hasGang) {
348  hasGang = UnitAttr::get(parser.getBuilder().getContext());
349  // optional gang operands
350  if (succeeded(parser.parseOptionalLParen())) {
351  if (succeeded(parser.parseOptionalKeyword(LoopOp::getGangNumKeyword()))) {
352  if (parser.parseEqual())
353  return failure();
354  gangNum = OpAsmParser::UnresolvedOperand{};
355  if (parser.parseOperand(*gangNum) || parser.parseColonType(gangNumType))
356  return failure();
357  } else {
358  gangNum = std::nullopt;
359  }
360  // FIXME: Comma should require subsequent operands.
361  (void)parser.parseOptionalComma();
362  if (succeeded(
363  parser.parseOptionalKeyword(LoopOp::getGangStaticKeyword()))) {
364  gangStatic = OpAsmParser::UnresolvedOperand{};
365  if (parser.parseEqual())
366  return failure();
367  gangStatic = OpAsmParser::UnresolvedOperand{};
368  if (parser.parseOperand(*gangStatic) ||
369  parser.parseColonType(gangStaticType))
370  return failure();
371  }
372  // FIXME: Why allow optional last commas?
373  (void)parser.parseOptionalComma();
374  if (failed(parser.parseRParen()))
375  return failure();
376  }
377  return success();
378 }

438 LogicalResult acc::LoopOp::verify() {
439  // auto, independent and seq attribute are mutually exclusive.
440  if ((getAuto_() && (getIndependent() || getSeq())) ||
441  (getIndependent() && getSeq())) {
442  return emitError() << "only one of \"" << acc::LoopOp::getAutoAttrStrName()
443  << "\", " << getIndependentAttrName() << ", "
444  << getSeqAttrName()
445  << " can be present at the same time";
446  }
447 
448  // Gang, worker and vector are incompatible with seq.
449  if (getSeq() && (getHasGang() || getHasWorker() || getHasVector()))
450  return emitError("gang, worker or vector cannot appear with the seq attr");
451 
452  // Check non-empty body().
453  if (getRegion().empty())
454  return emitError("expected non-empty body.");
455 
456  return success();
457 }
