24 void transform::detail::checkImplementsTransformOpInterface(
25  StringRef name, MLIRContext *context) {
26  // Since the operation is being inserted into the Transform dialect and the
27  // dialect does not implement the interface fallback, only check for the op
28  // itself having the interface implementation.
29  RegisteredOperationName opName =
30  *RegisteredOperationName::lookup(name, context);
31  assert((opName.hasInterface<TransformOpInterface>() ||
32  opName.hasTrait<OpTrait::IsTerminator>()) &&
33  "non-terminator ops injected into the transform dialect must "
34  "implement TransformOpInterface");
35  assert(opName.hasInterface<MemoryEffectOpInterface>() &&
36  "ops injected into the transform dialect must implement "
37  "MemoryEffectsOpInterface");
38 }

91 Type transform::TransformDialect::parseType(DialectAsmParser &parser) const {
92  StringRef keyword;
93  SMLoc loc = parser.getCurrentLocation();
94  if (failed(parser.parseKeyword(&keyword)))
95  return nullptr;
96 
97  auto it = typeParsingHooks.find(keyword);
98  if (it == typeParsingHooks.end()) {
99  parser.emitError(loc) << "unknown type mnemonic: " << keyword;
100  return nullptr;
101  }
102 
103  return it->getValue()(parser);
104 }

133 LogicalResult transform::TransformDialect::verifyOperationAttribute(
134  Operation *op, NamedAttribute attribute) {
135  if (attribute.getName().getValue() == kWithNamedSequenceAttrName) {
136  if (!op->hasTrait<OpTrait::SymbolTable>()) {
137  return emitError(op->getLoc()) << attribute.getName()
138  << " attribute can only be attached to "
139  "operations with symbol tables";
140  }
141 
142  const mlir::CallGraph callgraph(op);
143  for (auto scc = llvm::scc_begin(&callgraph); !scc.isAtEnd(); ++scc) {
144  if (!scc.hasCycle())
145  continue;
146 
147  // Need to check this here additionally because this verification may run
148  // before we check the nested operations.
149  if ((*scc->begin())->isExternal())
150  return op->emitOpError() << "contains a call to an external operation, "
151  "which is not allowed";
152 
153  Operation *first = (*scc->begin())->getCallableRegion()->getParentOp();
154  InFlightDiagnostic diag = emitError(first->getLoc())
155  << "recursion not allowed in named sequences";
156  for (auto it = std::next(scc->begin()); it != scc->end(); ++it) {
157  // Need to check this here additionally because this verification may
158  // run before we check the nested operations.
159  if ((*it)->isExternal()) {
160  return op->emitOpError() << "contains a call to an external "
161  "operation, which is not allowed";
162  }
163 
164  Operation *current = (*it)->getCallableRegion()->getParentOp();
165  diag.attachNote(current->getLoc()) << "operation on recursion stack";
166  }
167  return diag;
168  }
169  return success();
170  }
171  if (attribute.getName().getValue() == kTargetTagAttrName) {
172  if (!attribute.getValue().isa<StringAttr>()) {
173  return op->emitError()
174  << attribute.getName() << " attribute must be a string";
175  }
176  return success();
177  }
178  if (attribute.getName().getValue() == kArgConsumedAttrName ||
179  attribute.getName().getValue() == kArgReadOnlyAttrName) {
180  if (!attribute.getValue().isa<UnitAttr>()) {
181  return op->emitError()
182  << attribute.getName() << " must be a unit attribute";
183  }
184  return success();
185  }
186  return emitError(op->getLoc())
187  << "unknown attribute: " << attribute.getName();
188 }
