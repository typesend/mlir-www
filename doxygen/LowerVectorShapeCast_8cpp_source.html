43 /// ShapeOp 2D -> 1D downcast serves the purpose of flattening 2-D to 1-D
44 /// vectors progressively on the way to target llvm.matrix intrinsics.
45 /// This iterates over the most major dimension of the 2-D vector and performs
46 /// rewrites into:
47 /// vector.extract from 2-D + vector.insert_strided_slice offset into 1-D
48 class ShapeCastOp2DDownCastRewritePattern
49  : public OpRewritePattern<vector::ShapeCastOp> {
50 public:
51  using OpRewritePattern::OpRewritePattern;
52 
53  LogicalResult matchAndRewrite(vector::ShapeCastOp op,
54  PatternRewriter &rewriter) const override {
55  auto sourceVectorType = op.getSourceVectorType();
56  auto resultVectorType = op.getResultVectorType();
57  if (sourceVectorType.getRank() != 2 || resultVectorType.getRank() != 1)
58  return failure();
59 
60  auto loc = op.getLoc();
61  Value desc = rewriter.create<arith::ConstantOp>(
62  loc, resultVectorType, rewriter.getZeroAttr(resultVectorType));
63  unsigned mostMinorVectorSize = sourceVectorType.getShape()[1];
64  for (int64_t i = 0, e = sourceVectorType.getShape().front(); i != e; ++i) {
65  Value vec = rewriter.create<vector::ExtractOp>(loc, op.getSource(), i);
66  desc = rewriter.create<vector::InsertStridedSliceOp>(
67  loc, vec, desc,
68  /*offsets=*/i * mostMinorVectorSize, /*strides=*/1);
69  }
70  rewriter.replaceOp(op, desc);
71  return success();
72  }
73 };

75 /// ShapeOp 1D -> 2D upcast serves the purpose of unflattening 2-D from 1-D
76 /// vectors progressively.
77 /// This iterates over the most major dimension of the 2-D vector and performs
78 /// rewrites into:
79 /// vector.extract_strided_slice from 1-D + vector.insert into 2-D
80 /// Note that 1-D extract_strided_slice are lowered to efficient vector.shuffle.
81 class ShapeCastOp2DUpCastRewritePattern
82  : public OpRewritePattern<vector::ShapeCastOp> {
83 public:
84  using OpRewritePattern::OpRewritePattern;
85 
86  LogicalResult matchAndRewrite(vector::ShapeCastOp op,
87  PatternRewriter &rewriter) const override {
88  auto sourceVectorType = op.getSourceVectorType();
89  auto resultVectorType = op.getResultVectorType();
90  if (sourceVectorType.getRank() != 1 || resultVectorType.getRank() != 2)
91  return failure();
92 
93  auto loc = op.getLoc();
94  Value desc = rewriter.create<arith::ConstantOp>(
95  loc, resultVectorType, rewriter.getZeroAttr(resultVectorType));
96  unsigned mostMinorVectorSize = resultVectorType.getShape()[1];
97  for (int64_t i = 0, e = resultVectorType.getShape().front(); i != e; ++i) {
98  Value vec = rewriter.create<vector::ExtractStridedSliceOp>(
99  loc, op.getSource(), /*offsets=*/i * mostMinorVectorSize,
100  /*sizes=*/mostMinorVectorSize,
101  /*strides=*/1);
102  desc = rewriter.create<vector::InsertOp>(loc, vec, desc, i);
103  }
104  rewriter.replaceOp(op, desc);
105  return success();
106  }
107 };

109 // We typically should not lower general shape cast operations into data
110 // movement instructions, since the assumption is that these casts are
111 // optimized away during progressive lowering. For completeness, however,
112 // we fall back to a reference implementation that moves all elements
113 // into the right place if we get here.
114 class ShapeCastOpRewritePattern : public OpRewritePattern<vector::ShapeCastOp> {
115 public:
116  using OpRewritePattern::OpRewritePattern;
117 
118  LogicalResult matchAndRewrite(vector::ShapeCastOp op,
119  PatternRewriter &rewriter) const override {
120  Location loc = op.getLoc();
121  auto sourceVectorType = op.getSourceVectorType();
122  auto resultVectorType = op.getResultVectorType();
123 
124  // Special case 2D / 1D lowerings with better implementations.
125  // TODO: make is ND / 1D to allow generic ND -> 1D -> MD.
126  int64_t srcRank = sourceVectorType.getRank();
127  int64_t resRank = resultVectorType.getRank();
128  if ((srcRank == 2 && resRank == 1) || (srcRank == 1 && resRank == 2))
129  return failure();
130 
131  // Generic ShapeCast lowering path goes all the way down to unrolled scalar
132  // extract/insert chains.
133  // TODO: consider evolving the semantics to only allow 1D source or dest and
134  // drop this potentially very expensive lowering.
135  // Compute number of elements involved in the reshape.
136  int64_t numElts = 1;
137  for (int64_t r = 0; r < srcRank; r++)
138  numElts *= sourceVectorType.getDimSize(r);
139  // Replace with data movement operations:
140  // x[0,0,0] = y[0,0]
141  // x[0,0,1] = y[0,1]
142  // x[0,1,0] = y[0,2]
143  // etc., incrementing the two index vectors "row-major"
144  // within the source and result shape.
145  SmallVector<int64_t> srcIdx(srcRank);
146  SmallVector<int64_t> resIdx(resRank);
147  Value result = rewriter.create<arith::ConstantOp>(
148  loc, resultVectorType, rewriter.getZeroAttr(resultVectorType));
149  for (int64_t i = 0; i < numElts; i++) {
150  if (i != 0) {
151  incIdx(srcIdx, sourceVectorType, srcRank - 1);
152  incIdx(resIdx, resultVectorType, resRank - 1);
153  }
154  Value e = rewriter.create<vector::ExtractOp>(loc, op.getSource(), srcIdx);
155  result = rewriter.create<vector::InsertOp>(loc, e, result, resIdx);
156  }
157  rewriter.replaceOp(op, result);
158  return success();
159  }
160 
161 private:
162  static void incIdx(SmallVector<int64_t> &idx, VectorType tp, int64_t r) {
163  assert(0 <= r && r < tp.getRank());
164  if (++idx[r] == tp.getDimSize(r)) {
165  idx[r] = 0;
166  incIdx(idx, tp, r - 1);
167  }
168  }
169 };
170 } // namespace
